35 class TFDispatcher : public o2::framework::Task
36 {
37  public:
38  TFDispatcher(int slot, int ngen, int nlanes, int latency) : mSlot(slot), mNGen(ngen), mNLanes(nlanes), mLatency(latency) {}
39 
40  void init(o2::framework::InitContext& ic) final
41  {
42  mMaxTF = ic.options().get<int64_t>("max-timeframes");
43  }
44 
45  void run(o2::framework::ProcessingContext& pc) final
46  {
47  for (auto& input : pc.inputs()) {
48  auto tfid = header::get<o2::framework::DataProcessingHeader*>(input.header)->startTime;
49  if (tfid >= mMaxTF - 1) {
50  LOG(INFO) << "Data generator reached TF " << tfid << ", stopping";
51  pc.services().get<o2::framework::ControlService>().endOfStream();
52  pc.services().get<o2::framework::ControlService>().readyToQuit(o2::framework::QuitRequest::Me);
53  if (!acceptTF(tfid)) {
54  return;
55  }
56  break;
57  }
58  if (!acceptTF(tfid)) {
59  return;
60  }
61  }
62  int size = 100 + gRandom->Integer(100); // push dummy output
63  usleep(mLatency);
64  pc.outputs().snapshot(o2::framework::OutputRef{"output", 0}, size);
65  }
66 
67  bool acceptTF(int tfid)
68  {
69 
70  // check if the current TF should be processed by this instance of the generator
71 
72  int targetSlot = (tfid / mNLanes) % mNGen;
73  if (targetSlot != mSlot) {
74  LOG(INFO) << "tfid = " << tfid << ", mNLanes = " << mNLanes << ", mNGen = " << mNGen << ", mSlot = " << mSlot << " target slot = " << targetSlot << ": discarded";
75  return false;
76  }
77  LOG(INFO) << "tfid = " << tfid << ", mNLanes = " << mNLanes << ", mNGen = " << mNGen << ", mSlot = " << mSlot << " target slot = " << targetSlot << ": accepted";
78  return true;
79  }
80 
81  private:
82  uint64_t mMaxTF = 1;
83  int mSlot = 1;
84  int mNGen = 1;
85  int mNLanes = 1;
86  int mLatency = 0;
87 };

89 class TFProcessor : public o2::framework::Task
90 {
91  public:
92  TFProcessor(int latency, int latencyRMS) : mMeanLatency(latency), mLatencyRMS(latencyRMS){};
93 
94  void init(o2::framework::InitContext& ic) final
95  {
96  mDevCopy = ic.services().get<const o2::framework::DeviceSpec>().inputTimesliceId;
97  gRandom->SetSeed(mDevCopy);
98  mTOFChannelCalib = ic.options().get<bool>("do-TOF-channel-calib");
99  mTOFChannelCalibInTestMode = ic.options().get<bool>("do-TOF-channel-calib-in-test-mode");
100  LOG(INFO) << "TFProcessorCopy: " << mDevCopy << " MeanLatency: " << mMeanLatency << " LatencyRMS: " << mLatencyRMS << " DoTOFChannelCalib: " << mTOFChannelCalib
101  << " DoTOFChannelCalibInTestMode: " << mTOFChannelCalibInTestMode;
102 
103  for (int i = 0; i < o2::tof::Geo::NCHANNELS; i++) {
104  mChannelShifts[i] = (2000. / o2::tof::Geo::NCHANNELS) * i + (-1000.); // shift needs to be always the same for a channel; in this way we make them all in [-1000, 1000], and shifting with the channel index
105  }
106  }
107 
108  void run(o2::framework::ProcessingContext& pc) final
109  {
110  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
111  // introduceDelay
112  uint32_t delay = std::abs(gRandom->Gaus(mMeanLatency, mLatencyRMS));
113  LOG(INFO) << "TFProcessorCopy: " << mDevCopy << " Simulate latency of " << delay << " mcs for TF " << tfcounter;
114  usleep(delay);
115 
116  // push dummy output
117  auto size = pc.inputs().get<int>("input");
118  auto& output = pc.outputs().make<std::vector<o2::dataformats::CalibInfoTOF>>(o2::framework::OutputRef{"output", 0});
119  output.reserve(size);
120 
121  double clockShift = 1e3 * std::sin(tfcounter / 100. * o2::constants::math::PI); // in ps
122 
123  for (int i = size; i--;) {
124  if (!mTOFChannelCalib) {
125  output.emplace_back(gRandom->Integer(o2::tof::Geo::NCHANNELS), 0, gRandom->Gaus(clockShift, 100.), 0, 0);
126  } else {
127  int channel = mTOFChannelCalibInTestMode ? gRandom->Integer(100) : gRandom->Integer(o2::tof::Geo::NCHANNELS);
128  double value = gRandom->Gaus(mChannelShifts[channel], 100.); // in ps
129  double tot = gRandom->Gaus(12, 2); // in ns
130  output.emplace_back(channel, 0, value, tot, 0);
131  }
132  }
133  }
134 
135  private:
136  int mDevCopy = 0;
137  uint32_t mMeanLatency = 0;
138  uint32_t mLatencyRMS = 1;
139  bool mTOFChannelCalib = false;
140  bool mTOFChannelCalibInTestMode = false;
141  double mChannelShifts[o2::tof::Geo::NCHANNELS];
142 };
