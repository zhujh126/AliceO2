35 void TrackerTraitsCPU::computeLayerTracklets()
36 {
37  PrimaryVertexContext* primaryVertexContext = mPrimaryVertexContext;
38  for (int iLayer{0}; iLayer < mTrkParams.TrackletsPerRoad(); ++iLayer) {
39  if (primaryVertexContext->getClusters()[iLayer].empty() || primaryVertexContext->getClusters()[iLayer + 1].empty()) {
40  continue;
41  }
42 
43  const float3& primaryVertex = primaryVertexContext->getPrimaryVertex();
44  const int currentLayerClustersNum{static_cast<int>(primaryVertexContext->getClusters()[iLayer].size())};
45 
46  for (int iCluster{0}; iCluster < currentLayerClustersNum; ++iCluster) {
47  const Cluster& currentCluster{primaryVertexContext->getClusters()[iLayer][iCluster]};
48 
49  if (primaryVertexContext->isClusterUsed(iLayer, currentCluster.clusterId)) {
50  continue;
51  }
52 
53  const float tanLambda{(currentCluster.zCoordinate - primaryVertex.z) / currentCluster.rCoordinate};
54  const float zAtRmin{tanLambda * (mPrimaryVertexContext->getMinR(iLayer + 1) -
55  currentCluster.rCoordinate) +
56  currentCluster.zCoordinate};
57  const float zAtRmax{tanLambda * (mPrimaryVertexContext->getMaxR(iLayer + 1) -
58  currentCluster.rCoordinate) +
59  currentCluster.zCoordinate};
60 
61  const int4 selectedBinsRect{getBinsRect(currentCluster, iLayer, zAtRmin, zAtRmax,
62  mTrkParams.TrackletMaxDeltaZ[iLayer], mTrkParams.TrackletMaxDeltaPhi)};
63 
64  if (selectedBinsRect.x == 0 && selectedBinsRect.y == 0 && selectedBinsRect.z == 0 && selectedBinsRect.w == 0) {
65  continue;
66  }
67 
68  int phiBinsNum{selectedBinsRect.w - selectedBinsRect.y + 1};
69 
70  if (phiBinsNum < 0) {
71  phiBinsNum += mTrkParams.PhiBins;
72  }
73 
74  for (int iPhiBin{selectedBinsRect.y}, iPhiCount{0}; iPhiCount < phiBinsNum;
75  iPhiBin = ++iPhiBin == mTrkParams.PhiBins ? 0 : iPhiBin, iPhiCount++) {
76  const int firstBinIndex{primaryVertexContext->mIndexTableUtils.getBinIndex(selectedBinsRect.x, iPhiBin)};
77  const int maxBinIndex{firstBinIndex + selectedBinsRect.z - selectedBinsRect.x + 1};
78  const int firstRowClusterIndex = primaryVertexContext->getIndexTables()[iLayer][firstBinIndex];
79  const int maxRowClusterIndex = primaryVertexContext->getIndexTables()[iLayer][maxBinIndex];
80 
81  for (int iNextLayerCluster{firstRowClusterIndex}; iNextLayerCluster < maxRowClusterIndex;
82  ++iNextLayerCluster) {
83 
84  if (iNextLayerCluster >= (int)primaryVertexContext->getClusters()[iLayer + 1].size()) {
85  break;
86  }
87 
88  const Cluster& nextCluster{primaryVertexContext->getClusters()[iLayer + 1][iNextLayerCluster]};
89 
90  if (primaryVertexContext->isClusterUsed(iLayer + 1, nextCluster.clusterId)) {
91  continue;
92  }
93 
94  const float deltaZ{o2::gpu::GPUCommonMath::Abs(tanLambda * (nextCluster.rCoordinate - currentCluster.rCoordinate) +
95  currentCluster.zCoordinate - nextCluster.zCoordinate)};
96  const float deltaPhi{o2::gpu::GPUCommonMath::Abs(currentCluster.phiCoordinate - nextCluster.phiCoordinate)};
97 
98  if (deltaZ < mTrkParams.TrackletMaxDeltaZ[iLayer] &&
99  (deltaPhi < mTrkParams.TrackletMaxDeltaPhi ||
100  o2::gpu::GPUCommonMath::Abs(deltaPhi - constants::math::TwoPi) < mTrkParams.TrackletMaxDeltaPhi)) {
101 
102  if (iLayer > 0 &&
103  primaryVertexContext->getTrackletsLookupTable()[iLayer - 1][iCluster] == constants::its::UnusedIndex) {
104 
105  primaryVertexContext->getTrackletsLookupTable()[iLayer - 1][iCluster] =
106  primaryVertexContext->getTracklets()[iLayer].size();
107  }
108 
109  primaryVertexContext->getTracklets()[iLayer].emplace_back(iCluster, iNextLayerCluster, currentCluster,
110  nextCluster);
111  }
112  }
113  }
114  }
115  if (iLayer > 0 && iLayer < mTrkParams.TrackletsPerRoad() - 1 &&
116  primaryVertexContext->getTracklets()[iLayer].size() > primaryVertexContext->getCellsLookupTable()[iLayer - 1].size()) {
117  throw std::runtime_error(fmt::format("not enough memory in the CellsLookupTable, increase the tracklet memory coefficients: {} tracklets on L{}, lookup table size {} on L{}",
118  primaryVertexContext->getTracklets()[iLayer].size(), iLayer, primaryVertexContext->getCellsLookupTable()[iLayer - 1].size(), iLayer - 1));
119  }
120  }
121 #ifdef CA_DEBUG
122  std::cout << "+++ Number of tracklets per layer: ";
123  for (int iLayer{0}; iLayer < mTrkParams.TrackletsPerRoad(); ++iLayer) {
124  std::cout << primaryVertexContext->getTracklets()[iLayer].size() << "\t";
125  }
126  std::cout << std::endl;
127 #endif
128 }

130 void TrackerTraitsCPU::computeLayerCells()
131 {
132  PrimaryVertexContext* primaryVertexContext = mPrimaryVertexContext;
133  for (int iLayer{0}; iLayer < mTrkParams.CellsPerRoad(); ++iLayer) {
134 
135  if (primaryVertexContext->getTracklets()[iLayer + 1].empty() ||
136  primaryVertexContext->getTracklets()[iLayer].empty()) {
137 
138  return;
139  }
140 
141  const float3& primaryVertex = primaryVertexContext->getPrimaryVertex();
142  const int currentLayerTrackletsNum{static_cast<int>(primaryVertexContext->getTracklets()[iLayer].size())};
143 
144  for (int iTracklet{0}; iTracklet < currentLayerTrackletsNum; ++iTracklet) {
145 
146  const Tracklet& currentTracklet{primaryVertexContext->getTracklets()[iLayer][iTracklet]};
147  const int nextLayerClusterIndex{currentTracklet.secondClusterIndex};
148  const int nextLayerFirstTrackletIndex{
149  primaryVertexContext->getTrackletsLookupTable()[iLayer][nextLayerClusterIndex]};
150 
151  if (nextLayerFirstTrackletIndex == constants::its::UnusedIndex) {
152 
153  continue;
154  }
155 
156  const Cluster& firstCellCluster{primaryVertexContext->getClusters()[iLayer][currentTracklet.firstClusterIndex]};
157  const Cluster& secondCellCluster{
158  primaryVertexContext->getClusters()[iLayer + 1][currentTracklet.secondClusterIndex]};
159  const float firstCellClusterQuadraticRCoordinate{firstCellCluster.rCoordinate * firstCellCluster.rCoordinate};
160  const float secondCellClusterQuadraticRCoordinate{secondCellCluster.rCoordinate *
161  secondCellCluster.rCoordinate};
162  const float3 firstDeltaVector{secondCellCluster.xCoordinate - firstCellCluster.xCoordinate,
163  secondCellCluster.yCoordinate - firstCellCluster.yCoordinate,
164  secondCellClusterQuadraticRCoordinate - firstCellClusterQuadraticRCoordinate};
165  const int nextLayerTrackletsNum{static_cast<int>(primaryVertexContext->getTracklets()[iLayer + 1].size())};
166 
167  for (int iNextLayerTracklet{nextLayerFirstTrackletIndex};
168  iNextLayerTracklet < nextLayerTrackletsNum &&
169  primaryVertexContext->getTracklets()[iLayer + 1][iNextLayerTracklet].firstClusterIndex ==
170  nextLayerClusterIndex;
171  ++iNextLayerTracklet) {
172 
173  const Tracklet& nextTracklet{primaryVertexContext->getTracklets()[iLayer + 1][iNextLayerTracklet]};
174  const float deltaTanLambda{std::abs(currentTracklet.tanLambda - nextTracklet.tanLambda)};
175  const float deltaPhi{std::abs(currentTracklet.phiCoordinate - nextTracklet.phiCoordinate)};
176 
177  if (deltaTanLambda < mTrkParams.CellMaxDeltaTanLambda &&
178  (deltaPhi < mTrkParams.CellMaxDeltaPhi ||
179  std::abs(deltaPhi - constants::math::TwoPi) < mTrkParams.CellMaxDeltaPhi)) {
180 
181  const float averageTanLambda{0.5f * (currentTracklet.tanLambda + nextTracklet.tanLambda)};
182  const float directionZIntersection{-averageTanLambda * firstCellCluster.rCoordinate +
183  firstCellCluster.zCoordinate};
184  const float deltaZ{std::abs(directionZIntersection - primaryVertex.z)};
185 
186  if (deltaZ < mTrkParams.CellMaxDeltaZ[iLayer]) {
187 
188  const Cluster& thirdCellCluster{
189  primaryVertexContext->getClusters()[iLayer + 2][nextTracklet.secondClusterIndex]};
190 
191  const float thirdCellClusterQuadraticRCoordinate{thirdCellCluster.rCoordinate *
192  thirdCellCluster.rCoordinate};
193 
194  const float3 secondDeltaVector{thirdCellCluster.xCoordinate - firstCellCluster.xCoordinate,
195  thirdCellCluster.yCoordinate - firstCellCluster.yCoordinate,
196  thirdCellClusterQuadraticRCoordinate -
197  firstCellClusterQuadraticRCoordinate};
198 
199  float3 cellPlaneNormalVector{math_utils::crossProduct(firstDeltaVector, secondDeltaVector)};
200 
201  const float vectorNorm{std::sqrt(cellPlaneNormalVector.x * cellPlaneNormalVector.x +
202  cellPlaneNormalVector.y * cellPlaneNormalVector.y +
203  cellPlaneNormalVector.z * cellPlaneNormalVector.z)};
204 
205  if (vectorNorm < constants::math::FloatMinThreshold ||
206  std::abs(cellPlaneNormalVector.z) < constants::math::FloatMinThreshold) {
207 
208  continue;
209  }
210 
211  const float inverseVectorNorm{1.0f / vectorNorm};
212  const float3 normalizedPlaneVector{cellPlaneNormalVector.x * inverseVectorNorm,
213  cellPlaneNormalVector.y * inverseVectorNorm,
214  cellPlaneNormalVector.z * inverseVectorNorm};
215  const float planeDistance{-normalizedPlaneVector.x * (secondCellCluster.xCoordinate - primaryVertex.x) -
216  (normalizedPlaneVector.y * secondCellCluster.yCoordinate - primaryVertex.y) -
217  normalizedPlaneVector.z * secondCellClusterQuadraticRCoordinate};
218  const float normalizedPlaneVectorQuadraticZCoordinate{normalizedPlaneVector.z * normalizedPlaneVector.z};
219  const float cellTrajectoryRadius{std::sqrt(
220  (1.0f - normalizedPlaneVectorQuadraticZCoordinate - 4.0f * planeDistance * normalizedPlaneVector.z) /
221  (4.0f * normalizedPlaneVectorQuadraticZCoordinate))};
222  const float2 circleCenter{-0.5f * normalizedPlaneVector.x / normalizedPlaneVector.z,
223  -0.5f * normalizedPlaneVector.y / normalizedPlaneVector.z};
224  const float distanceOfClosestApproach{std::abs(
225  cellTrajectoryRadius - std::sqrt(circleCenter.x * circleCenter.x + circleCenter.y * circleCenter.y))};
226 
227  if (distanceOfClosestApproach >
228  mTrkParams.CellMaxDCA[iLayer]) {
229 
230  continue;
231  }
232 
233  const float cellTrajectoryCurvature{1.0f / cellTrajectoryRadius};
234  if (iLayer > 0 &&
235  primaryVertexContext->getCellsLookupTable()[iLayer - 1][iTracklet] == constants::its::UnusedIndex) {
236 
237  primaryVertexContext->getCellsLookupTable()[iLayer - 1][iTracklet] =
238  primaryVertexContext->getCells()[iLayer].size();
239  }
240 
241  primaryVertexContext->getCells()[iLayer].emplace_back(
242  currentTracklet.firstClusterIndex, nextTracklet.firstClusterIndex, nextTracklet.secondClusterIndex,
243  iTracklet, iNextLayerTracklet, normalizedPlaneVector, cellTrajectoryCurvature);
244  }
245  }
246  }
247  }
248  }
249 #ifdef CA_DEBUG
250  std::cout << "+++ Number of cells per layer: ";
251  for (int iLayer{0}; iLayer < mTrkParams.CellsPerRoad(); ++iLayer) {
252  std::cout << primaryVertexContext->getCells()[iLayer].size() << "\t";
253  }
254  std::cout << std::endl;
255 #endif
256 }
