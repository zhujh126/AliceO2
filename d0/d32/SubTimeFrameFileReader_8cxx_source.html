66 std::size_t SubTimeFrameFileReader::getHeaderStackSize() // throws ios_base::failure
67 {
68  // Expect valid Stack in the file.
69  // First Header must be DataHeader. The size is unknown since there are multiple versions.
70  // Each header in the stack extends BaseHeader
71 
72  // Read first the base header then the rest of the extended header. Keep going until the next flag is set.
73  // reset the file pointer to the original incoming position, so the complete Stack can be read in
74 
75  bool readNextHeader = true;
76  std::size_t lStackSize = 0;
77  DataHeader lBaseHdr; // Use DataHeader since the BaseHeader has no default contructor.
78 
79  const auto lFilePosStart = position();
80 
81  const auto cMaxHeaders = 16; /* make sure we don't loop forever */
82  auto lNumHeaders = 0;
83  while (readNextHeader && (++lNumHeaders <= cMaxHeaders)) {
84  // read BaseHeader only!
85  if (!read_advance(&lBaseHdr, sizeof(BaseHeader))) {
86  return 0;
87  }
88 
89  if (!ignore_nbytes(lBaseHdr.size())) {
90  return 0;
91  }
92 
93  lStackSize += lBaseHdr.size();
94  readNextHeader = (lBaseHdr.next() != nullptr);
95  }
96  // reset the file pointer
97  set_position(lFilePosStart);
98 
99  if (lNumHeaders >= cMaxHeaders) {
100  LOGP(ERROR, "FileReader: Reached max number of headers allowed: {}.", cMaxHeaders);
101  return 0;
102  }
103 
104  return lStackSize;
105 }

107 Stack SubTimeFrameFileReader::getHeaderStack(std::size_t& pOrigsize)
108 {
109  const auto lStackSize = getHeaderStackSize();
110  pOrigsize = lStackSize;
111 
112  if (lStackSize < sizeof(BaseHeader)) {
113  // error in the stream
114  pOrigsize = 0;
115  return Stack{};
116  }
117 
118  std::byte* lStackMem = reinterpret_cast<std::byte*>(peek());
119  if (!ignore_nbytes(lStackSize)) {
120  // error in the stream
121  pOrigsize = 0;
122  return Stack{};
123  }
124 
125  // This must handle different versions of DataHeader
126  // check if DataHeader needs an upgrade by looking at the version number
127  const BaseHeader* lBaseOfDH = BaseHeader::get(lStackMem);
128  if (!lBaseOfDH) {
129  return Stack{};
130  }
131 
132  if (lBaseOfDH->headerVersion < DataHeader::sVersion) {
133  DataHeader lNewDh;
134 
135  // Write over the new DataHeader. We need to update some of the BaseHeader values.
136  assert(sizeof(DataHeader) > lBaseOfDH->size()); // current DataHeader must be larger
137  std::memcpy(&lNewDh, (void*)lBaseOfDH->data(), lBaseOfDH->size());
138 
139  // make sure to bump the version in the BaseHeader.
140  // TODO: Is there a better way?
141  lNewDh.headerSize = sizeof(DataHeader);
142  lNewDh.headerVersion = DataHeader::sVersion;
143 
144  if (lBaseOfDH->headerVersion == 1 || lBaseOfDH->headerVersion == 2) {
145  /* nothing to do for the upgrade */
146  } else {
147  LOGP(ERROR, "FileReader: DataHeader v{} read from file is not upgraded to the current version {}",
148  lBaseOfDH->headerVersion, DataHeader::sVersion);
149  LOGP(ERROR, "Try using a newer version of DataDistribution or file a BUG");
150  }
151 
152  if (lBaseOfDH->size() == lStackSize) {
153  return Stack(lNewDh);
154  } else {
155  assert(lBaseOfDH->size() < lStackSize);
156 
157  return Stack(
158  lNewDh,
159  Stack(lStackMem + lBaseOfDH->size()));
160  }
161  }
162 
163  return Stack(lStackMem);
164 }

168 std::unique_ptr<MessagesPerRoute> SubTimeFrameFileReader::read(FairMQDevice* device, const std::vector<o2f::OutputRoute>& outputRoutes,
169  const std::string& rawChannel, int verbosity)
170 {
171  std::unique_ptr<MessagesPerRoute> messagesPerRoute = std::make_unique<MessagesPerRoute>();
172  auto& msgMap = *messagesPerRoute.get();
173  assert(device);
174 
175  auto findOutputChannel = [&outputRoutes, &rawChannel](const o2::header::DataHeader* h) {
176  if (!rawChannel.empty()) {
177  return std::string{rawChannel};
178  } else {
179  for (auto& oroute : outputRoutes) {
180  LOG(DEBUG) << "comparing with matcher to route " << oroute.matcher << " TSlice:" << oroute.timeslice;
181  if (o2f::DataSpecUtils::match(oroute.matcher, h->dataOrigin, h->dataDescription, h->subSpecification) && ((h->tfCounter % oroute.maxTimeslices) == oroute.timeslice)) {
182  LOG(DEBUG) << "picking the route:" << o2f::DataSpecUtils::describe(oroute.matcher) << " channel " << oroute.channel;
183  return std::string{oroute.channel};
184  }
185  }
186  }
187  LOGP(ERROR, "Failed to find output channel for {}/{}/{} @ timeslice {}", h->dataOrigin.str, h->dataDescription.str, h->subSpecification, h->tfCounter);
188  return std::string{};
189  };
190 
191  auto addPart = [&msgMap](FairMQMessagePtr hd, FairMQMessagePtr pl, const std::string& fairMQChannel) {
192  FairMQParts* parts = nullptr;
193  parts = msgMap[fairMQChannel].get(); // FairMQParts*
194  if (!parts) {
195  msgMap[fairMQChannel] = std::make_unique<FairMQParts>();
196  parts = msgMap[fairMQChannel].get();
197  }
198  parts->AddPart(std::move(hd));
199  parts->AddPart(std::move(pl));
200  };
201 
202  // record current position
203  const auto lTfStartPosition = position();
204 
205  if (lTfStartPosition == size() || !mFileMap.is_open() || eof()) {
206  return nullptr;
207  }
208  auto tfID = sStfId++;
209  std::size_t lMetaHdrStackSize = 0;
210  const DataHeader* lStfMetaDataHdr = nullptr;
211  SubTimeFrameFileMeta lStfFileMeta;
212 
213  auto printStack = [tfID](const o2::header::Stack& st) {
214  auto dph = o2::header::get<o2f::DataProcessingHeader*>(st.data());
215  auto dh = o2::header::get<o2::header::DataHeader*>(st.data());
216  LOGP(INFO, "TF#{} Header for {}/{}/{} @ timeslice {} run {} | {} of {} size {}, TForbit {} | DPH: {}/{}/{}", tfID,
217  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->runNumber,
218  dh->splitPayloadIndex, dh->splitPayloadParts, dh->payloadSize, dh->firstTForbit,
219  dph ? dph->startTime : 0, dph ? dph->duration : 0, dph ? dph->creation : 0);
220  };
221 
222  // Read DataHeader + SubTimeFrameFileMeta
223  auto lMetaHdrStack = getHeaderStack(lMetaHdrStackSize);
224  if (lMetaHdrStackSize == 0) {
225  LOG(ERROR) << "Failed to read the TF file header. The file might be corrupted.";
226  mFileMap.close();
227  return nullptr;
228  }
229 
230  lStfMetaDataHdr = o2::header::DataHeader::Get(lMetaHdrStack.first());
231  if (!read_advance(&lStfFileMeta, sizeof(SubTimeFrameFileMeta))) {
232  return nullptr;
233  }
234 
235  // verify we're actually reading the correct data in
236  if (!(SubTimeFrameFileMeta::getDataHeader().dataDescription == lStfMetaDataHdr->dataDescription)) {
237  LOGP(WARNING, "Reading bad data: SubTimeFrame META header");
238  mFileMap.close();
239  return nullptr;
240  }
241 
242  // prepare to read the TF data
243  const auto lStfSizeInFile = lStfFileMeta.mStfSizeInFile;
244  if (lStfSizeInFile == (sizeof(DataHeader) + sizeof(SubTimeFrameFileMeta))) {
245  LOGP(WARNING, "Reading an empty TF from file. Only meta information present");
246  mFileMap.close();
247  return nullptr;
248  }
249 
250  // check there's enough data in the file
251  if ((lTfStartPosition + lStfSizeInFile) > this->size()) {
252  LOGP(WARNING, "Not enough data in file for this TF. Required: {}, available: {}", lStfSizeInFile, (this->size() - lTfStartPosition));
253  mFileMap.close();
254  return nullptr;
255  }
256 
257  // Index
258  std::size_t lStfIndexHdrStackSize = 0;
259  const DataHeader* lStfIndexHdr = nullptr;
260 
261  // Read DataHeader + SubTimeFrameFileMeta
262  auto lStfIndexHdrStack = getHeaderStack(lStfIndexHdrStackSize);
263  if (lStfIndexHdrStackSize == 0) {
264  mFileMap.close();
265  return nullptr;
266  }
267  lStfIndexHdr = o2::header::DataHeader::Get(lStfIndexHdrStack.first());
268  if (!lStfIndexHdr) {
269  LOG(ERROR) << "Failed to read the TF index structure. The file might be corrupted.";
270  return nullptr;
271  }
272 
273  if (!ignore_nbytes(lStfIndexHdr->payloadSize)) {
274  return nullptr;
275  }
276 
277  // Remaining data size of the TF:
278  // total size in file - meta (hdr+struct) - index (hdr + payload)
279  const auto lStfDataSize = lStfSizeInFile - (lMetaHdrStackSize + sizeof(SubTimeFrameFileMeta)) - (lStfIndexHdrStackSize + lStfIndexHdr->payloadSize);
280 
281  std::int64_t lLeftToRead = lStfDataSize;
282  STFHeader stfHeader{tfID, -1u, -1u};
283 
284  // read <hdrStack + data> pairs
285  while (lLeftToRead > 0) {
286 
287  // allocate and read the Headers
288  std::size_t lDataHeaderStackSize = 0;
289  Stack lDataHeaderStack = getHeaderStack(lDataHeaderStackSize);
290  if (lDataHeaderStackSize == 0) {
291  mFileMap.close();
292  return nullptr;
293  }
294  const DataHeader* lDataHeader = o2::header::DataHeader::Get(lDataHeaderStack.first());
295  if (!lDataHeader) {
296  LOG(ERROR) << "Failed to read the TF HBF DataHeader structure. The file might be corrupted.";
297  mFileMap.close();
298  return nullptr;
299  }
300 
301  o2::header::Stack headerStack{*lDataHeader, o2f::DataProcessingHeader{tfID, 1}};
302  if (stfHeader.runNumber == -1) {
303  stfHeader.id = lDataHeader->tfCounter;
304  stfHeader.runNumber = lDataHeader->runNumber;
305  stfHeader.firstOrbit = lDataHeader->firstTForbit;
306  }
307  const auto fmqChannel = findOutputChannel(lDataHeader);
308  if (fmqChannel.empty()) { // no output channel
309  mFileMap.close();
310  return nullptr;
311  }
312  auto fmqFactory = device->GetChannel(fmqChannel, 0).Transport();
313  auto lHdrStackMsg = fmqFactory->CreateMessage(headerStack.size(), fair::mq::Alignment{64});
314  memcpy(lHdrStackMsg->GetData(), headerStack.data(), headerStack.size());
315  // read the data
316  const std::uint64_t lDataSize = lDataHeader->payloadSize;
317 
318  auto lDataMsg = fmqFactory->CreateMessage(lDataSize, fair::mq::Alignment{64});
319  if (!read_advance(lDataMsg->GetData(), lDataSize)) {
320  return nullptr;
321  }
322  if (verbosity > 0) {
323  if (verbosity > 1 || lDataHeader->splitPayloadIndex == 0) {
324  printStack(headerStack);
325  if (o2::raw::RDHUtils::checkRDH(lDataMsg->GetData()) && verbosity > 2) {
326  o2::raw::RDHUtils::printRDH(lDataMsg->GetData());
327  }
328  }
329  }
330  addPart(std::move(lHdrStackMsg), std::move(lDataMsg), fmqChannel);
331 
332  // update the counter
333  lLeftToRead -= (lDataHeaderStackSize + lDataSize);
334  }
335 
336  if (lLeftToRead < 0) {
337  LOG(ERROR) << "FileRead: Read more data than it is indicated in the META header!";
338  return nullptr;
339  }
340 
341  // add TF acknowledge part
342  o2::header::DataHeader stfDistDataHeader(gDataDescSubTimeFrame, o2::header::gDataOriginFLP, 0, sizeof(STFHeader), 0, 1);
343  stfDistDataHeader.payloadSerializationMethod = o2::header::gSerializationMethodNone;
344  stfDistDataHeader.firstTForbit = stfHeader.firstOrbit;
345  stfDistDataHeader.runNumber = stfHeader.runNumber;
346  stfDistDataHeader.tfCounter = stfHeader.id;
347  stfHeader.id = tfID;
348  const auto fmqChannel = findOutputChannel(&stfDistDataHeader);
349  if (!fmqChannel.empty()) { // no output channel
350  auto fmqFactory = device->GetChannel(fmqChannel, 0).Transport();
351  o2::header::Stack headerStackSTF{stfDistDataHeader, o2f::DataProcessingHeader{tfID}};
352  if (verbosity > 0) {
353  printStack(headerStackSTF);
354  }
355  auto hdMessageSTF = fmqFactory->CreateMessage(headerStackSTF.size(), fair::mq::Alignment{64});
356  auto plMessageSTF = fmqFactory->CreateMessage(stfDistDataHeader.payloadSize, fair::mq::Alignment{64});
357  memcpy(hdMessageSTF->GetData(), headerStackSTF.data(), headerStackSTF.size());
358  memcpy(plMessageSTF->GetData(), &stfHeader, sizeof(STFHeader));
359  addPart(std::move(hdMessageSTF), std::move(plMessageSTF), fmqChannel);
360  }
361 
362  return messagesPerRoute;
363 }
