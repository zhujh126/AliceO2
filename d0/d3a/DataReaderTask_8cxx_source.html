236 bool DataReaderTask::isTimeFrameEmpty(ProcessingContext& pc)
237 {
238  constexpr auto origin = header::gDataOriginTRD;
239  o2::framework::InputSpec dummy{"dummy", framework::ConcreteDataMatcher{origin, header::gDataDescriptionRawData, 0xDEADBEEF}};
240  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload.
241  // frame detected we have no data and send this instead
242  // send empty output so as to not block workflow
243  for (const auto& ref : o2::framework::InputRecordWalker(pc.inputs(), {dummy})) {
244  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
245  if (dh->payloadSize == 0) {
246  LOGP(INFO, "Found blank input input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : ",
247  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize);
248  return true;
249  }
250  }
251  return false;
252 }

254 void DataReaderTask::run(ProcessingContext& pc)
255 {
256  //NB this is run per time frame on the epn.
257  LOG(info) << "TRD Translator Task run";
258  auto dataReadStart = std::chrono::high_resolution_clock::now();
259 
260  if (isTimeFrameEmpty(pc)) {
261  sendData(pc, true); //send the empty tf data.
262  return;
263  }
264  uint64_t total1 = 0, total2 = 0;
265  /* set encoder output buffer */
266  char bufferOut[o2::trd::constants::HBFBUFFERMAX];
267  int loopcounter = 0;
268  auto device = pc.services().get<o2::framework::RawDeviceService>().device();
269  auto outputRoutes = pc.services().get<o2::framework::RawDeviceService>().spec().outputs;
270  auto fairMQChannel = outputRoutes.at(0).channel;
271  /* loop over inputs routes */
272  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
273  if (!iit.isValid()) {
274  continue;
275  }
276  /* loop over input parts */
277  int inputpartscount = 0;
278  int emptyframe = 0;
279  for (auto const& ref : iit) {
280  auto inputprocessingstart = std::chrono::high_resolution_clock::now(); // measure total processing time
281  if (mVerbose) {
282  const auto dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
283  LOGP(info, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : ",
284  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize);
285  }
286  const auto* headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
287  auto payloadIn = ref.payload;
288  auto payloadInSize = headerIn->payloadSize;
289  // const auto dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
290  if (std::string(headerIn->dataDescription.str) != std::string("DISTSUBTIMEFRAMEFLP")) {
291  if (!mCompressedData) { //we have raw data coming in from flp
292  if (mVerbose) {
293  LOG(info) << " parsing non compressed data in the data reader task with a payload of " << payloadInSize << " payload size";
294  }
295  // LOG(info) << "start of data is at ref.payload=0x"<< std::hex << " headerIn->payloadSize:0x" << headerIn->payloadSize <<" headerIn->headerSize:0x" <<headerIn->headerSize;
296  total1 += headerIn->payloadSize;
297  total2 += headerIn->headerSize;
298  // LOG(info) << "start of data is at ref.payload=0x"<< std::hex << " total1:0x" << total1 <<" total2:0x" <<total2;
299  mReader.setDataBuffer(payloadIn);
300  mReader.setDataBufferSize(payloadInSize);
301  mReader.configure(mTrackletHCHeaderState, mOptions);
302  //mReader.setStats(&mTimeFrameStats);
303  mReader.run();
304  mWordsRead += mReader.getWordsRead();
305  mWordsRejected += mReader.getWordsRejected();
306  if (mVerbose) {
307  LOG(info) << "relevant vectors to read : " << mReader.sumTrackletsFound() << " tracklets and " << mReader.sumDigitsFound() << " compressed digits";
308  }
309  } else { // we have compressed data coming in from flp.
310  mCompressedReader.setDataBuffer(payloadIn);
311  mCompressedReader.setDataBufferSize(payloadInSize);
312  mCompressedReader.configure(mOptions);
313  mCompressedReader.run();
314  }
315  } // ignore the input of DISTSUBTIMEFRAMEFLP
316  // auto inputprocessingtime = std::chrono::high_resolution_clock::now() - inputprocessingstart;
317  // LOGP(info, "Input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : processed in {} us",
318  // dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,std::chrono::duration_cast<std::chrono::microseconds>(inputprocessingtime).count());
319  }
320 
321  std::chrono::duration<double, std::milli> dataReadTime = std::chrono::high_resolution_clock::now() - dataReadStart;
322  LOG(info) << "Processing time for Data reading " << std::chrono::duration_cast<std::chrono::milliseconds>(dataReadTime).count() << "ms";
323  if (mRootOutput) {
324  mTimeFrameTime->Fill((int)std::chrono::duration_cast<std::chrono::milliseconds>(dataReadTime).count());
325  }
326  //auto timeframe=mReader.getEventStorage();
327  // (*timeframe)->mTFStats.mTimeTaken= std::chrono::duration_cast<std::chrono::milliseconds>(dataReadTime).count();
328  // (*timeframe)->mTFStats.mDigitsFound = mReader.getDigitsFound();
329  // (*timeframe)->mTFStats.mTrackletsFound = mReader.getTrackletsFound();
330  // (*timeframe)->mTFStats.mDataWordsRead = mWordsRead * 4;
331  // (*timeframe)->mTFStats.mDataWordsRejected = mWordsRejected * 4;
332  if (!mCompressedData) {
333  LOG(info) << "Digits found : " << mReader.getDigitsFound();
334  LOG(info) << "Tracklets found : " << mReader.getTrackletsFound();
335  LOG(info) << "DataRead in :" << mWordsRead * 4 << " bytes";
336  LOG(info) << "DataRejected in :" << mWordsRejected * 4 << " bytes";
337  LOG(info) << "DataRetention :bad/good" << (double)mWordsRejected / (double)mWordsRead << "";
338  LOG(info) << "Total % good data bad/(good+bad)" << (double)mWordsRejected / ((double)mWordsRead + (double)mWordsRejected) * 100.0 << " %";
339  }
340  /* output */
341  sendData(pc, false);
342  }
343 }
