223 template <typename T, typename Iterator, typename BinCenterView>
224 std::vector<double> fitGaus(Iterator first, Iterator last, BinCenterView axisfirst)
225 {
226  TLinearFitter fitter(3, "pol2");
227  TMatrixD mat(3, 3);
228  Double_t kTol = mat.GetTol();
229  fitter.StoreData(kFALSE);
230  fitter.ClearPoints();
231  TVectorD par(3);
232  TVectorD sigma(3);
233  TMatrixD A(3, 3);
234  TMatrixD b(3, 1);
235  T rms = TMath::RMS(first, last);
236  // Markus: return type of std::max_element is an iterator, cannot cast implicitly to double
237  // better use auto
238  // pointer needs to be dereferenced afterwards
239  auto xMax = std::max_element(first, last);
240  auto xMin = std::min_element(first, last);
241  auto nbins = last - first;
242  const double binWidth = double(*xMax - *xMin) / double(nbins);
243 
244  Float_t meanCOG = 0;
245  Float_t rms2COG = 0;
246  Float_t sumCOG = 0;
247 
248  Float_t entries = 0;
249  Int_t nfilled = 0;
250 
251  for (auto iter = first; iter != last; iter++) {
252  entries += *iter;
253  if (*iter > 0) {
254  nfilled++;
255  }
256  }
257 
258  // TODO: Check why this is needed
259  if (*xMax < 4) {
260  throw FitGausError_t::FIT_ERROR;
261  }
262  if (entries < 12) {
263  throw FitGausError_t::FIT_ERROR;
264  }
265 
266  if (rms < kTol) {
267  throw FitGausError_t::FIT_ERROR;
268  }
269 
270  /*
271  fitResult<T, 4> result;
272  result.setParameter<3>(entries);
273  */
274  // create the result, first fill it with all 0's
275  std::vector<double> result;
276  for (int r = 0; r < 5; r++) {
277  result.push_back(0);
278  }
279  // then set the third parameter to entries
280  result.at(3) = entries;
281 
282  int ibin = 0;
283  Int_t npoints = 0;
284  for (auto iter = first, axisiter = axisfirst; iter != last; iter++, axisiter++) {
285  if (nbins > 1) {
286  // Markus: For this we implemented the BinCenterView
287  double x = *axisfirst;
288  double y = *iter;
289  double ey = std::sqrt(y);
290  fitter.AddPoint(&x, y, ey);
291  if (npoints < 3) {
292  A(npoints, 0) = 1;
293  A(npoints, 1) = x;
294  A(npoints, 2) = x * x;
295  b(npoints, 0) = y;
296  meanCOG += x * nbins;
297  rms2COG += x * nbins * x;
298  sumCOG += nbins;
299  }
300  npoints++;
301  }
302  }
303 
304  Double_t chi2 = 0;
305  if (npoints >= 3) {
306  if (npoints == 3) {
307  //analytic calculation of the parameters for three points
308  A.Invert();
309  TMatrixD res(1, 3);
310  res.Mult(A, b);
311  par[0] = res(0, 0);
312  par[1] = res(0, 1);
313  par[2] = res(0, 2);
314  chi2 = -3.;
315  } else {
316  // use fitter for more than three points
317  fitter.Eval();
318  fitter.GetParameters(par);
319  fitter.GetCovarianceMatrix(mat);
320  //result.setChi2(fitter.GetChisquare() / Double_t(npoints));
321  result.at(4) = (fitter.GetChisquare() / Double_t(npoints));
322  }
323  if (TMath::Abs(par[1]) < kTol) {
324  throw FitGausError_t::FIT_ERROR;
325  ;
326  }
327  if (TMath::Abs(par[2]) < kTol) {
328  throw FitGausError_t::FIT_ERROR;
329  ;
330  }
331 
332  T param1 = T(par[1] / (-2. * par[2]));
333  //result.setParameter<1>(param1);
334  //result.setParameter<2>(T(1. / TMath::Sqrt(TMath::Abs(-2. * par[2]))));
335  result.at(1) = param1;
336  result.at(2) = T(1. / TMath::Sqrt(TMath::Abs(-2. * par[2])));
337  auto lnparam0 = par[0] + par[1] * param1 + par[2] * param1 * param1;
338  if (lnparam0 > 307) {
339  throw FitGausError_t::FIT_ERROR;
340  ;
341  }
342  //result.setParameter<0>(TMath::Exp(lnparam0));
343  result.at(0) = TMath::Exp(lnparam0);
344  return result;
345  }
346 
347  if (npoints == 2) {
348  //use center of gravity for 2 points
349  meanCOG /= sumCOG;
350  rms2COG /= sumCOG;
351  /*
352  result.setParameter<0>(xMax);
353  result.setParameter<1>(meanCOG);
354  result.setParameter<2>(TMath::Sqrt(TMath::Abs(meanCOG * meanCOG - rms2COG)));
355  result.setChi2(-2);
356  */
357  result.at(0) = *xMax;
358  result.at(1) = meanCOG;
359  result.at(2) = TMath::Sqrt(TMath::Abs(meanCOG * meanCOG - rms2COG));
360  result.at(4) = -2;
361  }
362  if (npoints == 1) {
363  meanCOG /= sumCOG;
364  /*
365  result.setParameter<0>(xMax);
366  result.setParameter<1>(meanCOG);
367  result.setParameter<2>(binWidth / TMath::Sqrt(12));
368  result.setChi2(-1);
369  */
370  result.at(0) = *xMax;
371  result.at(1) = meanCOG;
372  result.at(2) = binWidth / TMath::Sqrt(12);
373  result.at(4) = -1;
374  }
375  return result;
376 }
