129  void EventManagerFrame::DoTimeTick()
130  {
131  std::unique_lock<std::mutex> lck(mtx, std::defer_lock);
132  bool inTick;
133  lck.lock();
134  inTick = this->inTick;
135  this->inTick = true;
136  lck.unlock();
137  if (inTick) {
138  return;
139  }
140 
141  if (mEventManager->getDataSource()->refresh()) {
142  mEventManager->displayCurrentEvent();
143  }
144  lck.lock();
145  this->inTick = false;
146  lck.unlock();
147  }
