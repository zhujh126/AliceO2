21 std::string NameConf::buildFileName(const std::string_view prefix, const std::string_view delimiter, const std::string_view defPrefix, const std::string_view defName,
22  const std::string_view extension, const std::string_view optDir)
23 {
24  if (o2::utils::Str::pathIsDirectory(prefix)) { // if path is directory, just add to default name, ignoring optional directory optDir argument
25  return o2::utils::Str::concat_string(prefix, "/", defPrefix, delimiter, defName, '.', extension);
26  } else if (!prefix.empty() && o2::utils::Str::pathExists(prefix)) { // explicit file path is provided, use it directly
27  return std::string(prefix);
28  }
29  auto dir = o2::utils::Str::rectifyDirectory(optDir); // directory might have been provided
30  // is the prefix really prefix or a file-name
31  if (!prefix.empty()) {
32  auto path = o2::utils::Str::concat_string(dir, prefix);
33  if (o2::utils::Str::pathExists(path)) {
34  return path;
35  }
36  }
37  return o2::utils::Str::concat_string(dir, prefix.empty() ? defPrefix : prefix, delimiter, defName, '.', extension);
38 }
