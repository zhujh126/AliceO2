81 bool Options::processCommandLine(int argc, char* argv[])
82 {
83  int opt;
84  bool save = false;
85  std::string optionsFileName = "o2eve.json"; // name with options to use
86 
87  // put ':' in the starting of the
88  // string so that program can
89  //distinguish between '?' and ':'
90  while ((opt = getopt(argc, argv, ":d:f:hijop:rsvt")) != -1) {
91  switch (opt) {
92  case 'd':
93  this->mDataFolder = optarg;
94  break;
95  case 'f':
96  this->mFileName = optarg;
97  break;
98  case 'i':
99  this->mIts = true;
100  break;
101  case 'h':
102  std::cout << usage() << std::endl;
103  return false;
104  case 'j':
105  this->mJSON = true;
106  break;
107  case 'o':
108  this->mOnline = true;
109  break;
110  case 'p':
111  optionsFileName = optarg;
112  break;
113  case 'r':
114  this->mRandomTracks = true;
115  break;
116  case 's':
117  save = true;
118  break;
119  case 't':
120  this->mTpc = true;
121  break;
122  case 'v':
123  this->mVsd = true;
124  break;
125  case ':':
126  LOG(ERROR) << "option needs a value: " << char(optopt);
127  LOG(INFO) << usage();
128  return false;
129  case '?':
130  LOG(ERROR) << "unknown option: " << char(optopt);
131  LOG(INFO) << usage();
132  return false;
133  }
134  }
135 
136  // optind is for the extra arguments
137  // which are not parsed
138  for (; optind < argc; optind++) {
139  LOG(ERROR) << "extra arguments: " << argv[optind];
140  LOG(INFO) << usage();
141  return false;
142  }
143 
144  if (save) {
145  this->saveToJSON("o2eve.json");
146  return false;
147  }
148 
149  return true;
150 }

157  rapidjson::Value json(rapidjson::kNumberType);
158  rapidjson::Value online(rapidjson::kNumberType);
159  rapidjson::Value randomTracks(rapidjson::kNumberType);
160  rapidjson::Value tpc(rapidjson::kNumberType);
161  rapidjson::Value vsd(rapidjson::kNumberType);
162 
163  dataFolder.SetString(rapidjson::StringRef(this->dataFolder().c_str()));
164  fileName.SetString(rapidjson::StringRef(this->fileName().c_str()));
165  its.SetBool(this->its());
166  json.SetBool(this->json());
167  online.SetBool(this->online());
168  randomTracks.SetBool(this->randomTracks());
169  tpc.SetBool(this->tpc());
170  vsd.SetBool(this->vsd());
171 
172  rapidjson::Document tree(rapidjson::kObjectType);
173  rapidjson::Document::AllocatorType& allocator = tree.GetAllocator();
174  tree.AddMember("dataFolder", dataFolder, allocator);
175  tree.AddMember("fileName", fileName, allocator);
176  tree.AddMember("its", its, allocator);
177  tree.AddMember("json", json, allocator);
178  tree.AddMember("online", online, allocator);
179  tree.AddMember("randomTracks", randomTracks, allocator);
180  tree.AddMember("tpc", tpc, allocator);
181  tree.AddMember("vsd", vsd, allocator);
182 
183  rapidjson::StringBuffer buffer;
184  rapidjson::PrettyWriter<rapidjson::StringBuffer> writer(buffer);
185  tree.Accept(writer);
186 
187  std::ofstream out(filename);
188  out << buffer.GetString();
189  out.close();
190  return true;
191 }
