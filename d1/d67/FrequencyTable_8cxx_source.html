24 void FrequencyTable::resizeFrequencyTable(symbol_t min, symbol_t max)
25 {
26  LOG(trace) << "start resizing frequency table";
27  internal::RANSTimer t;
28  t.start();
29  // calculate new dimensions
30  const symbol_t newMin = std::min(mMin, min);
31  const symbol_t newMax = std::max(mMax, max);
32  const size_t newSize = newMax - newMin + 1;
33 
34  // empty - init and prevent special treatment of corner cases when resizing.
35  if (mFrequencyTable.empty()) {
36  mFrequencyTable.resize(newSize, 0);
37  }
38 
39  // if the new size is bigger than the old one we need to resize the frequency table
40  if (newSize > mFrequencyTable.size()) {
41  const size_t offset = newMin < mMin ? std::abs(min - mMin) : 0;
42  std::vector<count_t> tmpFrequencyTable;
43  tmpFrequencyTable.reserve(newSize);
44  // insert initial offset if applicable
45  tmpFrequencyTable.insert(std::begin(tmpFrequencyTable), offset, 0);
46  // append current frequency table
47  tmpFrequencyTable.insert(std::end(tmpFrequencyTable), mFrequencyTable.begin(), mFrequencyTable.end());
48  //fill tail with zeroes if applicable
49  const size_t tail = newMax > mMax ? std::abs(max - mMax) : 0;
50  tmpFrequencyTable.insert(std::end(tmpFrequencyTable), tail, 0);
51 
52  mFrequencyTable = std::move(tmpFrequencyTable);
53  }
54 
55  assert(mFrequencyTable.size() == newSize);
56  mMin = newMin;
57  mMax = newMax;
58 
59  t.stop();
60  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
61  LOG(trace) << "done resizing frequency table";
62 }
