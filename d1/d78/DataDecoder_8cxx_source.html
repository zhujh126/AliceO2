75 {

163 void DataDecoder::setFirstOrbitInTF(uint32_t orbit)
164 {
165  constexpr int BCINORBIT = o2::constants::lhc::LHCMaxBunches;
166  constexpr int TWENTYBITSATONE = 0xFFFFF;
167  if (!mFirstOrbitInRun) {
168  LOG(ERROR) << "[setFirstOrbitInTF] first orbit in run not set!";
169  return;
170  }
171  if (orbit < mFirstOrbitInRun) {
172  LOG(ERROR) << "[setFirstOrbitInTF] first TF orbit smaller than first orbit in run!";
173  return;
174  }
175 
176  // the SAMPA BC value at the beginning of the TF is computed by counting the number of orbits
177  // since the beginning of the run, and then multiplying this number by the number of BC in one orbit.
178  // We then take the first 20 bits of the result to emulate the internal SAMPA BC counter.
179  uint64_t nOrbitsInRun = orbit - mFirstOrbitInRun.value();
180  uint64_t bc = (nOrbitsInRun * BCINORBIT + mSampaTimeOffset);
181  uint32_t bc20bits = bc & TWENTYBITSATONE;
182  mSampaTimeFrameStart = SampaTimeFrameStart(orbit, bc20bits);
183 }

187 void DataDecoder::decodeBuffer(gsl::span<const std::byte> buf)
188 {
189  if (mDebug) {
190  std::cout << "\n\n============================\nStart of new buffer\n";
191  }
192  size_t bufSize = buf.size();
193  size_t pageStart = 0;
194  while (bufSize > pageStart) {
195  RDH* rdh = reinterpret_cast<RDH*>(const_cast<std::byte*>(&(buf[pageStart])));
196  if (mDebug) {
197  if (pageStart == 0) {
198  std::cout << "+++\n[decodeBuffer]" << std::endl;
199  } else {
200  std::cout << "---\n[decodeBuffer]" << std::endl;
201  }
202  o2::raw::RDHUtils::printRDH(rdh);
203  }
204  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
205  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
206  if (rdhHeaderSize != 64) {
207  break;
208  }
209  auto pageSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
210 
211  gsl::span<const std::byte> page(reinterpret_cast<const std::byte*>(rdh), pageSize);
212  decodePage(page);
213 
214  pageStart += pageSize;
215  }
216 
217  if (mDebug) {
218  std::cout << "[decodeBuffer] mOrbits size: " << mOrbits.size() << std::endl;
219  dumpOrbits(mOrbits);
220  std::cout << "[decodeBuffer] mDigits size: " << mDigits.size() << std::endl;
221  dumpDigits();
222  }
223 }

227 void DataDecoder::dumpDigits()
228 {
229  for (size_t di = 0; di < mDigits.size(); di++) {
230  auto& d = mDigits[di];
231  auto detID = d.getDetID();
232  auto padID = d.getPadID();

242  uint32_t sampaTime = d.getSampaTime();

244  detID, padID, d.getADC(), orbit, bunchCrossing, sampaTime);
245  std::cout << fmt::format("\tC {} PAD_XY {:+2.2f} , {:+2.2f}", (bend ? (int)0 : (int)1), X, Y);
246  std::cout << std::endl;

262 int32_t DataDecoder::getMergerChannelId(const DsElecId& dsElecId, DualSampaChannelId channel)
263 {
264  auto solarId = dsElecId.solarId();
265  uint32_t dsId = static_cast<uint32_t>(dsElecId.elinkGroupId()) * 5 + dsElecId.elinkIndexInGroup();
266  if (solarId > DataDecoder::sMaxSolarId || dsId >= 40 || channel >= 64) {

275 bool DataDecoder::mergeDigits(uint32_t mergerChannelId, o2::mch::raw::SampaCluster& sc)
276 {
277  static constexpr uint32_t BCROLLOVER = (1 << 20);
278  static constexpr uint32_t ONEADCCLOCK = 4;
279  static constexpr uint32_t MAXNOFSAMPLES = 0x3FF;
280  static constexpr uint32_t TWENTYBITSATONE = 0xFFFFF;
281 
282  // only digits that start at the beginning of the SAMPA window can be merged
283  if (sc.sampaTime != 0) {
284  return false;
285  }
286 
287  auto& mergerCh = mMergerRecords[mergerChannelId];
288 
289  // if there is not previous digit for this channel then no merging is possible
290  if (mergerCh.digitId < 0) {
291  return false;
292  }
293 
294  // time stamp of the digits to be merged
295  uint32_t bcStart = sc.bunchCrossing;
296  // correct for bunch crossing counter rollover if needed
297  if (bcStart < mergerCh.bcEnd) {
298  bcStart += BCROLLOVER;
299  }
300 
301  // if the new digit starts just one ADC clock cycle after the end of the previous,
302  // the it can be merged into the existing one
303  if ((bcStart - mergerCh.bcEnd) != ONEADCCLOCK) {
304  return false;
305  }
306 
307  // add total charge and number of samples to existing digit
308  auto& digit = mDigits[mergerCh.digitId].digit;
309  digit.setADC(digit.getADC() + sc.sum());
310  uint32_t newNofSamples = digit.getNofSamples() + sc.nofSamples();
311  if (newNofSamples > MAXNOFSAMPLES) {
312  newNofSamples = MAXNOFSAMPLES;
313  }
314  digit.setNofSamples(newNofSamples);
315 
316  // update the time stamp of the signal's end
317  mergerCh.bcEnd = (bcStart & TWENTYBITSATONE) + (sc.nofSamples() - 1) * 4;
318 
319  return true;
320 }

334 bool DataDecoder::getPadMapping(const DsElecId& dsElecId, DualSampaChannelId channel, int& deId, int& dsIddet, int& padId)
335 {
336  deId = -1;
337  dsIddet = -1;
338  padId = -1;
339 
340  if (auto opt = mElec2Det(dsElecId); opt.has_value()) {
341  DsDetId dsDetId = opt.value();
342  dsIddet = dsDetId.dsId();
343  deId = dsDetId.deId();
344  }
345  if (mDebug) {
346  auto s = asString(dsElecId);
347  auto ch = fmt::format("{}-CH{:02d}", s, channel);
348  std::cout << ch << " "
349  << "deId " << deId << " dsIddet " << dsIddet << std::endl;
350  }
351 
352  if (deId < 0 || dsIddet < 0 || !isValidDeID(deId)) {
353  LOGP(error, "got invalid DsDetId from dsElecId={}", asString(dsElecId));
354  return false;
355  }
356

358  padId = segment.findPadByFEE(dsIddet, int(channel));
359

362  }
363  return true;

368 bool DataDecoder::addDigit(const DsElecId& dsElecId, DualSampaChannelId channel, const o2::mch::raw::SampaCluster& sc)
369 {
370  int deId, dsIddet, padId;
371  if (!getPadMapping(dsElecId, channel, deId, dsIddet, padId)) {
372  return false;
373  }
374 
375  uint32_t digitadc = sc.sum();
376 
377  if (mDebug) {
378  auto s = asString(dsElecId);
379  auto ch = fmt::format("{}-CH{:02d}", s, channel);
380  std::cout << ch << " "
381  << fmt::format("PAD ({:04d} {:04d} {:04d})\tADC {:06d} TIME ({} {} {:02d}) SIZE {} END {}",
382  deId, dsIddet, padId, digitadc, mOrbit, sc.bunchCrossing, sc.sampaTime, sc.nofSamples(), (sc.sampaTime + sc.nofSamples() - 1))
383  << (((sc.sampaTime + sc.nofSamples() - 1) >= 98) ? " *" : "") << std::endl;
384  }
385 
386  // skip channels not associated to any pad
387  if (padId < 0) {
388  LOGP(error, "got invalid padId from dsElecId={} dualSampaId={} channel={}", asString(dsElecId), dsIddet, channel);
389  return false;
390  }
391 
392  RawDigit digit;
393  digit.digit = o2::mch::Digit(deId, padId, digitadc, 0, sc.nofSamples());
394  digit.info.chip = channel / 32;
395  digit.info.ds = dsElecId.elinkId();
396  digit.info.solar = dsElecId.solarId();
397  digit.info.sampaTime = sc.sampaTime;
398  digit.info.bunchCrossing = sc.bunchCrossing;
399  digit.info.orbit = mOrbit;
400 
401  mDigits.emplace_back(digit);
402 
403  if (mDebug) {
404  RawDigit& lastDigit = mDigits.back();
405  LOGP(info, "DIGIT STORED: ORBIT {} ADC {} DE {} PADID {} TIME {} BXCOUNT {}",
406  mOrbit, lastDigit.getADC(), lastDigit.getDetID(), lastDigit.getPadID(),
407  lastDigit.getSampaTime(), lastDigit.getBunchCrossing());
408  }
409  return true;
410 }

495 int32_t DataDecoder::digitsTimeDiff(uint32_t orbit1, uint32_t bc1, uint32_t orbit2, uint32_t bc2)
496 {
497  // bunch crossings are stored with 20 bits
498  static const int32_t BCROLLOVER = (1 << 20);
499  // number of bunch crossings in one orbit
500  constexpr int BCINORBIT = o2::constants::lhc::LHCMaxBunches;
501 
502  // We use the difference of orbits values to estimate the minimum and maximum allowed
503  // difference in bunch crossings
504  int64_t dOrbit = static_cast<int64_t>(orbit2) - static_cast<int64_t>(orbit1);
505 
506  // Digits might be sent out later than the orbit in which they were recorded.
507  // We account for this by allowing an extra +/- 3 orbits when converting the
508  // difference from orbit numbers to bunch crossings.
509  int64_t dBcMin = (dOrbit - 10) * BCINORBIT;
510  int64_t dBcMax = (dOrbit + 3) * BCINORBIT;
511 
512  // Difference in bunch crossing values
513  int64_t dBc = static_cast<int64_t>(bc2) - static_cast<int64_t>(bc1);
514 
515  if (dBc < dBcMin) {
516  // the difference is too small, so we assume that it needs to be
517  // incremented by one rollover factor
518  dBc += BCROLLOVER;
519  } else if (dBc > dBcMax) {
520  // the difference is too big, so we assume that it needs to be
521  // decremented by one rollover factor
522  dBc -= BCROLLOVER;
523  }
524 
525  return static_cast<int32_t>(dBc);
526 }

530 void DataDecoder::computeDigitsTime(RawDigitVector& digits, SampaTimeFrameStart& sampaTimeFrameStart, bool debug)
531 {
532  constexpr int32_t bcInTF = 256 * o2::constants::lhc::LHCMaxBunches;
533  constexpr int32_t timeInvalid = DataDecoder::tfTimeInvalid;
534  auto setDigitTime = [&](Digit& d, int32_t tfTime) {
535  d.setTime(tfTime);
536  if (debug) {
537  std::cout << "[computeDigitsTime] hit time set to " << d.getTime() << std::endl;
538  }
539  };
540 
541  for (size_t di = 0; di < digits.size(); di++) {
542  Digit& d = digits[di].digit;
543  SampaInfo& info = digits[di].info;
544 
545  int32_t tfTime = 0;
546  uint32_t bc = sampaTimeFrameStart.mBunchCrossing;
547  uint32_t orbit = sampaTimeFrameStart.mOrbit;
548  tfTime = DataDecoder::digitsTimeDiff(orbit, bc, info.orbit, info.getBXTime());
549  if (tfTime >= bcInTF) {
550  LOGP(warning, "DE {} PAD {}: time {} exceeds TF length", d.getDetID(), d.getPadID(), tfTime);
551  }
552  if (debug) {
553  std::cout << "\n[computeDigitsTime] hit " << info.orbit << "," << info.getBXTime()
554  << " tfTime(1) " << orbit << "," << bc << " diff " << tfTime << std::endl;
555  }
556  setDigitTime(d, tfTime);
557  info.tfTime = tfTime;
558 
559  if (debug) {
560  std::cout << " solar " << info.solar << " ds " << info.ds << " chip " << info.chip << std::endl;
561  std::cout << " pad " << d.getDetID() << "," << d.getPadID() << " "
562  << info.orbit << " " << info.tfTime << " " << info.getBXTime() << std::endl;
563  }
564  }
