60  struct SampaInfo {
61  union {
62  uint32_t id = 0;
63  struct {
64  uint32_t chip : 1;
65  uint32_t ds : 6;
66  uint32_t solar : 16;
67  uint32_t unused : 9;
68  };
69  };
70 
71  union {
72  // default value
73  uint64_t time = 0x0000000000000000;
74  struct {
75  uint32_t sampaTime : 10;
76  uint32_t bunchCrossing : 20;
77  uint32_t reserved : 2;
78  uint32_t orbit;
79  };
80  };
81  uint32_t getBXTime() const
82  {
83  return (bunchCrossing + (sampaTime * 4));
84  }
85  int32_t tfTime;
86  bool timeValid() const
87  {
88  return (tfTime != tfTimeInvalid);
89  }
90 
91  bool operator==(const SampaInfo&) const;
92  bool operator<(const SampaInfo& rhs) const
93  {
94  if (id < rhs.id) {
95  return true;
96  } else if (time < rhs.time) {
97  return true;
98  }
99  return false;
100  }
101  };
