60 unsigned int Tracklet::getTrackletWord() const
61 {
62 
63  int rndAdd = 0;
64  int decPlaces = 5; // must be larger than 1 or change the following code
65  // if (decPlaces > 1)
66  rndAdd = (1 << (decPlaces - 1)) + 1;
67  uint32_t trackletWord;
68  unsigned int pid, padrow, slope, offset;
69 
70  pid = ((int)mPID) & 0xff; //8bits
71  padrow = ((int)getZ()) & 0xf; //4bits
72  slope = ((int)mdY) & 0x7ff; //7bits
73  offset = ((int)mY) & 0x1fff; //13bits
74  //mMCMT[cpu] = (pid << 24) | (padrow << 20) | (slope << 13) | offset;
75  trackletWord = (pid << 24) | (padrow << 20) | (slope << 13) | offset;
76 
77  return trackletWord;
78 }
