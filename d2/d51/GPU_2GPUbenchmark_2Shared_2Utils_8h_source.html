82  for (size_t iBuffAddress{0}; iBuffAddress < getMaxChunks(); ++iBuffAddress) {

150 inline void ResultWriter::addBenchmarkEntry(const std::string bName, const std::string type, const int nChunks)
151 {
152  mBenchmarkTrees.emplace_back(new TTree((bName + "_" + type).data(), (bName + "_" + type).data()));
153  mBenchmarkResults.clear();
154  mBenchmarkResults.resize(nChunks);
155  mBenchmarkTrees.back()->Branch("elapsed", &mBenchmarkResults);
156 }

164 {
