33 HBFUtilsInitializer::HBFUtilsInitializer(const o2f::ConfigContext& configcontext, o2f::WorkflowSpec& wf)
34 {
35  auto updateHBFUtils = [&configcontext]() {
36  static bool done = false;
37  if (!done) {
38  bool helpasked = configcontext.helpOnCommandLine(); // if help is asked, don't take for granted that the ini file is there, don't produce an error if it is not!
39  std::string conf = configcontext.options().isSet(HBFConfOpt) ? configcontext.options().get<std::string>(HBFConfOpt) : "";
40  if (!conf.empty() && conf != "none" && !(helpasked && !o2::utils::Str::pathExists(conf))) {
41  o2::conf::ConfigurableParam::updateFromFile(conf, "HBFUtils", true); // update only those values which were not touched yet (provenance == kCODE)
42  }
43  const auto& hbfu = o2::raw::HBFUtils::Instance();
44  hbfu.checkConsistency();
45  done = true;
46  }
47  };
48 
49  const auto& hbfu = o2::raw::HBFUtils::Instance();
50  for (auto& spec : wf) {
51  if (spec.inputs.empty()) {
52  updateHBFUtils();
53  o2f::ConfigParamsHelper::addOptionIfMissing(spec.options, o2f::ConfigParamSpec{"orbit-offset-enumeration", o2f::VariantType::Int64, int64_t(hbfu.getFirstIRofTF({0, hbfu.orbitFirstSampled}).orbit), {"1st injected orbit"}});
54  o2f::ConfigParamsHelper::addOptionIfMissing(spec.options, o2f::ConfigParamSpec{"orbit-multiplier-enumeration", o2f::VariantType::Int64, int64_t(hbfu.nHBFPerTF), {"orbits/TF"}});
55  }
56  }
57 }
