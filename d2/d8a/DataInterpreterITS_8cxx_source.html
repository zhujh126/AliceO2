50 VisualisationEvent DataInterpreterITS::interpretDataForType(TObject* data, EVisualisationDataType type)
51 {
52  TList* list = (TList*)data;
53  Int_t event = ((TVector2*)list->At(2))->X();
54 
55  VisualisationEvent ret_event({.eventNumber = 0,
56  .runNumber = 0,
57  .energy = 0,
58  .multiplicity = 0,
59  .collidingSystem = "",
60  .timeStamp = 0});
61 
62  if (type == Clusters) {
63  its::GeometryTGeo* gman = its::GeometryTGeo::Instance();
64 
65  TFile* clustFile = (TFile*)list->At(1);
66  TTree* clusters = (TTree*)clustFile->Get("o2sim");
67 
68  //Read all clusters to a buffer
69  std::vector<itsmft::Cluster>* clusArr = nullptr;
70  clusters->SetBranchAddress("ITSCluster", &clusArr);
71 
72  std::vector<itsmft::ROFRecord>* clusterROFrames = nullptr;
73  clusters->SetBranchAddress("ITSClustersROF", &clusterROFrames);
74  clusters->GetEntry(0);
75 
76  auto currentClusterROF = clusterROFrames->at(event);
77 
78  int first, last;
79  first = currentClusterROF.getFirstEntry();
80  last = first + currentClusterROF.getNEntries();
81 
82  gsl::span<itsmft::Cluster> mClusters = gsl::make_span(&(*clusArr)[first], last - first);
83 
84  for (const auto& c : mClusters) {
85  const auto& gloC = c.getXYZGloRot(*gman);
86  double xyz[3] = {gloC.X(), gloC.Y(), gloC.Z()};
87  ret_event.addCluster(xyz);
88  }
89  } else if (type == ESD) {
90  TFile* trackFile = (TFile*)list->At(0);
91  TFile* clustFile = (TFile*)list->At(1);
92 
93  TTree* tracks = (TTree*)trackFile->Get("o2sim");
94 
95  TTree* clusters = (TTree*)clustFile->Get("o2sim");
96 
97  //Read all tracks to a buffer
98  std::vector<its::TrackITS>* trkArr = nullptr;
99  tracks->SetBranchAddress("ITSTrack", &trkArr);
100 
101  //Read all track RO frames to a buffer
102  std::vector<itsmft::ROFRecord>* trackROFrames = nullptr;
103  tracks->SetBranchAddress("ITSTracksROF", &trackROFrames);
104 
105  tracks->GetEntry(0);
106 
107  //Read all clusters to a buffer
108  std::vector<itsmft::Cluster>* clusArr = nullptr;
109  clusters->SetBranchAddress("ITSCluster", &clusArr);
110 
111  std::vector<itsmft::ROFRecord>* clusterROFrames = nullptr;
112  clusters->SetBranchAddress("ITSClustersROF", &clusterROFrames);
113  clusters->GetEntry(0);
114 
115  TEveTrackList* trackList = new TEveTrackList("tracks");
116  trackList->IncDenyDestroy();
117  auto prop = trackList->GetPropagator();
118  prop->SetMagField(0.5);
119  prop->SetMaxR(50.);
120 
121  auto currentTrackROF = trackROFrames->at(event);
122 
123  int first, last;
124  first = currentTrackROF.getFirstEntry();
125  last = first + currentTrackROF.getNEntries();
126 
127  gsl::span<its::TrackITS> mTracks = gsl::make_span(&(*trkArr)[first], last - first);
128 
129  for (const auto& rec : mTracks) {
130  std::array<float, 3> p;
131  rec.getPxPyPzGlo(p);
132  TEveRecTrackD t;
133  t.fP = {p[0], p[1], p[2]};
134  t.fSign = (rec.getSign() < 0) ? -1 : 1;
135  TEveTrack* eve_track = new TEveTrack(&t, prop);
136  eve_track->MakeTrack();
137 
138  auto start = eve_track->GetLineStart();
139  auto end = eve_track->GetLineEnd();
140  VisualisationTrack* track = ret_event.addTrack({.charge = rec.getSign(),
141  .energy = 0.0,
142  .ID = 0,
143  .PID = 0,
144  .mass = 0.0,
145  .signedPT = 0.0,
146  .startXYZ = {start.fX, start.fY, start.fZ},
147  .endXYZ = {end.fX, end.fY, end.fZ},
148  .pxpypz = {p[0], p[1], p[2]},
149  .parentID = 0,
150  .phi = 0.0,
151  .theta = 0.0,
152  .helixCurvature = 0.0,
153  .type = 0,
154  .source = ITSSource});
155 
156  for (Int_t i = 0; i < eve_track->GetN(); ++i) {
157  Float_t x, y, z;
158  eve_track->GetPoint(i, x, y, z);
159  track->addPolyPoint(x, y, z);
160  }
161  delete eve_track;
162

169  // tpoints->SetNextPoint(gloC.X(), gloC.Y(), gloC.Z());
170  // }
171  }
172  delete trackList;
173  }
174  return ret_event;
175 }
