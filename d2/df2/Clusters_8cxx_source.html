31 template <class T>
32 bool Clusters::processCluster(const T& cluster, const o2::tpc::Sector sector, const int row)
33 {
34  const int nROC = row < 63 ? int(sector) : int(sector) + 36;
35  const int rocRow = row < 63 ? row : row - 63;
36 
37  const float pad = cluster.getPad();
38 
39  const auto qMax = cluster.getQmax();
40  const auto qTot = cluster.getQtot();
41  const auto sigmaPad = cluster.getSigmaPad();
42  const auto sigmaTime = cluster.getSigmaTime();
43  const auto timeBin = cluster.getTime();
44 
45  float count = mNClusters.getCalArray(nROC).getValue(rocRow, pad);
46  mNClusters.getCalArray(nROC).setValue(rocRow, pad, count + 1);
47 
48  float charge = mQMax.getCalArray(nROC).getValue(rocRow, pad);
49  mQMax.getCalArray(nROC).setValue(rocRow, pad, charge + qMax);
50 
51  charge = mQTot.getCalArray(nROC).getValue(rocRow, pad);
52  mQTot.getCalArray(nROC).setValue(rocRow, pad, charge + qTot);
53 
54  count = mSigmaTime.getCalArray(nROC).getValue(rocRow, pad);
55  mSigmaTime.getCalArray(nROC).setValue(rocRow, pad, count + sigmaTime);
56 
57  count = mSigmaPad.getCalArray(nROC).getValue(rocRow, pad);
58  mSigmaPad.getCalArray(nROC).setValue(rocRow, pad, count + sigmaPad);
59 
60  count = mTimeBin.getCalArray(nROC).getValue(rocRow, pad);
61  mTimeBin.getCalArray(nROC).setValue(rocRow, pad, count + timeBin);
62 
63  return true;
64 }

74  val = mQMax.getValue(cruID.sector(), rowInSector, padInRow);

76

78  mTimeBin.setValue(cruID.sector(), rowInSector, padInRow, val + timeBin);
