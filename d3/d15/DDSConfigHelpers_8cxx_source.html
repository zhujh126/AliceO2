33 void dumpDeviceSpec2DDS(std::ostream& out,
34  const std::vector<DeviceSpec>& specs,
35  const std::vector<DeviceExecution>& executions,
36  const CommandInfo& commandInfo)
37 {
38  out << R"(<topology name="o2-dataflow">)"
39  "\n";
40  assert(specs.size() == executions.size());
41 
42  for (size_t di = 0; di < specs.size(); ++di) {
43  auto& spec = specs[di];
44  auto& execution = executions[di];
45  if (execution.args.empty()) {
46  continue;
47  }
48  out << " "
49  << fmt::format("<decltask name=\"{}\">\n", spec.id);
50  out << " "
51  << R"(<exe reachable="true">)";
52  out << replaceFirstOccurrence(commandInfo.command, "--dds", "--dump") << " | ";
53  for (size_t ei = 0; ei < execution.environ.size(); ++ei) {
54  out << fmt::format(execution.environ[ei],
55  fmt::arg("timeslice0", spec.inputTimesliceId),
56  fmt::arg("timeslice1", spec.inputTimesliceId + 1),
57  fmt::arg("timeslice4", spec.inputTimesliceId + 4))
58  << " ";
59  }
60  std::string accumulatedChannelPrefix;
61  char* s = strdup(execution.args[0]);
62  out << basename(s) << " ";
63  free(s);
64  for (size_t ai = 1; ai < execution.args.size(); ++ai) {
65  const char* arg = execution.args[ai];
66  if (!arg) {
67  break;
68  }
69  // Do not print out the driver client explicitly
70  if (strcmp(arg, "--driver-client-backend") == 0) {
71  ai++;
72  continue;
73  }
74  if (strcmp(arg, "--control") == 0) {
75  ai++;
76  continue;
77  }
78  if (strcmp(arg, "--channel-prefix") == 0 &&
79  ai + 1 < execution.args.size() &&
80  *execution.args[ai + 1] == 0) {
81  ai++;
82  continue;
83  }
84  if (strpbrk(arg, "' ;@") != nullptr || arg[0] == 0) {
85  out << fmt::format(R"("{}" )", arg);
86  } else if (strpbrk(arg, "\"") != nullptr || arg[0] == 0) {
87  out << fmt::format(R"('{}' )", arg);
88  } else {
89  out << fmt::format(R"({} )", arg);
90  }
91  }
92  out << "--plugin dds";
93  if (accumulatedChannelPrefix.empty() == false) {
94  out << " --channel-config \"" << accumulatedChannelPrefix << "\"";
95  }
96  out << "</exe>\n";
97  out << " </decltask>\n";
98  }
99  out << " <declcollection name=\"DPL\">\n <tasks>\n";
100  for (size_t di = 0; di < specs.size(); ++di) {
101  out << fmt::format(" <name>{}</name>\n", specs[di].id);
102  }
103  out << " </tasks>\n </declcollection>\n";
104  out << "</topology>\n";
105 }
