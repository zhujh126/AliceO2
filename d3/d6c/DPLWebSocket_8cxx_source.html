54 void websocket_server_callback(uv_stream_t* stream, ssize_t nread, const uv_buf_t* buf)
55 {
56  WSDPLHandler* server = (WSDPLHandler*)stream->data;
57  assert(server);
58  if (nread == 0) {
59  return;
60  }
61  if (nread == UV_EOF) {
62  LOG(DEBUG) << "websocket_server_callback: communication with driver closed";
63  uv_close((uv_handle_t*)stream, websocket_server_close_callback);
64  return;
65  }
66  if (nread < 0) {
67  LOG(ERROR) << "websocket_server_callback: Error while reading from websocket";
68  uv_close((uv_handle_t*)stream, websocket_server_close_callback);
69  return;
70  }
71  try {
72  parse_http_request(buf->base, nread, server);
73  free(buf->base);
74  } catch (RuntimeErrorRef& ref) {
75  auto& err = o2::framework::error_from_ref(ref);
76  LOG(ERROR) << "Error while parsing request: " << err.what;
77  }
78 }

131 void WSDPLHandler::endHeaders()
132 {
134  if (mHeaders["upgrade"] != "websocket") {
135  throw WSError{400, "Bad Request: not a websocket upgrade"};
136  }
137  if (mHeaders["connection"] != "upgrade") {
138  throw WSError{400, "Bad Request: connection not for upgrade"};
139  }
140  if (mHeaders["sec-websocket-protocol"] != "dpl") {
141  throw WSError{400, "Bad Request: websocket protocol not \"dpl\"."};
142  }
143  if (mHeaders.count("sec-websocket-key") == 0) {
144  throw WSError{400, "Bad Request: sec-websocket-key missing"};
145  }
146  if (mHeaders["sec-websocket-version"] != "13") {
147  throw WSError{400, "Bad Request: wrong protocol version"};
148  }
149  mHandler->headers(mHeaders);
151  LOG(debug) << "Got upgrade request with nonce " << mHeaders["sec-websocket-key"].c_str();
152  std::string reply = encode_websocket_handshake_reply(mHeaders["sec-websocket-key"].c_str());
153  mHandshaken = true;
154 
155  uv_buf_t bfr = uv_buf_init(strdup(reply.data()), reply.size());
156  uv_write_t* info_req = (uv_write_t*)malloc(sizeof(uv_write_t));
157  uv_write(info_req, (uv_stream_t*)mStream, &bfr, 1, ws_handshake_done_callback);
158  auto header = mHeaders.find("x-dpl-pid");
159  if (header != mHeaders.end()) {
160  LOG(debug) << "Driver connected to PID : " << header->second;
161  for (size_t i = 0; i < mServerContext->infos->size(); ++i) {
162  if (std::to_string((*mServerContext->infos)[i].pid) == header->second) {
163  (*mServerContext->controls)[i].controller = new DeviceController{this};
164  break;
165  }
166  }
167  } else {
168  LOG(INFO) << "Connection not bound to a PID";
169  }
170 }

245 void websocket_client_callback(uv_stream_t* stream, ssize_t nread, const uv_buf_t* buf)
246 {
247  DriverClientContext* context = (DriverClientContext*)stream->data;
248  context->state->loopReason |= DeviceState::WS_COMMUNICATION;
249  assert(context->client);
250  if (nread == 0) {
251  return;
252  }
253  if (nread == UV_EOF) {
254  LOG(debug) << "EOF received from server, closing.";
255  uv_read_stop(stream);
256  uv_close((uv_handle_t*)stream, close_client_websocket);
257  return;
258  }
259  if (nread < 0) {
260  // FIXME: improve error message
261  // FIXME: should I close?
262  LOG(ERROR) << "Error while reading from websocket";
263  uv_read_stop(stream);
264  uv_close((uv_handle_t*)stream, close_client_websocket);
265  return;
266  }
267  try {
268  LOG(debug) << "Data received from server";
269  parse_http_request(buf->base, nread, context->client);
270  } catch (RuntimeErrorRef& ref) {
271  auto& err = o2::framework::error_from_ref(ref);
272  LOG(ERROR) << "Error while parsing request: " << err.what;
273  }
274 }

325 void WSDPLClient::endHeaders()
326 {
328  if (mHeaders["upgrade"] != "websocket") {
329  throw runtime_error_f("No websocket upgrade");
330  }
331  if (mHeaders["connection"] != "upgrade") {
332  throw runtime_error_f("No connection upgrade");
333  }
334  if (mHeaders.count("sec-websocket-accept") == 0) {
335  throw runtime_error("sec-websocket-accept not found");
336  }
337 
338  std::string expectedAccept = HTTPParserHelpers::calculateAccept(mNonce.c_str());
339  if (mHeaders["sec-websocket-accept"] != expectedAccept) {
340  throw runtime_error_f(R"(Invalid accept received: "%s", expected "%s")", mHeaders["sec-websocket-accept"].c_str(), expectedAccept.c_str());
341  }
342 
343  LOG(INFO) << "Correctly handshaken websocket connection.";
345  mHandshaken = true;
346  mHandshake();
347 }
