38 template <class T>
39 void o2::globaltracking::RecoContainer::createTracksVariadic(T creator) const
40 {
41  // We go from most complete tracks to least complete ones, taking into account that some track times
42  // do not bear their own kinematics but just constrain the time
43  // As we get more track types functional, this method should be completed
44  // If user provided function creator returns true, then the track is considered as consumed and its contributing
45  // simpler tracks will not be provided to the creator. If it returns false, the creator will be called also
46  // with this simpler contrubutors.
47  // The creator function is called with track kinematics, track GlobalTrackID and track timing information as 2 floats
48  // which depends on the track time:
49  // 1) For track types containing TimeStampWithError ts it is ts.getTimeStamp(), getTimeStampError()
50  // 2) For tracks with asymmetric time uncertainty, e.g. TPC: as mean time of t0-errBwd,t+errFwd and 0.5(errBwd+errFwd), all in TPC time bins
51  // 3) For tracks whose timing is provided as RO frame: as time in \mus for RO frame start since the start of TF, half-duration of RO window.
52 
53  auto start_time = std::chrono::high_resolution_clock::now();
54  constexpr float PS2MUS = 1e-6;
55  std::array<std::vector<uint8_t>, GTrackID::NSources> usedData;
56  auto flagUsed2 = [&usedData](int idx, int src) {
57  if (!usedData[src].empty()) {
58  usedData[src][idx] = 1;
59  }
60  };
61  auto flagUsed = [&usedData, &flagUsed2](const GTrackID gidx) { flagUsed2(gidx.getIndex(), gidx.getSource()); };
62  auto isUsed2 = [&usedData](int idx, int src) { return (!usedData[src].empty()) && (usedData[src][idx] != 0); };
63  auto isUsed = [&usedData, isUsed2](const GTrackID gidx) { return isUsed2(gidx.getIndex(), gidx.getSource()); };
64 
65  // create only for those data types which are used
66  const auto tracksITS = getITSTracks();
67  const auto trkITABRefs = getITSABRefs();
68  const auto tracksMFT = getMFTTracks();
69  const auto tracksMCH = getMCHTracks();
70  const auto tracksTPC = getTPCTracks();
71  const auto tracksTPCITS = getTPCITSTracks();
72  const auto tracksTPCTOF = getTPCTOFTracks(); // TOF-TPC tracks with refit
73  const auto matchesTPCTOF = getTPCTOFMatches(); // and corresponding matches
74  const auto tracksTPCTRD = getTPCTRDTracks<o2::trd::TrackTRD>();
75  const auto matchesITSTPCTOF = getITSTPCTOFMatches(); // just matches, no refit done
76  const auto tofClusters = getTOFClusters();
77  const auto tracksITSTPCTRD = getITSTPCTRDTracks<o2::trd::TrackTRD>();
78 
79  const auto trigTPCTRD = getTPCTRDTriggers();
80 
81  usedData[GTrackID::ITS].resize(tracksITS.size()); // to flag used ITS tracks
82  usedData[GTrackID::MCH].resize(tracksMCH.size()); // to flag used MCH tracks
83  usedData[GTrackID::MFT].resize(tracksMFT.size()); // to flag used MFT tracks
84  usedData[GTrackID::TPC].resize(tracksTPC.size()); // to flag used TPC tracks
85  usedData[GTrackID::ITSTPC].resize(tracksTPCITS.size()); // to flag used ITSTPC tracks
86  usedData[GTrackID::ITSTPCTRD].resize(tracksITSTPCTRD.size()); // to flag used ITSTPCTRD tracks
87  usedData[GTrackID::TPCTRD].resize(tracksTPCTRD.size()); // to flag used TPCTRD tracks
88  usedData[GTrackID::ITSTPCTOF].resize(getITSTPCTOFMatches().size()); // to flag used ITSTPC-TOF matches
89 
90  // ITS-TPC-TRD-TOF
91  // TODO, will flag used ITS-TPC-TRD
92 
93  // ITS-TPC-TRD
94  {
95  const auto trigITSTPCTRD = getITSTPCTRDTriggers();
96  for (unsigned itr = 0; itr < trigITSTPCTRD.size(); itr++) {
97  const auto& trig = trigITSTPCTRD[itr];
98  float t0 = trig.getBCData().differenceInBC(startIR) * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
99  for (unsigned i = trig.getTrackRefs().getFirstEntry(); i < trig.getTrackRefs().getEntriesBound(); i++) {
100  const auto& trc = tracksITSTPCTRD[i];
101  if (isUsed2(i, GTrackID::ITSTPCTRD)) {
102  flagUsed(trc.getRefGlobalTrackId()); // flag seeding ITS-TPC track
103  continue;
104  }
105  if (creator(trc, {i, GTrackID::ITSTPCTRD}, t0, 1e-3)) { // assign 1ns error to BC
106  flagUsed2(i, GTrackID::ITSTPCTRD); // flag itself (is it needed?)
107  flagUsed(trc.getRefGlobalTrackId()); // flag seeding ITS-TPC track
108  }
109  }
110  }
111  }
112 
113  // ITS-TPC-TOF matches, thes are just MatchInfoTOF objects, pointing on ITS-TPC match and TOF cl.
114  {
115 
116  if (matchesITSTPCTOF.size() && (!tofClusters.size() || !tracksTPCITS.size())) {
117  throw std::runtime_error(fmt::format("Global-TOF tracks ({}) require ITS-TPC tracks ({}) and TOF clusters ({})",
118  matchesITSTPCTOF.size(), tracksTPCITS.size(), tofClusters.size()));
119  }
120  for (unsigned i = 0; i < matchesITSTPCTOF.size(); i++) {
121  const auto& match = matchesITSTPCTOF[i];
122  auto gidx = match.getTrackRef(); // this should be corresponding ITS-TPC track
123  if (isUsed(gidx)) { // RS FIXME: THIS IS TEMPORARY, until the TOF matching will use ITS-TPC-TRD as an input
124  continue;
125  }
126  // no need to check isUsed: by construction this ITS-TPC was not used elsewhere
127  const auto& tofCl = tofClusters[match.getTOFClIndex()];
128  float timeTOFMUS = (tofCl.getTime() - match.getLTIntegralOut().getTOF(o2::track::PID::Pion)) * PS2MUS; // tof time in \mus, FIXME: account for time of flight to R TOF
129  const float timeErr = 0.010f; // assume 10 ns error FIXME
130  if (creator(tracksTPCITS[gidx.getIndex()], {i, GTrackID::ITSTPCTOF}, timeTOFMUS, timeErr)) {
131  //flagUsed2(i, GTrackID::TOF); // flag used TOF match // TODO might be not needed
132  flagUsed(gidx); // flag used ITS-TPC tracks
133  }
134  }
135  }
136 
137  // TPC-TRD
138  {
139  const auto trigTPCTRD = getTPCTRDTriggers();
140  for (unsigned itr = 0; itr < trigTPCTRD.size(); itr++) {
141  const auto& trig = trigTPCTRD[itr];
142  float t0 = trig.getBCData().differenceInBC(startIR) * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
143  for (unsigned i = trig.getTrackRefs().getFirstEntry(); i < trig.getTrackRefs().getEntriesBound(); i++) {
144  const auto& trc = tracksTPCTRD[i];
145  if (isUsed2(i, GTrackID::TPCTRD)) {
146  flagUsed(trc.getRefGlobalTrackId()); // flag seeding TPC track
147  continue;
148  }
149  if (creator(trc, {i, GTrackID::TPCTRD}, t0, 1e-6)) { // assign 1ns error to BC
150  flagUsed2(i, GTrackID::TPCTRD); // flag itself (is it needed?)
151  flagUsed(trc.getRefGlobalTrackId()); // flag seeding TPC track
152  }
153  }
154  }
155  }
156 
157  // ITS-TPC matches, may refer to ITS, TPC (TODO: something else?) tracks
158  {
159  for (unsigned i = 0; i < tracksTPCITS.size(); i++) {
160  const auto& matchTr = tracksTPCITS[i];
161  if (isUsed2(i, GTrackID::ITSTPC)) {
162  flagUsed(matchTr.getRefITS()); // flag used ITS tracks or AB tracklets (though the latter is not really necessary)
163  flagUsed(matchTr.getRefTPC()); // flag used TPC tracks
164  continue;
165  }
166  if (creator(matchTr, {i, GTrackID::ITSTPC}, matchTr.getTimeMUS().getTimeStamp(), matchTr.getTimeMUS().getTimeStampError())) {
167  flagUsed2(i, GTrackID::ITSTPC);
168  flagUsed(matchTr.getRefITS()); // flag used ITS tracks or AB tracklets (though the latter is not really necessary)
169  flagUsed(matchTr.getRefTPC()); // flag used TPC tracks
170  }
171  }
172  }
173 
174  // TPC-TOF matches, may refer to TPC (TODO: something else?) tracks
175  {
176  if (matchesTPCTOF.size() && !tracksTPCTOF.size()) {
177  throw std::runtime_error(fmt::format("TPC-TOF matched tracks ({}) require TPCTOF matches ({}) and TPCTOF tracks ({})",
178  matchesTPCTOF.size(), tracksTPCTOF.size()));
179  }
180  for (unsigned i = 0; i < matchesTPCTOF.size(); i++) {
181  const auto& match = matchesTPCTOF[i];
182  const auto& gidx = match.getTrackRef(); // TPC track global idx
183  if (isUsed(gidx)) { // is TPC track already used
184  continue;
185  }
186  const auto& trc = tracksTPCTOF[i];
187  if (creator(trc, {i, GTrackID::TPCTOF}, trc.getTimeMUS().getTimeStamp(), trc.getTimeMUS().getTimeStampError())) {
188  flagUsed(gidx); // flag used TPC tracks
189  }
190  }
191  }
192 
193  // TPC only tracks
194  {
195  int nacc = 0, noffer = 0;
196  for (unsigned i = 0; i < tracksTPC.size(); i++) {
197  if (isUsed2(i, GTrackID::TPC)) { // skip used tracks
198  continue;
199  }
200  const auto& trc = tracksTPC[i];
201  if (creator(trc, {i, GTrackID::TPC}, trc.getTime0() + 0.5 * (trc.getDeltaTFwd() - trc.getDeltaTBwd()), 0.5 * (trc.getDeltaTFwd() + trc.getDeltaTBwd()))) {
202  flagUsed2(i, GTrackID::TPC); // flag used TPC tracks
203  }
204  }
205  }
206 
207  // ITS only tracks
208  {
209  const auto& rofrs = getITSTracksROFRecords();
210  for (unsigned irof = 0; irof < rofrs.size(); irof++) {
211  const auto& rofRec = rofrs[irof];
212  float t0 = rofRec.getBCData().differenceInBC(startIR) * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
213  int trlim = rofRec.getFirstEntry() + rofRec.getNEntries();
214  for (int it = rofRec.getFirstEntry(); it < trlim; it++) {
215  if (isUsed2(it, GTrackID::ITS)) { // skip used tracks
216  continue;
217  }
218  GTrackID gidITS(it, GTrackID::ITS);
219  const auto& trc = getTrack<o2::its::TrackITS>(gidITS);
220  if (creator(trc, gidITS, t0, 0.5)) {
221  flagUsed2(it, GTrackID::ITS);
222  }
223  }
224  }
225  }
226 
227  // MFT only tracks
228  {
229  const auto& rofrs = getMFTTracksROFRecords();
230  for (unsigned irof = 0; irof < rofrs.size(); irof++) {
231  const auto& rofRec = rofrs[irof];
232  float t0 = rofRec.getBCData().differenceInBC(startIR) * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
233  int trlim = rofRec.getFirstEntry() + rofRec.getNEntries();
234  for (int it = rofRec.getFirstEntry(); it < trlim; it++) {
235 
236  GTrackID gidMFT(it, GTrackID::MFT);
237  const auto& trc = getTrack<o2::mft::TrackMFT>(gidMFT);
238  if (creator(trc, gidMFT, t0, 0.5)) {
239  flagUsed2(it, GTrackID::MFT);
240  }
241  }
242  }
243  }
244 
245  // MCH standalone tracks
246  {
247  const auto& rofs = getMCHTracksROFRecords();
248  constexpr float bc2ns = o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
249  for (const auto& rof : rofs) {
250  auto bcWidth = 56;
251  // FIXME (LA): should really be rof.getBCWidth() once
252  // getBCWidth is actually set to a meaningfull value.
253  // For now we hard-code a 1.4 microseconds window for all tracks
254  auto rofMeanBC = rof.getBCData().differenceInBC(startIR) + bcWidth / 2;
255  float t0 = rofMeanBC * bc2ns;
256  float t0err = bc2ns * bcWidth / 2;
257  for (int idx = rof.getFirstIdx(); idx <= rof.getLastIdx(); ++idx) {
258  GTrackID gidMCH(idx, GTrackID::MCH);
259  const auto& trc = getTrack<o2::mch::TrackMCH>(gidMCH);
260  if (creator(trc, gidMCH, t0, t0err)) {
261  flagUsed2(idx, GTrackID::MCH);
262  }
263  }
264  }
265  }
266 
267  auto current_time = std::chrono::high_resolution_clock::now();
268  LOG(INFO) << "RecoContainer::createTracks took " << std::chrono::duration_cast<std::chrono::microseconds>(current_time - start_time).count() * 1e-6 << " CPU s.";
269 }
