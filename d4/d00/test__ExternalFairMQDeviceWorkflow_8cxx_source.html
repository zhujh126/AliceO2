47 std::vector<DataProcessorSpec> defineDataProcessing(ConfigContext const& config)
48 {
49  std::string defaultTransportConfig = config.options().get<std::string>("default-transport");
50  int nRolls = config.options().get<int>("number-of-events");
51  if (defaultTransportConfig == "zeromq") {
52  // nothing to do for the moment
53  } else if (defaultTransportConfig == "shmem") {
54  // nothing to do for the moment
55  } else {
56  throw std::runtime_error("invalid argument for option --default-transport : '" + defaultTransportConfig + "'");
57  }
58  std::vector<DataProcessorSpec> workflow;
59 
61  // a producer process steered by a timer
62  //
63  // the compute callback of the producer
64  auto producerCallback = [nRolls, counter = std::make_shared<int>()](DataAllocator& outputs, ControlService& control) {
65  outputs.make<int>(OutputRef{"data", 0}) = *counter;
66  if (++(*counter) >= nRolls) {
67  // send the end of stream signal, this is transferred by the proxies
68  // and allows to properly terminate downstream devices
69  control.endOfStream();
70  }
71  };
72 
73  workflow.emplace_back(DataProcessorSpec{"producer",
74  {InputSpec{"timer", "TST", "TIMER", 0, Lifetime::Timer}},
75  {OutputSpec{{"data"}, "TST", "DATA", 0, Lifetime::Timeframe}},
76  AlgorithmSpec{adaptStateless(producerCallback)},
77  {ConfigParamSpec{"period-timer", VariantType::Int, 100000, {"period of timer"}}}});
78 
80  // the dpl sink proxy process
81 
82  // use the OutputChannelSpec as a tool to create the default configuration for the out-of-band channel
83  OutputChannelSpec externalChannelSpec;
84  // Note: the name is hardcoded for now
85  externalChannelSpec.name = "downstream";
86  externalChannelSpec.type = ChannelType::Push;
87  externalChannelSpec.method = ChannelMethod::Bind;
88  externalChannelSpec.hostname = "localhost";
89  externalChannelSpec.port = 42042;
90  externalChannelSpec.listeners = 0;
91  if (!defaultTransportConfig.empty()) {
92  if (defaultTransportConfig == "zeromq") {
93  externalChannelSpec.protocol = ChannelProtocol::Network;
94  } else if (defaultTransportConfig == "shmem") {
95  externalChannelSpec.protocol = ChannelProtocol::IPC;
96  }
97  }
98  std::string channelConfig = formatExternalChannelConfiguration(externalChannelSpec);
99  // at some point the formatting tool might add the transport as well so we have to check
100  if (!defaultTransportConfig.empty() && defaultTransportConfig.find("transport=") == std::string::npos) {
101  channelConfig += ",transport=" + defaultTransportConfig;
102  }
103 
104  Inputs sinkInputs = {InputSpec{"external", "TST", "DATA", 0, Lifetime::Timeframe}};
105  auto channelSelector = [](InputSpec const&, const std::unordered_map<std::string, std::vector<FairMQChannel>>&) -> std::string {
106  return "downstream";
107  };
108  workflow.emplace_back(std::move(specifyFairMQDeviceMultiOutputProxy("dpl-sink", sinkInputs, channelConfig.c_str(), channelSelector)));
109 
111  // a simple checker process subscribing to the output of the input proxy
112  //
113  // the compute callback of the checker
114  auto counter = std::make_shared<int>(0);
115  auto checkerCallback = [counter](InputRecord& inputs, ControlService& control) {
116  LOG(DEBUG) << "got inputs " << inputs.size();
117  ASSERT_ERROR(inputs.get<int>("datain") == *counter);
118  ++(*counter);
119  };
120  auto checkCounter = [counter, nRolls](EndOfStreamContext&) {
121  ASSERT_ERROR(*counter == nRolls);
122  if (*counter == nRolls) {
123  LOG(INFO) << "checker has received " << nRolls << " successful event(s)";
124  }
125  };
126  auto checkerInit = [checkerCallback, checkCounter](CallbackService& callbacks) {
127  callbacks.set(CallbackService::Id::EndOfStream, checkCounter);
128  return adaptStateless(checkerCallback);
129  };
130 
131  // the checker process connects to the proxy
132  workflow.emplace_back(DataProcessorSpec{"checker",
133  {InputSpec{"datain", "PRX", "DATA", 0, Lifetime::Timeframe}},
134  {},
135  AlgorithmSpec{adaptStateful(checkerInit)}});
136 
138  // the input proxy process
139  // reads the messages from the output proxy via the out-of-band channel
140 
141  // converter callback for the external FairMQ device proxy ProcessorSpec generator
142  auto converter = [](FairMQDevice& device, FairMQParts& inputs, ChannelRetriever channelRetriever) {
143  ASSERT_ERROR(inputs.Size() >= 2);
144  if (inputs.Size() < 2) {
145  return;
146  }
147  int msgidx = 0;
148  auto dh = o2::header::get<o2::header::DataHeader*>(inputs.At(msgidx)->GetData());
149  if (!dh) {
150  LOG(ERROR) << "data on input " << msgidx << " does not follow the O2 data model, DataHeader missing";
151  return;
152  }
153  auto dph = o2::header::get<DataProcessingHeader*>(inputs.At(msgidx)->GetData());
154  if (!dph) {
155  LOG(ERROR) << "data on input " << msgidx << " does not follow the O2 data model, DataProcessingHeader missing";
156  return;
157  }
158  // Note: we want to run both the output and input proxy in the same workflow and thus we need
159  // different data identifiers and change the data origin in the forwarding
160  OutputSpec query{"PRX", dh->dataDescription, dh->subSpecification};
161  auto channelName = channelRetriever(query, dph->startTime);
162  bool isData = DataSpecUtils::match(OutputSpec{"TST", "DATA", 0}, dh->dataOrigin, dh->dataDescription, dh->subSpecification);
163  // for the configured data channel we require the channel name, the EOS message containing
164  // the forwarded SourceInfoHeader created by the output proxy will be skipped here since the
165  // input proxy handles this internally
166  ASSERT_ERROR(!isData || !channelName.empty());
167  LOG(DEBUG) << "using channel '" << channelName << "' for " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification});
168  if (channelName.empty()) {
169  return;
170  }
171  // make a copy of the header message, get the data header and change origin
172  auto outHeaderMessage = device.NewMessageFor(channelName, 0, inputs.At(msgidx)->GetSize());
173  memcpy(outHeaderMessage->GetData(), inputs.At(msgidx)->GetData(), inputs.At(msgidx)->GetSize());
174  // this we obviously need to fix in the get API, const'ness of the returned header pointer
175  // should depend on const'ness of the buffer
176  auto odh = const_cast<o2::header::DataHeader*>(o2::header::get<o2::header::DataHeader*>(outHeaderMessage->GetData()));
177  odh->dataOrigin = o2::header::DataOrigin("PRX");
178  FairMQParts output;
179  output.AddPart(std::move(outHeaderMessage));
180  output.AddPart(std::move(inputs.At(msgidx + 1)));
181  LOG(DEBUG) << "sending " << DataSpecUtils::describe(OutputSpec{odh->dataOrigin, odh->dataDescription, odh->subSpecification});
182  o2::framework::sendOnChannel(device, output, channelName);
183  };
184 
185  // we use the same spec to build the configuration string, ideally we would have some helpers
186  // which convert an OutputChannelSpec to an InputChannelSpec replacing 'bind' <--> 'connect'
187  // and 'push' <--> 'pull'
188  //
189  // skip the name in the configuration string as it is added in specifyExternalFairMQDeviceProxy
190  externalChannelSpec.name = "";
191  externalChannelSpec.type = ChannelType::Pull;
192  externalChannelSpec.method = ChannelMethod::Connect;
193  channelConfig = formatExternalChannelConfiguration(externalChannelSpec);
194  if (!defaultTransportConfig.empty() && defaultTransportConfig.find("transport=") == std::string::npos) {
195  channelConfig += ",transport=" + defaultTransportConfig;
196  }
197 
198  // Note: in order to make the DPL output proxy and an input proxy working in the same
199  // workflow, we use different data description
200  Outputs inputProxyOutputs = {OutputSpec{"PRX", "DATA", 0, Lifetime::Timeframe}};
201  workflow.emplace_back(specifyExternalFairMQDeviceProxy(
202  "input-proxy",
203  std::move(inputProxyOutputs),
204  channelConfig.c_str(),
205  converter));
206 
207  if (config.options().get<bool>("output-proxy-only")) {
208  // remove the input proxy and checker from the workflow
209  workflow.pop_back();
210  workflow.pop_back();
211  }
212  return workflow;
213 }
