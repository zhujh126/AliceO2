34 void SVertexer::process(const o2::globaltracking::RecoContainer& recoData) // accessor to various reconstrucred data types
35 {
36  updateTimeDependentParams(); // TODO RS: strictly speaking, one should do this only in case of the CCDB objects update
37  mPVertices = recoData.getPrimaryVertices();
38  buildT2V(recoData); // build track->vertex refs from vertex->track (if other workflow will need this, consider producing a message in the VertexTrackMatcher)
39  int ntrP = mTracksPool[POS].size(), ntrN = mTracksPool[NEG].size(), iThread = 0;
40  mV0sTmp[0].clear();
41  mCascadesTmp[0].clear();
42 
43 #ifdef WITH_OPENMP
44  omp_set_num_threads(mNThreads);
45  int dynGrp = std::min(4, std::max(1, mNThreads / 2));
46 #pragma omp parallel for schedule(dynamic, dynGrp)
47 #endif
48  for (int itp = 0; itp < ntrP; itp++) {
49  auto& seedP = mTracksPool[POS][itp];
50  for (int itn = mVtxFirstTrack[NEG][seedP.vBracket.getMin()]; itn < ntrN; itn++) { // start from the 1st negative track of lowest-ID vertex of positive
51  auto& seedN = mTracksPool[NEG][itn];
52  if (seedN.vBracket > seedP.vBracket) { // all vertices compatible with seedN are in future wrt that of seedP
53  break;
54  }
55 #ifdef WITH_OPENMP
56  iThread = omp_get_thread_num();
57 #endif
58  checkV0(seedP, seedN, itp, itn, iThread);
59  }
60  }
61 #ifdef WITH_OPENMP
62  for (int i = 1; i < mNThreads; i++) { // merge results of all threads
63  for (auto& casc : mCascadesTmp[i]) { // before merging fix cascades references on v0
64  casc.setV0ID(casc.getV0ID() + mV0sTmp[0].size());
65  }
66  mV0sTmp[0].insert(mV0sTmp[0].end(), mV0sTmp[i].begin(), mV0sTmp[i].end());
67  mCascadesTmp[0].insert(mCascadesTmp[0].end(), mCascadesTmp[i].begin(), mCascadesTmp[i].end());
68  mV0sTmp[i].clear();
69  mCascadesTmp[i].clear();
70  }
71 #endif
72  LOG(INFO) << "DONE : " << mV0sTmp[0].size() << " " << mCascadesTmp[0].size();
73 }

118  return;
119  }

136  fitter.setBz(bz);
137  fitter.setUseAbsDCA(mSVParams->useAbsDCA);

148 void SVertexer::buildT2V(const o2::globaltracking::RecoContainer& recoData) // accessor to various tracks
149 {
150  // build track->vertices from vertices->tracks, rejecting vertex contributors
151  auto trackIndex = recoData.getPrimaryVertexMatchedTracks(); // Global ID's for associated tracks
152  auto vtxRefs = recoData.getPrimaryVertexMatchedTrackRefs(); // references from vertex to these track IDs
153 
154  // track selector: at the moment reject prompt tracks contributing to vertex fit and unconstrained TPC tracks
155  auto selTrack = [&](GIndex gid) {
156  return (gid.isPVContributor() || !recoData.isTrackSourceLoaded(gid.getSource())) ? false : true;
157  };
158 
159  std::unordered_map<GIndex, std::pair<int, int>> tmap;
160  int nv = vtxRefs.size() - 1; // The last entry is for unassigned tracks, ignore them
161  for (int i = 0; i < 2; i++) {
162  mTracksPool[i].clear();
163  mVtxFirstTrack[i].clear();
164  mVtxFirstTrack[i].resize(nv, -1);
165  }
166 
167  for (int iv = 0; iv < nv; iv++) {
168  const auto& vtref = vtxRefs[iv];
169  int it = vtref.getFirstEntry(), itLim = it + vtref.getEntries();
170  for (; it < itLim; it++) {
171  auto tvid = trackIndex[it];
172  if (!selTrack(tvid)) {
173  continue;
174  }
175  std::decay_t<decltype(tmap.find(tvid))> tref{};
176  if (tvid.isAmbiguous()) {
177  auto tref = tmap.find(tvid);
178  if (tref != tmap.end()) {
179  mTracksPool[tref->second.second][tref->second.first].vBracket.setMax(iv); // this track was already processed with other vertex, account the latter
180  continue;
181  }
182  }
183  const auto& trc = recoData.getTrackParam(tvid);
184  int posneg = trc.getSign() < 0 ? 1 : 0;
185  mTracksPool[posneg].emplace_back(TrackCand{trc, tvid, {iv, iv}});
186  if (tvid.isAmbiguous()) { // track attached to >1 vertex, remember that it was already processed
187  tmap[tvid] = {mTracksPool[posneg].size() - 1, posneg};
188  }
189  }
190  }
191  // register 1st track of each charge for each vertex
192 
193  for (int pn = 0; pn < 2; pn++) {
194  auto& vtxFirstT = mVtxFirstTrack[pn];
195  const auto& tracksPool = mTracksPool[pn];
196  for (unsigned i = 0; i < tracksPool.size(); i++) {
197  const auto& t = tracksPool[i];
198  if (vtxFirstT[t.vBracket.getMin()] == -1) {
199  vtxFirstT[t.vBracket.getMin()] = i;
200  }
201  }
202  }
203 
204  LOG(INFO) << "Collected " << mTracksPool[POS].size() << " positive and " << mTracksPool[NEG].size() << " negative seeds";
205 }

208 bool SVertexer::checkV0(TrackCand& seedP, TrackCand& seedN, int iP, int iN, int ithread)
209 {
210  auto& fitterV0 = mFitterV0[ithread];
211  int nCand = fitterV0.process(seedP, seedN);
212  if (nCand == 0) { // discard this pair
213  return false;
214  }
215  const auto& v0XYZ = fitterV0.getPCACandidate();
216  // check closeness to the beam-line
217  float dxv0 = v0XYZ[0] - mMeanVertex.getX(), dyv0 = v0XYZ[1] - mMeanVertex.getY(), r2v0 = dxv0 * dxv0 + dyv0 * dyv0;
218  if (r2v0 < mMinR2ToMeanVertex) {
219  return false;
220  }
221  if (!fitterV0.isPropagateTracksToVertexDone() && !fitterV0.propagateTracksToVertex()) {
222  return false;
223  }
224  int cand = 0;
225  auto& trPProp = fitterV0.getTrack(0, cand);
226  auto& trNProp = fitterV0.getTrack(1, cand);
227  std::array<float, 3> pP, pN;
228  trPProp.getPxPyPzGlo(pP);
229  trNProp.getPxPyPzGlo(pN);
230  // estimate DCA of neutral V0 track to beamline: straight line with parametric equation
231  // x = X0 + pV0[0]*t, y = Y0 + pV0[1]*t reaches DCA to beamline (Xv, Yv) at
232  // t = -[ (x0-Xv)*pV0[0] + (y0-Yv)*pV0[1]) ] / ( pT(pV0)^2 )
233  // Similar equation for 3D distance involving pV0[2]
234  std::array<float, 3> pV0 = {pP[0] + pN[0], pP[1] + pN[1], pP[2] + pN[2]};
235  float pt2V0 = pV0[0] * pV0[0] + pV0[1] * pV0[1], prodXYv0 = dxv0 * pV0[0] + dyv0 * pV0[1], tDCAXY = prodXYv0 / pt2V0;
236  float p2V0 = pt2V0 + pV0[2] * pV0[2], ptV0 = std::sqrt(pt2V0);
237  // apply mass selections
238  float p2Pos = pP[0] * pP[0] + pP[1] * pP[1] + pP[2] * pP[2], p2Neg = pN[0] * pN[0] + pN[1] * pN[1] + pN[2] * pN[2];
239 
240  bool goodHyp = false;
241  std::array<bool, NHypV0> hypCheckStatus{};
242  for (int ipid = 0; ipid < NHypV0; ipid++) {
243  if (mV0Hyps[ipid].check(p2Pos, p2Neg, p2V0, ptV0)) {
244  goodHyp = hypCheckStatus[ipid] = true;
245  }
246  }
247  if (!goodHyp) {
248  return false;
249  }
250 
251  bool checkForCascade = mEnableCascades && r2v0 < mMaxR2ToMeanVertexCascV0 && (hypCheckStatus[HypV0::Lambda] || hypCheckStatus[HypV0::AntiLambda]);
252  bool rejectIfNotCascade = false;
253  float dcaX = dxv0 - pV0[0] * tDCAXY, dcaY = dyv0 - pV0[1] * tDCAXY, dca2 = dcaX * dcaX + dcaY * dcaY;
254  float cosPAXY = prodXYv0 / std::sqrt(r2v0 * pt2V0);
255 
256  if (checkForCascade) { // use loser cuts for cascade v0 candidates
257  if (dca2 > mMaxDCAXY2ToMeanVertexV0Casc || cosPAXY < mSVParams->minCosPAXYMeanVertexCascV0) {
258  return false;
259  }
260  }
261  if (dca2 > mMaxDCAXY2ToMeanVertex || cosPAXY < mSVParams->minCosPAXYMeanVertex) {
262  if (checkForCascade) {
263  rejectIfNotCascade = true;
264  } else {
265  return false;
266  }
267  }
268 
269  auto vlist = seedP.vBracket.getOverlap(seedN.vBracket); // indices of vertices shared by both seeds
270  if (vlist.isInvalid()) {
271  LOG(WARNING) << "Incompatible tracks: V0 " << seedP.vBracket.asString() << " | V1 " << seedN.vBracket.asString();
272  return false;
273  }
274 
275  bool added = false;
276  auto bestCosPA = checkForCascade ? mSVParams->minCosPACascV0 : mSVParams->minCosPA;
277  for (int iv = vlist.getMin(); iv <= vlist.getMax(); iv++) {
278  const auto& pv = mPVertices[iv];
279  const auto v0XYZ = fitterV0.getPCACandidatePos(cand);
280  // check cos of pointing angle
281  float dx = v0XYZ[0] - pv.getX(), dy = v0XYZ[1] - pv.getY(), dz = v0XYZ[2] - pv.getZ(), prodXYZv0 = dx * pV0[0] + dy * pV0[1] + dz * pV0[2];
282  float cosPA = prodXYZv0 / std::sqrt((dx * dx + dy * dy + dz * dz) * p2V0);
283  if (cosPA < bestCosPA) {
284  continue;
285  }
286  if (!added) {
287  auto& v0new = mV0sTmp[ithread].emplace_back(v0XYZ, pV0, fitterV0.calcPCACovMatrixFlat(cand), trPProp, trNProp, seedP.gid, seedN.gid);
288  v0new.setDCA(fitterV0.getChi2AtPCACandidate());
289  added = true;
290  }
291  auto& v0 = mV0sTmp[ithread].back();
292  v0.setCosPA(cosPA);
293  v0.setVertexID(iv);
294  bestCosPA = cosPA;
295  }
296  if (!added) {
297  return false;
298  }
299  auto& v0 = mV0sTmp[ithread].back();
300 
301  // check cascades
302  if (checkForCascade) {
303  int nCascAdded = 0;
304  if (hypCheckStatus[HypV0::Lambda]) {
305  nCascAdded += checkCascades(r2v0, pV0, p2V0, iN, NEG, ithread);
306  }
307  if (hypCheckStatus[HypV0::AntiLambda]) {
308  nCascAdded += checkCascades(r2v0, pV0, p2V0, iP, POS, ithread);
309  }
310  if (!nCascAdded && rejectIfNotCascade) { // v0 would be accepted only if it creates a cascade
311  mV0sTmp[ithread].pop_back();
312  return false;
313  }
314  }
315 
316  return true;
317 }

320 int SVertexer::checkCascades(float r2v0, std::array<float, 3> pV0, float p2V0, int avoidTrackID, int posneg, int ithread)
321 {
322  // check last added V0 for belonging to cascade
323  auto& fitterCasc = mFitterCasc[ithread];
324  const auto& v0 = mV0sTmp[ithread].back();
325  auto& tracks = mTracksPool[posneg];
326  const auto& pv = mPVertices[v0.getVertexID()];
327  int nCascIni = mCascadesTmp[ithread].size();
328  // start from the 1st track compatible with V0's primary vertex
329  for (unsigned it = mVtxFirstTrack[posneg][v0.getVertexID()]; it < tracks.size(); it++) {
330  if (it == avoidTrackID) {
331  continue; // skip the track used by V0
332  }
333  auto& bach = tracks[it];
334  if (bach.vBracket > v0.getVertexID()) {
335  break; // all other bachelor candidates will be also not compatible with this PV
336  }
337  if (bach.vBracket.isOutside(v0.getVertexID())) {
338  LOG(ERROR) << "Incompatible bachelor: PV " << bach.vBracket.asString() << " vs V0 " << v0.getVertexID();
339  }
340  int nCandC = fitterCasc.process(v0, bach);
341  if (nCandC == 0) { // discard this pair
342  continue;
343  }
344  int candC = 0;
345  const auto& cascXYZ = fitterCasc.getPCACandidatePos(candC);
346  // make sure the cascade radius is smaller than that of the vertex
347  float dxc = cascXYZ[0] - pv.getX(), dyc = cascXYZ[1] - pv.getY(), dzc = cascXYZ[2] - pv.getZ(), r2casc = dxc * dxc + dyc * dyc;
348  if (r2v0 - r2casc < mMinR2DiffV0Casc || r2casc < mMinR2ToMeanVertex) {
349  continue;
350  }
351  // do we want to apply mass cut ?
352  //
353  if (!fitterCasc.isPropagateTracksToVertexDone() && !fitterCasc.propagateTracksToVertex()) {
354  continue;
355  }
356  auto& trNeut = fitterCasc.getTrack(0, candC);
357  auto& trBach = fitterCasc.getTrack(1, candC);
358  trNeut.setPID(o2::track::PID::Lambda);
359  trBach.setPID(o2::track::PID::Pion);
360  std::array<float, 3> pNeut, pBach;
361  trNeut.getPxPyPzGlo(pNeut);
362  trBach.getPxPyPzGlo(pBach);
363  std::array<float, 3> pCasc = {pNeut[0] + pBach[0], pNeut[1] + pBach[1], pNeut[2] + pBach[2]};
364  auto prodPPos = pV0[0] * dxc + pV0[1] * dyc + pV0[2] * dzc;
365  if (prodPPos < 0.) { // causality cut
366  continue;
367  }
368  float pt2Casc = pCasc[0] * pCasc[0] + pCasc[1] * pCasc[1], p2Casc = pt2Casc + pCasc[2] * pCasc[2];
369  float cosPA = (pCasc[0] * dxc + pCasc[1] * dyc + pCasc[2] * dzc) / std::sqrt(p2Casc * (r2casc + dzc * dzc));
370  if (cosPA < mSVParams->minCosPACasc) {
371  continue;
372  }
373  float p2Bach = pBach[0] * pBach[0] + pBach[1] * pBach[1] + pBach[2] * pBach[2];
374  float ptCasc = std::sqrt(pt2Casc);
375  bool goodHyp = false;
376  for (int ipid = 0; ipid < NHypCascade; ipid++) {
377  if (mCascHyps[ipid].check(p2V0, p2Bach, p2Casc, ptCasc)) {
378  goodHyp = true;
379  break;
380  }
381  }
382  if (!goodHyp) {
383  continue;
384  }
385 
386  auto& casc = mCascadesTmp[ithread].emplace_back(cascXYZ, pCasc, fitterCasc.calcPCACovMatrixFlat(candC), trNeut, trBach, mV0sTmp[ithread].size() - 1, bach.gid);
387  o2::track::TrackParCov trc = casc;
388  o2::dataformats::DCA dca;
389  if (!trc.propagateToDCA(pv, fitterCasc.getBz(), &dca, 5.) ||
390  std::abs(dca.getY()) > mSVParams->maxDCAXYCasc || std::abs(dca.getZ()) > mSVParams->maxDCAZCasc) {
391  mCascadesTmp[ithread].pop_back();
392  continue;
393  }
394  casc.setCosPA(cosPA);
395  casc.setVertexID(v0.getVertexID());
396  casc.setDCA(fitterCasc.getChi2AtPCACandidate());
397  }
398  return mCascadesTmp[ithread].size() - nCascIni;
399 }
