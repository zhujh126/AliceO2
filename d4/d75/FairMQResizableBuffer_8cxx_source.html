42 arrow::Status FairMQResizableBuffer::Resize(const int64_t newSize, bool shrink_to_fit)
43 {
44  if (newSize < this->capacity_ && shrink_to_fit == true) {
45  auto newMessage = mCreator(newSize);
46  memcpy(newMessage->GetData(), mMessage->GetData(), newSize);
47  mMessage = std::move(newMessage);
48 #if ARROW_VERSION_MAJOR > 4

50 #else

52  this->data_ = this->mutable_data_;
53 #endif
54  assert(this->data_);
55  this->capacity_ = static_cast<int64_t>(mMessage->GetSize());
56  assert(newSize == this->capacity_);
57  } else if (newSize > this->capacity_) {
58  auto status = this->Reserve(newSize);

61  }
62  }
63  assert(newSize <= this->capacity_);
64

70 arrow::Status FairMQResizableBuffer::Reserve(const int64_t capacity)
71 {
72  assert(!mMessage || this->size_ <= mMessage->GetSize());
73  assert(!mMessage || this->capacity_ == mMessage->GetSize());
74  if (capacity <= this->capacity_) {
75  return arrow::Status::OK();
76  }
77  auto newMessage = mCreator(capacity);
78  assert(!mMessage || capacity > mMessage->GetSize());
79  if (mMessage) {
80  memcpy(newMessage->GetData(), mMessage->GetData(), mMessage->GetSize());
81  }
82  mMessage = std::move(newMessage);
83  assert(mMessage);
84 #if ARROW_VERSION_MAJOR > 4
85  this->data_ = reinterpret_cast<uint8_t*>(mMessage->GetData());
86 #else
87  this->mutable_data_ = reinterpret_cast<uint8_t*>(mMessage->GetData());
88  this->data_ = this->mutable_data_;
89 #endif
90  this->capacity_ = static_cast<int64_t>(mMessage->GetSize());
