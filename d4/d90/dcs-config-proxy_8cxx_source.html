34 void sendAnswer(const std::string& what, const std::string& ack_chan, FairMQDevice& device)
35 {
36  if (!ack_chan.empty()) {
37  LOG(INFO) << "Sending acknowledgment " << what;
38  auto fmqFactory = device.GetChannel(ack_chan).Transport();
39  auto msg = fmqFactory->CreateMessage(what.size(), fair::mq::Alignment{64});
40  memcpy(msg->GetData(), what.c_str(), what.size());
41  FairMQParts outParts;
42  outParts.AddPart(std::move(msg));
43  sendOnChannel(device, outParts, ack_chan);
44  }
45 }

53 InjectorFunction dcs2dpl(const std::string& acknowledge)
54 {
55 
56  auto timesliceId = std::make_shared<size_t>(0);
57 
58  return [acknowledge, timesliceId](FairMQDevice& device, FairMQParts& parts, ChannelRetriever channelRetriever) {
59  // make sure just 2 messages received
60  if (parts.Size() != 2) {
61  LOG(ERROR) << "received " << parts.Size() << " instead of 2 expected";
62  sendAnswer("error0: wrong number of messages", acknowledge, device);
63  return;
64  }
65  std::string filename{static_cast<const char*>(parts.At(0)->GetData()), parts.At(0)->GetSize()};
66  size_t filesize = parts.At(1)->GetSize();
67  LOG(INFO) << "received file " << filename << " of size " << filesize;
68  int dID = getDetID(filename);
69  if (dID < 0) {
70  LOG(ERROR) << "unknown detector for " << filename;
71  sendAnswer("error1: unrecognized filename", acknowledge, device);
72  return;
73  }
74 
75  o2::header::DataHeader hdrF("DCS_CONFIG_FILE", DetID(dID).getDataOrigin(), 0);
76  o2::header::DataHeader hdrN("DCS_CONFIG_NAME", DetID(dID).getDataOrigin(), 0);
77  OutputSpec outsp{hdrN.dataOrigin, hdrN.dataDescription, hdrN.subSpecification};
78  auto channel = channelRetriever(outsp, *timesliceId);
79  if (channel.empty()) {
80  LOG(ERROR) << "No output channel found for OutputSpec " << outsp;
81  sendAnswer("error2: no channel to send", acknowledge, device);
82  return;
83  }
84 
85  hdrN.tfCounter = *timesliceId; // this also
86  hdrN.payloadSerializationMethod = o2::header::gSerializationMethodNone;
87  hdrN.splitPayloadParts = 1;
88  hdrN.splitPayloadIndex = 1;
89  hdrN.payloadSize = parts.At(0)->GetSize();
90  hdrN.firstTForbit = 0; // this should be irrelevant for DCS
91 
92  hdrF.tfCounter = *timesliceId; // this also
93  hdrF.payloadSerializationMethod = o2::header::gSerializationMethodNone;
94  hdrF.splitPayloadParts = 1;
95  hdrF.splitPayloadIndex = 1;
96  hdrF.payloadSize = filesize;
97  hdrF.firstTForbit = 0; // this should be irrelevant for DCS
98 
99  auto fmqFactory = device.GetChannel(channel).Transport();
100 
101  o2::header::Stack headerStackF{hdrF, DataProcessingHeader{*timesliceId, 0}};
102  auto hdMessageF = fmqFactory->CreateMessage(headerStackF.size(), fair::mq::Alignment{64});
103  auto plMessageF = fmqFactory->CreateMessage(hdrF.payloadSize, fair::mq::Alignment{64});
104  memcpy(hdMessageF->GetData(), headerStackF.data(), headerStackF.size());
105  memcpy(plMessageF->GetData(), parts.At(1)->GetData(), hdrF.payloadSize);
106 
107  o2::header::Stack headerStackN{hdrN, DataProcessingHeader{*timesliceId, 0}};
108  auto hdMessageN = fmqFactory->CreateMessage(headerStackN.size(), fair::mq::Alignment{64});
109  auto plMessageN = fmqFactory->CreateMessage(hdrN.payloadSize, fair::mq::Alignment{64});
110  memcpy(hdMessageN->GetData(), headerStackN.data(), headerStackN.size());
111  memcpy(plMessageN->GetData(), parts.At(0)->GetData(), hdrN.payloadSize);
112 
113  FairMQParts outParts;
114  outParts.AddPart(std::move(hdMessageF));
115  outParts.AddPart(std::move(plMessageF));
116  outParts.AddPart(std::move(hdMessageN));
117  outParts.AddPart(std::move(plMessageN));
118  sendOnChannel(device, outParts, channel);
119 
120  sendAnswer("OK", acknowledge, device);
121  LOG(INFO) << "Sent DPL message and acknowledgment for file " << filename;
122  };
123 }

134 WorkflowSpec defineDataProcessing(ConfigContext const& config)
135 {
136  auto setChanName = [](const std::string& chan, const std::string& name) {
137  size_t n = 0;
138  if (std::string(chan).find("name=") != std::string::npos) {
139  n = std::string(chan).find(",");
140  if (n == std::string::npos) {
141  throw std::runtime_error(fmt::format("wrongly formatted channel: {}", chan));
142  }
143  n++;
144  }
145  return o2::utils::Str::concat_string("name=", name, ",", chan.substr(n, chan.size()));
146  };
147 
148  const std::string devName = "dcs-config-proxy";
149  auto chan = config.options().get<std::string>("subscribe-to");
150  if (chan.empty()) {
151  throw std::runtime_error("input channel is not provided");
152  }
153  chan = setChanName(chan, devName);
154 
155  auto chanTo = config.options().get<std::string>("acknowlege-to");
156  std::string ackChan{};
157  if (!chanTo.empty()) {
158  ackChan = "ackChan";
159  chan = o2::utils::Str::concat_string(chan, ";", setChanName(chanTo, ackChan));
160  }
161  LOG(INFO) << "Channels setup: " << chan;
162  Outputs dcsOutputs;
163  for (int id = DetID::First; id <= DetID::Last; id++) {
164  dcsOutputs.emplace_back(DetID(id).getDataOrigin(), "DCS_CONFIG_FILE", 0, Lifetime::Timeframe);
165  dcsOutputs.emplace_back(DetID(id).getDataOrigin(), "DCS_CONFIG_NAME", 0, Lifetime::Timeframe);
166  }
167 
168  DataProcessorSpec dcsConfigProxy = specifyExternalFairMQDeviceProxy(
169  devName.c_str(),
170  std::move(dcsOutputs),
171  // this is just default, can be overriden by --dcs-config-proxy '--channel-config..'
172  chan.c_str(),
173  dcs2dpl(ackChan));
174 
175  WorkflowSpec workflow;
176  workflow.emplace_back(dcsConfigProxy);
177  return workflow;
178 }
