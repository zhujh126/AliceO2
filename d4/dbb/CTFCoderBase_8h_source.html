67  {
68  if (!freq.size()) {
69  LOG(WARNING) << "Empty dictionary provided for slot " << slot << ", " << (op == OpType::Encoder ? "encoding" : "decoding") << " will assume literal symbols only";
70  }
71 
72  switch (op) {

75  break;
76  case OpType::Decoder:
