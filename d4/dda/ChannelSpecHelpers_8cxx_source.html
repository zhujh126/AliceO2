69 std::string composeIPCName(std::string const& prefix, std::string const& hostname, short port)
70 {
71  if (prefix == "@") {
72  return fmt::format("ipc://@{}_{},transport=shmem", hostname, port);
73  }
74  if (prefix.back() == '/') {
75  return fmt::format("ipc://{}{}_{},transport=shmem", prefix, hostname, port);
76  }
77  return fmt::format("ipc://{}/{}_{},transport=shmem", prefix, hostname, port);
78 }

81 std::string ChannelSpecHelpers::channelUrl(OutputChannelSpec const& channel)
