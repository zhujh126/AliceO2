37 template <typename DataT>
38 void o2::tpc::IDCCCDBHelper<DataT>::drawSide(const o2::tpc::IDCType type, const o2::tpc::Side side, const unsigned int integrationInterval, const std::string filename) const
39 {
40  const auto coords = o2::tpc::painter::getPadCoordinatesSector();
41  TH2Poly* poly = o2::tpc::painter::makeSideHist(side);
42  poly->SetContour(255);
43  poly->SetTitle(nullptr);
44  poly->GetXaxis()->SetTitleOffset(1.2f);
45  poly->GetYaxis()->SetTitleOffset(1.3f);
46  poly->GetZaxis()->SetTitleOffset(1.2f);
47  poly->GetZaxis()->SetTitle(getZAxisTitle(type).data());
48  poly->GetZaxis()->SetMaxDigits(3); // force exponential axis
49  poly->SetStats(0);
50 
51  TCanvas* can = new TCanvas("can", "can", 650, 600);
52  can->SetTopMargin(0.04f);
53  can->SetRightMargin(0.14f);
54  can->SetLeftMargin(0.1f);
55  poly->Draw("colz");
56 
57  unsigned int sectorStart = (side == Side::A) ? 0 : o2::tpc::SECTORSPERSIDE;
58  unsigned int sectorEnd = (side == Side::A) ? o2::tpc::SECTORSPERSIDE : Mapper::NSECTORS;
59  for (unsigned int sector = sectorStart; sector < sectorEnd; ++sector) {
60  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
61  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
62  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
63  const auto padNum = Mapper::getGlobalPadNumber(irow, ipad, region);
64  const float angDeg = 10.f + sector * 20;
65  auto coordinate = coords[padNum];
66  coordinate.rotate(angDeg);
67  const float yPos = 0.25f * static_cast<float>(coordinate.yVals[0] + coordinate.yVals[1] + coordinate.yVals[2] + coordinate.yVals[3]);
68  const float xPos = 0.25f * static_cast<float>(coordinate.xVals[0] + coordinate.xVals[1] + coordinate.xVals[2] + coordinate.xVals[3]);
69  const auto padTmp = (side == Side::A) ? ipad : (Mapper::PADSPERROW[region][irow] - ipad); // C-Side is mirrored
70  switch (type) {
71  case IDCType::IDCZero:
72  default:
73  poly->Fill(xPos, yPos, getIDCZeroVal(sector, region, irow, padTmp));
74  break;
75  case IDCType::IDCDelta:
76  poly->Fill(xPos, yPos, getIDCDeltaVal(sector, region, irow, padTmp, integrationInterval));
77  break;
78  case IDCType::IDC:
79  break;
80  case IDCType::IDCOne:
81  break;
82  }
83  }
84  }
85  }
86  }
87  if (!filename.empty()) {
88  can->SaveAs(filename.data());
89  delete poly;
90  delete can;
91  }
92 }

94 template <typename DataT>
95 void o2::tpc::IDCCCDBHelper<DataT>::drawSector(const IDCType type, const unsigned int sector, const unsigned int integrationInterval, const std::string filename) const
96 {
97  const auto coords = o2::tpc::painter::getPadCoordinatesSector();
98  TH2Poly* poly = o2::tpc::painter::makeSectorHist("hSector", "Sector;local #it{x} (cm);local #it{y} (cm); #it{IDC}");
99  poly->SetContour(255);
100  poly->SetTitle(nullptr);
101  poly->GetYaxis()->SetTickSize(0.002f);
102  poly->GetYaxis()->SetTitleOffset(0.7f);
103  poly->GetZaxis()->SetTitleOffset(1.3f);
104  poly->SetStats(0);
105  poly->GetZaxis()->SetTitle(getZAxisTitle(type).data());
106 
107  TCanvas* can = new TCanvas("can", "can", 2000, 1400);
108  can->SetRightMargin(0.14f);
109  can->SetLeftMargin(0.06f);
110  can->SetTopMargin(0.04f);
111 
112  TLatex lat;
113  lat.SetTextFont(63);
114  lat.SetTextSize(2);
115  poly->Draw("colz");
116 
117  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
118  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
119  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
120  const auto padNum = Mapper::getGlobalPadNumber(irow, ipad, region);
121  const auto coordinate = coords[padNum];
122  const float yPos = -0.5f * static_cast<float>(coordinate.yVals[0] + coordinate.yVals[2]); // local coordinate system is mirrored
123  const float xPos = 0.5f * static_cast<float>(coordinate.xVals[0] + coordinate.xVals[2]);
124  switch (type) {
125  case IDCType::IDCZero:
126  default:
127  poly->Fill(xPos, yPos, getIDCZeroVal(sector, region, irow, ipad));
128  break;
129  case IDCType::IDCDelta:
130  poly->Fill(xPos, yPos, getIDCDeltaVal(sector, region, irow, ipad, integrationInterval));
131  break;
132  case IDCType::IDC:
133  break;
134  case IDCType::IDCOne:
135  break;
136  }
137  // draw global pad number
138  lat.SetTextAlign(12);
139  lat.DrawLatex(xPos, yPos, Form("%i", ipad));
140  }
141  }
142  }
143  if (!filename.empty()) {
144  can->SaveAs(filename.data());
145  delete poly;
146  delete can;
147  }
148 }
