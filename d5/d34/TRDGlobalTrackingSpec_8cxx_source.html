54 void TRDGlobalTracking::init(InitContext& ic)
55 {
56 
57  //-------- init geometry and field --------//
58  o2::base::GeometryManager::loadGeometry();
59  o2::base::Propagator::initFieldFromGRP();
60  auto geo = Geometry::instance();
61  geo->createPadPlaneArray();
62  geo->createClusterMatrixArray();
63  mFlatGeo = std::make_unique<GeometryFlat>(*geo);
64 
65  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
66  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
67  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
68  if (o2::utils::Str::pathExists(matLUTFile)) {
69  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
70  o2::base::Propagator::Instance()->setMatLUT(lut);
71  LOG(INFO) << "Loaded material LUT from " << matLUTFile;
72  } else {
73  LOG(INFO) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
74  }
75 
76  //-------- init GPU reconstruction --------//
77  GPURecoStepConfiguration cfgRecoStep;
78  cfgRecoStep.steps = GPUDataTypes::RecoStep::NoRecoStep;
79  cfgRecoStep.inputs.clear();
80  cfgRecoStep.outputs.clear();
81  mRec = GPUReconstruction::CreateInstance("CPU", true);
82  mRec->SetSettings(o2::base::Propagator::Instance()->getNominalBz(), &cfgRecoStep);
83 
84  mChainTracking = mRec->AddChain<GPUChainTracking>();
85 
86  mTracker = new GPUTRDTracker();
87  mTracker->SetNCandidates(mRec->GetProcessingSettings().trdNCandidates); // must be set before initialization
88  if (mStrict && mRec->GetProcessingSettings().trdNCandidates == 1) {
89  LOG(ERROR) << "Strict matching mode requested, but tracks with another close hypothesis will not be rejected. Please set trdNCandidates to at least 3.";
90  }
91  mTracker->SetProcessPerTimeFrame(true);
92  mTracker->SetGenerateSpacePoints(false); // set to true to force space point calculation by the TRD tracker itself
93 
94  mRec->RegisterGPUProcessor(mTracker, false);
95  mChainTracking->SetTRDGeometry(std::move(mFlatGeo));
96  if (mRec->Init()) {
97  LOG(FATAL) << "GPUReconstruction could not be initialized";
98  }
99 
100  mTracker->PrintSettings();
101 
102  mTimer.Stop();
103  mTimer.Reset();
104 }

117 void TRDGlobalTracking::fillMCTruthInfo(const TrackTRD& trk, o2::MCCompLabel lblSeed, std::vector<o2::MCCompLabel>& lblContainerTrd, std::vector<o2::MCCompLabel>& lblContainerMatch, const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* trkltLabels) const
118 {
119  // Check MC labels of the TRD tracklets attached to the track seed.
120  // Set TRD track label to the most frequent tracklet label.
121  // Fake flag is set if either not all TRD tracklets have most frequent label
122  // or if the seeding label is different from the most frequent TRD label.
123  // In case multiple tracklet labels occur most often we choose the one which matches the label of the seed, or,
124  // if that is not the case one of the most frequent labels is chosen arbitrarily
125  LOG(DEBUG) << "Checking seed with label: " << lblSeed;
126  std::unordered_map<o2::MCCompLabel, unsigned int> labelCounter;
127  int nTracklets = 0;
128  unsigned int maxOccurences = 0;
129  for (int iLy = 0; iLy < constants::NLAYER; ++iLy) {
130  auto trkltIndex = trk.getTrackletIndex(iLy);
131  if (trkltIndex == -1) {
132  // no tracklet in this layer
133  continue;
134  }
135  const auto& lblsTrklt = trkltLabels->getLabels(trkltIndex);
136  for (const auto lblTrklt : lblsTrklt) {
137  int nOcc = ++labelCounter[lblTrklt];
138  if (nOcc > maxOccurences) {
139  maxOccurences = nOcc;
140  }
141  }
142  }
143  o2::MCCompLabel mostFrequentLabel;
144  for (const auto& [lbl, count] : labelCounter) {
145  LOG(DEBUG) << "Label " << lbl << " occured " << count << " times.";
146  if (count == maxOccurences) {
147  if (lblSeed == lbl) {
148  // most frequent label matches seed label
149  mostFrequentLabel = lbl;
150  mostFrequentLabel.setFakeFlag(maxOccurences != trk.getNtracklets());
151  lblContainerTrd.push_back(mostFrequentLabel);
152  lblContainerMatch.push_back(lblSeed); // is not fake by definition, since the seed label matches the TRD track label
153  return;
154  } else {
155  // maybe multiple labels occur with the same frequency and the next one might match the seed?
156  mostFrequentLabel = lbl;
157  }
158  }
159  }
160  mostFrequentLabel.setFakeFlag(maxOccurences != trk.getNtracklets());
161  lblContainerTrd.push_back(mostFrequentLabel);
162  lblSeed.setFakeFlag(lblSeed != mostFrequentLabel);
163  lblContainerMatch.push_back(lblSeed);
164 }

166 void TRDGlobalTracking::fillTrackTriggerRecord(const std::vector<TrackTRD>& tracks, std::vector<TrackTriggerRecord>& trigRec, const gsl::span<const o2::trd::TriggerRecord>& trackletTrigRec) const
167 {
168  int currTrigRec = 0;
169  int nTracksCurr = 0;
170  int iTrackFirst = 0;
171  for (const auto& trk : tracks) {
172  if (trk.getCollisionId() != currTrigRec) {
173  // new collision ID, create new track trigger record
174  trigRec.emplace_back(trackletTrigRec[currTrigRec].getBCData(), iTrackFirst, nTracksCurr);
175  currTrigRec = trk.getCollisionId();
176  iTrackFirst += nTracksCurr;
177  nTracksCurr = 0;
178  }
179  ++nTracksCurr;
180  }
181  if (nTracksCurr > 0) {
182  // create track trigger record for remaining track range
183  trigRec.emplace_back(trackletTrigRec[currTrigRec].getBCData(), iTrackFirst, nTracksCurr);
184  }

187 void TRDGlobalTracking::run(ProcessingContext& pc)
188 {
189  mTimer.Start(false);
190  mChainTracking->ClearIOPointers();
191  o2::globaltracking::RecoContainer inputTracks;
192  inputTracks.collectData(pc, *mDataRequest);
193  auto tmpInputContainer = getRecoInputContainer(pc, &mChainTracking->mIOPtrs, &inputTracks, mUseMC);
194  auto tmpContainer = GPUWorkflowHelper::fillIOPtr(mChainTracking->mIOPtrs, inputTracks, mUseMC, nullptr, GTrackID::getSourcesMask("TRD"), mTrkMask, GTrackID::mask_t{GTrackID::MASK_NONE});
195  LOGF(INFO, "There are %i tracklets in total from %i trigger records", mChainTracking->mIOPtrs.nTRDTracklets, mChainTracking->mIOPtrs.nTRDTriggerRecords);
196  LOGF(INFO, "As input seeds are available: %i ITS-TPC matched tracks and %i TPC tracks", mChainTracking->mIOPtrs.nTracksTPCITSO2, mChainTracking->mIOPtrs.nOutputTracksTPCO2);
197 
198  std::vector<o2::MCCompLabel> matchLabelsITSTPC;
199  std::vector<o2::MCCompLabel> trdLabelsITSTPC;
200  std::vector<o2::MCCompLabel> matchLabelsTPC;
201  std::vector<o2::MCCompLabel> trdLabelsTPC;
202  gsl::span<const o2::MCCompLabel> tpcTrackLabels;
203  gsl::span<const o2::MCCompLabel> itstpcTrackLabels;
204  if (mUseMC) {
205  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, mTrkMask)) {
206  itstpcTrackLabels = inputTracks.getTPCITSTracksMCLabels();
207  }
208  if (GTrackID::includesSource(GTrackID::Source::TPC, mTrkMask)) {
209  tpcTrackLabels = inputTracks.getTPCTracksMCLabels();
210  }
211  }
212 
213  mTracker->Reset();
214  updateTimeDependentParams();
215  mRec->PrepareEvent();
216  mRec->SetupGPUProcessor(mTracker, true);
217 
218  // check trigger record filter setting
219  bool foundFilteredTrigger = false;
220  for (int iTrig = 0; iTrig < mChainTracking->mIOPtrs.nTRDTriggerRecords; ++iTrig) {
221  if (mChainTracking->mIOPtrs.trdTrigRecMask[iTrig] == 0) {
222  foundFilteredTrigger = true;
223  }
224  }
225  if (!foundFilteredTrigger && mTrigRecFilter) {
226  LOG(WARNING) << "Trigger filtering requested, but no TRD trigger is actually masked. Can be that none needed to be masked or that the setting was not active for the tracklet transformer";
227  } else if (foundFilteredTrigger && !mTrigRecFilter) {
228  LOG(ERROR) << "Trigger filtering is not requested, but masked TRD triggers are found. Rerun tracklet transformer without trigger filtering";
229  }
230 
231  // load input tracks
232  LOG(DEBUG) << "Start loading input seeds into TRD tracker";
233  int nTracksLoadedITSTPC = 0;
234  int nTracksLoadedTPC = 0;
235  // load ITS-TPC matched tracks
236  for (int iTrk = 0; iTrk < mChainTracking->mIOPtrs.nTracksTPCITSO2; ++iTrk) {
237  const auto& trkITSTPC = mChainTracking->mIOPtrs.tracksTPCITSO2[iTrk];
238  GPUTRDTracker::HelperTrackAttributes trkAttribs;
239  trkAttribs.mTime = trkITSTPC.getTimeMUS().getTimeStamp();
240  trkAttribs.mTimeAddMax = trkITSTPC.getTimeMUS().getTimeStampError() * mRec->GetParam().rec.trd.nSigmaTerrITSTPC;
241  trkAttribs.mTimeSubMax = trkITSTPC.getTimeMUS().getTimeStampError() * mRec->GetParam().rec.trd.nSigmaTerrITSTPC;
242  GPUTRDTrack trkLoad(trkITSTPC);
243  auto trackGID = GTrackID(iTrk, GTrackID::ITSTPC);
244  if (mTracker->LoadTrack(trkLoad, trackGID.getRaw(), true, &trkAttribs)) {
245  continue;
246  }
247  ++nTracksLoadedITSTPC;
248  LOGF(DEBUG, "Loaded ITS-TPC track %i with time %f", nTracksLoadedITSTPC, trkAttribs.mTime);
249  }
250  // load TPC-only tracks
251  for (int iTrk = 0; iTrk < mChainTracking->mIOPtrs.nOutputTracksTPCO2; ++iTrk) {
252  if (mChainTracking->mIOPtrs.tpcLinkITS && mChainTracking->mIOPtrs.tpcLinkITS[iTrk] != -1) {
253  // this TPC tracks has already been matched to ITS and the ITS-TPC track has already been loaded in the tracker
254  continue;
255  }
256  const auto& trkTpc = mChainTracking->mIOPtrs.outputTracksTPCO2[iTrk];
257  GPUTRDTracker::HelperTrackAttributes trkAttribs;
258  trkAttribs.mTime = trkTpc.getTime0() * mTPCTBinMUS;
259  trkAttribs.mTimeAddMax = trkTpc.getDeltaTFwd() * mTPCTBinMUS;
260  trkAttribs.mTimeSubMax = trkTpc.getDeltaTBwd() * mTPCTBinMUS;
261  if (trkTpc.hasASideClustersOnly()) {
262  trkAttribs.mSide = -1;
263  } else if (trkTpc.hasCSideClustersOnly()) {
264  trkAttribs.mSide = 1;
265  }
266  GPUTRDTrack trkLoad(trkTpc);
267  auto trackGID = GTrackID(iTrk, GTrackID::TPC);
268  if (mTracker->LoadTrack(trkLoad, trackGID.getRaw(), true, &trkAttribs)) {
269  continue;
270  }
271  ++nTracksLoadedTPC;
272  LOGF(DEBUG, "Loaded TPC track %i with time %f", nTracksLoadedTPC, trkAttribs.mTime);
273  }
274  LOGF(INFO, "%i tracks are loaded into the TRD tracker. Out of those %i ITS-TPC tracks and %i TPC tracks", nTracksLoadedITSTPC + nTracksLoadedTPC, nTracksLoadedITSTPC, nTracksLoadedTPC);
275 
276  // start the tracking
277  //mTracker->DumpTracks();
278  mTracker->DoTracking(mChainTracking);
279  //mTracker->DumpTracks();
280 
281  // finished tracking, now collect the output
282  std::vector<TrackTRD> tracksOutITSTPC;
283  std::vector<TrackTRD> tracksOutTPC;
284  std::vector<TrackTriggerRecord> trackTrigRecITSTPC;
285  std::vector<TrackTriggerRecord> trackTrigRecTPC;
286  GPUTRDTrack* tracksOutRaw = mTracker->Tracks();
287  std::vector<unsigned int> trackIdxArray(mTracker->NTracks()); // track indices sorted by trigger record index
288  std::iota(trackIdxArray.begin(), trackIdxArray.end(), 0);
289  std::sort(trackIdxArray.begin(), trackIdxArray.end(), [tracksOutRaw](int lhs, int rhs) { return tracksOutRaw[lhs].getCollisionId() < tracksOutRaw[rhs].getCollisionId(); });
290 
291  int nTrackletsAttached = 0; // only used for debug information
292  for (int iTrk = 0; iTrk < mTracker->NTracks(); ++iTrk) {
293  const auto& trdTrack = mTracker->Tracks()[trackIdxArray[iTrk]];
294  if (trdTrack.getCollisionId() < 0) {
295  // skip tracks without TRD tracklets (the collision ID for the TRD tracks is initialized to -1 and only changed if a tracklet is attached to the track)
296  continue;
297  }
298  if (mStrict && (trdTrack.getIsAmbiguous() || trdTrack.getReducedChi2() > mTracker->Param().rec.trd.chi2StrictCut)) {
299  // skip tracks which have another hypothesis close to the best one or which do are above strict chi2 threshold
300  continue;
301  }
302  nTrackletsAttached += trdTrack.getNtracklets();
303  auto trackGID = trdTrack.getRefGlobalTrackId();
304  if (trackGID.includesDet(GTrackID::Source::ITS)) {
305  // this track is from an ITS-TPC seed
306  tracksOutITSTPC.push_back(trdTrack);
307  if (mUseMC) {
308  fillMCTruthInfo(trdTrack, itstpcTrackLabels[trackGID], trdLabelsITSTPC, matchLabelsITSTPC, inputTracks.getTRDTrackletsMCLabels());
309  }
310  } else {
311  // this track is from a TPC-only seed
312  tracksOutTPC.push_back(trdTrack);
313  if (mUseMC) {
314  fillMCTruthInfo(trdTrack, tpcTrackLabels[trackGID], trdLabelsTPC, matchLabelsTPC, inputTracks.getTRDTrackletsMCLabels());
315  }
316  }
317  }
318 
319  fillTrackTriggerRecord(tracksOutITSTPC, trackTrigRecITSTPC, tmpInputContainer->mTriggerRecords);
320  fillTrackTriggerRecord(tracksOutTPC, trackTrigRecTPC, tmpInputContainer->mTriggerRecords);
321 
322  LOGF(INFO, "The TRD tracker found %lu tracks from TPC seeds and %lu tracks from ITS-TPC seeds and attached in total %i tracklets out of %i",
323  tracksOutTPC.size(), tracksOutITSTPC.size(), nTrackletsAttached, mChainTracking->mIOPtrs.nTRDTracklets);
324 
325  uint32_t ss = o2::globaltracking::getSubSpec(mStrict ? o2::globaltracking::MatchingType::Strict : o2::globaltracking::MatchingType::Standard);
326  if (inputTracks.isTrackSourceLoaded(GTrackID::Source::ITSTPC)) {
327  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCH_ITSTPC", 0, Lifetime::Timeframe}, tracksOutITSTPC);
328  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "TRGREC_ITSTPC", 0, Lifetime::Timeframe}, trackTrigRecITSTPC);
329  if (mUseMC) {
330  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MCLB_ITSTPC", 0, Lifetime::Timeframe}, matchLabelsITSTPC);
331  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MCLB_ITSTPC_TRD", 0, Lifetime::Timeframe}, trdLabelsITSTPC);
332  }
333  }
334  if (inputTracks.isTrackSourceLoaded(GTrackID::Source::TPC)) {
335  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCH_TPC", ss, Lifetime::Timeframe}, tracksOutTPC);
336  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "TRGREC_TPC", ss, Lifetime::Timeframe}, trackTrigRecTPC);
337  if (mUseMC) {
338  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MCLB_TPC", ss, Lifetime::Timeframe}, matchLabelsTPC);
339  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MCLB_TPC_TRD", ss, Lifetime::Timeframe}, trdLabelsTPC);
340  }
341  }
342 
343  mTimer.Stop();
344 }

352 DataProcessorSpec getTRDGlobalTrackingSpec(bool useMC, GTrackID::mask_t src, bool trigRecFilterActive, bool strict)
353 {
354  std::vector<OutputSpec> outputs;
355  uint32_t ss = o2::globaltracking::getSubSpec(strict ? o2::globaltracking::MatchingType::Strict : o2::globaltracking::MatchingType::Standard);
356  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
357  if (strict) {
358  dataRequest->setMatchingInputStrict();
359  }
360  dataRequest->requestTracks(src, useMC);
361  dataRequest->requestClusters(GTrackID::getSourcesMask("TRD"), useMC);
362  auto& inputs = dataRequest->inputs;
363 
364 
365  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, src)) {
366  outputs.emplace_back(o2::header::gDataOriginTRD, "MATCH_ITSTPC", 0, Lifetime::Timeframe);
367  outputs.emplace_back(o2::header::gDataOriginTRD, "TRGREC_ITSTPC", 0, Lifetime::Timeframe);
368  if (useMC) {
369  outputs.emplace_back(o2::header::gDataOriginTRD, "MCLB_ITSTPC", 0, Lifetime::Timeframe);
370  outputs.emplace_back(o2::header::gDataOriginTRD, "MCLB_ITSTPC_TRD", 0, Lifetime::Timeframe);
371  }
372  }
373  if (GTrackID::includesSource(GTrackID::Source::TPC, src)) {
374  outputs.emplace_back(o2::header::gDataOriginTRD, "MATCH_TPC", ss, Lifetime::Timeframe);
375  outputs.emplace_back(o2::header::gDataOriginTRD, "TRGREC_TPC", ss, Lifetime::Timeframe);
376  if (useMC) {
377  outputs.emplace_back(o2::header::gDataOriginTRD, "MCLB_TPC", ss, Lifetime::Timeframe);
378  outputs.emplace_back(o2::header::gDataOriginTRD, "MCLB_TPC_TRD", ss, Lifetime::Timeframe);
379  }
380  if (trigRecFilterActive) {
381  LOG(ERROR) << "Matching to TPC-only tracks requested, but IR without ITS contribution are filtered out. This does not lead to a crash, but it deteriorates the matching efficiency.";
382  }
383  }
384 
385  std::string processorName = o2::utils::Str::concat_string("trd-globaltracking", GTrackID::getSourcesNames(src));
386  std::replace(processorName.begin(), processorName.end(), ',', '_');
387 
388  return DataProcessorSpec{
389  processorName,
390  inputs,
391  outputs,
392  AlgorithmSpec{adaptFromTask<TRDGlobalTracking>(useMC, dataRequest, src, trigRecFilterActive, strict)},
393  Options{{"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
394 }
