46 void TrackBasedCalib::calculateAngResHistos(const o2::globaltracking::RecoContainer& input)
47 {
48 
49  mTracksIn = input.getITSTPCTRDTracks<TrackTRD>();
50  mTrackletsRaw = input.getTRDTracklets();
51  mTrackletsCalib = input.getTRDCalibratedTracklets();
52 
53  if (mTrackletsRaw.size() != mTrackletsCalib.size()) {
54  LOG(ERROR) << "TRD raw tracklet container size differs from calibrated tracklet container size";
55  return;
56  }
57 
58  int nTracksSuccess = 0;
59  int nAngularResidualsCollected = 0;
60  for (const auto& trkIn : mTracksIn) {
61  if (trkIn.getNtracklets() < 3) {
62  // with less than 3 tracklets the TRD-only refit not meaningful
63  continue;
64  }
65  auto trkWork = trkIn; // input is const, so we need to create a copy
66  bool trackFailed = false;
67 
68  trkWork.setChi2(0.f);
69  trkWork.resetCovariance(20);
70 
71  // first inward propagation (TRD track fit)
72  int currLayer = NLAYER;
73  for (int iLayer = NLAYER - 1; iLayer >= 0; --iLayer) {
74  if (trkWork.getTrackletIndex(iLayer) == -1) {
75  continue;
76  }
77  if (propagateAndUpdate(trkWork, iLayer, true)) {
78  trackFailed = true;
79  break;
80  }
81  currLayer = iLayer;
82  }
83  if (trackFailed) {
84  continue;
85  }
86 
87  // outward propagation (smoothing)
88  for (int iLayer = currLayer + 1; iLayer < NLAYER; ++iLayer) {
89  if (trkWork.getTrackletIndex(iLayer) == -1) {
90  continue;
91  }
92  if (propagateAndUpdate(trkWork, iLayer, true)) {
93  trackFailed = true;
94  break;
95  }
96  currLayer = iLayer;
97  }
98  if (trackFailed) {
99  continue;
100  }
101 
102  // second inward propagation (collect angular differences between tracklets + TRD track)
103  for (int iLayer = currLayer; iLayer >= 0; --iLayer) {
104  if (trkWork.getTrackletIndex(iLayer) == -1) {
105  continue;
106  }
107  if (propagateAndUpdate(trkWork, iLayer, false)) {
108  trackFailed = true;
109  break;
110  }
111 
112  float trkAngle = o2::math_utils::asin(trkWork.getSnp()) * TMath::RadToDeg();
113  float trkltAngle = o2::math_utils::atan(mTrackletsCalib[trkWork.getTrackletIndex(iLayer)].getDy() / Geometry::cdrHght()) * TMath::RadToDeg();
114  float angleDeviation = trkltAngle - trkAngle;
115  if (mAngResHistos.addEntry(angleDeviation, trkAngle, mTrackletsRaw[trkWork.getTrackletIndex(iLayer)].getDetector())) {
116  // track impact angle out of histogram range
117  continue;
118  }
119  ++nAngularResidualsCollected;
120  }
121 
122  // here we can count the number of successfully processed tracks
123  ++nTracksSuccess;
124  } // end of track loop
125  LOGF(INFO, "Successfully processed %i tracks and collected %i angular residuals", nTracksSuccess, nAngularResidualsCollected);
126  //mAngResHistos.print();
127 }

129 bool TrackBasedCalib::propagateAndUpdate(TrackTRD& trk, int iLayer, bool doUpdate) const
130 {
131  // Propagates the track to TRD layer iLayer and updates the track
132  // parameters (if requested)
133  // returns 0 in case of success
134 
135  auto propagator = o2::base::Propagator::Instance();
136 
137  int trkltId = trk.getTrackletIndex(iLayer);
138  int trkltDet = mTrackletsRaw[trkltId].getDetector();
139  int trkltSec = trkltDet / (NLAYER * NSTACK);
140 
141  if (trkltSec != o2::math_utils::angle2Sector(trk.getAlpha())) {
142  if (!trk.rotate(o2::math_utils::sector2Angle(trkltSec))) {
143  LOGF(DEBUG, "Track could not be rotated in tracklet coordinate system");
144  return 1;
145  }
146  }
147 
148  if (!propagator->PropagateToXBxByBz(trk, mTrackletsCalib[trkltId].getX(), mMaxSnp, mMaxStep, mMatCorr)) {
149  LOGF(DEBUG, "Track propagation failed in layer %i (pt=%f, xTrk=%f, xToGo=%f)", iLayer, trk.getPt(), trk.getX(), mTrackletsCalib[trkltId].getX());
150  return 1;
151  }
152 
153  if (!doUpdate) {
154  // nothing more to be done
155  return 0;
156  }
157 
158  const PadPlane* pad = Geometry::instance()->getPadPlane(trkltDet);
159  float tilt = tan(TMath::DegToRad() * pad->getTiltingAngle()); // tilt is signed! and returned in degrees
160  float tiltCorrUp = tilt * (mTrackletsCalib[trkltId].getZ() - trk.getZ());
161  float zPosCorrUp = mTrackletsCalib[trkltId].getZ() + mRecoParam.getZCorrCoeffNRC() * trk.getTgl();
162  float padLength = pad->getRowSize(mTrackletsRaw[trkltId].getPadRow());
163  if (!((trk.getSigmaZ2() < (padLength * padLength / 12.f)) && (std::fabs(mTrackletsCalib[trkltId].getZ() - trk.getZ()) < padLength))) {
164  tiltCorrUp = 0.f;
165  }
166 
167  std::array<float, 2> trkltPosUp{mTrackletsCalib[trkltId].getY() - tiltCorrUp, zPosCorrUp};
168  std::array<float, 3> trkltCovUp;
169  mRecoParam.recalcTrkltCov(tilt, trk.getSnp(), pad->getRowSize(mTrackletsRaw[trkltId].getPadRow()), trkltCovUp);
170 
171  if (!trk.update(trkltPosUp, trkltCovUp)) {
172  LOGF(INFO, "Failed to update track with space point in layer %i", iLayer);
173  return 1;
174  }
175  return 0;
176 }
