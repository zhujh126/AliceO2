48 std::string VisualisationEvent::toJson()
49 {
50  Document tree(kObjectType);
51  Document::AllocatorType& allocator = tree.GetAllocator();
52 
53  // Tracks
54  rapidjson::Value trackCount(rapidjson::kNumberType);
55  trackCount.SetInt(this->getTrackCount());
56  tree.AddMember("trackCount", trackCount, allocator);
57  Value jsonTracks(kArrayType);
58  for (size_t i = 0; i < this->getTrackCount(); i++) {
59  jsonTracks.PushBack(this->mTracks[i].jsonTree(allocator), allocator);
60  }
61  tree.AddMember("mTracks", jsonTracks, allocator);
62 
63  // Clusters
64  rapidjson::Value clusterCount(rapidjson::kNumberType);
65  clusterCount.SetInt(this->getClusterCount());
66  tree.AddMember("clusterCount", clusterCount, allocator);
67  Value jsonClusters(kArrayType);
68  for (size_t i = 0; i < this->getClusterCount(); i++) {
69  jsonClusters.PushBack(this->mClusters[i].jsonTree(allocator), allocator);
70  }
71  tree.AddMember("mClusters", jsonClusters, allocator);
72 
73  // stringify
74  StringBuffer buffer;
75  Writer<StringBuffer> writer(buffer);
76  tree.Accept(writer);
77  return buffer.GetString();
78 }
