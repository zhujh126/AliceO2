117  void expandPreCluster(uint32_t ip, uint16_t row, int preClusIndex)
118  {
119  auto& firstIndex = preClusterHeads[preClusterIndices[preClusIndex]];
120  pixels.emplace_back(firstIndex, ip);
121  firstIndex = pixels.size() - 1;
122  curr[row] = preClusIndex;
123  }
124 
126  void addNewPrecluster(uint32_t ip, uint16_t row)
127  {
128  preClusterHeads.push_back(pixels.size());
129  // new head does not point yet (-1) on other pixels, store just the entry of the pixel in the ChipPixelData
130  pixels.emplace_back(-1, ip);
131  int lastIndex = preClusterIndices.size();
132  preClusterIndices.push_back(lastIndex);
133  curr[row] = lastIndex; // store index of the new precluster in the current column buffer
134  }
