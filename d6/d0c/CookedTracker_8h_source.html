81  template <typename U, typename V>
82  void process(gsl::span<const CompClusterExt> clusters, gsl::span<const unsigned char>::iterator& it, const o2::itsmft::TopologyDictionary& dict, U& tracks, V& clusIdx, o2::itsmft::ROFRecord& rof)
83  {
84  TrackInserter inserter = [&tracks, &clusIdx, this](const TrackITSExt& t) -> int {
85  // convert internal track to output format
86  auto& trackNew = tracks.emplace_back(t);
87  int noc = t.getNumberOfClusters();
88  int clEntry = clusIdx.size();
89  for (int i = 0; i < noc; i++) {
90  const Cluster* c = this->getCluster(t.getClusterIndex(i));
91  Int_t idx = c - &mClusterCache[0]; // Index of this cluster in event
92  clusIdx.emplace_back(this->mFirstInFrame + idx);
93  }
94  trackNew.setClusterRefs(clEntry, noc);
95  trackNew.setPattern(0x7f); // this tracker finds only complete tracks
96  return tracks.size();
97  };
98  process(clusters, it, dict, inserter, rof);
99  }
