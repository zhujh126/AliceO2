53 template <class Mapping>
54 void STFDecoder<Mapping>::init(InitContext& ic)
55 {
56  try {
57  mDecoder = std::make_unique<RawPixelDecoder<Mapping>>();
58  mDecoder->init();
59  } catch (const std::exception& e) {
60  LOG(ERROR) << "exception was thrown in decoder creation: " << e.what();
61  throw;
62  } catch (...) {
63  LOG(ERROR) << "non-std::exception was thrown in decoder creation";
64  throw;
65  }
66 
67  auto detID = Mapping::getDetID();
68  try {
69  mNThreads = std::max(1, ic.options().get<int>("nthreads"));
70  mDecoder->setNThreads(mNThreads);
71  mDecoder->setFormat(ic.options().get<bool>("old-format") ? GBTLink::OldFormat : GBTLink::NewFormat);
72  mDecoder->setVerbosity(ic.options().get<int>("decoder-verbosity"));
73  mDecoder->setFillCalibData(mDoCalibData);
74  std::string noiseFile = o2::base::NameConf::getNoiseFileName(detID, mNoiseName, "root");
75  if (o2::utils::Str::pathExists(noiseFile)) {
76  TFile* f = TFile::Open(noiseFile.data(), "old");
77  auto pnoise = (NoiseMap*)f->Get("Noise");
78  AlpideCoder::setNoisyPixels(pnoise);
79  LOG(INFO) << mSelfName << " loading noise map file: " << noiseFile;
80  } else {
81  LOG(INFO) << mSelfName << " Noise file " << noiseFile << " is absent, " << Mapping::getName() << " running without noise suppression";
82  }
83  } catch (const std::exception& e) {
84  LOG(ERROR) << "exception was thrown in decoder configuration: " << e.what();
85  throw;
86  } catch (...) {
87  LOG(ERROR) << "non-std::exception was thrown in decoder configuration";
88  throw;
89  }
90 
91  if (mDoClusters) {
92  try {
93  mClusterer = std::make_unique<Clusterer>();
94  mClusterer->setNChips(Mapping::getNChips());
95  const auto grp = o2::parameters::GRPObject::loadFrom();
96  if (grp) {
97  mClusterer->setContinuousReadOut(grp->isDetContinuousReadOut(detID));
98  } else {
99  throw std::runtime_error("failed to retrieve GRP");
100  }
101 
102  // settings for the fired pixel overflow masking
103  const auto& alpParams = DPLAlpideParam<Mapping::getDetID()>::Instance();
104  const auto& clParams = ClustererParam<Mapping::getDetID()>::Instance();
105  auto nbc = clParams.maxBCDiffToMaskBias;
106  nbc += mClusterer->isContinuousReadOut() ? alpParams.roFrameLengthInBC : (alpParams.roFrameLengthTrig / o2::constants::lhc::LHCBunchSpacingNS);
107  mClusterer->setMaxBCSeparationToMask(nbc);
108  mClusterer->setMaxRowColDiffToMask(clParams.maxRowColDiffToMask);
109 
110  std::string dictFile = o2::base::NameConf::getAlpideClusterDictionaryFileName(detID, mDictName, "bin");
111  if (o2::utils::Str::pathExists(dictFile)) {
112  mClusterer->loadDictionary(dictFile);
113  LOG(INFO) << mSelfName << " clusterer running with a provided dictionary: " << dictFile;
114  } else {
115  LOG(INFO) << mSelfName << " Dictionary " << dictFile << " is absent, " << Mapping::getName() << " clusterer expects cluster patterns";
116  }
117  mClusterer->print();
118  } catch (const std::exception& e) {
119  LOG(ERROR) << "exception was thrown in clustrizer configuration: " << e.what();
120  throw;
121  } catch (...) {
122  LOG(ERROR) << "non-std::exception was thrown in clusterizer configuration";
123  throw;
124  }
125  }

129 template <class Mapping>
130 void STFDecoder<Mapping>::run(ProcessingContext& pc)
131 {
132  int nSlots = pc.inputs().getNofParts(0);
133  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
134  mTimer.Start(false);
135  mDecoder->startNewTF(pc.inputs());
136  auto orig = Mapping::getOrigin();
137  std::vector<o2::itsmft::CompClusterExt> clusCompVec;
138  std::vector<o2::itsmft::ROFRecord> clusROFVec;
139  std::vector<unsigned char> clusPattVec;
140 
141  std::vector<Digit> digVec;
142  std::vector<GBTCalibData> calVec;
143  std::vector<ROFRecord> digROFVec;
144 
145  if (mDoDigits) {
146  digVec.reserve(mEstNDig);
147  digROFVec.reserve(mEstNROF);
148  }
149  if (mDoClusters) {
150  clusCompVec.reserve(mEstNClus);
151  clusROFVec.reserve(mEstNROF);
152  clusPattVec.reserve(mEstNClusPatt);
153  }
154  if (mDoCalibData) {
155  calVec.reserve(mEstNCalib);
156  }
157 
158  mDecoder->setDecodeNextAuto(false);
159  while (mDecoder->decodeNextTrigger()) {
160  if (mDoDigits) { // call before clusterization, since the latter will hide the digits
161  mDecoder->fillDecodedDigits(digVec, digROFVec); // lot of copying involved
162  if (mDoCalibData) {
163  mDecoder->fillCalibData(calVec);
164  }
165  }
166  if (mDoClusters) { // !!! THREADS !!!
167  mClusterer->process(mNThreads, *mDecoder.get(), &clusCompVec, mDoPatterns ? &clusPattVec : nullptr, &clusROFVec);
168  }
169  }
170 
171  if (mDoDigits) {
172  pc.outputs().snapshot(Output{orig, "DIGITS", 0, Lifetime::Timeframe}, digVec);
173  pc.outputs().snapshot(Output{orig, "DIGITSROF", 0, Lifetime::Timeframe}, digROFVec);
174  mEstNDig = std::max(mEstNDig, size_t(digVec.size() * 1.2));
175  mEstNROF = std::max(mEstNROF, size_t(digROFVec.size() * 1.2));
176  if (mDoCalibData) {
177  pc.outputs().snapshot(Output{orig, "GBTCALIB", 0, Lifetime::Timeframe}, calVec);
178  mEstNCalib = std::max(mEstNCalib, size_t(calVec.size() * 1.2));
179  }
180  }
181 
182  if (mDoClusters) { // we are not obliged to create vectors which are not requested, but other devices might not know the options of this one
183  pc.outputs().snapshot(Output{orig, "COMPCLUSTERS", 0, Lifetime::Timeframe}, clusCompVec);
184  pc.outputs().snapshot(Output{orig, "PATTERNS", 0, Lifetime::Timeframe}, clusPattVec);
185  pc.outputs().snapshot(Output{orig, "CLUSTERSROF", 0, Lifetime::Timeframe}, clusROFVec);
186  mEstNClus = std::max(mEstNClus, size_t(clusCompVec.size() * 1.2));
187  mEstNClusPatt = std::max(mEstNClusPatt, size_t(clusPattVec.size() * 1.2));
188  mEstNROF = std::max(mEstNROF, size_t(clusROFVec.size() * 1.2));
189  }
190 
191  if (mDoClusters) {
192  LOG(INFO) << mSelfName << " Built " << clusCompVec.size() << " clusters in " << clusROFVec.size() << " ROFs";
193  }
194  if (mDoDigits) {
195  LOG(INFO) << mSelfName << " Decoded " << digVec.size() << " Digits in " << digROFVec.size() << " ROFs";
196  }
197  mTimer.Stop();
198  auto tfID = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true))->tfCounter;
199  LOG(INFO) << mSelfName << " Total time for TF " << tfID << '(' << mTFCounter << ") : CPU: " << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0;
200  mTFCounter++;
201 }

218 DataProcessorSpec getSTFDecoderITSSpec(bool doClusters, bool doPatterns, bool doDigits, bool doCalib, bool askDISTSTF, const std::string& dict, const std::string& noise)
219 {
220  std::vector<OutputSpec> outputs;
221  auto orig = o2::header::gDataOriginITS;
222 
223  if (doDigits) {
224  outputs.emplace_back(orig, "DIGITS", 0, Lifetime::Timeframe);
225  outputs.emplace_back(orig, "DIGITSROF", 0, Lifetime::Timeframe);
226  if (doCalib) {
227  outputs.emplace_back(orig, "GBTCALIB", 0, Lifetime::Timeframe);
228  }
229  }
230  if (doClusters) {
231  outputs.emplace_back(orig, "COMPCLUSTERS", 0, Lifetime::Timeframe);
232  outputs.emplace_back(orig, "CLUSTERSROF", 0, Lifetime::Timeframe);
233  outputs.emplace_back(orig, "PATTERNS", 0, Lifetime::Timeframe);
234  }
235 
236  std::vector<InputSpec> inputs{{"stf", ConcreteDataTypeMatcher{orig, "RAWDATA"}, Lifetime::Optional}};
237  if (askDISTSTF) {
238  inputs.emplace_back("stdDist", "FLP", "DISTSUBTIMEFRAME", 0, Lifetime::Timeframe);
239  }
240 
241  return DataProcessorSpec{
242  "its-stf-decoder",
243  inputs,
244  outputs,
245  AlgorithmSpec{adaptFromTask<STFDecoder<ChipMappingITS>>(doClusters, doPatterns, doDigits, doCalib, dict, noise)},
246  Options{
247  {"nthreads", VariantType::Int, 1, {"Number of decoding/clustering threads"}},
248  {"old-format", VariantType::Bool, false, {"Use old format (1 trigger per CRU page)"}},
249  {"decoder-verbosity", VariantType::Int, 0, {"Verbosity level (-1: silent, 0: errors, 1: headers, 2: data)"}}}};
250 }
