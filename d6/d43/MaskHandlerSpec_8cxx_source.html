79  void run(o2::framework::ProcessingContext& pc)
80  {
81  auto tStart = std::chrono::high_resolution_clock::now();
82 
83  std::array<gsl::span<const ColumnData>, 2> masks;
84 
85  masks[0] = pc.inputs().get<gsl::span<ColumnData>>("mid_noise_mask");
86  masks[1] = pc.inputs().get<gsl::span<ColumnData>>("mid_dead_mask");
87 
88  auto tAlgoStart = std::chrono::high_resolution_clock::now();
89 
90  bool isChanged = false;
91  for (int itype = 0; itype < 2; ++itype) {
92  ChannelMasksHandler chMasks;
93  for (auto& mask : masks[itype]) {
94  chMasks.setFromChannelMask(mask);
95  }
96  if (!(chMasks == mMasksHandlers[itype])) {
97  mMasksHandlers[itype] = chMasks;
98  isChanged = true;
99  }
100  }
101 
102  if (isChanged) {
103  printSummary();
104  }
105 
106  mTimerMaskHandler += std::chrono::high_resolution_clock::now() - tAlgoStart;
107 
108  mTimer += std::chrono::high_resolution_clock::now() - tStart;
109  }
110 
111  private:
112  std::array<ChannelMasksHandler, 2> mMasksHandlers{};
113  std::chrono::duration<double> mTimer{0};
114  std::chrono::duration<double> mTimerMaskHandler{0};
115 };
