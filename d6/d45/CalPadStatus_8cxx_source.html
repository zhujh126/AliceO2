145 TH1F* CalPadStatus::makeHisto1D()
146 {
147  //
148  // Make 1D histo
149  //
150 
151  char name[1000];
152  snprintf(name, 1000, "%s Pad 1D", getTitle().c_str());
153  TH1F* his = new TH1F(name, name, 6, -0.5, 5.5);
154  his->GetXaxis()->SetBinLabel(1, "Good");
155  his->GetXaxis()->SetBinLabel(2, "Masked");
156  his->GetXaxis()->SetBinLabel(3, "PadBridgedLeft");
157  his->GetXaxis()->SetBinLabel(4, "PadBridgedRight");
158  his->GetXaxis()->SetBinLabel(5, "ReadSecond");
159  his->GetXaxis()->SetBinLabel(6, "NotConnected");
160 
161  for (int idet = 0; idet < MAXCHAMBER; idet++) {
162  if (mROC[idet]) {
163  for (int ichannel = 0; ichannel < mROC[idet]->getNchannels(); ichannel++) {
164  int status = (int)mROC[idet]->getStatus(ichannel);
165  if (status == 2)
166  status = 1;
167  if (status == 4)
168  status = 2;
169  if (status == 8)
170  status = 3;
171  if (status == 16)
172  status = 4;
173  if (status == 32)
174  status = 5;
175  his->Fill(status);
176  }
177  }
178  }
179 
180  return his;
181 }

184 TH2F* CalPadStatus::makeHisto2DSmPl(int sm, int pl)
185 {
186  //
187  // Make 2D graph
188  //
189 
190  gStyle->SetPalette(1);
191  Geometry* trdGeo = new Geometry();
192  const PadPlane* padPlane0 = trdGeo->getPadPlane(pl, 0);
193  Double_t row0 = padPlane0->getRow0();
194  Double_t col0 = padPlane0->getCol0();
195 
196  char name[1000];
197  snprintf(name, 1000, "%s Pad 2D sm %d pl %d", getTitle().c_str(), sm, pl);
198  TH2F* his = new TH2F(name, name, 88, -TMath::Abs(row0), TMath::Abs(row0), 148, -TMath::Abs(col0), TMath::Abs(col0));
199 
200  // Where we begin
201  int offsetsmpl = 30 * sm + pl;
202 
203  for (int k = 0; k < NSTACK; k++) {
204  int det = offsetsmpl + k * 6;
205  if (mROC[det]) {
206  CalSingleChamberStatus* calRoc = mROC[det];
207  for (int icol = 0; icol < calRoc->getNcols(); icol++) {
208  for (int irow = 0; irow < calRoc->getNrows(); irow++) {
209  int binz = 0;
210  int kb = NSTACK - 1 - k;
211  int krow = calRoc->getNrows() - 1 - irow;
212  int kcol = calRoc->getNcols() - 1 - icol;
213  if (kb > 2)
214  binz = 16 * (kb - 1) + 12 + krow + 1 + 2 * (kb + 1);
215  else
216  binz = 16 * kb + krow + 1 + 2 * (kb + 1);
217  int biny = kcol + 1 + 2;
218  Float_t value = calRoc->getStatus(icol, irow);
219  his->SetBinContent(binz, biny, value);
220  }
221  }
222  for (int icol = 1; icol < 147; icol++) {
223  for (int l = 0; l < 2; l++) {
224  int binz = 0;
225  int kb = NSTACK - 1 - k;
226  if (kb > 2)
227  binz = 16 * (kb - 1) + 12 + 1 + 2 * (kb + 1) - (l + 1);
228  else
229  binz = 16 * kb + 1 + 2 * (kb + 1) - (l + 1);
230  his->SetBinContent(binz, icol, 50.0);
231  }
232  }
233  }
234  }
235  for (int icol = 1; icol < 147; icol++) {
236  his->SetBinContent(88, icol, 50.0);
237  his->SetBinContent(87, icol, 50.0);
238  }
239  for (int irow = 1; irow < 89; irow++) {
240  his->SetBinContent(irow, 1, 50.0);
241  his->SetBinContent(irow, 2, 50.0);
242  his->SetBinContent(irow, 147, 50.0);
243  his->SetBinContent(irow, 148, 50.0);
244  }
245 
246  his->SetXTitle("z (cm)");
247  his->SetYTitle("y (cm)");
248  his->SetMaximum(50);
249  his->SetMinimum(0.0);
250  his->SetStats(0);
251 
252  return his;
253 }
