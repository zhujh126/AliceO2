83  template <typename Arg>
84  static void doAppendInputWithMetadata(const char* name, bool value, std::vector<InputSpec>& inputs)
85  {
86  using metadata = typename aod::MetadataTrait<std::decay_t<Arg>>::metadata;
87  static_assert(std::is_same_v<metadata, void> == false,
88  "Could not find metadata. Did you register your type?");
89  std::vector<ConfigParamSpec> inputMetadata;
90  inputMetadata.emplace_back(ConfigParamSpec{std::string{"control:"} + name, VariantType::Bool, value, {"\"\""}});
91  if constexpr (soa::is_soa_index_table_t<std::decay_t<Arg>>::value) {
92  auto inputSources = getIndexSources<std::decay_t<Arg>>();
93  std::sort(inputSources.begin(), inputSources.end(), [](ConfigParamSpec const& a, ConfigParamSpec const& b) { return a.name < b.name; });
94  auto last = std::unique(inputSources.begin(), inputSources.end(), [](ConfigParamSpec const& a, ConfigParamSpec const& b) { return a.name == b.name; });
95  inputSources.erase(last, inputSources.end());
96  inputMetadata.insert(inputMetadata.end(), inputSources.begin(), inputSources.end());
97  }
98  auto locate = std::find_if(inputs.begin(), inputs.end(), [](InputSpec& input) { return input.binding == metadata::tableLabel(); });
99  if (locate != inputs.end()) {
100  // amend entry
101  auto& entryMetadata = locate->metadata;
102  entryMetadata.insert(entryMetadata.end(), inputMetadata.begin(), inputMetadata.end());
103  std::sort(entryMetadata.begin(), entryMetadata.end(), [](ConfigParamSpec const& a, ConfigParamSpec const& b) { return a.name < b.name; });
104  auto new_end = std::unique(entryMetadata.begin(), entryMetadata.end(), [](ConfigParamSpec const& a, ConfigParamSpec const& b) { return a.name == b.name; });
105  entryMetadata.erase(new_end, entryMetadata.end());
106  } else {
107  // add entry
108  inputs.push_back(InputSpec{metadata::tableLabel(), metadata::origin(), metadata::description(), Lifetime::Timeframe, inputMetadata});
109  }
110  }

199  static auto extractSomethingFromRecord(InputRecord& record, std::vector<ExpressionInfo> const infos, size_t phash)

255  auto getLabelFromType()
256  {
257  if constexpr (soa::is_soa_index_table_t<std::decay_t<G>>::value) {
258  using T = typename std::decay_t<G>::first_t;
259  if constexpr (soa::is_type_with_originals_v<std::decay_t<T>>) {
260  using O = typename framework::pack_element_t<0, typename std::decay_t<G>::originals>;
261  using groupingMetadata = typename aod::MetadataTrait<O>::metadata;
262  return std::string("fIndex") + groupingMetadata::tableLabel();
263  } else {
264  using groupingMetadata = typename aod::MetadataTrait<T>::metadata;
265  return std::string("fIndex") + groupingMetadata::tableLabel();
266  }
267  } else if constexpr (soa::is_type_with_originals_v<std::decay_t<G>>) {
268  using T = typename framework::pack_element_t<0, typename std::decay_t<G>::originals>;
269  using groupingMetadata = typename aod::MetadataTrait<T>::metadata;
270  return std::string("fIndex") + groupingMetadata::tableLabel();
271  } else {
272  using groupingMetadata = typename aod::MetadataTrait<std::decay_t<G>>::metadata;
273  return std::string("fIndex") + groupingMetadata::tableLabel();
274  }
275  }

277  GroupSlicerIterator(G& gt, std::tuple<A...>& at)
278  : mAt{&at},
279  mGroupingElement{gt.begin()},
280  position{0}
281  {
282  if constexpr (soa::is_soa_filtered_t<std::decay_t<G>>::value) {
283  groupSelection = &gt.getSelectedRows();
284  }
285  auto indexColumnName = getLabelFromType();
289  auto splitter = [&](auto&& x) {
290  using xt = std::decay_t<decltype(x)>;
291  constexpr auto index = framework::has_type_at_v<std::decay_t<decltype(x)>>(associated_pack_t{});
292  if (x.size() != 0 && hasIndexTo<std::decay_t<G>>(typename xt::persistent_columns_t{})) {
293  auto result = o2::framework::sliceByColumn(indexColumnName.c_str(),
294  x.asArrowTable(),
295  static_cast<int32_t>(gt.tableSize()),
296  &groups[index],
297  &offsets[index],
298  &sizes[index]);
299  if (result.ok() == false) {
300  throw runtime_error("Cannot split collection");
301  }
302  if (groups[index].size() > gt.tableSize()) {
303  throw runtime_error_f("Splitting collection resulted in a larger group number (%d) than there is rows in the grouping table (%d).", groups[index].size(), gt.tableSize());
304  };
305  }
306  };
307 
308  std::apply(
309  [&](auto&&... x) -> void {
310  (splitter(x), ...);
311  },
312  at);
314  auto extractor = [&](auto&& x) {
315  using xt = std::decay_t<decltype(x)>;
316  if constexpr (soa::is_soa_filtered_t<xt>::value) {
317  constexpr auto index = framework::has_type_at_v<std::decay_t<decltype(x)>>(associated_pack_t{});
318  selections[index] = &x.getSelectedRows();
319  starts[index] = selections[index]->begin();
320  offsets[index].push_back(std::get<xt>(at).tableSize());
321  }
322  };
323  std::apply(
324  [&](auto&&... x) -> void {
325  (extractor(x), ...);
326  },
327  at);
328  }

336  template <typename B, typename C>
337  constexpr bool isIndexTo()
338  {
339  if constexpr (soa::is_type_with_binding_v<C>) {
340  if constexpr (soa::is_soa_index_table_t<std::decay_t<B>>::value) {
341  using T = typename std::decay_t<B>::first_t;
342  if constexpr (soa::is_type_with_originals_v<std::decay_t<T>>) {
343  using TT = typename framework::pack_element_t<0, typename std::decay_t<T>::originals>;
344  return std::is_same_v<typename C::binding_t, TT>;
345  } else {
346  using TT = std::decay_t<T>;
347  return std::is_same_v<typename C::binding_t, TT>;
348  }
349  } else {
350  if constexpr (soa::is_type_with_originals_v<std::decay_t<B>>) {
351  using TT = typename framework::pack_element_t<0, typename std::decay_t<B>::originals>;
352  return std::is_same_v<typename C::binding_t, TT>;
353  } else {
354  using TT = std::decay_t<B>;
355  return std::is_same_v<typename C::binding_t, TT>;
356  }
357  }
358  }
359  return false;
360  }

394  template <typename A1>
395  auto prepareArgument()
396  {
397  constexpr auto index = framework::has_type_at_v<A1>(associated_pack_t{});
398  if (std::get<A1>(*mAt).size() != 0 && hasIndexTo<G>(typename std::decay_t<A1>::persistent_columns_t{})) {
399  uint64_t pos;
400  if constexpr (soa::is_soa_filtered_t<std::decay_t<G>>::value) {
401  pos = (*groupSelection)[position];
402  } else {
403  pos = position;
404  }
405  if constexpr (soa::is_soa_filtered_t<std::decay_t<A1>>::value) {
406  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(((groups[index])[pos]).value);
407 
408  // for each grouping element we need to slice the selection vector
409  auto start_iterator = std::lower_bound(starts[index], selections[index]->end(), (offsets[index])[pos]);
410  auto stop_iterator = std::lower_bound(start_iterator, selections[index]->end(), (offsets[index])[pos] + (sizes[index])[pos]);
411  starts[index] = stop_iterator;
412  soa::SelectionVector slicedSelection{start_iterator, stop_iterator};
413  std::transform(slicedSelection.begin(), slicedSelection.end(), slicedSelection.begin(),
414  [&](int64_t idx) {
415  return idx - static_cast<int64_t>((offsets[index])[pos]);
416  });
417 
418  std::decay_t<A1> typedTable{{groupedElementsTable}, std::move(slicedSelection), (offsets[index])[pos]};
419  typedTable.bindInternalIndicesTo(&std::get<A1>(*mAt));
420  return typedTable;
421  } else {
422  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(((groups[index])[pos]).value);
423  std::decay_t<A1> typedTable{{groupedElementsTable}, (offsets[index])[pos]};
424  typedTable.bindInternalIndicesTo(&std::get<A1>(*mAt));
425  return typedTable;
426  }
427  } else {
428  return std::get<A1>(*mAt);
429  }
430  O2_BUILTIN_UNREACHABLE();
431  }
432 
433  std::tuple<A...>* mAt;
434  typename grouping_t::iterator mGroupingElement;
435  uint64_t position = 0;
436  soa::SelectionVector const* groupSelection = nullptr;
437  std::array<std::vector<arrow::Datum>, sizeof...(A)> groups;
438  std::array<std::vector<uint64_t>, sizeof...(A)> offsets;
439  std::array<std::vector<int>, sizeof...(A)> sizes;
440  std::array<soa::SelectionVector const*, sizeof...(A)> selections;
441  std::array<soa::SelectionVector::const_iterator, sizeof...(A)> starts;
442  };

463  template <typename Task, typename R, typename C, typename Grouping, typename... Associated>
464  static void invokeProcess(Task& task, InputRecord& inputs, R (C::*processingFunction)(Grouping, Associated...), std::vector<ExpressionInfo> const& infos)
465  {
466  using G = std::decay_t<Grouping>;
467  auto groupingTable = AnalysisDataProcessorBuilder::bindGroupingTable(inputs, processingFunction, infos);
468 
469  // set filtered tables for partitions with grouping
470  homogeneous_apply_refs([&groupingTable](auto& x) {
471  PartitionManager<std::decay_t<decltype(x)>>::setPartition(x, groupingTable);
472  PartitionManager<std::decay_t<decltype(x)>>::bindInternalIndices(x, &groupingTable);
473  return true;
474  },
475  task);
476 
477  if constexpr (sizeof...(Associated) == 0) {
478  // single argument to process
479  homogeneous_apply_refs([&groupingTable](auto& x) {
480  PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable);
481  PartitionManager<std::decay_t<decltype(x)>>::getBoundToExternalIndices(x, groupingTable);
482  return true;
483  },
484  task);
485  if constexpr (soa::is_soa_iterator_t<G>::value) {
486  for (auto& element : groupingTable) {
487  std::invoke(processingFunction, task, *element);
488  }
489  } else {
490  static_assert(soa::is_soa_table_like_t<G>::value,
491  "Single argument of process() should be a table-like or an iterator");
492  std::invoke(processingFunction, task, groupingTable);
493  }
494  } else {
495  // multiple arguments to process
496  static_assert(((soa::is_soa_iterator_t<std::decay_t<Associated>>::value == false) && ...),
497  "Associated arguments of process() should not be iterators");
498  auto associatedTables = AnalysisDataProcessorBuilder::bindAssociatedTables(inputs, processingFunction, infos);
499  //pre-bind self indices
500  std::apply(
501  [&](auto&... t) {
502  (homogeneous_apply_refs(
503  [&](auto& p) {
504  PartitionManager<std::decay_t<decltype(p)>>::bindInternalIndices(p, &t);
505  return true;
506  },
507  task),
508  ...);
509  },
510  associatedTables);
511 
512  auto binder = [&](auto&& x) {
513  x.bindExternalIndices(&groupingTable, &std::get<std::decay_t<Associated>>(associatedTables)...);
514  homogeneous_apply_refs([&x](auto& t) {
515  PartitionManager<std::decay_t<decltype(t)>>::setPartition(t, x);
516  PartitionManager<std::decay_t<decltype(t)>>::bindExternalIndices(t, &x);
517  PartitionManager<std::decay_t<decltype(t)>>::getBoundToExternalIndices(t, x);
518  return true;
519  },
520  task);
521  };
522  groupingTable.bindExternalIndices(&std::get<std::decay_t<Associated>>(associatedTables)...);
523 
524  // always pre-bind full tables to support index hierarchy
525  std::apply(
526  [&](auto&&... x) {
527  (binder(x), ...);
528  },
529  associatedTables);
530 
531  if constexpr (soa::is_soa_iterator_t<std::decay_t<G>>::value) {
532  // grouping case
533  auto slicer = GroupSlicer(groupingTable, associatedTables);
534  for (auto& slice : slicer) {
535  auto associatedSlices = slice.associatedTables();
536 
537  std::apply(
538  [&](auto&&... x) {
539  (binder(x), ...);
540  },
541  associatedSlices);
542

718 template <typename T, typename... Args>
719 DataProcessorSpec adaptAnalysisTask(ConfigContext const& ctx, Args&&... args)
720 {
721  TH1::AddDirectory(false);
722 
723  auto [name_str, task] = getTaskNameSetProcesses<T>(args...);
724 
725  auto suffix = ctx.options().get<std::string>("workflow-suffix");
726  if (!suffix.empty()) {
727  name_str += suffix;
728  }
729  const char* name = name_str.c_str();
730 
731  auto hash = compile_time_hash(name);
732 
733  std::vector<OutputSpec> outputs;
734  std::vector<InputSpec> inputs;
735  std::vector<ConfigParamSpec> options;
736  std::vector<ExpressionInfo> expressionInfos;
737 
739  homogeneous_apply_refs([&options, &hash](auto& x) { return OptionManager<std::decay_t<decltype(x)>>::appendOption(options, x); }, *task.get());
740 
742  if constexpr (has_process_v<T>) {
743  AnalysisDataProcessorBuilder::inputsFromArgs(&T::process, "default", true, inputs, expressionInfos);
744  }
745  homogeneous_apply_refs(
746  [name = name_str, &expressionInfos, &inputs](auto& x) {
747  using D = std::decay_t<decltype(x)>;
748  if constexpr (is_base_of_template<ProcessConfigurable, D>::value) {
749  // this pushes (argumentIndex,processHash,schemaPtr,nullptr) into expressionInfos for arguments that are Filtered/filtered_iterators
750  AnalysisDataProcessorBuilder::inputsFromArgs(x.process, (name + "/" + x.name).c_str(), x.value, inputs, expressionInfos);
751  return true;
752  }
753  return false;
754  },
755  *task.get());
756 
757  // avoid self-forwarding if process methods subscribe to same tables
758  std::sort(inputs.begin(), inputs.end(), [](InputSpec const& a, InputSpec const& b) { return a.binding < b.binding; });
759  auto last = std::unique(inputs.begin(), inputs.end(), [](InputSpec const& a, InputSpec const& b) { return a.binding == b.binding; });
760  inputs.erase(last, inputs.end());
761 
762  // request base tables for spawnable extended tables
763  // this checks for duplications
764  homogeneous_apply_refs([&inputs](auto& x) {
765  return SpawnManager<std::decay_t<decltype(x)>>::requestInputs(inputs, x);
766  },
767  *task.get());
768 
769  //request base tables for indices to be built
770  homogeneous_apply_refs([&inputs](auto& x) {
771  return IndexManager<std::decay_t<decltype(x)>>::requestInputs(inputs, x);
772  },
773  *task.get());
774 
775  // no static way to check if the task defines any processing, we can only make sure it subscribes to at least something
776  if (inputs.empty() == true) {
777  LOG(WARN) << "Task " << name_str << " has no inputs";
778  }
779 
780  homogeneous_apply_refs([&outputs, &hash](auto& x) { return OutputManager<std::decay_t<decltype(x)>>::appendOutput(outputs, x, hash); }, *task.get());
781 
782  std::vector<ServiceSpec> requiredServices = CommonServices::defaultServices();
783  homogeneous_apply_refs([&requiredServices](auto& x) { return ServiceManager<std::decay_t<decltype(x)>>::add(requiredServices, x); }, *task.get());
784 
785  auto algo = AlgorithmSpec::InitCallback{[task = task, expressionInfos](InitContext& ic) mutable {
786  homogeneous_apply_refs([&ic](auto&& x) { return OptionManager<std::decay_t<decltype(x)>>::prepare(ic, x); }, *task.get());
787  homogeneous_apply_refs([&ic](auto&& x) { return ServiceManager<std::decay_t<decltype(x)>>::prepare(ic, x); }, *task.get());
788 
789  auto& callbacks = ic.services().get<CallbackService>();
790  auto endofdatacb = [task](EndOfStreamContext& eosContext) {
791  homogeneous_apply_refs([&eosContext](auto&& x) { return OutputManager<std::decay_t<decltype(x)>>::postRun(eosContext, x); }, *task.get());
792  eosContext.services().get<ControlService>().readyToQuit(QuitRequest::Me);
793  };
794  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
795 
797  homogeneous_apply_refs(
798  [&ic](auto& x) -> bool { return FilterManager<std::decay_t<decltype(x)>>::updatePlaceholders(x, ic); },
799  *task.get());
801  homogeneous_apply_refs(
802  [&ic](auto& x) -> bool { PartitionManager<std::decay_t<decltype(x)>>::updatePlaceholders(x, ic); return true; },
803  *task.get());
805  homogeneous_apply_refs([&expressionInfos](auto& x) {
806  return FilterManager<std::decay_t<decltype(x)>>::createExpressionTrees(x, expressionInfos);
807  },
808  *task.get());
809 
810  if constexpr (has_init_v<T>) {
811  task->init(ic);
812  }
813 
814  return [task, expressionInfos](ProcessingContext& pc) {
815  homogeneous_apply_refs([&pc](auto&& x) { return OutputManager<std::decay_t<decltype(x)>>::prepare(pc, x); }, *task.get());
816  if constexpr (has_run_v<T>) {
817  task->run(pc);
818  }
819  if constexpr (has_process_v<T>) {
820  AnalysisDataProcessorBuilder::invokeProcess(*(task.get()), pc.inputs(), &T::process, expressionInfos);
821  }
822  homogeneous_apply_refs(
823  [&pc, &expressionInfos, &task](auto& x) {
824  if constexpr (is_base_of_template<ProcessConfigurable, std::decay_t<decltype(x)>>::value) {
825  if (x.value == true) {
826  AnalysisDataProcessorBuilder::invokeProcess(*task.get(), pc.inputs(), x.process, expressionInfos);
827  return true;
828  }
829  }
830  return false;
831  },
832  *task.get());
833 
834  homogeneous_apply_refs([&pc](auto&& x) { return OutputManager<std::decay_t<decltype(x)>>::finalize(pc, x); }, *task.get());
835  };
836  }};
837 
838  DataProcessorSpec spec{
839  name,
840  // FIXME: For the moment we hardcode this. We could build
841  // this list from the list of methods actually implemented in the
842  // task itself.
843  inputs,
844  outputs,
845  algo,
846  options,
847  requiredServices};
848  return spec;
849 }
