28 unsigned int o2::tpc::IDCGroupHelperRegion::getGroupedPad(const unsigned int upad, const unsigned int ulrow, const unsigned int region, const unsigned int groupPads, const unsigned int groupRows, const unsigned int groupedrows, const std::vector<unsigned int>& padsPerRow)
29 {
30  const int relPadHalf = static_cast<int>(std::floor((upad - 0.5f * Mapper::PADSPERROW[region][ulrow]) / groupPads));
31  const unsigned int nGroupedPads = padsPerRow[getGroupedRow(ulrow, groupRows, groupedrows)];
32  const unsigned int nGroupedPadsHalf = (nGroupedPads / 2);
33  if (std::abs(relPadHalf) >= nGroupedPadsHalf) {
34  return std::signbit(relPadHalf) ? 0 : nGroupedPads - 1;
35  }
36  return static_cast<unsigned int>(static_cast<int>(nGroupedPadsHalf) + relPadHalf);
37 }

57 unsigned int o2::tpc::IDCGroupHelperRegion::getLastPad(const unsigned int ulrow) const
58 {
59  const unsigned int nPads = Mapper::PADSPERROW[mRegion][ulrow] / 2;
60  const unsigned int padsRemainder = nPads % mGroupPads;
61  int unsigned lastPad = (padsRemainder == 0) ? nPads - mGroupPads : nPads - padsRemainder;
62  if (padsRemainder && padsRemainder <= mGroupLastPadsThreshold) {
63  lastPad -= mGroupPads;
64  }
65  return lastPad;

68 void o2::tpc::IDCGroupHelperRegion::initIDCGroupHelperRegion()
69 {
70  const unsigned int nRows = getLastRow() / mGroupRows + 1;
71  setRows(nRows);
72  for (unsigned int irow = 0; irow < nRows; ++irow) {
73  const unsigned int row = irow * mGroupRows;
74  mPadsPerRow[irow] = 2 * (getLastPad(row) / mGroupPads + 1);
75  }
76 
77  mNIDCsPerCRU = std::accumulate(mPadsPerRow.begin(), mPadsPerRow.end(), decltype(mPadsPerRow)::value_type(0));
78  for (unsigned int i = 1; i < mRows; ++i) {
79  const unsigned int lastInd = i - 1;
80  mOffsRow[i] = mOffsRow[lastInd] + mPadsPerRow[lastInd];
81  }
82 }
