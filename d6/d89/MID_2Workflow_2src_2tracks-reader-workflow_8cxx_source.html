35 class TrackSamplerTask
36 {
37  public:
39  void init(InitContext& ic)
40  {
41  auto inputFileName = ic.options().get<std::string>("infile");
42  mMinNumberOfROFsPerTF = ic.options().get<int>("repack-rofs");
43 
44  mReader = std::make_unique<RootTreeReader>("midreco", inputFileName.c_str(), -1,
45  RootTreeReader::PublishingMode::Single,
46  RootTreeReader::BranchDefinition<std::vector<char>>{
47  Output{"MID", "TRACKS", 0, Lifetime::Timeframe}, "MIDTrack"},
48  RootTreeReader::BranchDefinition<std::vector<ROFRecord>>{
49  Output{"MID", "TRACKSROF", 0, Lifetime::Timeframe}, "MIDTrackROF"},
50  &mAccumulator);
51  }
52 
54  void run(ProcessingContext& pc)
55  {
56  if (mReader->next()) {
57  (*mReader)(pc);
58  if (mROFs.size() >= mMinNumberOfROFsPerTF) {
59  publish(pc.outputs());
60  }
61  } else {
62  if (mROFs.size() > 0) {
63  publish(pc.outputs());
64  }
65  pc.services().get<ControlService>().endOfStream();
66  }
67  }
68 
69  private:
71  bool accumulate(std::string_view name, char* data)
72  {
73  if (name == "MIDTrackROF") {
74 
75  auto rofs = reinterpret_cast<std::vector<ROFRecord>*>(data);
76 
77  // accumulate the ROFs, shifting the track indexing accordingly
78  size_t offset = (mROFs.size() > 0) ? mROFs.back().firstEntry + mROFs.back().nEntries : 0;
79  std::transform(rofs->begin(), rofs->end(), std::back_inserter(mROFs), [offset](const ROFRecord& rof) {
80  return ROFRecord{rof, rof.firstEntry + offset, rof.nEntries};
81  });
82 
83  } else if (name == "MIDTrack") {
84 
85  auto tracks = reinterpret_cast<std::vector<char>*>(data);
86 
87  // accumulate the tracks, in Track format
88  if (tracks->size() % sizeof(Track) != 0) {
89  throw std::length_error("invalid track format");
90  }
91  size_t offset = mTracks.size();
92  mTracks.resize(offset + tracks->size() / sizeof(Track));
93  std::memcpy(&(mTracks[offset]), tracks->data(), tracks->size());
94 
95  } else {
96  throw std::invalid_argument("invalid branch");
97  }
98 
99  return true;
100  }
101 
103  void publish(DataAllocator& out)
104  {
105  out.snapshot(OutputRef{"rofs"}, mROFs);
106  out.snapshot(OutputRef{"tracks"}, mTracks);
107  mROFs.clear();
108  mTracks.clear();
109  }
110 
111  size_t mMinNumberOfROFsPerTF = 1;
112  std::vector<ROFRecord> mROFs{};
113  std::vector<Track> mTracks{};
114  std::unique_ptr<RootTreeReader> mReader{};
115  RootTreeReader::SpecialPublishHook mAccumulator{
117  [this](std::string_view name, ProcessingContext&, Output const&, char* data) -> bool {
118  return this->accumulate(name, data);
119  }};
120 };
121

123 {

131  {"repack-rofs", VariantType::Int, 1, {"min number of rofs per timeframe"}}}}};
132 }
