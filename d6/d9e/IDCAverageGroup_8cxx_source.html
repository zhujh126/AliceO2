49 void o2::tpc::IDCAverageGroup::processIDCs()
50 {
51  const static auto& paramIDCGroup = ParameterIDCGroup::Instance();
52 
53 #pragma omp parallel for num_threads(sNThreads)
54  for (unsigned int integrationInterval = 0; integrationInterval < getNIntegrationIntervals(); ++integrationInterval) {
55  const unsigned int threadNum = omp_get_thread_num();
56  const unsigned int lastRow = mIDCsGrouped.getLastRow();
57  unsigned int rowGrouped = 0;
58  for (unsigned int iRow = 0; iRow <= lastRow; iRow += mIDCsGrouped.getGroupRows()) {
59  // the sectors is divide in to two parts around ylocal=0 to get the same simmetric grouping around ylocal=0
60  for (int iYLocalSide = 0; iYLocalSide < 2; ++iYLocalSide) {
61  const unsigned int region = mIDCsGrouped.getRegion();
62  const unsigned int nPads = Mapper::PADSPERROW[region][iRow] / 2;
63  const unsigned int endPads = mIDCsGrouped.getLastPad(iRow) + nPads;
64 
65  const unsigned int halfPadsInRow = mIDCsGrouped.getPadsPerRow(rowGrouped) / 2;
66  unsigned int padGrouped = iYLocalSide ? halfPadsInRow : halfPadsInRow - 1;
67  for (unsigned int ipad = nPads; ipad <= endPads; ipad += mIDCsGrouped.getGroupPads()) {
68  const unsigned int endRows = (iRow == lastRow) ? (Mapper::ROWSPERREGION[region] - iRow) : mIDCsGrouped.getGroupRows();
69  mRobustAverage[threadNum].clear();
70  for (unsigned int iRowMerge = 0; iRowMerge < endRows; ++iRowMerge) {
71  const unsigned int iRowTmp = iRow + iRowMerge;
72  const auto offs = Mapper::ADDITIONALPADSPERROW[region][iRowTmp] - Mapper::ADDITIONALPADSPERROW[region][iRow];
73  const auto padStart = (ipad == 0) ? 0 : offs;
74  const unsigned int endPadsTmp = (ipad == endPads) ? (Mapper::PADSPERROW[region][iRowTmp] - ipad) : mIDCsGrouped.getGroupPads() + offs;
75  for (unsigned int ipadMerge = padStart; ipadMerge < endPadsTmp; ++ipadMerge) {
76  const unsigned int iPadTmp = ipad + ipadMerge;
77  const unsigned int iPadSide = iYLocalSide ? iPadTmp : Mapper::PADSPERROW[region][iRowTmp] - iPadTmp - 1;
78  const unsigned int indexIDC = integrationInterval * Mapper::PADSPERREGION[region] + Mapper::OFFSETCRULOCAL[region][iRowTmp] + iPadSide;
79  mRobustAverage[threadNum].addValue(mIDCsUngrouped[indexIDC] * Mapper::PADAREA[region]);
80  }
81  }
82 
83  switch (paramIDCGroup.Method) {
84  case o2::tpc::AveragingMethod::SLOW:
85  default:
86  mIDCsGrouped(rowGrouped, padGrouped, integrationInterval) = mRobustAverage[threadNum].getFilteredAverage(mSigma);
87  break;
88  case o2::tpc::AveragingMethod::FAST:
89  mIDCsGrouped(rowGrouped, padGrouped, integrationInterval) = mRobustAverage[threadNum].getMean();
90  break;
91  }
92 
93  iYLocalSide ? ++padGrouped : --padGrouped;
94  }
95  }
96  ++rowGrouped;
97  }
98  }
99 }

124 void o2::tpc::IDCAverageGroup::drawUngroupedIDCs(const unsigned int integrationInterval, const std::string filename) const
125 {
126  const auto coords = o2::tpc::painter::getPadCoordinatesSector();
127  TH2Poly* poly = o2::tpc::painter::makeSectorHist("hSector", "Sector;local #it{x} (cm);local #it{y} (cm); #it{IDC}");
128  poly->SetContour(255);
129  poly->SetTitle(nullptr);
130  poly->GetYaxis()->SetTickSize(0.002f);
131  poly->GetYaxis()->SetTitleOffset(0.7f);
132  poly->GetZaxis()->SetTitleOffset(1.3f);
133  poly->SetStats(0);
134 
135  TCanvas* can = new TCanvas("can", "can", 2000, 1400);
136  can->SetRightMargin(0.14f);
137  can->SetLeftMargin(0.06f);
138  can->SetTopMargin(0.04f);
139 
140  TLatex lat;
141  lat.SetTextFont(63);
142  lat.SetTextSize(2);
143 
144  poly->Draw("colz");
145  const unsigned int region = mIDCsGrouped.getRegion();
146  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
147  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
148  const auto padNum = Mapper::getGlobalPadNumber(irow, ipad, region);
149  const auto coordinate = coords[padNum];
150  const float yPos = -0.5 * (coordinate.yVals[0] + coordinate.yVals[2]); // local coordinate system is mirrored
151  const float xPos = 0.5 * (coordinate.xVals[0] + coordinate.xVals[2]);
152  const unsigned int indexIDC = integrationInterval * Mapper::PADSPERREGION[region] + Mapper::OFFSETCRULOCAL[region][irow] + ipad;
153  const float idc = mIDCsUngrouped[indexIDC] * Mapper::PADAREA[region];
154  poly->Fill(xPos, yPos, idc);
155  lat.SetTextAlign(12);
156  lat.DrawLatex(xPos, yPos, Form("%i", ipad));
157  }
158  }
159 
160  if (!filename.empty()) {
161  can->SaveAs(filename.data());
162  delete poly;
163  delete can;
164  }
165 }

199 void o2::tpc::IDCAverageGroup::createDebugTree(const IDCAverageGroup& idcavg, o2::utils::TreeStreamRedirector& pcstream)
200 {
201  const Mapper& mapper = Mapper::instance();
202  unsigned int sector = idcavg.getSector();
203  unsigned int cru = sector * Mapper::NREGIONS + idcavg.getRegion();
204  const o2::tpc::CRU cruTmp(cru);
205  unsigned int region = cruTmp.region();
206 
207  for (unsigned int integrationInterval = 0; integrationInterval < idcavg.getNIntegrationIntervals(); ++integrationInterval) {
208  const unsigned long padsPerCRU = Mapper::PADSPERREGION[region];
209  std::vector<unsigned int> vRow(padsPerCRU);
210  std::vector<unsigned int> vPad(padsPerCRU);
211  std::vector<float> vXPos(padsPerCRU);
212  std::vector<float> vYPos(padsPerCRU);
213  std::vector<float> vGlobalXPos(padsPerCRU);
214  std::vector<float> vGlobalYPos(padsPerCRU);
215  std::vector<float> idcsPerIntegrationInterval(padsPerCRU); // idcs for one time bin
216  std::vector<float> groupedidcsPerIntegrationInterval(padsPerCRU); // idcs for one time bin
217  std::vector<float> invPadArea(padsPerCRU);
218 
219  for (unsigned int iPad = 0; iPad < padsPerCRU; ++iPad) {
220  const GlobalPadNumber globalNum = Mapper::GLOBALPADOFFSET[region] + iPad;
221  const auto& padPosLocal = mapper.padPos(globalNum);
222  vRow[iPad] = padPosLocal.getRow();
223  vPad[iPad] = padPosLocal.getPad();
224  vXPos[iPad] = mapper.getPadCentre(padPosLocal).X();
225  vYPos[iPad] = mapper.getPadCentre(padPosLocal).Y();
226  invPadArea[iPad] = Mapper::PADAREA[region];
227  const GlobalPosition2D globalPos = mapper.LocalToGlobal(LocalPosition2D(vXPos[iPad], vYPos[iPad]), cruTmp.sector());
228  vGlobalXPos[iPad] = globalPos.X();
229  vGlobalYPos[iPad] = globalPos.Y();
230  idcsPerIntegrationInterval[iPad] = idcavg.getUngroupedIDCVal(iPad, integrationInterval);
231  groupedidcsPerIntegrationInterval[iPad] = idcavg.getGroupedIDCValGlobal(vRow[iPad], vPad[iPad], integrationInterval);
232  }
233 
234  pcstream << "tree"
235  << "cru=" << cru
236  << "sector=" << sector
237  << "region=" << region
238  << "integrationInterval=" << integrationInterval
239  << "IDCUngrouped.=" << idcsPerIntegrationInterval
240  << "IDCGrouped.=" << groupedidcsPerIntegrationInterval
241  << "invPadArea.=" << invPadArea
242  << "pad.=" << vPad
243  << "row.=" << vRow
244  << "lx.=" << vXPos
245  << "ly.=" << vYPos
246  << "gx.=" << vGlobalXPos
247  << "gy.=" << vGlobalYPos
248  << "\n";
249  }
250 }
