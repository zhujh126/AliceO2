66 void AODProducerWorkflowDPL::collectBCs(gsl::span<const o2::ft0::RecPoints>& ft0RecPoints,
67  gsl::span<const o2::dataformats::PrimaryVertex>& primVertices,
68  const std::vector<o2::InteractionTimeRecord>& mcRecords,
69  std::map<uint64_t, int>& bcsMap)
70 {
71  // collecting non-empty BCs and enumerating them
72  for (auto& rec : mcRecords) {
73  uint64_t globalBC = rec.toLong();
74  bcsMap[globalBC] = 1;
75  }
76 
77  for (auto& ft0RecPoint : ft0RecPoints) {
78  uint64_t globalBC = ft0RecPoint.getInteractionRecord().toLong();
79  bcsMap[globalBC] = 1;
80  }
81 
82  for (auto& vertex : primVertices) {
83  auto& timeStamp = vertex.getTimeStamp();

85  uint64_t globalBC = std::round(tsTimeStamp / o2::constants::lhc::LHCBunchSpacingNS);
86  bcsMap[globalBC] = 1;
87  }
88 
89  int bcID = 0;
90  for (auto& item : bcsMap) {
91  item.second = bcID;
92  bcID++;
93  }

96 uint64_t AODProducerWorkflowDPL::getTFNumber(const o2::InteractionRecord& tfStartIR, int runNumber)
97 {
98  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
99  o2::ccdb::CcdbApi ccdb_api;
100  const std::string rct_path = "RCT/RunInformation/";
101  const std::string start_orbit_path = "Trigger/StartOrbit";
102  const std::string url = "http://ccdb-test.cern.ch:8080";
103 
104  mgr.setURL(url);
105  ccdb_api.init(url);
106 
107  std::map<int, int>* mapStartOrbit = mgr.get<std::map<int, int>>(start_orbit_path);
108  int64_t ts = 0;
109  std::map<std::string, std::string> metadata;
110  std::map<std::string, std::string> headers;
111  const std::string run_path = Form("%s/%i", rct_path.data(), runNumber);
112  headers = ccdb_api.retrieveHeaders(run_path, metadata, -1);
113  ts = atol(headers["SOR"].c_str());
114 
115  // ccdb returns timestamp in mus
116  // mus to ms
117  ts = ts / 1000;
118 
119  uint32_t initialOrbit = mapStartOrbit->at(runNumber);
120  uint16_t firstRecBC = tfStartIR.bc;
121  uint32_t firstRecOrbit = tfStartIR.orbit;
122  const o2::InteractionRecord firstRec(firstRecBC, firstRecOrbit);
123  ts += firstRec.bc2ns() / 1000000;
124 
125  return ts;

205 template <typename TracksCursorType, typename TracksCovCursorType, typename TracksExtraCursorType, typename MftTracksCursorType, typename FwdTracksCursorType>
206 void AODProducerWorkflowDPL::fillTrackTablesPerCollision(int collisionID,
207  double interactionTime,
208  const o2::dataformats::VtxTrackRef& trackRef,
209  gsl::span<const GIndex>& GIndices,
210  o2::globaltracking::RecoContainer& data,
211  TracksCursorType& tracksCursor,
212  TracksCovCursorType& tracksCovCursor,
213  TracksExtraCursorType& tracksExtraCursor,
214  MftTracksCursorType& mftTracksCursor,
215  FwdTracksCursorType& fwdTracksCursor,
216  const dataformats::PrimaryVertex& vertex)
217 {
218  const auto& tpcClusRefs = data.getTPCTracksClusterRefs();
219  const auto& tpcClusShMap = data.clusterShMapTPC;
220  const auto& tpcClusAcc = data.getTPCClusters();
221  const auto& tpcTracks = data.getTPCTracks();
222  const auto& itsTracks = data.getITSTracks();
223  const auto& itsABRefs = data.getITSABRefs();
224 
225  for (int src = GIndex::NSources; src--;) {
226  int start = trackRef.getFirstEntryOfSource(src);
227  int end = start + trackRef.getEntriesOfSource(src);
228  LOG(DEBUG) << "Unassigned tracks: src = " << src << ", start = " << start << ", end = " << end;
229  for (int ti = start; ti < end; ti++) {
230  TrackExtraInfo extraInfoHolder;
231  auto& trackIndex = GIndices[ti];
232  if (GIndex::includesSource(src, mInputSources)) {
233  if (src == GIndex::Source::MFT) { // MFT tracks are treated separately since they are stored in a different table
234  const auto& track = data.getMFTTrack(trackIndex);
235  addToMFTTracksTable(mftTracksCursor, track, collisionID);
236  } else if (src == GIndex::Source::MCH) {
237  // FwdTracks tracks are treated separately since they are stored in a different table
238  const auto& track = data.getMCHTrack(trackIndex);
239  if (collisionID < 0) {
240  InteractionRecord meanIR;
241  auto rofsMCH = data.getMCHTracksROFRecords();
242  for (const auto& rof : rofsMCH) {
243  if (trackIndex >= rof.getFirstIdx() && trackIndex <= rof.getLastIdx()) {
244  meanIR = rof.getBCData() + rof.getBCWidth() / 2;
245  }
246  math_utils::Point3D<double> vertex{};
247  // FIXME: should we get better
248  // than {0,0,0} as vertex here ?
249  addToFwdTracksTable(fwdTracksCursor, track, -1, meanIR.toLong(), vertex);
250  }
251  } else {
252  long bcID{-1}; // FIXME: how do I get bcID ?
253  math_utils::Point3D<double> vtx{vertex.getX(),
254  vertex.getY(), vertex.getZ()};
255  addToFwdTracksTable(fwdTracksCursor, track, collisionID, bcID, vtx);
256  }
257  } else {
258  auto contributorsGID = data.getSingleDetectorRefs(trackIndex);
259  const auto& trackPar = data.getTrackParam(trackIndex);
260  if (contributorsGID[GIndex::Source::ITS].isIndexSet()) {
261  extraInfoHolder.itsClusterMap = itsTracks[contributorsGID[GIndex::ITS].getIndex()].getPattern();
262  } else if (contributorsGID[GIndex::Source::ITSAB].isIndexSet()) { // this is an ITS-TPC afterburner contributor
263  extraInfoHolder.itsClusterMap = itsABRefs[contributorsGID[GIndex::Source::ITSAB].getIndex()].pattern;
264  }
265  if (contributorsGID[GIndex::Source::TPC].isIndexSet()) {
266  const auto& tpcOrig = tpcTracks[contributorsGID[GIndex::TPC].getIndex()];
267  extraInfoHolder.tpcInnerParam = tpcOrig.getP();
268  extraInfoHolder.tpcChi2NCl = tpcOrig.getNClusters() ? tpcOrig.getChi2() / tpcOrig.getNClusters() : 0;
269  extraInfoHolder.tpcSignal = tpcOrig.getdEdx().dEdxTotTPC;
270  uint8_t shared, found, crossed; // fixme: need to switch from these placeholders to something more reasonable
271  countTPCClusters(tpcOrig, tpcClusRefs, tpcClusShMap, tpcClusAcc, shared, found, crossed);
272  extraInfoHolder.tpcNClsFindable = tpcOrig.getNClusters();
273  extraInfoHolder.tpcNClsFindableMinusFound = tpcOrig.getNClusters() - found;
274  extraInfoHolder.tpcNClsFindableMinusCrossedRows = tpcOrig.getNClusters() - crossed;
275  extraInfoHolder.tpcNClsShared = shared;
276  }
277  if (contributorsGID[GIndex::Source::ITSTPCTOF].isIndexSet()) {
278  const auto& tofMatch = data.getTOFMatch(contributorsGID[GIndex::Source::ITSTPCTOF]);
279  extraInfoHolder.tofChi2 = tofMatch.getChi2();
280  const auto& tofInt = tofMatch.getLTIntegralOut();
281  float intLen = tofInt.getL();
282  extraInfoHolder.length = intLen;
283  if (interactionTime > 0) {
284  extraInfoHolder.tofSignal = static_cast<float>(tofMatch.getSignal() - interactionTime);
285  }
286  const float mass = o2::constants::physics::MassPionCharged; // default pid = pion
287  if (tofInt.getTOF(o2::track::PID::Pion) > 0.f) {
288  const float expBeta = (intLen / (tofInt.getTOF(o2::track::PID::Pion) * cSpeed));
289  extraInfoHolder.tofExpMom = mass * expBeta / std::sqrt(1.f - expBeta * expBeta);
290  }
291  }
292  if (src == GIndex::Source::TPCTRD || src == GIndex::Source::ITSTPCTRD) {
293  const auto& trdOrig = data.getTrack<o2::trd::TrackTRD>(src, contributorsGID[src].getIndex());
294  extraInfoHolder.trdChi2 = trdOrig.getChi2();
295  extraInfoHolder.trdPattern = getTRDPattern(trdOrig);
296  }
297  addToTracksTable(tracksCursor, tracksCovCursor, trackPar, collisionID, src);
298  addToTracksExtraTable(tracksExtraCursor, extraInfoHolder);
299  // collecting table indices of barrel tracks for V0s table
300  mGIDToTableID.emplace(trackIndex, mTableTrID);
301  mTableTrID++;
302  }
303  }
304  }
305  }
306 }

308 template <typename FwdTracksCursorType>
309 void AODProducerWorkflowDPL::addToFwdTracksTable(FwdTracksCursorType& fwdTracksCursor,
310  const o2::mch::TrackMCH& track, int collisionID,
311  int bcID,
312  const math_utils::Point3D<double>& vertex)
313 
314 {
315  // mch standalone tracks extrapolated to vertex
316 
317  // compute 3 sets of tracks parameters :
318  // - at vertex
319  // - at DCA
320  // - at the end of the absorber
321 
322  // extrapolate to vertex
323  o2::mch::TrackParam trackParamAtVertex(track.getZ(), track.getParameters());
324  double errVtx{0.0}; // FIXME: get errors associated with vertex if available
325  double errVty{0.0};
326  if (!o2::mch::TrackExtrap::extrapToVertex(trackParamAtVertex, vertex.x(), vertex.y(), vertex.z(), errVtx, errVty)) {
327  return;
328  }
329 
330  // extrapolate to DCA
331  o2::mch::TrackParam trackParamAtDCA(track.getZ(), track.getParameters());
332  if (!o2::mch::TrackExtrap::extrapToVertexWithoutBranson(trackParamAtDCA, vertex.z())) {
333  return;
334  }
335 
336  // extrapolate to the end of the absorber
337  o2::mch::TrackParam trackParamAtRAbs(track.getZ(), track.getParameters());
338  if (!o2::mch::TrackExtrap::extrapToZ(trackParamAtRAbs, -505.)) { // FIXME: replace hardcoded 505
339  return;
340  }
341 
342  double dcaX = trackParamAtDCA.getNonBendingCoor() - vertex.x();
343  double dcaY = trackParamAtDCA.getBendingCoor() - vertex.y();
344  double dca = std::sqrt(dcaX * dcaX + dcaY * dcaY);
345 
346  double xAbs = trackParamAtRAbs.getNonBendingCoor();
347  double yAbs = trackParamAtRAbs.getBendingCoor();
348 
349  double px = trackParamAtVertex.px();
350  double py = trackParamAtVertex.py();
351  double pz = trackParamAtVertex.pz();
352 
353  double pt = std::sqrt(px * px + py * py);
354  double dphi = std::asin(py / pt);
355  double dtanl = pz / pt;
356  double dinvqpt = 1.0 / (trackParamAtVertex.getCharge() * pt);
357  double dpdca = trackParamAtVertex.p() * dca;
358  double dchi2 = track.getChi2OverNDF();
359 
360  // table columns must be floats, not double
361  float x = trackParamAtVertex.getNonBendingCoor();
362  float y = trackParamAtVertex.getBendingCoor();
363  float z = trackParamAtVertex.getZ();
364  float rabs = std::sqrt(xAbs * xAbs + yAbs * yAbs);
365  float phi = dphi;
366  float tanl = dtanl;
367  float invqpt = dinvqpt;
368  float chi2 = dchi2;
369  float pdca = dpdca;
370 
371  fwdTracksCursor(0,
372  collisionID,
373  bcID,
374  o2::aod::fwdtrack::MCHStandaloneTrack,
375  x,
376  y,
377  z,
378  phi,
379  tanl,
380  invqpt,
381  track.getNClusters(),
382  chi2,
383  pdca,
384  rabs,
385  -1.0, // chi2matchmchmid,
386  -1.0, // chi2matchmchmft,
387  -1.0, // matchscoremchmft,
388  -1, // matchmfttrackid,
389  -1 // matchmchtrackid
390  );
391 }

393 template <typename MCParticlesCursorType>
394 void AODProducerWorkflowDPL::fillMCParticlesTable(o2::steer::MCKinematicsReader& mcReader,
395  const MCParticlesCursorType& mcParticlesCursor,
396  gsl::span<const o2::dataformats::VtxTrackRef>& primVer2TRefs,
397  gsl::span<const GIndex>& GIndices,
398  o2::globaltracking::RecoContainer& data,
399  std::vector<std::pair<int, int>> const& mccolid_to_eventandsource)
400 {
401  // mark reconstructed MC particles to store them into the table
402  for (auto& trackRef : primVer2TRefs) {
403  for (int src = GIndex::NSources; src--;) {
404  int start = trackRef.getFirstEntryOfSource(src);
405  int end = start + trackRef.getEntriesOfSource(src);
406  for (int ti = start; ti < end; ti++) {
407  auto& trackIndex = GIndices[ti];
408  if (GIndex::includesSource(src, mInputSources)) {
409  auto mcTruth = data.getTrackMCLabel(trackIndex);
410  if (!mcTruth.isValid()) {
411  continue;
412  }
413  int source = mcTruth.getSourceID();
414  int event = mcTruth.getEventID();
415  int particle = mcTruth.getTrackID();
416  mToStore[Triplet_t(source, event, particle)] = 1;
417  // treating contributors of global tracks
418  auto contributorsGID = data.getSingleDetectorRefs(trackIndex);
419  if (contributorsGID[GIndex::Source::ITS].isIndexSet() && contributorsGID[GIndex::Source::TPC].isIndexSet()) {
420  auto mcTruthITS = data.getTrackMCLabel(contributorsGID[GIndex::Source::ITS]);
421  if (mcTruthITS.isValid()) {
422  source = mcTruthITS.getSourceID();
423  event = mcTruthITS.getEventID();
424  particle = mcTruthITS.getTrackID();
425  mToStore[Triplet_t(source, event, particle)] = 1;
426  }
427  auto mcTruthTPC = data.getTrackMCLabel(contributorsGID[GIndex::Source::TPC]);
428  if (mcTruthTPC.isValid()) {
429  source = mcTruthTPC.getSourceID();
430  event = mcTruthTPC.getEventID();
431  particle = mcTruthTPC.getTrackID();
432  mToStore[Triplet_t(source, event, particle)] = 1;
433  }
434  }
435  }
436  }
437  }
438  }
439  int tableIndex = 1;
440  for (int mccolid = 0; mccolid < mccolid_to_eventandsource.size(); ++mccolid) {
441  auto event = mccolid_to_eventandsource[mccolid].first;
442  auto source = mccolid_to_eventandsource[mccolid].second;
443  std::vector<MCTrack> const& mcParticles = mcReader.getTracks(source, event);
444  // mark tracks to be stored per event
445  // loop over stack of MC particles from end to beginning: daughters are stored after mothers
446  if (mRecoOnly) {
447  for (int particle = mcParticles.size() - 1; particle >= 0; particle--) {
448  int mother0 = mcParticles[particle].getMotherTrackId();
449  if (mother0 == -1) {
450  mToStore[Triplet_t(source, event, particle)] = 1;
451  }
452  if (mToStore.find(Triplet_t(source, event, particle)) == mToStore.end()) {
453  continue;
454  }
455  if (mother0 != -1) {
456  mToStore[Triplet_t(source, event, mother0)] = 1;
457  }
458  int mother1 = mcParticles[particle].getSecondMotherTrackId();
459  if (mother1 != -1) {
460  mToStore[Triplet_t(source, particle, mother1)] = 1;
461  }
462  int daughter0 = mcParticles[particle].getFirstDaughterTrackId();
463  if (daughter0 != -1) {
464  mToStore[Triplet_t(source, event, daughter0)] = 1;
465  }
466  int daughterL = mcParticles[particle].getLastDaughterTrackId();
467  if (daughterL != -1) {
468  mToStore[Triplet_t(source, event, daughterL)] = 1;
469  }
470  }
471  // enumerate reconstructed mc particles and their relatives to get mother/daughter relations
472  for (int particle = 0; particle < mcParticles.size(); particle++) {
473  auto mapItem = mToStore.find(Triplet_t(source, event, particle));
474  if (mapItem != mToStore.end()) {
475  mapItem->second = tableIndex - 1;
476  tableIndex++;
477  }
478  }
479  }
480  // if all mc particles are stored, all mc particles will be enumerated
481  if (!mRecoOnly) {
482  for (int particle = 0; particle < mcParticles.size(); particle++) {
483  mToStore[Triplet_t(source, event, particle)] = tableIndex - 1;
484  tableIndex++;
485  }
486  }
487  // fill survived mc tracks into the table
488  for (int particle = 0; particle < mcParticles.size(); particle++) {
489  if (mToStore.find(Triplet_t(source, event, particle)) == mToStore.end()) {
490  continue;
491  }
492  int statusCode = 0;
493  uint8_t flags = 0;

547 void AODProducerWorkflowDPL::countTPCClusters(const o2::tpc::TrackTPC& track,
548  const gsl::span<const o2::tpc::TPCClRefElem>& tpcClusRefs,
549  const gsl::span<const unsigned char>& tpcClusShMap,
550  const o2::tpc::ClusterNativeAccess& tpcClusAcc,
551  uint8_t& shared, uint8_t& found, uint8_t& crossed)
552 {
553  constexpr int maxRows = 152;
554  constexpr int neighbour = 2;
555  std::array<bool, maxRows> clMap{}, shMap{};
556  uint8_t sectorIndex;
557  uint8_t rowIndex;
558  uint32_t clusterIndex;
559  shared = 0;
560  for (int i = 0; i < track.getNClusterReferences(); i++) {
561  o2::tpc::TrackTPC::getClusterReference(tpcClusRefs, i, sectorIndex, rowIndex, clusterIndex, track.getClusterRef());
562  unsigned int absoluteIndex = tpcClusAcc.clusterOffset[sectorIndex][rowIndex] + clusterIndex;
563  clMap[rowIndex] = true;
564  if (tpcClusShMap[absoluteIndex] > 1) {
565  if (!shMap[rowIndex]) {
566  shared++;
567  }
568  shMap[rowIndex] = true;
569  }
570  }
571 
572  crossed = 0;
573  found = 0;
574  int last = -1;
575  for (int i = 0; i < maxRows; i++) {
576  if (clMap[i]) {
577  crossed++;
578  found++;
579  last = i;
580  } else if ((i - last) <= neighbour) {
581  crossed++;
582  } else {
583  int lim = std::min(i + 1 + neighbour, maxRows);
584  for (int j = i + 1; j < lim; j++) {
585  if (clMap[j]) {
586  crossed++;
587  }
588  }
589  }
590  }
