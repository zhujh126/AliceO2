51  void init(o2::framework::InitContext& ic) final
52  {
53  // set up ADC value filling
54  // TODO: clean up to not go via RawReaderCRUManager
55  mCalibPedestal.init(); // initialize configuration via configKeyValues
56  mRawReader.createReader("");
57 
58  mRawReader.setADCDataCallback([this](const PadROCPos& padROCPos, const CRU& cru, const gsl::span<const uint32_t> data) -> int {
59  const int timeBins = mCalibPedestal.update(padROCPos, cru, data);
60  mCalibPedestal.setNumberOfProcessedTimeBins(std::max(mCalibPedestal.getNumberOfProcessedTimeBins(), size_t(timeBins)));
61  return timeBins;
62  });
63 
64  mRawReader.setLinkZSCallback([this](int cru, int rowInSector, int padInRow, int timeBin, float adcValue) -> bool {
65  CRU cruID(cru);
66  mCalibPedestal.updateROC(cruID.roc(), rowInSector - (rowInSector > 62) * 63, padInRow, timeBin, adcValue);
67  return true;
68  });
69 
70  mMaxEvents = static_cast<uint32_t>(ic.options().get<int>("max-events"));
71  mUseOldSubspec = ic.options().get<bool>("use-old-subspec");
72  mForceQuit = ic.options().get<bool>("force-quit");
73  mDirectFileDump = ic.options().get<bool>("direct-file-dump");
74  if (mUseOldSubspec) {
75  LOGP(info, "Using old subspecification (CruId << 16) | ((LinkId + 1) << (CruEndPoint == 1 ? 8 : 0))");
76  }
77  }

79  void run(o2::framework::ProcessingContext& pc) final
80  {
81  // in case the maximum number of events was reached don't do further processing
82  if (mReadyToQuit) {
83  return;
84  }
85 
86  auto& reader = mRawReader.getReaders()[0];
87  calib_processing_helper::processRawData(pc.inputs(), reader, mUseOldSubspec, mSectors);
88 
89  mCalibPedestal.incrementNEvents();
90  const auto nTFs = mCalibPedestal.getNumberOfProcessedEvents();
91  LOGP(info, "Number of processed TFs: {} ({})", nTFs, mMaxEvents);
92 
93  if ((mPublishAfter && (nTFs % mPublishAfter) == 0)) {
94  LOGP(info, "Publishing after {} TFs", nTFs);
95  mCalibPedestal.analyse();
96  dumpCalibData();
97  sendOutput(pc.outputs());
98  }
99 
100  if (mMaxEvents && (nTFs >= mMaxEvents) && !mCalibDumped) {
101  LOGP(info, "Maximm number of TFs reached ({}), no more processing will be done", mMaxEvents);
102  mReadyToQuit = true;
103  mCalibPedestal.analyse();
104  dumpCalibData();
105  if (mForceQuit) {
106  pc.services().get<ControlService>().endOfStream();
107  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
108  } else {
109  //pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
110  }
111  }
112  }
113 
114  void endOfStream(o2::framework::EndOfStreamContext& ec) final
115  {
116  LOGP(info, "endOfStream");
117  mCalibPedestal.analyse();
118  dumpCalibData();
119  sendOutput(ec.outputs());
120  ec.services().get<ControlService>().readyToQuit(QuitRequest::Me);
121  }

136  //____________________________________________________________________________
137  void sendOutput(DataAllocator& output)
138  {
139 
140  std::array<const CalDet<float>*, 2> data = {&mCalibPedestal.getPedestal(), &mCalibPedestal.getNoise()};
141  std::array<CDBType, 2> dataType = {CDBType::CalPedestal, CDBType::CalNoise};
142 
143  for (size_t i = 0; i < data.size(); ++i) {
144  auto cal = data[i];
145  auto image = o2::utils::MemFileHelper::createFileImage(cal, typeid(*cal), cal->getName(), "data");
146  int type = int(dataType[i]);
147 
148  header::DataHeader::SubSpecificationType subSpec{(header::DataHeader::SubSpecificationType)((mLane << 4) + i)};
149  output.snapshot(Output{gDataOriginTPC, "CLBPART", subSpec}, *image.get());
150  output.snapshot(Output{gDataOriginTPC, "CLBPARTINFO", subSpec}, type);
151  }
152  }
153 
154  //____________________________________________________________________________
155  void dumpCalibData()
156  {
157  if (mDirectFileDump && !mCalibDumped) {
158  LOGP(info, "Dumping output");
159  mCalibPedestal.dumpToFile(fmt::format("pedestals_{:02}.root", mLane));
160  mCalibDumped = true;
161  }
162  }

165 DataProcessorSpec getTPCCalibPedestalSpec(const std::string inputSpec, uint32_t ilane = 0, std::vector<int> sectors = {}, uint32_t publishAfterTFs = 0)
166 {
167  std::vector<o2::framework::OutputSpec> outputs;
168  outputs.emplace_back(ConcreteDataTypeMatcher{gDataOriginTPC, "CLBPART"});
169  outputs.emplace_back(ConcreteDataTypeMatcher{gDataOriginTPC, "CLBPARTINFO"});
170 
171  const auto id = fmt::format("calib-tpc-pedestal-{:02}", ilane);
172  return DataProcessorSpec{
173  id.data(),
174  select(inputSpec.data()),
175  outputs,
176  AlgorithmSpec{adaptFromTask<TPCCalibPedestalDevice>(ilane, sectors, publishAfterTFs)},
177  Options{
178  {"max-events", VariantType::Int, 0, {"maximum number of events to process"}},
179  {"use-old-subspec", VariantType::Bool, false, {"use old subsecifiation definition"}},
180  {"force-quit", VariantType::Bool, false, {"force quit after max-events have been reached"}},
181  {"direct-file-dump", VariantType::Bool, false, {"directly dump calibration to file"}},
182  } // end Options
183  }; // end DataProcessorSpec
