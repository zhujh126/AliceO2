251 void DataRequest::requestTracks(GTrackID::mask_t src, bool useMC)
252 {
253  // request tracks for sources probided by the mask
254  if (src[GTrackID::ITS]) {
255  requestITSTracks(useMC);
256  }
257  if (src[GTrackID::MFT]) {
258  requestMFTTracks(useMC);
259  }
260  if (src[GTrackID::MCH]) {
261  requestMCHTracks(useMC);
262  }
263  if (src[GTrackID::TPC]) {
264  requestTPCTracks(useMC);
265  }
266  if (src[GTrackID::ITSTPC] || src[GTrackID::ITSTPCTOF]) {
267  requestITSTPCTracks(useMC);
268  }
269  if (src[GTrackID::TPCTOF]) {
270  requestTPCTOFTracks(useMC);
271  }
272  if (src[GTrackID::ITSTPCTOF]) {
273  requestTOFMatches(useMC);
274  requestTOFClusters(false); // RSTODO Needed just to set the time of ITSTPC track, consider moving to MatchInfoTOF
275  // NOTE: Getting TOF Clusters is carried over to InputHelper::addInputSpecs. If changed here, please fix there.
276  }
277  if (src[GTrackID::ITSTPCTRD]) {
278  requestITSTPCTRDTracks(useMC);
279  }
280  if (src[GTrackID::TPCTRD]) {
281  requestTPCTRDTracks(useMC);
282  }
283 }

285 void DataRequest::requestClusters(GTrackID::mask_t src, bool useMC)
286 {
287  // request clusters for detectors of the sources probided by the mask
288 
289  // clusters needed for refits
290  if (GTrackID::includesDet(DetID::ITS, src)) {
291  requestITSClusters(useMC);
292  }
293  if (GTrackID::includesDet(DetID::TPC, src)) {
294  requestTPCClusters(useMC);
295  }
296  if (GTrackID::includesDet(DetID::TOF, src)) {
297  requestTOFClusters(useMC);
298  }
299  if (GTrackID::includesDet(DetID::TRD, src)) {
300  requestTRDTracklets(useMC);
301  }
302 }

305 void RecoContainer::collectData(ProcessingContext& pc, const DataRequest& requests)
306 {
307  auto& reqMap = requests.requestMap;
308 
309  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().getFirstValid(true).header);
310  startIR = {0, dh->firstTForbit};
311 
312  auto req = reqMap.find("trackITS");
313  if (req != reqMap.end()) {
314  addITSTracks(pc, req->second);
315  }
316 
317  req = reqMap.find("trackMFT");
318  if (req != reqMap.end()) {
319  addMFTTracks(pc, req->second);
320  }
321 
322  req = reqMap.find("trackMCH");
323  if (req != reqMap.end()) {
324  addMCHTracks(pc, req->second);
325  }
326 
327  req = reqMap.find("trackTPC");
328  if (req != reqMap.end()) {
329  addTPCTracks(pc, req->second);
330  }
331 
332  req = reqMap.find("trackITSTPC");
333  if (req != reqMap.end()) {
334  addITSTPCTracks(pc, req->second);
335  }
336 
337  req = reqMap.find("trackITSTPCTRD");
338  if (req != reqMap.end()) {
339  addITSTPCTRDTracks(pc, req->second);
340  }
341 
342  req = reqMap.find("trackTPCTRD");
343  if (req != reqMap.end()) {
344  addTPCTRDTracks(pc, req->second);
345  }
346 
347  req = reqMap.find("trackTPCTOF");
348  if (req != reqMap.end()) {
349  addTPCTOFTracks(pc, req->second);
350  }
351 
352  req = reqMap.find("matchTOF");
353  if (req != reqMap.end()) {
354  addTOFMatches(pc, req->second);
355  }
356 
357  req = reqMap.find("clusITS");
358  if (req != reqMap.end()) {
359  addITSClusters(pc, req->second);
360  }
361 
362  req = reqMap.find("clusTPC");
363  if (req != reqMap.end()) {
364  addTPCClusters(pc, req->second, reqMap.find("trackTPC") != reqMap.end());
365  }
366 
367  req = reqMap.find("clusTOF");
368  if (req != reqMap.end()) {
369  addTOFClusters(pc, req->second);
370  }
371 
372  req = reqMap.find("FT0");
373  if (req != reqMap.end()) {
374  addFT0RecPoints(pc, req->second);
375  }
376 
377  req = reqMap.find("trackletTRD");
378  if (req != reqMap.end()) {
379  addTRDTracklets(pc, req->second);
380  }
381 
382  req = reqMap.find("Cosmics");
383  if (req != reqMap.end()) {
384  addCosmicTracks(pc, req->second);
385  }
386 
387  req = reqMap.find("PVertex");
388  if (req != reqMap.end()) {
389  addPVertices(pc, req->second);
390  }
391 
392  req = reqMap.find("PVertexTMP");
393  if (req != reqMap.end()) {
394  addPVerticesTMP(pc, req->second);
395  }
396 
397  req = reqMap.find("SVertex");
398  if (req != reqMap.end()) {
399  addSVertices(pc, req->second);
400  }
401 
402  req = reqMap.find("IRFramesITS");
403  if (req != reqMap.end()) {
404  addIRFramesITS(pc);
405  }
406 }

707 // get contributors from single detectors
708 RecoContainer::GlobalIDSet RecoContainer::getSingleDetectorRefs(GTrackID gidx) const
709 {
710  GlobalIDSet table;
711  auto src = gidx.getSource();
712  table[src] = gidx;
713  if (src == GTrackID::ITSTPCTRD) {
714  const auto& parent0 = getITSTPCTRDTrack<o2::trd::TrackTRD>(gidx);
715  const auto& parent1 = getTPCITSTrack(parent0.getRefGlobalTrackId());
716  table[GTrackID::ITSTPC] = parent0.getRefGlobalTrackId();
717  table[parent1.getRefITS().getSource()] = parent1.getRefITS();
718  table[GTrackID::TPC] = parent1.getRefTPC();
719  table[GTrackID::TRD] = gidx; // there is no standalone TRD track, so use the index for the ITSTPCTRD track array
720  }
721  if (src == GTrackID::ITSTPCTOF) {
722  const auto& parent0 = getTOFMatch(gidx); //ITS/TPC : TOF
723  const auto& parent1 = getTPCITSTrack(parent0.getTrackRef());
724  table[GTrackID::ITSTPC] = parent0.getTrackRef();
725  table[GTrackID::TOF] = {unsigned(parent0.getIdxTOFCl()), GTrackID::TOF};
726  table[GTrackID::TPC] = parent1.getRefTPC();
727  table[parent1.getRefITS().getSource()] = parent1.getRefITS(); // ITS source might be an ITS track or ITSAB tracklet
728  } else if (src == GTrackID::TPCTOF) {
729  const auto& parent0 = getTPCTOFMatch(gidx); //TPC : TOF
730  table[GTrackID::TOF] = {unsigned(parent0.getIdxTOFCl()), GTrackID::TOF};
731  table[GTrackID::TPC] = parent0.getTrackRef();
732  } else if (src == GTrackID::ITSTPC) {
733  const auto& parent0 = getTPCITSTrack(gidx);
734  table[GTrackID::TPC] = parent0.getRefTPC();
735  table[parent0.getRefITS().getSource()] = parent0.getRefITS(); // ITS source might be an ITS track or ITSAB tracklet
736  }
737  return std::move(table);
738 }
