41 bool TrackFitter::fit(TrackLTF& track, bool outward)
42 {
45 
46  auto nClusters = track.getNumberOfPoints();
47  auto lastLayer = track.getLayers()[(outward ? 0 : nClusters - 1)];
48 
49  if (mVerbose) {
50  std::cout << "Seed covariances: \n"
51  << track.getCovariances() << std::endl
52  << std::endl;
53  }
54 
55  // recursively compute clusters, updating the track parameters
56  if (!outward) { // Inward for vertexing
57  while (nClusters-- > 0) {
58  if (!computeCluster(track, nClusters, lastLayer)) {
59  return false;
60  }
61  }
62  } else { // Outward for MCH matching
63  int ncl = 0;
64  while (ncl < nClusters) {
65  if (!computeCluster(track, ncl, lastLayer)) {
66  return false;
67  }
68  ncl++;
69  }
70  }
71  if (mVerbose) {
72  std::cout << "Track Chi2 = " << track.getTrackChi2() << std::endl;
73  std::cout << " ***************************** Done fitting *****************************\n";
74  }
75 
76  return true;
77 }

80 bool TrackFitter::initTrack(TrackLTF& track, bool outward)
81 {
82 
83  // initialize the starting track parameters and cluster
84  double sigmainvQPtsq;
85  double chi2invqptquad;
86  auto invQPt0 = invQPtFromFCF(track, mBZField, sigmainvQPtsq);
87  auto nPoints = track.getNumberOfPoints();
88  auto k = TMath::Abs(o2::constants::math::B2C * mBZField);
89  auto Hz = std::copysign(1, mBZField);
90 
91  if (mVerbose) {
92  std::cout << "\n ***************************** Start Fitting new track ***************************** \n";
93  std::cout << "N Clusters = " << nPoints << std::endl;
94  }
95 
96  track.setInvQPtSeed(invQPt0);
97  track.setChi2QPtSeed(chi2invqptquad);
98  track.setInvQPt(invQPt0);
99 
101  int first_cls, last_cls;
102  if (outward) { // MCH matching
103  first_cls = 1;
104  last_cls = 0;
105  } else { // Vertexing
106  first_cls = nPoints - 1;
107  last_cls = nPoints - 2;
108  }
109 
110  auto x0 = track.getXCoordinates()[first_cls];
111  auto y0 = track.getYCoordinates()[first_cls];
112  auto z0 = track.getZCoordinates()[first_cls];
113 
114  //Compute tanl using first two clusters
115  auto deltaX = track.getXCoordinates()[1] - track.getXCoordinates()[0];
116  auto deltaY = track.getYCoordinates()[1] - track.getYCoordinates()[0];
117  auto deltaZ = track.getZCoordinates()[1] - track.getZCoordinates()[0];
118  auto deltaR = TMath::Sqrt(deltaX * deltaX + deltaY * deltaY);
119  auto tanl0 = 0.5 * TMath::Sqrt2() * (deltaZ / deltaR) *
120  TMath::Sqrt(TMath::Sqrt((invQPt0 * deltaR * k) * (invQPt0 * deltaR * k) + 1) + 1);
121 
122  // Compute phi at the last cluster using two last clusters
123  deltaX = track.getXCoordinates()[first_cls] - track.getXCoordinates()[last_cls];
124  deltaY = track.getYCoordinates()[first_cls] - track.getYCoordinates()[last_cls];
125  deltaZ = track.getZCoordinates()[first_cls] - track.getZCoordinates()[last_cls];
126  deltaR = TMath::Sqrt(deltaX * deltaX + deltaY * deltaY);
127  auto phi0 = TMath::ATan2(deltaY, deltaX) - 0.5 * Hz * invQPt0 * deltaZ * k / tanl0;
128  auto sigmax0sq = track.getSigmasX2()[first_cls];
129  auto sigmay0sq = track.getSigmasY2()[first_cls];
130  auto sigmax1sq = track.getSigmasX2()[last_cls];
131  auto sigmay1sq = track.getSigmasY2()[last_cls];
132  auto sigmaDeltaXsq = sigmax0sq + sigmax1sq;
133  auto sigmaDeltaYsq = sigmay0sq + sigmay1sq;
134 
135  track.setX(x0);
136  track.setY(y0);
137  track.setZ(z0);
138  track.setPhi(phi0);
139  track.setTanl(tanl0);
140 
141  if (mVerbose) {
142  std::cout << " Init " << (track.isCA() ? "CA Track " : "LTF Track") << std::endl;
143  auto model = (mTrackModel == Helix) ? "Helix" : (mTrackModel == Quadratic) ? "Quadratic"
144  : (mTrackModel == Optimized) ? "Optimized"
145  : "Linear";
146  std::cout << "Track Model: " << model << std::endl;
147  std::cout << " initTrack: X = " << x0 << " Y = " << y0 << " Z = " << z0 << " Tgl = " << tanl0 << " Phi = " << phi0 << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
148  }
149 
150  SMatrix55Sym lastParamCov;
151  float qptsigma = TMath::Max(std::abs(track.getInvQPt()), .5);
152  float tanlsigma = TMath::Max(std::abs(track.getTanl()), .5);
153 
154  lastParamCov(0, 0) = 1; // <X,X>
155  lastParamCov(1, 1) = 1; // <Y,X>
156  lastParamCov(2, 2) = TMath::Pi() * TMath::Pi() / 16; // <PHI,X>
157  lastParamCov(3, 3) = 10 * tanlsigma * tanlsigma; // <TANL,X>
158  lastParamCov(4, 4) = 10 * qptsigma * qptsigma; // <INVQPT,X>
159 
160  track.setCovariances(lastParamCov);
161  track.setTrackChi2(0.);
162 
163  return true;
164 }

167 bool TrackFitter::propagateToZ(TrackLTF& track, double z)
168 {
169  // Propagate track to the z position of the new cluster
170  switch (mTrackModel) {
171  case Linear:
172  track.propagateToZlinear(z);
173  break;
174  case Quadratic:
175  track.propagateToZquadratic(z, mBZField);
176  break;
177  case Helix:
178  track.propagateToZhelix(z, mBZField);
179  break;
180  case Optimized:
181  track.propagateToZ(z, mBZField);
182  break;
183  default:
184  std::cout << " Invalid track model.\n";
185  return false;
186  break;
187  }
188  return true;
189 }

192 bool TrackFitter::propagateToNextClusterWithMCS(TrackLTF& track, double z, int& startingLayerID, const int& newLayerID)
193 {
194 
195  // Propagate track to the next cluster z position, adding angular MCS effects at the center of
196  // each disk crossed by the track. This method is valid only for track propagation between
197  // clusters at MFT layers positions.
198 
199  if (startingLayerID == newLayerID) { // Same layer, nothing to do.
200  if (mVerbose) {
201  std::cout << " => Propagate to next cluster with MCS : startingLayerID = " << startingLayerID << " = > "
202  << " newLayerID = " << newLayerID << " (NLayers = " << std::abs(newLayerID - startingLayerID);
203  std::cout << ") ; track.getZ() = " << track.getZ() << " => ";
204  std::cout << "destination cluster z = " << z << " ; => Same layer: no MCS effects." << std::endl;
205  }
206  if (z != track.getZ()) {
207  propagateToZ(track, z);
208  }
209  return true;
210  }
211 
212  using o2::mft::constants::LayerZPosition;
213  auto startingZ = track.getZ();
214 
215  //https://stackoverflow.com/questions/1903954/is-there-a-standard-sign-function-signum-sgn-in-c-c
216  auto signum = [](auto a) {
217  return (0 < a) - (a < 0);
218  };
219  int direction = signum(newLayerID - startingLayerID); // takes values +1, 0, -1
220  auto currentLayer = startingLayerID;
221 
222  if (mVerbose) {
223  std::cout << " => Propagate to next cluster with MCS : startingLayerID = " << startingLayerID << " = > "
224  << " newLayerID = " << newLayerID << " (NLayers = " << std::abs(newLayerID - startingLayerID);
225  std::cout << ") ; track.getZ() = " << track.getZ() << " => ";
226  std::cout << "destination cluster z = " << z << " ; " << std::endl;
227  }
228 
229  // Number of disks crossed by this track segment
230  while (currentLayer != newLayerID) {
231  auto nextlayer = currentLayer + direction;
232  auto nextZ = LayerZPosition[nextlayer];
233 
234  int NDisksMS;
235  if (nextZ - track.getZ() > 0) {
236  NDisksMS = (currentLayer % 2 == 0) ? (currentLayer - nextlayer) / 2 : (currentLayer - nextlayer + 1) / 2;
237  } else {
238  NDisksMS = (currentLayer % 2 == 0) ? (nextlayer - currentLayer + 1) / 2 : (nextlayer - currentLayer) / 2;
239  }
240 
241  if (mVerbose) {
242  std::cout << "currentLayer = " << currentLayer << " ; "
243  << "nextlayer = " << nextlayer << " ; ";
244  std::cout << "track.getZ() = " << track.getZ() << " ; ";
245  std::cout << "nextZ = " << nextZ << " ; ";
246  std::cout << "NDisksMS = " << NDisksMS << std::endl;
247  }
248 
249  if ((NDisksMS * mMFTDiskThicknessInX0) != 0) {
250  track.addMCSEffect(NDisksMS * mMFTDiskThicknessInX0);
251  if (mVerbose) {
252  std::cout << "Track covariances after MCS effects: \n"
253  << track.getCovariances() << std::endl
254  << std::endl;
255  }
256  }
257 
258  if (mVerbose) {
259  std::cout << " BeforeExtrap: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
260  }
261 
262  propagateToZ(track, nextZ);
263 
264  currentLayer = nextlayer;
265  }
266  if (z != track.getZ()) {
267  propagateToZ(track, z);
268  }
269  startingLayerID = newLayerID;
270  return true;
271 }

285  const auto& sigmaY2 = track.getSigmasY2()[cluster];
286  const auto& newLayerID = track.getLayers()[cluster];
287 
288  if (mVerbose) {
289  std::cout << "computeCluster: X = " << clx << " Y = " << cly << " Z = " << clz << " nCluster = " << cluster << " Layer = " << newLayerID << std::endl;
290  }
291 
292  if (!propagateToNextClusterWithMCS(track, clz, startingLayerID, newLayerID)) {
293  return false;
294  }
295 
296  if (mVerbose) {
297  std::cout << " AfterExtrap: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
298  std::cout << "Track covariances after extrap: \n"
299  << track.getCovariances() << std::endl

312  std::cout << "Track covariances after Kalman update: \n"
313  << track.getCovariances() << std::endl
314  << std::endl;
315  }
316  return true;
317  }
318  return false;

322 Double_t invQPtFromFCF(const TrackLTF& track, Double_t bFieldZ, Double_t& sigmainvqptsq)
323 {
324 
325  const std::array<Float_t, constants::mft::LayersNumber>& xPositions = track.getXCoordinates();
326  const std::array<Float_t, constants::mft::LayersNumber>& yPositions = track.getYCoordinates();
327  const std::array<Float_t, constants::mft::LayersNumber>& zPositions = track.getZCoordinates();
328  const std::array<Float_t, constants::mft::LayersNumber>& SigmasX2 = track.getSigmasX2();
329  const std::array<Float_t, constants::mft::LayersNumber>& SigmasY2 = track.getSigmasY2();
330 
331  // Fast Circle Fit (Hansroul, Jeremie, Savard, 1987)
332  auto nPoints = track.getNumberOfPoints();
333  std::vector<double> xVal(nPoints);
334  std::vector<double> yVal(nPoints);
335  std::vector<double> zVal(nPoints);
336  std::vector<double> xErr(nPoints);
337  std::vector<double> yErr(nPoints);
338  std::vector<double> uVal(nPoints - 1);
339  std::vector<double> vVal(nPoints - 1);
340  std::vector<double> vErr(nPoints - 1);
341  std::vector<double> fweight(nPoints - 1);
342  std::vector<double> Rn(nPoints - 1);
343  std::vector<double> Pn(nPoints - 1);
344  Double_t A, Aerr, B, Berr, x2, y2, invx2y2, a, b, r, sigmaRsq, u2, sigma;
345  Double_t F0, F1, F2, F3, F4, SumSRn, SumSPn, SumRn, SumUPn, SumRP;
346 
347  SumSRn = SumSPn = SumRn = SumUPn = SumRP = 0.0;
348  F0 = F1 = F2 = F3 = F4 = 0.0;
349 
350  for (auto np = 0; np < nPoints; np++) {
351  xErr[np] = SigmasX2[np];
352  yErr[np] = SigmasY2[np];
353  if (np > 0) {
354  xVal[np] = xPositions[np] - xVal[0];
355  yVal[np] = yPositions[np] - yVal[0];
356  xErr[np] *= std::sqrt(2.);
357  yErr[np] *= std::sqrt(2.);
358  } else {
359  xVal[np] = 0.;
360  yVal[np] = 0.;
361  }
362  zVal[np] = zPositions[np];
363  }
364  for (int i = 0; i < (nPoints - 1); i++) {
365  x2 = xVal[i + 1] * xVal[i + 1];
366  y2 = yVal[i + 1] * yVal[i + 1];
367  invx2y2 = 1. / (x2 + y2);
368  uVal[i] = xVal[i + 1] * invx2y2;
369  vVal[i] = yVal[i + 1] * invx2y2;
370  vErr[i] = std::sqrt(8. * xErr[i + 1] * xErr[i + 1] * x2 * y2 + 2. * yErr[i + 1] * yErr[i + 1] * (x2 - y2) * (x2 - y2)) * invx2y2 * invx2y2;
371  u2 = uVal[i] * uVal[i];
372  fweight[i] = 1. / vErr[i];
373  F0 += fweight[i];
374  F1 += fweight[i] * uVal[i];
375  F2 += fweight[i] * u2;
376  F3 += fweight[i] * uVal[i] * u2;
377  F4 += fweight[i] * u2 * u2;
378  }
379 
380  double Rn_det1 = F2 * F4 - F3 * F3;
381  double Rn_det2 = F1 * F4 - F2 * F3;
382  double Rn_det3 = F1 * F3 - F2 * F2;
383  double Pn_det1 = Rn_det2;
384  double Pn_det2 = F0 * F4 - F2 * F2;
385  double Pn_det3 = F0 * F3 - F1 * F2;
386 
387  for (int j = 0; j < (nPoints - 1); j++) {
388  Rn[j] = fweight[j] * (Rn_det1 - uVal[j] * Rn_det2 + uVal[j] * uVal[j] * Rn_det3);
389  SumSRn += Rn[j] * Rn[j] * vErr[j] * vErr[j];
390  SumRn += Rn[j];
391 
392  Pn[j] = fweight[j] * (-Pn_det1 + uVal[j] * Pn_det2 - uVal[j] * uVal[j] * Pn_det3);
393  SumSPn += Pn[j] * Pn[j] * vErr[j] * vErr[j];
394  SumUPn += uVal[j] * Pn[j];
395 
396  SumRP += Rn[j] * Pn[j] * vErr[j] * vErr[j] * vErr[j];
397  }
398 
399  Double_t invqpt_fcf;
400  Int_t qfcf;
401  // chi2 = 0.;
402  if (LinearRegression((nPoints - 1), uVal, vVal, vErr, B, Berr, A, Aerr)) {
403  // v = a * u + b
404  // circle passing through (0,0):
405  // (x - rx)^2 + (y - ry)^2 = r^2
406  // ---> a = - rx / ry;
407  // ---> b = 1 / (2 * ry)
408  b = 1. / (2. * A);
409  a = -B * b;
410  r = std::sqrt(a * a + b * b);
411  double_t invR = 1. / r;
412 
413  // pt --->
414  Double_t invpt = 1. / (o2::constants::math::B2C * bFieldZ * r);
415 
416  // sign(q) --->
417  // rotate around the first point (0,0) to bring the last point
418  // on the x axis (y = 0) and check the y sign of the rotated
419  // center of the circle
420  Double_t x = xVal[nPoints - 1], y = yVal[nPoints - 1], z = zVal[nPoints - 1];
421  Double_t slope = TMath::ATan2(y, x);
422  Double_t cosSlope = TMath::Cos(slope);
423  Double_t sinSlope = TMath::Sin(slope);
424  Double_t rxRot = a * cosSlope + b * sinSlope;
425  Double_t ryRot = a * sinSlope - b * cosSlope;
426  qfcf = (ryRot > 0.) ? -1 : +1;
427 
428  Double_t alpha = 2.0 * std::abs(TMath::ATan2(rxRot, ryRot));
429  Double_t x0 = xVal[0], y0 = yVal[0], z0 = zVal[0];
430  Double_t dxyz2 = (x - x0) * (x - x0) + (y - y0) * (y - y0) + (z - z0) * (z - z0);
431  Double_t cst = 1000.;
432  Double_t c_alpha = cst * alpha;
433  Double_t p, pt, pz;
434  pt = 1. / invpt;
435  p = std::sqrt(dxyz2) / c_alpha;
436  pz = std::sqrt(p * p - pt * pt);
437 
438  invqpt_fcf = qfcf * invpt;
439 
440  //error calculations:
441  double invA2 = 1. / (A * A);
442 
443  double sigmaAsq = SumSRn / (SumRn * SumRn);
444  double sigmaBsq = SumSPn / (SumUPn * SumUPn);
445  double sigmaAB = SumRP / (SumRn * SumUPn);
446 
447  double sigmaasq_FCF = TMath::Abs(0.25 * invA2 * invA2 * (B * B * sigmaAsq + A * A * sigmaBsq - A * B * sigmaAB));
448  double sigmabsq_FCF = TMath::Abs(0.25 * invA2 * invA2 * sigmaAsq);
449  double sigma2R = invR * invR * (b * b * sigmaasq_FCF + a * a * sigmabsq_FCF + 2 * a * b * TMath::Sqrt(sigmaasq_FCF) * TMath::Sqrt(sigmabsq_FCF));
450 
451  sigmainvqptsq = sigma2R * invpt * invpt * invR * invR;
452 
453  } else { // the linear regression failed...
454  LOG(WARN) << "LinearRegression failed!";
455  invqpt_fcf = 1. / 100.;
456  }
457 
458  return invqpt_fcf;
459 }

462 Bool_t LinearRegression(Int_t nVal, std::vector<double>& xVal, std::vector<double>& yVal, std::vector<double>& yErr, Double_t& B, Double_t& Berr, Double_t& A, Double_t& Aerr)
463 {
464  // linear regression y = B * x + A
465 
466  Double_t S1, SXY, SX, SY, SXX, SsXY, SsXX, SsYY, Xm, Ym, s, delta, difx;
467  Double_t invYErr2;
468 
469  S1 = SXY = SX = SY = SXX = 0.0;
470  SsXX = SsYY = SsXY = Xm = Ym = 0.;
471  difx = 0.;
472  for (Int_t i = 0; i < nVal; i++) {
473  invYErr2 = 1. / (yErr[i] * yErr[i]);
474  S1 += invYErr2;
475  SXY += xVal[i] * yVal[i] * invYErr2;
476  SX += xVal[i] * invYErr2;
477  SY += yVal[i] * invYErr2;
478  SXX += xVal[i] * xVal[i] * invYErr2;
479  if (i > 0) {
480  difx += TMath::Abs(xVal[i] - xVal[i - 1]);
481  }
482  Xm += xVal[i];
483  Ym += yVal[i];
484  SsXX += xVal[i] * xVal[i];
485  SsYY += yVal[i] * yVal[i];
486  SsXY += xVal[i] * yVal[i];
487  }
488  delta = SXX * S1 - SX * SX;
489  if (delta == 0.) {
490  return kFALSE;
491  }
492  B = (SXY * S1 - SX * SY) / delta;
493  A = (SY * SXX - SX * SXY) / delta;
494 
495  Ym /= (Double_t)nVal;
496  Xm /= (Double_t)nVal;
497  SsYY -= (Double_t)nVal * (Ym * Ym);
498  SsXX -= (Double_t)nVal * (Xm * Xm);
499  SsXY -= (Double_t)nVal * (Ym * Xm);
500  Double_t eps = 1.E-24;
501  if ((nVal > 2) && (TMath::Abs(difx) > eps) && ((SsYY - (SsXY * SsXY) / SsXX) > 0.)) {
502  s = TMath::Sqrt((SsYY - (SsXY * SsXY) / SsXX) / (nVal - 2));
503  Aerr = s * TMath::Sqrt(1. / (Double_t)nVal + (Xm * Xm) / SsXX);
504  Berr = s / TMath::Sqrt(SsXX);
505  } else {
506  Aerr = 0.;
507  Berr = 0.;
508  }
509  return kTRUE;
510 }
