42 void CTFCoder::compress(CompressedDigits& cd, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
43 {
44  // convert digits/channel to their compressed version
45  cd.clear();
46  if (!digitVec.size()) {
47  return;
48  }
49  const auto& dig0 = digitVec[0];
50  cd.header.nTriggers = digitVec.size();
51  cd.header.firstOrbit = dig0.getOrbit();
52  cd.header.firstBC = dig0.getBC();
53  cd.header.triggerGate = DigitizationParameters::Instance().mTime_trg_gate;
54 
55  cd.trigger.resize(cd.header.nTriggers);
56  cd.bcInc.resize(cd.header.nTriggers);
57  cd.orbitInc.resize(cd.header.nTriggers);
58  cd.eventFlags.resize(cd.header.nTriggers);
59  cd.nChan.resize(cd.header.nTriggers);
60 
61  cd.idChan.resize(channelVec.size());
62  cd.qtcChain.resize(channelVec.size());
63  cd.cfdTime.resize(channelVec.size());
64  cd.qtcAmpl.resize(channelVec.size());
65 
66  uint16_t prevBC = cd.header.firstBC;
67  uint32_t prevOrbit = cd.header.firstOrbit;
68  uint32_t ccount = 0;
69  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
70  const auto& digit = digitVec[idig];
71  const auto chanels = digit.getBunchChannelData(channelVec); // we assume the channels are sorted
72 
73  // fill trigger info
74  cd.trigger[idig] = digit.getTriggers().triggersignals;
75  cd.eventFlags[idig] = digit.getEventStatusWord();
76  if (prevOrbit == digit.getOrbit()) {
77  cd.bcInc[idig] = digit.getBC() - prevBC;
78  cd.orbitInc[idig] = 0;
79  } else {
80  cd.bcInc[idig] = digit.getBC();
81  cd.orbitInc[idig] = digit.getOrbit() - prevOrbit;
82  }
83  prevBC = digit.getBC();
84  prevOrbit = digit.getOrbit();
85  // fill channels info
86  cd.nChan[idig] = chanels.size();
87  if (!cd.nChan[idig]) {
88  LOG(ERROR) << "Digits with no channels";
89  continue;
90  }
91  uint8_t prevChan = 0;
92  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
93  assert(prevChan <= chanels[ic].ChId);
94  cd.idChan[ccount] = chanels[ic].ChId - prevChan;
95  cd.qtcChain[ccount] = chanels[ic].ChainQTC;
96  cd.cfdTime[ccount] = chanels[ic].CFDTime;
97  cd.qtcAmpl[ccount] = chanels[ic].QTCAmpl;
98  prevChan = chanels[ic].ChId;
99  ccount++;
100  }
101  }
102 }

105 void CTFCoder::createCoders(const std::string& dictPath, o2::ctf::CTFCoderBase::OpType op)
106 {
107  bool mayFail = true; // RS FIXME if the dictionary file is not there, do not produce exception
108  auto buff = readDictionaryFromFile<CTF>(dictPath, mayFail);
109  if (!buff.size()) {
110  if (mayFail) {
111  return;
112  }
113  throw std::runtime_error("Failed to create CTF dictionaty");
114  }
115  const auto* ctf = CTF::get(buff.data());
116 
117  auto getFreq = [ctf](CTF::Slots slot) -> o2::rans::FrequencyTable {
118  o2::rans::FrequencyTable ft;
119  auto bl = ctf->getBlock(slot);
120  auto md = ctf->getMetadata(slot);
121  ft.addFrequencies(bl.getDict(), bl.getDict() + bl.getNDict(), md.min, md.max);
122  return std::move(ft);
123  };
124  auto getProbBits = [ctf](CTF::Slots slot) -> int {
125  return ctf->getMetadata(slot).probabilityBits;
126  };
127 
128  CompressedDigits cd; // just to get member types
129 #define MAKECODER(part, slot) createCoder<decltype(part)::value_type>(op, getFreq(slot), getProbBits(slot), int(slot))
130  // clang-format off
131  MAKECODER(cd.trigger, CTF::BLC_trigger);
132  MAKECODER(cd.bcInc, CTF::BLC_bcInc);
133  MAKECODER(cd.orbitInc, CTF::BLC_orbitInc);
134  MAKECODER(cd.nChan, CTF::BLC_nChan);
135  MAKECODER(cd.eventFlags, CTF::BLC_flags);
136  MAKECODER(cd.idChan, CTF::BLC_idChan);
137  MAKECODER(cd.qtcChain, CTF::BLC_qtcChain);
138  MAKECODER(cd.cfdTime, CTF::BLC_cfdTime);
139  MAKECODER(cd.qtcAmpl, CTF::BLC_qtcAmpl);
140  // clang-format on
141 }

144 size_t CTFCoder::estimateCompressedSize(const CompressedDigits& cd)
145 {
146  size_t sz = 0;
147  // clang-format off
148  // RS FIXME this is very crude estimate, instead, an empirical values should be used
149 #define VTP(vec) typename std::remove_reference<decltype(vec)>::type::value_type
150 #define ESTSIZE(vec, slot) mCoders[int(slot)] ? \
151  rans::calculateMaxBufferSize(vec.size(), reinterpret_cast<const o2::rans::LiteralEncoder64<VTP(vec)>*>(mCoders[int(slot)].get())->getAlphabetRangeBits(), sizeof(VTP(vec)) ) : vec.size()*sizeof(VTP(vec))
152  sz += ESTSIZE(cd.trigger, CTF::BLC_trigger);
153  sz += ESTSIZE(cd.bcInc, CTF::BLC_bcInc);
154  sz += ESTSIZE(cd.orbitInc, CTF::BLC_orbitInc);
155  sz += ESTSIZE(cd.nChan, CTF::BLC_nChan);
156  sz += ESTSIZE(cd.eventFlags, CTF::BLC_flags);
157  sz += ESTSIZE(cd.idChan, CTF::BLC_idChan);
158  sz += ESTSIZE(cd.qtcChain, CTF::BLC_qtcChain);
159  sz += ESTSIZE(cd.cfdTime, CTF::BLC_cfdTime);
160  sz += ESTSIZE(cd.qtcAmpl, CTF::BLC_qtcAmpl);
161  // clang-format on
162 
163  LOG(INFO) << "Estimated output size is " << sz << " bytes";
164  return sz;
165 }
