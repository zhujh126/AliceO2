150 template <typename... T>
151 float Tracker::evaluateTask(void (Tracker::*task)(T...), const char* taskName, std::ostream& ostream,
152  T&&... args)
153 {
154  float diff{0.f};
155 
156  if (constants::DoTimeBenchmarks) {
157  auto start = std::chrono::high_resolution_clock::now();
158  (this->*task)(std::forward<T>(args)...);
159  auto end = std::chrono::high_resolution_clock::now();
160 
161  std::chrono::duration<double, std::milli> diff_t{end - start};
162  diff = diff_t.count();
163 
164  if (fair::Logger::Logging(fair::Severity::info)) {
165  if (taskName == nullptr) {
166  ostream << diff << "\t";
167  } else {
168  ostream << std::setw(2) << " - " << taskName << " completed in: " << diff << " ms" << std::endl;
169  }
170  }
171  } else {
172  (this->*task)(std::forward<T>(args)...);
173  }
174 
175  return diff;
176 }
