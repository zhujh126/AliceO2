188 inline void TPCdEdxCalibrationSplines::setSplinesFromFile(TFile& inpf)
189 {
190  FlatObject::startConstruction();
191 
192  int buffSize = 0;
193  int offsets1[FSplines];
194  int offsets2[FSplines];
195 
196  for (unsigned int ireg = 0; ireg < FSplines; ++ireg) {
197  std::string splinename = fmt::format("spline_qMax_region{}", ireg);
198  o2::gpu::Spline2D<float, 1>* splineTmpqMax = o2::gpu::Spline2D<float, 1>::readFromFile(inpf, splinename.data());
199  mCalibSplinesqMax[ireg] = *splineTmpqMax;
200  buffSize = alignSize(buffSize, mCalibSplinesqMax[ireg].getBufferAlignmentBytes());
201  offsets1[ireg] = buffSize;
202  buffSize += mCalibSplinesqMax[ireg].getFlatBufferSize();
203  }
204 
205  for (unsigned int ireg = 0; ireg < FSplines; ++ireg) {
206  std::string splinename = fmt::format("spline_qTot_region{}", ireg);
207  o2::gpu::Spline2D<float, 1>* splineTmpqTot = o2::gpu::Spline2D<float, 1>::readFromFile(inpf, splinename.data());
208  mCalibSplinesqTot[ireg] = *splineTmpqTot;
209  buffSize = alignSize(buffSize, mCalibSplinesqTot[ireg].getBufferAlignmentBytes());
210  offsets2[ireg] = buffSize;
211  buffSize += mCalibSplinesqTot[ireg].getFlatBufferSize();
212  }
213 
214  FlatObject::finishConstruction(buffSize);
215 
216  for (unsigned int i = 0; i < FSplines; i++) {
217  mCalibSplinesqMax[i].moveBufferTo(mFlatBufferPtr + offsets1[i]);
218  }
219  for (unsigned int i = 0; i < FSplines; i++) {
220  mCalibSplinesqTot[i].moveBufferTo(mFlatBufferPtr + offsets2[i]);
221  }
222 }
