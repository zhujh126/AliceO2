50 void AltroDecoder::readChannels()
51 {
52  mChannelsInitialized = false;
53  mChannels.clear();
54  int currentpos = 0;
55  auto& buffer = mRawReader.getPayload().getPayloadWords();
56  while (currentpos < buffer.size() - mRCUTrailer.getTrailerSize()) {
57  auto currentword = buffer[currentpos++];
58  if (currentword >> 30 != 1) {
59  continue;
60  }
61  // starting a new channel
62  mChannels.emplace_back(currentword & 0xFFF, (currentword >> 16) & 0x3FF);
63  auto& currentchannel = mChannels.back();
64  currentchannel.setBadChannel((currentword >> 29) & 0x1);
65 
67  int numberofwords = (currentchannel.getPayloadSize() + 2) / 3;
68  std::vector<uint16_t> bunchwords;
69  for (int iword = 0; iword < numberofwords; iword++) {
70  currentword = buffer[currentpos++];
71  if ((currentword >> 30) != 0) {
72  // AliceO2::InfoLogger::InfoLogger logger;
73  // logger << "Unexpected end of payload in altro channel payload! DDL=" << std::setw(3) << std::setfill(0) << mRawReader.getRawHeader().getLink()
74  // << ", Address=0x" << std::hex << current.getHardwareAddress() << ", word=0x" << currentword << std::dec;
75  currentpos--;
76  continue;
77  }
78  bunchwords.push_back((currentword >> 20) & 0x3FF);
79  bunchwords.push_back((currentword >> 10) & 0x3FF);
80  bunchwords.push_back(currentword & 0x3FF);
81  }
82 
83  // decode bunches
84  int currentsample = 0;
85  while (currentsample < currentchannel.getPayloadSize() && bunchwords.size() > currentsample + 2) {
86  int bunchlength = bunchwords[currentsample] - 2, // remove words for bunchlength and starttime
87  starttime = bunchwords[currentsample + 1];
88  auto& currentbunch = currentchannel.createBunch(bunchlength, starttime);
89  currentbunch.initFromRange(gsl::span<uint16_t>(&bunchwords[currentsample + 2], std::min((unsigned long)bunchlength, bunchwords.size() - currentsample - 2)));
90  currentsample += bunchlength + 2;
91  }
92  }
93  mChannelsInitialized = true;
94 }
