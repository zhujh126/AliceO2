58 void Tracker::clustersToTracks(const ROframe& event, std::ostream& timeBenchmarkOutputStream)
59 {
60  const int verticesNum = event.getPrimaryVerticesNum();
61  mTracks.clear();
62  mTrackLabels.clear();
63 
64  for (int iVertex = 0; iVertex < verticesNum; ++iVertex) {
65 
66  float total{0.f};
67 
68  for (int iteration = 0; iteration < mTrkParams.size(); ++iteration) {
69 
70  int numCls = 0;
71  for (unsigned int iLayer{0}; iLayer < mTrkParams[iteration].NLayers; ++iLayer) {
72  numCls += event.getClusters()[iLayer].size();
73  }
74  if (numCls < mTrkParams[iteration].MinTrackLength) {
75  continue;
76  }
77 
78  mTraits->UpdateTrackingParameters(mTrkParams[iteration]);
80  int pass = iteration + iVertex;
81  std::array<float, 3> pV = {event.getPrimaryVertex(iVertex).x, event.getPrimaryVertex(iVertex).y, event.getPrimaryVertex(iVertex).z};
82  total += evaluateTask(&Tracker::initialisePrimaryVertexContext, "Context initialisation",
83  timeBenchmarkOutputStream, mMemParams[iteration], mTrkParams[iteration], event.getClusters(), pV, pass);
84  total += evaluateTask(&Tracker::computeTracklets, "Tracklet finding", timeBenchmarkOutputStream);
85  total += evaluateTask(&Tracker::computeCells, "Cell finding", timeBenchmarkOutputStream);
86  total += evaluateTask(&Tracker::findCellsNeighbours, "Neighbour finding", timeBenchmarkOutputStream, iteration);
87  total += evaluateTask(&Tracker::findRoads, "Road finding", timeBenchmarkOutputStream, iteration);
88  total += evaluateTask(&Tracker::findTracks, "Track finding", timeBenchmarkOutputStream, event);
89  }
90  if (constants::DoTimeBenchmarks && fair::Logger::Logging(fair::Severity::info)) {
91  timeBenchmarkOutputStream << std::setw(2) << " - "
92  << "Vertex processing completed in: " << total << "ms" << std::endl;
93  }
94  }
95  if (event.hasMCinformation()) {
96  computeTracksMClabels(event);
97  } else {
98  rectifyClusterIndices(event);
99  }
100 }

112 void Tracker::findCellsNeighbours(int& iteration)
113 {
114  for (int iLayer{0}; iLayer < mTrkParams[iteration].CellsPerRoad() - 1; ++iLayer) {
115 
116  if (mPrimaryVertexContext->getCells()[iLayer + 1].empty() ||
117  mPrimaryVertexContext->getCellsLookupTable()[iLayer].empty()) {
118  continue;
119  }
120 
121  int layerCellsNum{static_cast<int>(mPrimaryVertexContext->getCells()[iLayer].size())};
122  const int nextLayerCellsNum{static_cast<int>(mPrimaryVertexContext->getCells()[iLayer + 1].size())};
123  mPrimaryVertexContext->getCellsNeighbours()[iLayer].resize(nextLayerCellsNum);
124 
125  for (int iCell{0}; iCell < layerCellsNum; ++iCell) {
126 
127  const Cell& currentCell{mPrimaryVertexContext->getCells()[iLayer][iCell]};
128  const int nextLayerTrackletIndex{currentCell.getSecondTrackletIndex()};
129  const int nextLayerFirstCellIndex{mPrimaryVertexContext->getCellsLookupTable()[iLayer][nextLayerTrackletIndex]};
130  if (nextLayerFirstCellIndex != constants::its::UnusedIndex &&
131  mPrimaryVertexContext->getCells()[iLayer + 1][nextLayerFirstCellIndex].getFirstTrackletIndex() ==
132  nextLayerTrackletIndex) {
133 
134  for (int iNextLayerCell{nextLayerFirstCellIndex}; iNextLayerCell < nextLayerCellsNum; ++iNextLayerCell) {
135 
136  Cell& nextCell{mPrimaryVertexContext->getCells()[iLayer + 1][iNextLayerCell]};
137  if (nextCell.getFirstTrackletIndex() != nextLayerTrackletIndex) {
138  break;
139  }
140 
141  const float3 currentCellNormalVector{currentCell.getNormalVectorCoordinates()};
142  const float3 nextCellNormalVector{nextCell.getNormalVectorCoordinates()};
143  const float3 normalVectorsDeltaVector{currentCellNormalVector.x - nextCellNormalVector.x,
144  currentCellNormalVector.y - nextCellNormalVector.y,
145  currentCellNormalVector.z - nextCellNormalVector.z};
146 
147  const float deltaNormalVectorsModulus{(normalVectorsDeltaVector.x * normalVectorsDeltaVector.x) +
148  (normalVectorsDeltaVector.y * normalVectorsDeltaVector.y) +
149  (normalVectorsDeltaVector.z * normalVectorsDeltaVector.z)};
150  const float deltaCurvature{std::abs(currentCell.getCurvature() - nextCell.getCurvature())};
151 
152  if (deltaNormalVectorsModulus < mTrkParams[iteration].NeighbourMaxDeltaN[iLayer] &&
153  deltaCurvature < mTrkParams[iteration].NeighbourMaxDeltaCurvature[iLayer]) {
154 
155  mPrimaryVertexContext->getCellsNeighbours()[iLayer][iNextLayerCell].push_back(iCell);
156 
157  const int currentCellLevel{currentCell.getLevel()};
158 
159  if (currentCellLevel >= nextCell.getLevel()) {
160 
161  nextCell.setLevel(currentCellLevel + 1);
162  }
163  }
164  }
165  }
166  }
167  }

170 void Tracker::findRoads(int& iteration)
171 {
172  for (int iLevel{mTrkParams[iteration].CellsPerRoad()}; iLevel >= mTrkParams[iteration].CellMinimumLevel(); --iLevel) {
173  CA_DEBUGGER(int nRoads = -mPrimaryVertexContext->getRoads().size());
174  const int minimumLevel{iLevel - 1};
175 
176  for (int iLayer{mTrkParams[iteration].CellsPerRoad() - 1}; iLayer >= minimumLevel; --iLayer) {
177 
178  const int levelCellsNum{static_cast<int>(mPrimaryVertexContext->getCells()[iLayer].size())};
179 
180  for (int iCell{0}; iCell < levelCellsNum; ++iCell) {
181 
182  Cell& currentCell{mPrimaryVertexContext->getCells()[iLayer][iCell]};
183 
184  if (currentCell.getLevel() != iLevel) {
185  continue;
186  }
187 
188  mPrimaryVertexContext->getRoads().emplace_back(iLayer, iCell);
189 
192  if (iLevel == 1) {
193  continue;
194  }
195 
196  const int cellNeighboursNum{static_cast<int>(
197  mPrimaryVertexContext->getCellsNeighbours()[iLayer - 1][iCell].size())};
198  bool isFirstValidNeighbour = true;
199 
200  for (int iNeighbourCell{0}; iNeighbourCell < cellNeighboursNum; ++iNeighbourCell) {
201 
202  const int neighbourCellId = mPrimaryVertexContext->getCellsNeighbours()[iLayer - 1][iCell][iNeighbourCell];
203  const Cell& neighbourCell = mPrimaryVertexContext->getCells()[iLayer - 1][neighbourCellId];
204 
205  if (iLevel - 1 != neighbourCell.getLevel()) {
206  continue;
207  }
208 
209  if (isFirstValidNeighbour) {
210 
211  isFirstValidNeighbour = false;
212 
213  } else {
214 
215  mPrimaryVertexContext->getRoads().emplace_back(iLayer, iCell);
216  }
217 
218  traverseCellsTree(neighbourCellId, iLayer - 1);
219  }
220 
221  // TODO: crosscheck for short track iterations
222  // currentCell.setLevel(0);
223  }
224  }
225 #ifdef CA_DEBUG
226  nRoads += mPrimaryVertexContext->getRoads().size();
227  std::cout << "+++ Roads with " << iLevel + 2 << " clusters: " << nRoads << " / " << mPrimaryVertexContext->getRoads().size() << std::endl;
228 #endif
229  }

244 #endif
245 
246  for (auto& road : mPrimaryVertexContext->getRoads()) {
247  std::vector<int> clusters(mTrkParams[0].NLayers, constants::its::UnusedIndex);
248  int lastCellLevel = constants::its::UnusedIndex;
249  CA_DEBUGGER(int nClusters = 2);
250 
251  for (int iCell{0}; iCell < mTrkParams[0].CellsPerRoad(); ++iCell) {
252  const int cellIndex = road[iCell];
253  if (cellIndex == constants::its::UnusedIndex) {
254  continue;
255  } else {
256  clusters[iCell] = mPrimaryVertexContext->getCells()[iCell][cellIndex].getFirstClusterIndex();
257  clusters[iCell + 1] = mPrimaryVertexContext->getCells()[iCell][cellIndex].getSecondClusterIndex();
258  clusters[iCell + 2] = mPrimaryVertexContext->getCells()[iCell][cellIndex].getThirdClusterIndex();
259  assert(clusters[iCell] != constants::its::UnusedIndex &&
260  clusters[iCell + 1] != constants::its::UnusedIndex &&
261  clusters[iCell + 2] != constants::its::UnusedIndex);
262  lastCellLevel = iCell;
263  CA_DEBUGGER(nClusters++);
264  }
265  }
266 
267  CA_DEBUGGER(assert(nClusters >= mTrkParams[0].MinTrackLength));
268  CA_DEBUGGER(roadCounters[nClusters - 4]++);
269 
270  if (lastCellLevel == constants::its::UnusedIndex) {
271  continue;
272  }
273 
275  for (int iC{0}; iC < clusters.size(); iC++) {
276  if (clusters[iC] != constants::its::UnusedIndex) {
277  clusters[iC] = mPrimaryVertexContext->getClusters()[iC][clusters[iC]].clusterId;
278  }
279  }
280 
282  const auto& cluster1_glo = event.getClustersOnLayer(lastCellLevel + 2).at(clusters[lastCellLevel + 2]);
283  const auto& cluster2_glo = event.getClustersOnLayer(lastCellLevel + 1).at(clusters[lastCellLevel + 1]);
284  const auto& cluster3_glo = event.getClustersOnLayer(lastCellLevel).at(clusters[lastCellLevel]);
285 
286  const auto& cluster3_tf = event.getTrackingFrameInfoOnLayer(lastCellLevel).at(clusters[lastCellLevel]);
287 
289  TrackITSExt temporaryTrack{buildTrackSeed(cluster1_glo, cluster2_glo, cluster3_glo, cluster3_tf)};
290  for (size_t iC = 0; iC < clusters.size(); ++iC) {
291  temporaryTrack.setExternalClusterIndex(iC, clusters[iC], clusters[iC] != constants::its::UnusedIndex);
292  }
293  bool fitSuccess = fitTrack(event, temporaryTrack, mTrkParams[0].NLayers - 4, -1, -1);
294  if (!fitSuccess) {
295  continue;
296  }
297  CA_DEBUGGER(fitCounters[nClusters - 4]++);
298  temporaryTrack.resetCovariance();
299  fitSuccess = fitTrack(event, temporaryTrack, 0, mTrkParams[0].NLayers, 1, mTrkParams[0].FitIterationMaxChi2[0]);
300  if (!fitSuccess) {
301  continue;
302  }
303  CA_DEBUGGER(backpropagatedCounters[nClusters - 4]++);
304  temporaryTrack.getParamOut() = temporaryTrack;
305  temporaryTrack.resetCovariance();
306  fitSuccess = fitTrack(event, temporaryTrack, mTrkParams[0].NLayers - 1, -1, -1, mTrkParams[0].FitIterationMaxChi2[1]);
307 #ifdef CA_DEBUG
308  mDebugger->dumpTrackToBranchWithInfo("testBranch", temporaryTrack, event, mPrimaryVertexContext, true);
309 #endif
310  if (!fitSuccess) {
311  continue;
312  }
313  CA_DEBUGGER(refitCounters[nClusters - 4]++);
314  tracks.emplace_back(temporaryTrack);
315  CA_DEBUGGER(assert(nClusters == temporaryTrack.getNumberOfClusters()));
316  }
317  //mTraits->refitTracks(event.getTrackingFrameInfo(), tracks);
318 
319  std::sort(tracks.begin(), tracks.end(),
320  [](TrackITSExt& track1, TrackITSExt& track2) { return track1.isBetter(track2, 1.e6f); });
321

329  // std::array<int, 4> xcheckCounters{0};
330 #endif
331 
332  for (auto& track : tracks) {
333  CA_DEBUGGER(int nClusters = 0);
334  int nShared = 0;
335  for (int iLayer{0}; iLayer < mTrkParams[0].NLayers; ++iLayer) {

361  if (track.getClusterIndex(iLayer) == constants::its::UnusedIndex) {

413 bool Tracker::fitTrack(const ROframe& event, TrackITSExt& track, int start, int end, int step, const float chi2cut)
414 {
415  auto propInstance = o2::base::Propagator::Instance();
416  track.setChi2(0);
417  for (int iLayer{start}; iLayer != end; iLayer += step) {
418  if (track.getClusterIndex(iLayer) == constants::its::UnusedIndex) {
419  continue;
420  }
421  const TrackingFrameInfo& trackingHit = event.getTrackingFrameInfoOnLayer(iLayer).at(track.getClusterIndex(iLayer));
422 
423  if (!track.rotate(trackingHit.alphaTrackingFrame)) {
424  return false;
425  }
426 
427  if (!propInstance->propagateToX(track, trackingHit.xTrackingFrame, getBz(), o2::base::PropagatorImpl<float>::MAX_SIN_PHI, o2::base::PropagatorImpl<float>::MAX_STEP, mCorrType)) {
428  return false;
429  }
430 
431  auto predChi2{track.getPredictedChi2(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame)};
432  if (predChi2 > chi2cut) {
433  return false;
434  }
435  track.setChi2(track.getChi2() + predChi2);
436  if (!track.o2::track::TrackParCov::update(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame)) {
437  return false;
438  }
439  }
440  return true;
441 }

444 {

446  const int currentCellLevel = currentCell.getLevel();
447 
448  mPrimaryVertexContext->getRoads().back().addCell(currentLayerId, currentCellId);

451  const int cellNeighboursNum{static_cast<int>(
452  mPrimaryVertexContext->getCellsNeighbours()[currentLayerId - 1][currentCellId].size())};
453  bool isFirstValidNeighbour = true;
454

456 
457  const int neighbourCellId =

479 void Tracker::computeRoadsMClabels(const ROframe& event)
480 {
482  if (!event.hasMCinformation()) {
483  return;
484  }
485 
486  mPrimaryVertexContext->initialiseRoadLabels();
487 
488  int roadsNum{static_cast<int>(mPrimaryVertexContext->getRoads().size())};
489 
490  for (int iRoad{0}; iRoad < roadsNum; ++iRoad) {
491 
492  Road& currentRoad{mPrimaryVertexContext->getRoads()[iRoad]};
493  MCCompLabel maxOccurrencesValue{constants::its::UnusedIndex, constants::its::UnusedIndex,
494  constants::its::UnusedIndex, false};
495  int count{0};
496  bool isFakeRoad{false};
497  bool isFirstRoadCell{true};
498 
499  for (int iCell{0}; iCell < mTrkParams[0].CellsPerRoad(); ++iCell) {
500  const int currentCellIndex{currentRoad[iCell]};
501 
502  if (currentCellIndex == constants::its::UnusedIndex) {
503  if (isFirstRoadCell) {
504  continue;
505  } else {
506  break;
507  }
508  }
509 
510  const Cell& currentCell{mPrimaryVertexContext->getCells()[iCell][currentCellIndex]};
511 
512  if (isFirstRoadCell) {
513 
514  const int cl0index{mPrimaryVertexContext->getClusters()[iCell][currentCell.getFirstClusterIndex()].clusterId};
515  auto& cl0labs{event.getClusterLabels(iCell, cl0index)};
516  maxOccurrencesValue = cl0labs;
517  count = 1;
518 
519  const int cl1index{mPrimaryVertexContext->getClusters()[iCell + 1][currentCell.getSecondClusterIndex()].clusterId};
520  const auto& cl1labs{event.getClusterLabels(iCell + 1, cl1index)};
521 
522  if (cl1labs == maxOccurrencesValue) {
523  ++count;
524  } else {
525  maxOccurrencesValue = cl1labs;
526  count = 1;
527  isFakeRoad = true;
528  }
529 
530  isFirstRoadCell = false;
531  }
532 
533  const int cl2index{mPrimaryVertexContext->getClusters()[iCell + 2][currentCell.getThirdClusterIndex()].clusterId};
534  const auto& cl2labs{event.getClusterLabels(iCell + 2, cl2index)};
535 
536  if (cl2labs == maxOccurrencesValue) {
537  ++count;
538  } else {
539  --count;
540  isFakeRoad = true;
541  }
542 
543  if (count == 0) {
544  maxOccurrencesValue = cl2labs;
545  count = 1;
546  }
547  }
548 
549  mPrimaryVertexContext->setRoadLabel(iRoad, maxOccurrencesValue.getRawValue(), isFakeRoad);
550  }

553 void Tracker::computeTracksMClabels(const ROframe& event)
554 {
556  if (!event.hasMCinformation()) {
557  return;
558  }
559 
560  int tracksNum{static_cast<int>(mTracks.size())};
561 
562  for (auto& track : mTracks) {
563 
564  MCCompLabel maxOccurrencesValue{constants::its::UnusedIndex, constants::its::UnusedIndex,
565  constants::its::UnusedIndex, false};
566  int count{0};
567  bool isFakeTrack{false};
568 
569  for (int iCluster = 0; iCluster < TrackITSExt::MaxClusters; ++iCluster) {
570  const int index = track.getClusterIndex(iCluster);
571  if (index == constants::its::UnusedIndex) {
572  continue;
573  }
574  const MCCompLabel& currentLabel = event.getClusterLabels(iCluster, index);
575  if (currentLabel == maxOccurrencesValue) {
576  ++count;
577  } else {
578  if (count != 0) { // only in the first iteration count can be 0 at this point
579  isFakeTrack = true;
580  --count;
581  }
582  if (count == 0) {
583  maxOccurrencesValue = currentLabel;
584  count = 1;
585  }
586  }
587  track.setExternalClusterIndex(iCluster, event.getClusterExternalIndex(iCluster, index));
588  }
589 
590  if (isFakeTrack) {
591  maxOccurrencesValue.setFakeFlag();
592  }
593  mTrackLabels.emplace_back(maxOccurrencesValue);
594  }

614 track::TrackParCov Tracker::buildTrackSeed(const Cluster& cluster1, const Cluster& cluster2,
615  const Cluster& cluster3, const TrackingFrameInfo& tf3)
616 {
617  const float ca = std::cos(tf3.alphaTrackingFrame), sa = std::sin(tf3.alphaTrackingFrame);
618  const float x1 = cluster1.xCoordinate * ca + cluster1.yCoordinate * sa;
619  const float y1 = -cluster1.xCoordinate * sa + cluster1.yCoordinate * ca;
620  const float z1 = cluster1.zCoordinate;
621  const float x2 = cluster2.xCoordinate * ca + cluster2.yCoordinate * sa;
622  const float y2 = -cluster2.xCoordinate * sa + cluster2.yCoordinate * ca;
623  const float z2 = cluster2.zCoordinate;
624  const float x3 = tf3.xTrackingFrame;
625  const float y3 = tf3.positionTrackingFrame[0];
626  const float z3 = tf3.positionTrackingFrame[1];
627 
628  const float crv = math_utils::computeCurvature(x1, y1, x2, y2, x3, y3);
629  const float x0 = math_utils::computeCurvatureCentreX(x1, y1, x2, y2, x3, y3);
630  const float tgl12 = math_utils::computeTanDipAngle(x1, y1, x2, y2, z1, z2);
631  const float tgl23 = math_utils::computeTanDipAngle(x2, y2, x3, y3, z2, z3);
632 
633  const float fy = 1. / (cluster2.rCoordinate - cluster3.rCoordinate);
634  const float& tz = fy;
635  const float cy = (math_utils::computeCurvature(x1, y1, x2, y2 + constants::its::Resolution, x3, y3) - crv) /
636  (constants::its::Resolution * getBz() * o2::constants::math::B2C) *
637  20.f; // FIXME: MS contribution to the cov[14] (*20 added)
638  constexpr float s2 = constants::its::Resolution * constants::its::Resolution;
639 
640  return track::TrackParCov(tf3.xTrackingFrame, tf3.alphaTrackingFrame,
641  {y3, z3, crv * (x3 - x0), 0.5f * (tgl12 + tgl23),
642  std::abs(getBz()) < o2::constants::math::Almost0 ? o2::constants::math::Almost0
643  : crv / (getBz() * o2::constants::math::B2C)},
644  {s2, 0.f, s2, s2 * fy, 0.f, s2 * fy * fy, 0.f, s2 * tz, 0.f, s2 * tz * tz, s2 * cy, 0.f,
645  s2 * fy * cy, 0.f, s2 * cy * cy});
