92 std::vector<float> o2::tpc::IDCGroup::get1DIDCs() const
93 {
94  // integrate IDCs for each interval
95  std::vector<float> idc;
96  const unsigned int nIntervals = getNIntegrationIntervals();
97  idc.reserve(nIntervals);
98  for (unsigned int i = 0; i < nIntervals; ++i) {
99  // set integration range for one integration interval
100  const auto start = mIDCsGrouped.begin() + i * getNIDCsPerIntegrationInterval();
101  const auto end = start + getNIDCsPerIntegrationInterval();
102  idc.emplace_back(std::accumulate(start, end, decltype(mIDCsGrouped)::value_type(0)));
103  }
104  // normalize 1D-IDCs to absolute space charge
105  const float norm = Mapper::REGIONAREA[mRegion] / getNIDCsPerIntegrationInterval();
106  std::transform(idc.begin(), idc.end(), idc.begin(), [norm](auto& val) { return val * norm; });
107 
108  return idc;
109 }
