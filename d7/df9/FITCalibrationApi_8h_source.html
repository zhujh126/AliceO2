99 template <typename CalibrationObjectType>
100 const CalibrationObjectType& FITCalibrationApi::getCalibrationObjectForGivenTimestamp(long timestamp)
101 {
102 
103  o2::ccdb::BasicCCDBManager::instance().setTimestamp(timestamp);
104  auto calibObjectPtr = o2::ccdb::BasicCCDBManager::instance().get<CalibrationObjectType>(getObjectPath<CalibrationObjectType>());
105  if (nullptr == calibObjectPtr) {
106  throw std::runtime_error("Cannot read requested calibration object");
107  }
108  return *calibObjectPtr;
109 }
