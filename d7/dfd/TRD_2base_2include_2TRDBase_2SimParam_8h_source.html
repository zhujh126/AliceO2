125 inline double SimParam::timeResponse(double time) const
126 {
127  //
128  // Applies the preamp shaper time response
129  // (We assume a signal rise time of 0.2us = fTRFlo/2.
130  //
131 
132  double rt = (time - .5 * mTRFlo) * mInvTRFwid;
133  int iBin = (int)rt;
134  double dt = rt - iBin;
135  if ((iBin >= 0) && (iBin + 1 < mTRFbin)) {
136  return mTRFsmp[iBin] + (mTRFsmp[iBin + 1] - mTRFsmp[iBin]) * dt;
137  } else {
138  return 0.0;
139  }
140 }
141 
142 inline double SimParam::crossTalk(double time) const
143 {
144  //
145  // Applies the pad-pad capacitive cross talk
146  //
147 
148  double rt = (time - mTRFlo) * mInvTRFwid;
149  int iBin = (int)rt;
150  double dt = rt - iBin;
151  if ((iBin >= 0) && (iBin + 1 < mTRFbin)) {
152  return mCTsmp[iBin] + (mCTsmp[iBin + 1] - mCTsmp[iBin]) * dt;
153  } else {
154  return 0.0;
155  }
156 }
