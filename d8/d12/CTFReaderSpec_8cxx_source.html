51 template <typename T>
52 bool readFromTree(TTree& tree, const std::string brname, T& dest, int ev = 0)
53 {
54  auto* br = tree.GetBranch(brname.c_str());
55  if (br && br->GetEntries() > ev) {
56  auto* ptr = &dest;
57  br->SetAddress(&ptr);
58  br->GetEntry(ev);
59  br->ResetAddress();
60  return true;
61  }
62  return false;
63 }

107 void CTFReaderSpec::openCTFFile(const std::string& flname)
108 {
109  mCTFFile.reset(TFile::Open(flname.c_str()));
110  if (!mCTFFile->IsOpen() || mCTFFile->IsZombie()) {
111  LOG(ERROR) << "Failed to open file " << flname;
112  throw std::runtime_error("failed to open CTF file");
113  }
114  mCTFTree.reset((TTree*)mCTFFile->Get(std::string(o2::base::NameConf::CTFTREENAME).c_str()));
115  if (!mCTFTree) {
116  throw std::runtime_error("failed to load CTF tree");
117  }
118  mCurrEntry = 0;
119 }

122 void CTFReaderSpec::run(ProcessingContext& pc)
123 {
124  if (mNextToProcess >= mInput.size()) {
125  return;
126  }
127  if (mDelayMUS && mCTFCounter > 0) {
128  usleep(mDelayMUS);
129  }
130 
131  auto cput = mTimer.CpuTime();
132  mTimer.Start(false);
133 
134  if (!mCTFTree) { // there is still a tree open with multiple entries
135  std::string inputFile = o2::utils::Str::concat_string(mCTFDir, mInput[mNextToProcess]);
136  LOG(INFO) << "Reading CTF input " << mNextToProcess << ' ' << inputFile;
137  openCTFFile(inputFile);
138  }
139  CTFHeader ctfHeader;
140  if (!readFromTree(*(mCTFTree.get()), "CTFHeader", ctfHeader, mCurrEntry)) {
141  throw std::runtime_error("did not find CTFHeader");
142  }
143  LOG(INFO) << ctfHeader;
144 
145  auto setFirstTFOrbit = [&pc, &ctfHeader, this](const std::string& label) {
146  auto* hd = pc.outputs().findMessageHeader({label});
147  if (!hd) {
148  throw std::runtime_error(o2::utils::Str::concat_string("failed to find output message header for ", label));
149  }
150  hd->firstTForbit = ctfHeader.firstTForbit;
151  hd->tfCounter = this->mCTFCounter;
152  };
153 
154  // send CTF Header
155  pc.outputs().snapshot({"header"}, ctfHeader);
156  setFirstTFOrbit("header");
157 
158  DetID::mask_t detsTF = mDets & ctfHeader.detectors;
159  DetID det;
160 
161  det = DetID::ITS;
162  if (detsTF[det]) {
163  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::itsmft::CTF));
164  o2::itsmft::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
165  setFirstTFOrbit(det.getName());
166  }
167 
168  det = DetID::MFT;
169  if (detsTF[det]) {
170  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::itsmft::CTF));
171  o2::itsmft::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
172  setFirstTFOrbit(det.getName());
173  }
174 
175  det = DetID::TPC;
176  if (detsTF[det]) {
177  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::tpc::CTF));
178  o2::tpc::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
179  setFirstTFOrbit(det.getName());
180  }
181 
182  det = DetID::TRD;
183  if (detsTF[det]) {
184  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::trd::CTF));
185  o2::trd::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
186  setFirstTFOrbit(det.getName());
187  }
188 
189  det = DetID::FT0;
190  if (detsTF[det]) {
191  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::ft0::CTF));
192  o2::ft0::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
193  setFirstTFOrbit(det.getName());
194  }
195 
196  det = DetID::FV0;
197  if (detsTF[det]) {
198  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::fv0::CTF));
199  o2::fv0::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
200  setFirstTFOrbit(det.getName());
201  }
202 
203  det = DetID::FDD;
204  if (detsTF[det]) {
205  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::fdd::CTF));

207  setFirstTFOrbit(det.getName());
208  }
209 
210  det = DetID::TOF;
211  if (detsTF[det]) {
212  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::tof::CTF));

220  o2::mid::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
221  setFirstTFOrbit(det.getName());
222  }
223 
224  det = DetID::MCH;
225  if (detsTF[det]) {
226  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::mch::CTF));
227  o2::mch::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
228  setFirstTFOrbit(det.getName());
229  }
230 
231  det = DetID::EMC;
232  if (detsTF[det]) {
233  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::emcal::CTF));
234  o2::emcal::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
235  setFirstTFOrbit(det.getName());
236  }
237 
238  det = DetID::PHS;
239  if (detsTF[det]) {
240  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::phos::CTF));
241  o2::phos::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
242  setFirstTFOrbit(det.getName());
243  }
244 
245  det = DetID::CPV;
246  if (detsTF[det]) {
247  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::cpv::CTF));
248  o2::cpv::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
249  setFirstTFOrbit(det.getName());
250  }
251 
252  det = DetID::ZDC;
253  if (detsTF[det]) {
254  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::zdc::CTF));
255  o2::zdc::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
256  setFirstTFOrbit(det.getName());
257  }
258 
259  det = DetID::HMP;
260  if (detsTF[det]) {
261  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({det.getName()}, sizeof(o2::hmpid::CTF));
262  o2::hmpid::CTF::readFromTree(bufVec, *(mCTFTree.get()), det.getName(), mCurrEntry);
263  setFirstTFOrbit(det.getName());
264  }
265 
266  mTimer.Stop();
267  LOGP(INFO, "Read CTF#{} ({} of {} in {}) in {:.3f} s", mCTFCounter, mCurrEntry, mCTFTree->GetEntries(), mCTFFile->GetName(), mTimer.CpuTime() - cput);
268 
269  bool moreToProcess = (++mCurrEntry < mCTFTree->GetEntries());
270  if (!moreToProcess) { // this file is done, check if there are other files
271  mCTFTree.reset();
272  mCTFFile->Close();
273  mCTFFile.reset();
274  moreToProcess = true;
275  if (++mNextToProcess >= mInput.size()) {
276  if (++mLoopsCounter >= mLoops) {
277  moreToProcess = false;
278  } else {
279  mNextToProcess = 0;
280  LOG(INFO) << "Starting new loop " << mNextToProcess << " of " << mLoops;
281  }
282  }
283  }
284 
285  mCTFCounter++;
286 
287  if (!moreToProcess) {
288  pc.services().get<ControlService>().endOfStream();
289  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
290  LOGP(INFO, "CTF reading total timing: Cpu: {:.3f} Real: {:.3f} s for {} TFs in {} loops",
291  mTimer.CpuTime(), mTimer.RealTime(), mCTFCounter, mLoops);
292  }
293 }

296 DataProcessorSpec getCTFReaderSpec(DetID::mask_t dets, const std::string& inp, int loop, int delayMUS)
297 {
298  std::vector<OutputSpec> outputs;
299 
300  outputs.emplace_back(OutputLabel{"header"}, "CTF", "HEADER", 0, Lifetime::Timeframe);
301  for (auto id = DetID::First; id <= DetID::Last; id++) {
302  if (dets[id]) {
303  DetID det(id);
304  outputs.emplace_back(OutputLabel{det.getName()}, det.getDataOrigin(), "CTFDATA", 0, Lifetime::Timeframe);
305  }
306  }
307  return DataProcessorSpec{
308  "ctf-reader",
309  Inputs{},
310  outputs,
311  AlgorithmSpec{adaptFromTask<CTFReaderSpec>(dets, inp, loop, delayMUS)},
312  Options{{"input-dir", VariantType::String, "none", {"CTF input directory"}}}};
313 }
