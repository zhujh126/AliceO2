81 void TFReaderSpec::run(o2f::ProcessingContext& ctx)
82 {
83  if (!mDevice) {
84  mDevice = ctx.services().get<o2f::RawDeviceService>().device();
85  mOutputRoutes = ctx.services().get<o2f::RawDeviceService>().spec().outputs; // copy!!!
86  // start TFBuilder thread
87  mRunning = true;
88  mTFBuilderThread = std::thread(&TFReaderSpec::TFBuilder, this);
89  }
90  static auto tLastTF = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
91  static long deltaSending = 0; // time correction for sending
92  auto device = ctx.services().get<o2f::RawDeviceService>().device();
93  assert(device);
94  if (device != mDevice) {
95  throw std::runtime_error(fmt::format("FMQDevice has changed, old={} new={}", fmt::ptr(mDevice), fmt::ptr(device)));
96  }
97 
98  while (1) {
99  if (mTFCounter >= mInput.maxTFs) { // done
100  stopProcessing(ctx);
101  break;
102  }
103  if (mTFQueue.size()) {
104  auto tfPtr = std::move(mTFQueue.front());
105  mTFQueue.pop();
106  if (!tfPtr) {
107  LOG(ERROR) << "Builder provided nullptr TF pointer";
108  continue;
109  }
110  auto tNow = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
111  auto tDiff = tNow - tLastTF + 2 * deltaSending;
112  if (mTFCounter && tDiff < mInput.delay_us) {
113  usleep(mInput.delay_us - tDiff); // respect requested delay before sending
114  }
115  size_t nparts = 0;
116  auto tSend = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
117  for (auto& msgIt : *tfPtr.get()) {
118  nparts += msgIt.second->Size() / 2;
119  device->Send(*msgIt.second.get(), msgIt.first);
120  }
121  tNow = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
122  deltaSending = mTFCounter ? tNow - tLastTF : 0;
123  LOGP(INFO, "Sent TF {} of {} parts, {:.4f} s elapsed from previous TF.", mTFCounter, nparts, double(deltaSending) * 1e-6);
124  deltaSending -= mInput.delay_us;
125  if (!mTFCounter || deltaSending < 0) {
126  deltaSending = 0; // correction for next delay
127  }
128  tLastTF = tNow;
129  ++mTFCounter;
130  break;
131  }
132  if (!mRunning) { // no more TFs will be provided
133  stopProcessing(ctx);
134  break;
135  }
136  usleep(10); // wait for new TF to be built
137  }
138 }

154 {

156  mRunning = false;

163  ctx.services().get<o2f::ControlService>().readyToQuit(o2f::QuitRequest::Me);

167 void TFReaderSpec::TFBuilder()
168 {
169  // build TFs and add to the queue
170  std::string tfFileName;
171  auto sleepTime = std::chrono::microseconds(mInput.delay_us > 10000 ? mInput.delay_us : 10000);
172  while (mRunning && mDevice) {
173  if (mTFQueue.size() >= size_t(mInput.maxTFCache)) {
174  std::this_thread::sleep_for(sleepTime);
175  continue;
176  }
177  tfFileName = mFileFetcher ? mFileFetcher->getNextFileInQueue() : "";
178  if (!mRunning || (tfFileName.empty() && !mFileFetcher->isRunning()) || mTFBuilderCounter >= mInput.maxTFs) {
179  // stopped or no more files in the queue is expected or needed
180  LOG(INFO) << "TFBuilder stops processing";
181  if (mFileFetcher) {
182  mFileFetcher->stop();
183  }
184  mRunning = false;
185  break;
186  }
187  if (tfFileName.empty()) {
188  std::this_thread::sleep_for(10ms); // fait for the files cache to be filled
189  continue;
190  }
191  LOG(INFO) << "Processing file " << tfFileName;
192  SubTimeFrameFileReader reader(tfFileName);
193  size_t locID = 0;
194  //try
195  {
196  while (mRunning && mTFBuilderCounter < mInput.maxTFs) {
197  if (mTFQueue.size() >= size_t(mInput.maxTFCache)) {
198  std::this_thread::sleep_for(sleepTime);
199  continue;
200  }
201  auto tf = reader.read(mDevice, mOutputRoutes, mInput.rawChannelConfig, mInput.verbosity);
202  if (tf) {
203  mTFBuilderCounter++;
204  }
205  if (mRunning && tf) {
206  mTFQueue.push(std::move(tf));
207  locID++;
208  } else {
209  break;
210  }
211  }
212  // remove already processed file from the queue, unless they are needed for further looping
213  if (mFileFetcher) {
214  mFileFetcher->popFromQueue(mFileFetcher->getNLoops() >= mInput.maxLoops);
215  }
216  } /*catch (...) {
217  LOGP(ERROR, "Error when building {}-th TF from file {}", locID, tfFileName);
218  mFileFetcher->popFromQueue(mFileFetcher->getNLoops() >= mInput.maxLoops); // remove faile TF file
219  } */
220  }
221 }

224 o2f::DataProcessorSpec o2::rawdd::getTFReaderSpec(o2::rawdd::TFReaderInp& rinp)
225 {
226  // check which inputs are present in files to read
227  o2f::DataProcessorSpec spec;
228  spec.name = "tf-reader";
229  const DetID::mask_t DEFMask = DetID::getMask("ITS,TPC,TRD,TOF,PHS,CPV,EMC,HMP,MFT,MCH,MID,ZDC,FT0,FV0,FDD,CTP");
230  DetID::mask_t detMask = DEFMask; //DetID::getMask(rinp.detList) & DEFMask; // RS TODO
231 
232  if (rinp.rawChannelConfig.empty()) {
233  // we don't know a priori what will be the content of the TF data, so we create all possible outputs
234  for (DetID::ID id = DetID::First; id <= DetID::Last; id++) {
235  if (detMask[id]) {
236  spec.outputs.emplace_back(o2f::OutputSpec(o2f::ConcreteDataTypeMatcher{DetID::getDataOrigin(id), "RAWDATA"}));
237  }
238  }
239  // in case compessed TOF is present
240  if (detMask[DetID::TOF]) {
241  spec.outputs.emplace_back(o2f::OutputSpec(o2f::ConcreteDataTypeMatcher{DetID::getDataOrigin(DetID::TOF), "CRAWDATA"}));
242  }
243  spec.outputs.emplace_back(o2f::OutputSpec{{"stfDist"}, o2::header::gDataOriginFLP, "DISTSUBTIMEFRAME", 0});
244  } else {
245  auto nameStart = rinp.rawChannelConfig.find("name=");
246  if (nameStart == std::string::npos) {
247  throw std::runtime_error("raw channel name is not provided");
248  }
249  nameStart += strlen("name=");
250  auto nameEnd = rinp.rawChannelConfig.find(",", nameStart + 1);
251  if (nameEnd == std::string::npos) {
252  nameEnd = rinp.rawChannelConfig.size();
253  }
254  spec.options = {o2f::ConfigParamSpec{"channel-config", o2f::VariantType::String, rinp.rawChannelConfig, {"Out-of-band channel config"}}};
255  rinp.rawChannelConfig = rinp.rawChannelConfig.substr(nameStart, nameEnd - nameStart);
256  }
257 
258  spec.algorithm = o2f::adaptFromTask<TFReaderSpec>(rinp);
259 
260  return spec;
261 }
