48 GPUhd() const int4 getBinsRect(const Cluster& currentCluster, const int layerIndex,
49  const float z1, float maxdeltaz, float maxdeltaphi)
50 {
51  const float zRangeMin = z1 - maxdeltaz;
52  const float phiRangeMin = currentCluster.phiCoordinate - maxdeltaphi;
53  const float zRangeMax = z1 + maxdeltaz;
54  const float phiRangeMax = currentCluster.phiCoordinate + maxdeltaphi;
55 
56  if (zRangeMax < -LayersZCoordinate()[layerIndex + 1] ||
57  zRangeMin > LayersZCoordinate()[layerIndex + 1] || zRangeMin > zRangeMax) {
58  return constants::its2::getEmptyBinsRect();
59  }
60 
61  return int4{o2::gpu::GPUCommonMath::Max(0, getZBinIndex(layerIndex + 1, zRangeMin)),
62  getPhiBinIndex(phiRangeMin),
63  o2::gpu::GPUCommonMath::Min(ZBins - 1, getZBinIndex(layerIndex + 1, zRangeMax)),
64  getPhiBinIndex(phiRangeMax)};
65 }

155 GPUg() void trackleterKernel(
156  DeviceStoreVertexerHIP* store,
157  const TrackletingLayerOrder layerOrder,
158  const float phiCut)
159 {
160  const size_t nClustersMiddleLayer = store->getClusters()[1].size();
161  for (unsigned int currentClusterIndex = blockIdx.x * blockDim.x + threadIdx.x; currentClusterIndex < nClustersMiddleLayer; currentClusterIndex += blockDim.x * gridDim.x) {
162  if (currentClusterIndex < nClustersMiddleLayer) {
163  int storedTracklets{0};
164  const unsigned int stride{currentClusterIndex * store->getConfig().maxTrackletsPerCluster};
165  const Cluster& currentCluster = store->getClusters()[1][currentClusterIndex]; // assign-constructor may be a problem, check
166  const VertexerLayerName adjacentLayerIndex{layerOrder == TrackletingLayerOrder::fromInnermostToMiddleLayer ? VertexerLayerName::innermostLayer : VertexerLayerName::outerLayer};
167  const int4 selectedBinsRect{getBinsRect(currentCluster, static_cast<int>(adjacentLayerIndex), 0.f, 50.f, phiCut / 2)};
168  if (selectedBinsRect.x != 0 || selectedBinsRect.y != 0 || selectedBinsRect.z != 0 || selectedBinsRect.w != 0) {
169  int phiBinsNum{selectedBinsRect.w - selectedBinsRect.y + 1};
170  if (phiBinsNum < 0) {
171  phiBinsNum += PhiBins;
172  }
173  const int nClustersAdjacentLayer = store->getClusters()[static_cast<int>(adjacentLayerIndex)].size();
174  for (size_t iPhiBin{static_cast<size_t>(selectedBinsRect.y)}, iPhiCount{0}; (int)iPhiCount < phiBinsNum; iPhiBin = ++iPhiBin == PhiBins ? 0 : iPhiBin, iPhiCount++) {
175  const int firstBinIndex{constants::its2::getBinIndex(selectedBinsRect.x, iPhiBin)};
176  const int firstRowClusterIndex{store->getIndexTable(adjacentLayerIndex)[firstBinIndex]};
177  const int maxRowClusterIndex{store->getIndexTable(adjacentLayerIndex)[firstBinIndex + selectedBinsRect.z - selectedBinsRect.x + 1]};
178  for (int iAdjacentCluster{firstRowClusterIndex}; iAdjacentCluster < maxRowClusterIndex && iAdjacentCluster < nClustersAdjacentLayer; ++iAdjacentCluster) {
179  const Cluster& adjacentCluster = store->getClusters()[static_cast<int>(adjacentLayerIndex)][iAdjacentCluster]; // assign-constructor may be a problem, check
180  if (o2::gpu::GPUCommonMath::Abs(currentCluster.phiCoordinate - adjacentCluster.phiCoordinate) < phiCut) {
181  if (storedTracklets < store->getConfig().maxTrackletsPerCluster) {
182  if (layerOrder == TrackletingLayerOrder::fromInnermostToMiddleLayer) {
183  store->getDuplets01().emplace(stride + storedTracklets, iAdjacentCluster, currentClusterIndex, adjacentCluster, currentCluster);
184  } else {
185  store->getDuplets12().emplace(stride + storedTracklets, currentClusterIndex, iAdjacentCluster, currentCluster, adjacentCluster);
186  }
187  ++storedTracklets;
188  } else {
189  printf("debug: leaving tracklet behind\n");
190  }
191  }
192  }
193  }
194  }
195  store->getNFoundTracklets(layerOrder).emplace(currentClusterIndex, storedTracklets);
196  }
197  }
198 }

200 GPUg() void trackletSelectionKernel(
201  DeviceStoreVertexerHIP* store,
202  const unsigned char isInitRun = false,
203  const float tanLambdaCut = 0.025f,
204  const float phiCut = 0.002f)
205 {
206  const size_t nClustersMiddleLayer = store->getClusters()[1].size();
207  for (size_t currentClusterIndex = blockIdx.x * blockDim.x + threadIdx.x; currentClusterIndex < nClustersMiddleLayer; currentClusterIndex += blockDim.x * gridDim.x) {
208  const int stride{static_cast<int>(currentClusterIndex * store->getConfig().maxTrackletsPerCluster)};
209  int validTracklets{0};
210  for (int iTracklet12{0}; iTracklet12 < store->getNFoundTracklets(TrackletingLayerOrder::fromMiddleToOuterLayer)[currentClusterIndex]; ++iTracklet12) {
211  for (int iTracklet01{0}; iTracklet01 < store->getNFoundTracklets(TrackletingLayerOrder::fromInnermostToMiddleLayer)[currentClusterIndex] && validTracklets < store->getConfig().maxTrackletsPerCluster; ++iTracklet01) {
212  const float deltaTanLambda{o2::gpu::GPUCommonMath::Abs(store->getDuplets01()[stride + iTracklet01].tanLambda - store->getDuplets12()[stride + iTracklet12].tanLambda)};
213  const float deltaPhi{o2::gpu::GPUCommonMath::Abs(store->getDuplets01()[stride + iTracklet01].phiCoordinate - store->getDuplets12()[stride + iTracklet12].phiCoordinate)};
214  if (deltaTanLambda < tanLambdaCut && deltaPhi < phiCut && validTracklets != store->getConfig().maxTrackletsPerCluster) {
215  assert(store->getDuplets01()[stride + iTracklet01].secondClusterIndex == store->getDuplets12()[stride + iTracklet12].firstClusterIndex);
216  if (!isInitRun) {
217  store->getLines().emplace(store->getNExclusiveFoundLines()[currentClusterIndex] + validTracklets, store->getDuplets01()[stride + iTracklet01], store->getClusters()[0].get(), store->getClusters()[1].get());
218 #ifdef _ALLOW_DEBUG_TREES_ITS_
219  store->getDupletIndices()[0].emplace(store->getNExclusiveFoundLines()[currentClusterIndex] + validTracklets, stride + iTracklet01);
220  store->getDupletIndices()[1].emplace(store->getNExclusiveFoundLines()[currentClusterIndex] + validTracklets, stride + iTracklet12);
221 #endif
222  }
223  ++validTracklets;
224  }
225  }
226  }
227  if (isInitRun) {
228  store->getNFoundLines().emplace(currentClusterIndex, validTracklets);
229  if (validTracklets >= store->getConfig().maxTrackletsPerCluster) {
230  printf("Warning: not enough space for tracklet selection, some lines will be left behind\n");
231  }
232  }
233  }
234  if (blockIdx.x * blockDim.x + threadIdx.x == 0) {
235  // first thread I want to write an empty line after last found, as adebug flag. Might delete later
236  store->getLines().emplace(store->getNExclusiveFoundLines()[store->getClusters()[1].size() - 1] + store->getNFoundLines()[store->getClusters()[1].size() - 1]);
237  }
238 }

240 GPUg() void computeCentroidsKernel(DeviceStoreVertexerHIP* store,
241  const float pairCut)
242 {
243  const unsigned int nLines = store->getNExclusiveFoundLines()[store->getClusters()[1].size() - 1] + store->getNFoundLines()[store->getClusters()[1].size() - 1];
244  const unsigned int maxIterations{nLines * (nLines - 1) / 2};
245  for (unsigned int currentThreadIndex = blockIdx.x * blockDim.x + threadIdx.x; currentThreadIndex < maxIterations; currentThreadIndex += blockDim.x * gridDim.x) {
246  int iFirstLine = currentThreadIndex / nLines;
247  int iSecondLine = currentThreadIndex % nLines;
248  if (iSecondLine <= iFirstLine) {
249  iFirstLine = nLines - iFirstLine - 2;
250  iSecondLine = nLines - iSecondLine - 1;
251  }
252  if (Line::getDCA(store->getLines()[iFirstLine], store->getLines()[iSecondLine]) < pairCut) {
253  ClusterLinesHIP cluster{store->getLines()[iFirstLine], store->getLines()[iSecondLine]};
254  if (cluster.getVertex()[0] * cluster.getVertex()[0] + cluster.getVertex()[1] * cluster.getVertex()[1] < 1.98f * 1.98f) {
255  // printOnThread(0, "xCentr: %f, yCentr: %f \n", cluster.getVertex()[0], cluster.getVertex()[1]);
256  store->getXYCentroids().emplace(2 * currentThreadIndex, cluster.getVertex()[0]);
257  store->getXYCentroids().emplace(2 * currentThreadIndex + 1, cluster.getVertex()[1]);
258  } else {
259  // writing some data anyway outside the histogram, they will not be put in the histogram, by construction.
260  store->getXYCentroids().emplace(2 * currentThreadIndex, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[0]);
261  store->getXYCentroids().emplace(2 * currentThreadIndex + 1, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[1]);
262  }
263  } else {
264  // writing some data anyway outside the histogram, they will not be put in the histogram, by construction.
265  store->getXYCentroids().emplace(2 * currentThreadIndex, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[0]);
266  store->getXYCentroids().emplace(2 * currentThreadIndex + 1, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[1]);
267  }
268  }

271 GPUg() void computeZCentroidsKernel(DeviceStoreVertexerHIP* store,
272  const float pairCut, const int binOpeningX, const int binOpeningY)
273 {
274  const unsigned int nLines = store->getNExclusiveFoundLines()[store->getClusters()[1].size() - 1] + store->getNFoundLines()[store->getClusters()[1].size() - 1];
275  for (unsigned int currentThreadIndex = blockIdx.x * blockDim.x + threadIdx.x; currentThreadIndex < nLines; currentThreadIndex += blockDim.x * gridDim.x) {
276  // printOnThread(0, "Max X: %d Max Y %d \n", store->getTmpVertexPositionBins()[0].value, store->getTmpVertexPositionBins()[1].value);
277  if (store->getTmpVertexPositionBins()[0].value || store->getTmpVertexPositionBins()[1].value) {
278  float tmpX{store->getConfig().histConf.lowHistBoundariesXYZ[0] + store->getTmpVertexPositionBins()[0].key * store->getConfig().histConf.binSizeHistX + store->getConfig().histConf.binSizeHistX / 2};
279  int sumWX{store->getTmpVertexPositionBins()[0].value};
280  float wX{tmpX * store->getTmpVertexPositionBins()[0].value};
281  for (int iBin{o2::gpu::GPUCommonMath::Max(0, store->getTmpVertexPositionBins()[0].key - binOpeningX)}; iBin < o2::gpu::GPUCommonMath::Min(store->getTmpVertexPositionBins()[0].key + binOpeningX + 1, store->getConfig().histConf.nBinsXYZ[0] - 1); ++iBin) {
282  if (iBin != store->getTmpVertexPositionBins()[0].key) {
283  wX += (store->getConfig().histConf.lowHistBoundariesXYZ[0] + iBin * store->getConfig().histConf.binSizeHistX + store->getConfig().histConf.binSizeHistX / 2) * store->getHistogramXYZ()[0].get()[iBin];
284  sumWX += store->getHistogramXYZ()[0].get()[iBin];
285  }
286  }
287  float tmpY{store->getConfig().histConf.lowHistBoundariesXYZ[1] + store->getTmpVertexPositionBins()[1].key * store->getConfig().histConf.binSizeHistY + store->getConfig().histConf.binSizeHistY / 2};
288  int sumWY{store->getTmpVertexPositionBins()[1].value};
289  float wY{tmpY * store->getTmpVertexPositionBins()[1].value};
290  for (int iBin{o2::gpu::GPUCommonMath::Max(0, store->getTmpVertexPositionBins()[1].key - binOpeningY)}; iBin < o2::gpu::GPUCommonMath::Min(store->getTmpVertexPositionBins()[1].key + binOpeningY + 1, store->getConfig().histConf.nBinsXYZ[1] - 1); ++iBin) {
291  if (iBin != store->getTmpVertexPositionBins()[1].key) {
292  wY += (store->getConfig().histConf.lowHistBoundariesXYZ[1] + iBin * store->getConfig().histConf.binSizeHistY + store->getConfig().histConf.binSizeHistY / 2) * store->getHistogramXYZ()[1].get()[iBin];
293  sumWY += store->getHistogramXYZ()[1].get()[iBin];
294  }
295  }
296  store->getBeamPosition().emplace(0, wX / sumWX);
297  store->getBeamPosition().emplace(1, wY / sumWY);
298  float fakeBeamPoint1[3] = {store->getBeamPosition()[0], store->getBeamPosition()[1], -1}; // get two points laying at different z, to create line object
299  float fakeBeamPoint2[3] = {store->getBeamPosition()[0], store->getBeamPosition()[1], 1};
300  Line pseudoBeam{fakeBeamPoint1, fakeBeamPoint2};
301  if (Line::getDCA(store->getLines()[currentThreadIndex], pseudoBeam) < pairCut) {
302  ClusterLinesHIP cluster{store->getLines()[currentThreadIndex], pseudoBeam};
303  store->getZCentroids().emplace(currentThreadIndex, cluster.getVertex()[2]);
304  } else {
305  store->getZCentroids().emplace(currentThreadIndex, 2 * store->getConfig().histConf.lowHistBoundariesXYZ[2]);
306  }
307  }
308  }
309 }

311 GPUg() void computeVertexKernel(DeviceStoreVertexerHIP* store, const int vertIndex, const int minContributors, const int binOpeningZ)
312 {
313  for (unsigned int currentThreadIndex = blockIdx.x * blockDim.x + threadIdx.x; (int)currentThreadIndex < binOpeningZ; currentThreadIndex += blockDim.x * gridDim.x) {
314  if (currentThreadIndex == 0) {
315  if (store->getTmpVertexPositionBins()[2].value > 1 && (store->getTmpVertexPositionBins()[0].value || store->getTmpVertexPositionBins()[1].value)) {
316  float z{store->getConfig().histConf.lowHistBoundariesXYZ[2] + store->getTmpVertexPositionBins()[2].key * store->getConfig().histConf.binSizeHistZ + store->getConfig().histConf.binSizeHistZ / 2};
317  float ex{0.f};
318  float ey{0.f};
319  float ez{0.f};
320  int sumWZ{store->getTmpVertexPositionBins()[2].value};
321  float wZ{z * store->getTmpVertexPositionBins()[2].value};
322  for (int iBin{o2::gpu::GPUCommonMath::Max(0, store->getTmpVertexPositionBins()[2].key - binOpeningZ)}; iBin < o2::gpu::GPUCommonMath::Min(store->getTmpVertexPositionBins()[2].key + binOpeningZ + 1, store->getConfig().histConf.nBinsXYZ[2] - 1); ++iBin) {
323  if (iBin != store->getTmpVertexPositionBins()[2].key) {
324  wZ += (store->getConfig().histConf.lowHistBoundariesXYZ[2] + iBin * store->getConfig().histConf.binSizeHistZ + store->getConfig().histConf.binSizeHistZ / 2) * store->getHistogramXYZ()[2].get()[iBin];

328  }
329  if (sumWZ > minContributors || vertIndex == 0) {
330  store->getVertices().emplace(vertIndex, store->getBeamPosition()[0], store->getBeamPosition()[1], wZ / sumWZ, ex, ey, ez, sumWZ);
331  } else {
332  store->getVertices().emplace(vertIndex);
333  }
334  } else {
335  store->getVertices().emplace(vertIndex);
336  }
337  }
338  }

421 void VertexerTraitsHIP::computeVertices()
422 {
423  if (!mClusters[1].size()) {
424  std::cout << "\t\tno clusters on layer 1. Returning.\n";

429  size_t bufferSize = mStoreVertexerGPU.getConfig().tmpCUBBufferSize * sizeof(int);
