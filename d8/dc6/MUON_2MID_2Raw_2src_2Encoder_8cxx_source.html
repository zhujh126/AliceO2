30 void Encoder::init(const char* filename, bool perLink, int verbosity, bool debugMode)
31 {
33 
34  CrateMasks masks;
35  auto linkUniqueIds = mFEEIdConfig.getConfiguredLinkUniqueIDs();
36 
37  // Initialises the GBT link encoders
38  for (auto& linkUniqueId : linkUniqueIds) {
39  auto gbtUniqueId = mFEEIdConfig.getGBTUniqueId(linkUniqueId);
40  mGBTEncoders[gbtUniqueId].setGBTUniqueId(gbtUniqueId);
41  mGBTEncoders[gbtUniqueId].setMask(masks.getMask(gbtUniqueId));
42  mGBTIds[gbtUniqueId] = linkUniqueId;
43  }
44 
45  // Initializes the output link
46  auto ir = getOrbitIR(0);
47  mRawWriter.setVerbosity(verbosity);
48  for (uint16_t cruId = 0; cruId < 2; ++cruId) {
49  for (uint8_t epId = 0; epId < 2; ++epId) {
50  uint16_t feeId = 2 * cruId + epId;
51  mRawWriter.registerLink(feeId, cruId, raw::sUserLogicLinkID, epId, perLink ? fmt::format("{:s}_L{:d}.raw", filename, feeId) : fmt::format("{:s}.raw", filename));
52 
53  for (auto gbtUniqueId : mFEEIdConfig.getGBTUniqueIdsInLink(feeId)) {
54  // Initializes the trigger response to be added to the empty HBs
55  mGBTEncoders[gbtUniqueId].processTrigger(ir, raw::sORB);
56  mGBTEncoders[gbtUniqueId].flush(mOrbitResponse[feeId], ir);
57  }
58 
59  mOrbitResponseWord[feeId] = mOrbitResponse[feeId];
60  completeWord(mOrbitResponseWord[feeId]);
61  }
62  }
63 
64  mRawWriter.setEmptyPageCallBack(this);
65 
66  mConverter.setDebugMode(debugMode);
67 }

75 void Encoder::onOrbitChange(uint32_t orbit)
76 {
78  auto ir = getOrbitIR(orbit);

85 void Encoder::completeWord(std::vector<char>& buffer)
86 {
88  size_t dataSize = buffer.size();
89  size_t cruWord = 2 * o2::raw::RDHUtils::GBTWord;
90  size_t modulo = dataSize % cruWord;
91  if (modulo) {
92  dataSize += cruWord - modulo;
93  buffer.resize(dataSize, static_cast<char>(0));
94  }
95 }

97 void Encoder::writePayload(uint16_t feeId, const InteractionRecord& ir)
98 {
100 
101  std::vector<char> buf;
102  for (auto& gbtUniqueId : mFEEIdConfig.getGBTUniqueIdsInLink(feeId)) {
103  if (!mGBTEncoders[gbtUniqueId].isEmpty()) {
104  mGBTEncoders[gbtUniqueId].flush(buf, ir);
105  }
106  }
107  if (buf.empty()) {
108  return;
109  }
110 
111  // Add the orbit response
112  buf.insert(buf.begin(), mOrbitResponse[feeId].begin(), mOrbitResponse[feeId].end());
113  completeWord(buf);
114  mRawWriter.addData(feeId, feeId / 2, raw::sUserLogicLinkID, feeId % 2, ir, buf);
115 }

117 void Encoder::finalize(bool closeFile)
118 {
120  if (mLastIR.isDummy()) {
121  mLastIR.bc = 0;
122  mLastIR.orbit = mRawWriter.getHBFUtils().orbitFirst;
123  }
124  auto ir = getOrbitIR(mLastIR.orbit);
125  auto nextIr = getOrbitIR(mLastIR.orbit + 1);
126  for (uint16_t feeId = 0; feeId < 4; ++feeId) {
127  auto ir = getOrbitIR(mLastIR.orbit);
128  // Write the last payload
129  writePayload(feeId, ir);
130  // Since the regional response comes after few clocks,
131  // we might have the corresponding regional cards in the next orbit.
132  // If this is the case, we flush all data of the next orbit
133  writePayload(feeId, nextIr);
134  }
135  if (closeFile) {
136  mRawWriter.close();
137  }
138 }

140 void Encoder::process(gsl::span<const ColumnData> data, const InteractionRecord& ir, EventType eventType)
141 {
143  if (ir.orbit != mLastIR.orbit) {
144  onOrbitChange(mLastIR.orbit);
145  }
146 
147  mConverter.process(data);
148 
149  for (auto& item : mConverter.getData()) {
150  mGBTEncoders[item.first].process(item.second, ir);
151  }
152  mLastIR = ir;
153 }
