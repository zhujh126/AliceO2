65 template <typename T>
66 SymbolTable<T>::SymbolTable(const SymbolStatistics& symbolStats) : mMin{symbolStats.getMinSymbol()}
67 {
68  LOG(trace) << "start building symbol table";
69 
70  mIndex.reserve(symbolStats.size());
71  mSymbols.reserve(symbolStats.getNUsedAlphabetSymbols());
72 
73  mEscapeSymbol = [&]() -> std::unique_ptr<T> {
74  const auto [symFrequency, symCumulated] = symbolStats.getEscapeSymbol();
75  return std::make_unique<T>(symFrequency, symCumulated, symbolStats.getSymbolTablePrecision());
76  }();
77 
78  for (auto it = symbolStats.begin(); it != --symbolStats.end(); ++it) {
79  const auto [symFrequency, symCumulated] = *it;
80  if (symFrequency) {
81  mSymbols.emplace_back(symFrequency, symCumulated, symbolStats.getSymbolTablePrecision());
82  mIndex.emplace_back(&mSymbols.back());
83  } else {
84  mIndex.emplace_back(mEscapeSymbol.get());
85  }
86  }
87 
88  //escape symbol
89  mIndex.emplace_back(mEscapeSymbol.get());
90 
91 // advanced diagnostics for debug builds
92 #if !defined(NDEBUG)
93  LOG(debug2) << "SymbolTableProperties: {"
94  << "entries:" << mSymbols.size() << ", "
95  << "sizeB: " << mSymbols.size() * sizeof(T) + mIndex.size() * sizeof(T*) << "}";
96 #endif
97 
98  LOG(trace) << "done building symbol table";
99 }
