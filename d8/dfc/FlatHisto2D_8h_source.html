34 /*
35  Fast 2D histo class which can be messages as
36  FlatHisto2D<float> histo(nbinsX, xmin, xmax, nbinsY, ymin, ymax);
37  histo.fill(...);
38  pc.outputs().snapshot(Output{"Origin", "Desc", 0, Lifetime::Timeframe}, histo.getBase());
39 
40  and received (read only!) as
41  const auto hdata = pc.inputs().get<gsl::span<float>>("histodata");
42  FlatHisto2D<float> histoView;
43  histoView.adoptExternal(hdata);
44  or directly
45  FlatHisto2D<float> histoView(pc.inputs().get<gsl::span<float>>("histodata"));
46 */

164  int fill(T x, T y)
165  {
166  uint32_t bin = getBin(x, y);
167  if (isValidBin(bin)) {
168  mDataPtr[bin]++;
169  return bin;
170  }
171  return -1;
172  }
173 
174  int fill(T x, T y, T w)
175  {
176  uint32_t bin = getBin(x, y);
177  if (isValidBin(bin)) {
178  mDataPtr[bin] += w;
179  return bin;
180  }
181  return -1;
182  }

184  void fillBin(uint32_t bin, T w)
185  {
186  if (isValidBin(bin)) {
187  mDataPtr[bin] += w;
188  }
189  }

191  void fillBin(uint32_t bx, uint32_t by, T w)
192  {
193  auto bin = getGlobalBin(bx, by);
194  if (isValidBin(bin)) {
195  mDataPtr[bin] += w;
196  }
197  }

199  uint32_t getBinX(T x) const
200  {
201  auto dx = x - getXMin();
202  return dx < 0 ? 0xffffffff : uint32_t(dx * getBinSizeXInv());
203  }
204 
205  uint32_t getBinY(T y) const
206  {
207  auto dy = y - getYMin();
208  return dy < 0 ? 0xffffffff : uint32_t(dy * getBinSizeYInv());
209  }
210 
211  uint32_t getBin(T x, T y) const
212  {
213  auto bx = getBinX(x), by = getBinY(y);
214  return bx < getNBinsX() && by < getNBinsY() ? getGlobalBin(bx, by) : 0xffffffff;
215  }
216 
217  bool canFill() const
218  {
219  // histo can be filled only if hase its own data, otherwise only query can be done on the view
220  return mContainer.size() > NServiceSlots;
221  }

223  gsl::span<const T> getSliceY(uint32_t binX) const

225  int offs = binX * getNBinsY();
226  return binX < getNBinsX() ? gsl::span<const T>(&mDataPtr[offs], getNBinsY()) : gsl::span<const T>();
227  }
