67 std::vector<MetricIndices> createDefaultIndices(std::vector<DeviceMetricsInfo>& allDevicesMetrics)

224  totalBytesDestroyedMetric(driverMetrics, totalBytesDestroyed, timestamp);

230  }

232  static int stateTransitions = 0;
233  static int signalsCount = 0;
234  static int skippedCount = 0;
235  static uint64_t now = 0;
236  now = uv_hrtime();
237  static RateLimitingState lastReportedState = RateLimitingState::UNKNOWN;
238  static uint64_t lastReportTime = 0;
239  static int64_t MAX_SHARED_MEMORY = calculateAvailableSharedMemory(registry);
240  constexpr int64_t MAX_QUANTUM_SHARED_MEMORY = 300;
241  constexpr int64_t MIN_QUANTUM_SHARED_MEMORY = 50;
242 
243  static int64_t availableSharedMemory = MAX_SHARED_MEMORY;
244  static int64_t offeredSharedMemory = 0;
245  static int64_t lastDeviceOffered = 0;
248  int64_t lastCandidate = -1;
249  static int enoughSharedMemoryCount = availableSharedMemory - MIN_QUANTUM_SHARED_MEMORY > 0 ? 1 : 0;
250  static int lowSharedMemoryCount = availableSharedMemory - MIN_QUANTUM_SHARED_MEMORY > 0 ? 0 : 1;
251  int64_t possibleOffer = MIN_QUANTUM_SHARED_MEMORY;
252  for (size_t di = 0; di < specs.size(); di++) {
253  if (availableSharedMemory < possibleOffer) {
254  if (lowSharedMemoryCount == 0) {
255  LOGP(INFO, "We do not have enough shared memory ({}MB) to offer {}MB", availableSharedMemory, possibleOffer);
256  }
257  lowSharedMemoryCount++;
258  enoughSharedMemoryCount = 0;
259  break;
260  } else {
261  if (enoughSharedMemoryCount == 0) {
262  LOGP(INFO, "We are back in a state where we enough shared memory: {}MB", availableSharedMemory);
263  }
264  enoughSharedMemoryCount++;
265  lowSharedMemoryCount = 0;
266  }
267  size_t candidate = (lastDeviceOffered + di) % specs.size();
268 
269  auto& spec = specs[candidate];
270  auto& info = infos[candidate];
271  // Do not bother for inactive devices
272  // FIXME: there is probably a race condition if the device died and we did not
273  // took notice yet...
274  if (info.active == false || info.readyToQuit) {
275  continue;
276  }
277  if (specs[candidate].name != "internal-dpl-aod-reader") {
278  continue;
279  }
280  possibleOffer = std::min(MAX_QUANTUM_SHARED_MEMORY, availableSharedMemory);
281  LOGP(info, "Offering {}MB out of {} to {}", possibleOffer, availableSharedMemory, specs[candidate].id);
282  manager.queueMessage(specs[candidate].id.c_str(), fmt::format("/shm-offer {}", possibleOffer).data());
283  availableSharedMemory -= possibleOffer;
284  offeredSharedMemory += possibleOffer;
285  lastCandidate = candidate;
286  }
287  // We had at least a valid candidate, so
288  // next time we offer to the next device.
289  if (lastCandidate >= 0) {

291  }

294  // not used so far. So we need to account for the amount which got actually read (readerBytesCreated)
295  // and the amount which we know was given back.

297  static int64_t lastUnusedOfferedMemory = 0;
298  if (shmOfferConsumed != lastShmOfferConsumed) {
299  LOGP(INFO, "Offer consumed so far {}", shmOfferConsumed);
300  lastShmOfferConsumed = shmOfferConsumed;
301  }

303  if (lastUnusedOfferedMemory != unusedOfferedMemory) {
304  LOGP(INFO, "unusedOfferedMemory:{} = offered:{} - (expired:{} + consumed:{}) / 1000000", unusedOfferedMemory, offeredSharedMemory, totalBytesExpired / 1000000, shmOfferConsumed / 1000000);
305  lastUnusedOfferedMemory = unusedOfferedMemory;
306  }
307  // availableSharedMemory is the amount of memory which we know is available to be offered.
308  // We subtract the amount which we know was already offered but it's unused and we then balance how
309  // much was created with how much was destroyed.
310  availableSharedMemory = MAX_SHARED_MEMORY + ((totalBytesDestroyed - totalBytesCreated) / 1000000) - unusedOfferedMemory;
311  availableSharedMemoryMetric(driverMetrics, availableSharedMemory, timestamp);
312  unusedOfferedMemoryMetric(driverMetrics, unusedOfferedMemory, timestamp);
313 
314  offeredSharedMemoryMetric(driverMetrics, offeredSharedMemory, timestamp);
315  },
316  [](ProcessingContext& ctx, void* service) {
317  using DataHeader = o2::header::DataHeader;
318  ArrowContext* arrow = reinterpret_cast<ArrowContext*>(service);
319  auto totalBytes = 0;
320  auto totalMessages = 0;
321  for (auto& input : ctx.inputs()) {
322  if (input.header == nullptr) {
323  continue;
324  }
325  auto dh = o2::header::get<DataHeader*>(input.header);
326  if (dh->serialization != o2::header::gSerializationMethodArrow) {
327  LOGP(DEBUG, "Message {}/{} is not of kind arrow, therefore we are not accounting its shared memory", dh->dataOrigin, dh->dataDescription);
328  continue;
329  }
330  auto dph = o2::header::get<DataProcessingHeader*>(input.header);
331  bool forwarded = false;
332  for (auto const& forward : ctx.services().get<DeviceSpec const>().forwards) {
333  if (DataSpecUtils::match(forward.matcher, dh->dataOrigin, dh->dataDescription, dh->subSpecification)) {
334  forwarded = true;
335  break;
336  }
337  }
338  if (forwarded) {
339  LOGP(DEBUG, "Message {}/{} is forwarded so we are not returning its memory.", dh->dataOrigin, dh->dataDescription);
340  continue;
341  }
342  LOGP(DEBUG, "Message {}/{} is being deleted. We will return {}MB.", dh->dataOrigin, dh->dataDescription, dh->payloadSize / 1000000.);
343  totalBytes += dh->payloadSize;
344  totalMessages += 1;
345  }
346  arrow->updateBytesDestroyed(totalBytes);
347  LOGP(DEBUG, "{}MB bytes being given back to reader, totaling {}MB", totalBytes / 1000000., arrow->bytesDestroyed() / 1000000.);
348  arrow->updateMessagesDestroyed(totalMessages);
349  auto& monitoring = ctx.services().get<Monitoring>();
350  monitoring.send(Metric{(uint64_t)arrow->bytesDestroyed(), "arrow-bytes-destroyed"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
351  monitoring.send(Metric{(uint64_t)arrow->messagesDestroyed(), "arrow-messages-destroyed"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
352  monitoring.flushBuffer();
353  },
354  nullptr,
355  nullptr,
356  [](ServiceRegistry& registry, boost::program_options::variables_map const& vm) {
