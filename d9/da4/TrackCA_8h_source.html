79 inline void TrackLTF::setPoint(const Cluster& cl, const Int_t layer, const Int_t clusterId, const MCCompLabel label, const Int_t extClsIndex)
80 {
81  auto nPoints = getNumberOfPoints();
82  if (nPoints > 0) {
83  if (mZ[nPoints - 1] == cl.getZ()) {
84  LOG(WARN) << "MFT TrackLTF: skipping setPoint (1 cluster per layer!)";
85  return;
86  }
87  }
88  if (nPoints > constants::mft::LayersNumber) {
89  LOG(WARN) << "MFT TrackLTF Overflow";
90  return;
91  }
92  mX[nPoints] = cl.getX();
93  mY[nPoints] = cl.getY();
94  mZ[nPoints] = cl.getZ();
95  mSigmaX2[nPoints] = cl.sigmaX2;
96  mSigmaY2[nPoints] = cl.sigmaY2;
97  mLayer[nPoints] = layer;
98  mClusterId[nPoints] = clusterId;
99  mMCCompLabels[nPoints] = label;
100  setExternalClusterIndex(nPoints, extClsIndex);
101  setNumberOfPoints(nPoints + 1);
102 }

105 inline void TrackLTF::sort()
106 {
107  // Orders elements along z position
108  struct ClusterData {
109  Float_t x;
110  Float_t y;
111  Float_t z;
112  Float_t sigmaX2;
113  Float_t sigmaY2;
114  Int_t layer;
115  Int_t clusterId;
116  MCCompLabel label;
117  Int_t extClsIndex;
118  };
119  std::vector<ClusterData> points;
120 
121  // Loading cluster data
122  for (Int_t point = 0; point < getNumberOfPoints(); ++point) {
123  auto& somepoint = points.emplace_back();
124  somepoint.x = mX[point];
125  somepoint.y = mY[point];
126  somepoint.z = mZ[point];
127  somepoint.sigmaX2 = mSigmaX2[point];
128  somepoint.sigmaY2 = mSigmaY2[point];
129  somepoint.layer = mLayer[point];
130  somepoint.clusterId = mClusterId[point];
131  somepoint.label = mMCCompLabels[point];
132  somepoint.extClsIndex = mExtClsIndex[point];
133  }
134 
135  // Sorting cluster data
136  std::sort(points.begin(), points.end(), [](ClusterData a, ClusterData b) { return a.z > b.z; });
137 
138  // Storing sorted cluster data
139  for (Int_t point = 0; point < getNumberOfPoints(); ++point) {
140  mX[point] = points[point].x;
141  mY[point] = points[point].y;
142  mZ[point] = points[point].z;
143  mSigmaX2[point] = points[point].sigmaX2;
144  mSigmaY2[point] = points[point].sigmaY2;
145  mLayer[point] = points[point].layer;
146  mClusterId[point] = points[point].clusterId;
147  mMCCompLabels[point] = points[point].label;
148  mExtClsIndex[point] = points[point].extClsIndex;
149  }
150 }
