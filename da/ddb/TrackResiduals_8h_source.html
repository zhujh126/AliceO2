662 //_____________________________________________________
663 inline float TrackResiduals::getDXI(int ix) const
664 {
665  if (mUniformBins[VoxX]) {
666  return mDXI;
667  } else {
668  if (ix < param::NRowsPerROC[0]) {
669  // we are in the IROC
670  return 1.f / param::RowDX[0];
671  } else if (ix > param::NRowsAccumulated[param::NROCTypes - 1]) {
672  // we are in the last OROC
673  return 1.f / param::RowDX[param::NROCTypes - 1];
674  }
675 #ifdef TPC_RUN2
676  else {
677  // we are in OROC1
678  return 1.f / param::RowDX[1];
679  }
680 #else
681  else if (ix < param::NRowsAccumulated[2]) {
682  // OROC1
683  return 1.f / param::RowDX[1];
684  } else {
685  // OROC2
686  return 1.f / param::RowDX[2];
687  }
688 #endif
689  }
690 }

698 //_____________________________________________________
699 inline float TrackResiduals::getY2X(int ix, int ip) const
700 {
701  if (mUniformBins[VoxF]) {
702  return (0.5f + ip) * mDY2X[ix] - mMaxY2X[ix];
703  }
704  return mMaxY2X[ix] * (mY2XBinsCenter[ip] - mY2XBinsDH[ip]);
705 }

709 {
710  if (mUniformBins[VoxZ]) {
711  return (0.5f + iz) * getDZ2X();
712  }
713  return mZ2XBinsCenter[iz];
714 }

721  }
722  return mY2XBinsDI[iy] / mMaxY2X[ix];

751 //_____________________________________________________
752 inline int TrackResiduals::getXBinExact(float x) const
753 {
754  if (mUniformBins[VoxX]) {
755  int ix = (x - param::MinX) * mDXI;
756  return (ix < 0 || ix >= mNXBins) ? -1 : ix;
757  }
758  return getRowID(x);
759 }

761 //_____________________________________________________
762 inline int TrackResiduals::getXBin(float x) const
763 {
764  int bx = getXBinExact(x);
765  return bx > -1 ? (bx < mNXBins ? bx : mNXBins - 1) : 0;
766 }
767 
768 //_____________________________________________________
769 inline int TrackResiduals::getY2XBinExact(float y2x, int ix) const
770 {
771  if (y2x < -mMaxY2X[ix]) {
772  return -1;
773  }
774  if (y2x > mMaxY2X[ix]) {
775  return mNY2XBins;
776  }
777  if (mUniformBins[VoxF]) {
778  return static_cast<int>((y2x + mMaxY2X[ix]) * getDY2XI(ix));
779  }
780  y2x /= mMaxY2X[ix];
781  for (int iBin = 0; iBin < mNY2XBins; ++iBin) {
782  if (y2x < mY2XBinsCenter[iBin] + mY2XBinsDH[iBin]) {
783  return iBin;
784  }
785  }
786  return mNY2XBins;
787 }

789 //_____________________________________________________
790 inline int TrackResiduals::getY2XBin(float y2x, int ix) const
791 {
792  int bp = getY2XBinExact(y2x, ix);
793  return bp > -1 ? (bp < mNY2XBins ? bp : mNY2XBins - 1) : 0;
794 }
795 
796 //_____________________________________________________
797 inline int TrackResiduals::getZ2XBinExact(float z2x) const
798 {
799  if (mUniformBins[VoxZ]) {
800  float bz = z2x * getDZ2XI();
801  if (bz >= mNZ2XBins) {
802  return -1;
803  }
804  if (bz < 0) {
