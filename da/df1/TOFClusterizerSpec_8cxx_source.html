47 // use the tasking system of DPL
48 // just need to implement 2 special methods init + run (there is no need to inherit from anything)
49 class TOFDPLClustererTask
50 {
51  bool mUseMC = true;
52  bool mUseCCDB = false;
53  bool mIsCalib = false;
54  bool mIsCosmic = false;
55  int mTimeWin = 5000;
56 
57  public:
58  explicit TOFDPLClustererTask(bool useMC, bool useCCDB, bool doCalib, bool isCosmic) : mUseMC(useMC), mUseCCDB(useCCDB), mIsCalib(doCalib), mIsCosmic(isCosmic) {}
59  void init(framework::InitContext& ic)
60  {
61  // nothing special to be set up
62  mTimer.Stop();
63  mTimer.Reset();
64 
65  mTimeWin = ic.options().get<int>("cluster-time-window");
66  LOG(INFO) << "Is calibration from cluster on? " << mIsCalib;
67  LOG(INFO) << "DeltaTime for clusterization = " << mTimeWin << " ps";
68  LOG(INFO) << "Is cosmics? " << mIsCosmic;
69 
70  mClusterer.setCalibFromCluster(mIsCalib);
71  mClusterer.setDeltaTforClustering(mTimeWin);
72  }
73 
74  void run(framework::ProcessingContext& pc)
75  {
76  mTimer.Start(false);
77  // get digit data
78  auto digits = pc.inputs().get<gsl::span<o2::tof::Digit>>("tofdigits");
79  auto row = pc.inputs().get<gsl::span<o2::tof::ReadoutWindowData>>("readoutwin");
80 
81  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().getFirstValid(true).header);
82  mClusterer.setFirstOrbit(dh->firstTForbit);
83 
84  auto labelvector = std::make_shared<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>>();
85  if (mUseMC) {
86  auto digitlabels = pc.inputs().get<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>*>("tofdigitlabels");
87  *labelvector.get() = std::move(*digitlabels);
88  mClusterer.setMCTruthContainer(&mClsLabels);
89  mClsLabels.clear();
90  }
91 
92  o2::dataformats::CalibLHCphaseTOF lhcPhaseObj;
93  auto channelCalibObj = std::make_unique<o2::dataformats::CalibTimeSlewingParamTOF>();
94 
95  if (mUseCCDB) { // read calibration objects from ccdb
96  // check LHC phase
97  auto lhcPhase = pc.inputs().get<o2::dataformats::CalibLHCphaseTOF*>("tofccdbLHCphase");
98  auto channelCalib = pc.inputs().get<o2::dataformats::CalibTimeSlewingParamTOF*>("tofccdbChannelCalib");
99 
100  o2::dataformats::CalibLHCphaseTOF lhcPhaseObjTmp = std::move(*lhcPhase);
101  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObjTmp = std::move(*channelCalib);
102 
103  // make a copy in global scope
104  lhcPhaseObj = lhcPhaseObjTmp;
105  *channelCalibObj = channelCalibObjTmp;
106  } else { // calibration objects set to zero
107  lhcPhaseObj.addLHCphase(0, 0);
108  lhcPhaseObj.addLHCphase(2000000000, 0);
109 
110  for (int ich = 0; ich < o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELS; ich++) {
111  channelCalibObj->addTimeSlewingInfo(ich, 0, 0);
112  int sector = ich / o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
113  int channelInSector = ich % o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
114  channelCalibObj->setFractionUnderPeak(sector, channelInSector, 1);
115  }
116  }
117 
118  o2::tof::CalibTOFapi calibapi(long(0), &lhcPhaseObj, channelCalibObj.get());
119 
120  mClusterer.setCalibApi(&calibapi);
121 
122  // call actual clustering routine
123  mClustersArray.clear();
124  if (mIsCalib) {
125  mClusterer.getInfoFromCluster()->clear();
126  }
127 
128  if (mIsCosmic) {
129  mCosmicProcessor.clear();
130  }
131 
132  for (int i = 0; i < row.size(); i++) {
133  //printf("# TOF readout window for clusterization = %d/%lu (N digits = %d)\n", i, row.size(), row[i].size());
134  auto digitsRO = row[i].getBunchChannelData(digits);
135  mReader.setDigitArray(&digitsRO);
136 
137  if (mIsCosmic) {
138  mCosmicProcessor.process(mReader, i != 0);
139  }
140 
141  if (mUseMC) {
142  mClusterer.process(mReader, mClustersArray, &(labelvector->at(i)));
143  } else {
144  mClusterer.process(mReader, mClustersArray, nullptr);
145  }
146  }
147  LOG(DEBUG) << "TOF CLUSTERER : TRANSFORMED " << digits.size()
148  << " DIGITS TO " << mClustersArray.size() << " CLUSTERS";
149 
150  // send clusters
151  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe}, mClustersArray);
152  // send labels
153  if (mUseMC) {
154  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe}, mClsLabels);
155  }
156 
157  if (mIsCalib) {
158  std::vector<CalibInfoCluster>* clusterCalInfo = mClusterer.getInfoFromCluster();
159  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "INFOCALCLUS", 0, Lifetime::Timeframe}, *clusterCalInfo);
160  }
161 
162  if (mIsCosmic) {
163  std::vector<CosmicInfo>* cosmicInfo = mCosmicProcessor.getCosmicInfo();
164  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "INFOCOSMICS", 0, Lifetime::Timeframe}, *cosmicInfo);
165  std::vector<CalibInfoTrackCl>* cosmicTrack = mCosmicProcessor.getCosmicTrack();
166  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "INFOTRACKCOS", 0, Lifetime::Timeframe}, *cosmicTrack);
167  std::vector<int>* cosmicTrackSize = mCosmicProcessor.getCosmicTrackSize();
168  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "INFOTRACKSIZE", 0, Lifetime::Timeframe}, *cosmicTrackSize);
169  }
170 
171  mTimer.Stop();
172  }
173 
174  void endOfStream(EndOfStreamContext& ec)
175  {
176  LOGF(INFO, "TOF Clusterer total timing: Cpu: %.3e Real: %.3e s in %d slots",
177  mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1);
178  }
179 
180  private:
181  DigitDataReader mReader;
182  Clusterer mClusterer;
183  CosmicProcessor mCosmicProcessor;
184  TStopwatch mTimer;
185 
186  std::vector<Cluster> mClustersArray;
187  MCLabelContainer mClsLabels;
188  std::vector<CalibInfoCluster> mClusterCalInfo;
189 };

191 o2::framework::DataProcessorSpec getTOFClusterizerSpec(bool useMC, bool useCCDB, bool doCalib, bool isCosmic)
192 {
193  std::vector<InputSpec> inputs;
194  inputs.emplace_back("tofdigits", o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe);
195  inputs.emplace_back("readoutwin", o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe);
196  if (useCCDB) {
197  inputs.emplace_back("tofccdbLHCphase", o2::header::gDataOriginTOF, "LHCphase");
198  inputs.emplace_back("tofccdbChannelCalib", o2::header::gDataOriginTOF, "ChannelCalib");
199  }
200  if (useMC) {
201  inputs.emplace_back("tofdigitlabels", o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe);
202  }
203 
204  std::vector<OutputSpec> outputs;
205  outputs.emplace_back(o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe);
206  if (useMC) {
207  outputs.emplace_back(o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe);
208  }
209 
210  if (doCalib) {
211  outputs.emplace_back(o2::header::gDataOriginTOF, "INFOCALCLUS", 0, Lifetime::Timeframe);
212  }
213  if (isCosmic) {
214  outputs.emplace_back(o2::header::gDataOriginTOF, "INFOCOSMICS", 0, Lifetime::Timeframe);
215  outputs.emplace_back(o2::header::gDataOriginTOF, "INFOTRACKCOS", 0, Lifetime::Timeframe);
216  outputs.emplace_back(o2::header::gDataOriginTOF, "INFOTRACKSIZE", 0, Lifetime::Timeframe);
217  }
218 
219  return DataProcessorSpec{
220  "TOFClusterer",
221  inputs,
222  outputs,
223  AlgorithmSpec{adaptFromTask<TOFDPLClustererTask>(useMC, useCCDB, doCalib, isCosmic)},
224  Options{{"cluster-time-window", VariantType::Int, 5000, {"time window for clusterization in ps"}}}};
225 }
