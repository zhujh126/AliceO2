37 template <typename RawReaderType>
38 class FITDataReaderDPLSpec : public Task
39 {
40  public:
41  FITDataReaderDPLSpec(const RawReaderType& rawReader) : mRawReader(rawReader) {}
42  FITDataReaderDPLSpec() = delete;
43  ~FITDataReaderDPLSpec() override = default;
44  typedef RawReaderType RawReader_t;
45  RawReader_t mRawReader;
46  void init(InitContext& ic) final { RawReader_t::LookupTable_t::Instance().printFullMap(); }
47  void run(ProcessingContext& pc) final
48  {
49  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
50  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
51  {
52  std::vector<InputSpec> dummy{InputSpec{"dummy", ConcreteDataMatcher{mRawReader.mDataOrigin, o2::header::gDataDescriptionRawData, 0xDEADBEEF}}};
53  for (const auto& ref : InputRecordWalker(pc.inputs(), dummy)) {
54  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
55  if (dh->payloadSize == 0) {
56  LOGP(WARNING, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF",
57  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize);
58  mRawReader.makeSnapshot(pc); // send empty output
59  return;
60  }
61  }
62  }
63  std::vector<InputSpec> filter{InputSpec{"filter", ConcreteDataTypeMatcher{mRawReader.mDataOrigin, o2::header::gDataDescriptionRawData}, Lifetime::Timeframe}};
64  DPLRawParser parser(pc.inputs(), filter);
65  std::size_t count = 0;
66  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
67  //Proccessing each page
68  count++;
69  auto rdhPtr = it.get_if<o2::header::RAWDataHeader>();
70  gsl::span<const uint8_t> payload(it.data(), it.size());
71  mRawReader.process(payload, int(rdhPtr->linkID), int(rdhPtr->endPointID));
72  }
73  LOG(INFO) << "Pages: " << count;
74  mRawReader.accumulateDigits();
75  mRawReader.makeSnapshot(pc);
76  mRawReader.clear();
77  }
78 };

80 template <typename RawReaderType>
81 framework::DataProcessorSpec getFITDataReaderDPLSpec(const RawReaderType& rawReader, bool askSTFDist)
82 {
83  std::vector<OutputSpec> outputSpec;
84  rawReader.configureOutputSpec(outputSpec);
85  std::vector<InputSpec> inputSpec{{"STF", ConcreteDataTypeMatcher{rawReader.mDataOrigin, "RAWDATA"}, Lifetime::Optional}};
86  if (askSTFDist) {
87  inputSpec.emplace_back("STFDist", "FLP", "DISTSUBTIMEFRAME", 0, Lifetime::Timeframe);
88  }
89  std::string dataProcName = rawReader.mDataOrigin.template as<std::string>();
90  std::for_each(dataProcName.begin(), dataProcName.end(), [](char& c) { c = ::tolower(c); });
91  dataProcName += "-datareader-dpl";
92  LOG(INFO) << dataProcName;
93  return DataProcessorSpec{
94  dataProcName,
95  inputSpec,
96  outputSpec,
97  adaptFromTask<FITDataReaderDPLSpec<RawReaderType>>(rawReader),
98  Options{}};
99 }
