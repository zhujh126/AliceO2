72  {

211 struct IndexExclusive {
213  template <typename... Cs, typename Key, typename T1, typename... T>
214  static auto indexBuilder(const char* label, framework::pack<Cs...>, Key const&, std::tuple<T1, T...> tables)
215  {
216  static_assert(sizeof...(Cs) == sizeof...(T) + 1, "Number of columns does not coincide with number of supplied tables");
217  using tables_t = framework::pack<T...>;
218  using first_t = T1;
219  auto tail = tuple_tail(tables);
220  TableBuilder builder;
221  auto cursor = framework::FFL(builder.cursor<o2::soa::Table<Cs...>>());
222 
223  std::array<int32_t, sizeof...(T)> values;
224  iterator_tuple_t<std::decay_t<T>...> iterators = std::apply(
225  [](auto&&... x) {
226  return std::make_tuple(x.begin()...);
227  },
228  tail);
229 
230  using rest_it_t = decltype(pack_from_tuple(iterators));
231 
232  int32_t idx = -1;
233  auto setValue = [&](auto& x) -> bool {
234  using type = std::decay_t<decltype(x)>;
235  constexpr auto position = framework::has_type_at_v<type>(rest_it_t{});
236 
237  lowerBound<Key>(idx, x);
238  if (x == soa::RowViewSentinel{static_cast<uint64_t>(x.mMaxRow)}) {
239  return false;
240  } else if (x.template getId<Key>() != idx) {
241  return false;
242  } else {
243  values[position] = x.globalIndex();
244  ++x;
245  return true;
246  }
247  };
248 
249  auto first = std::get<first_t>(tables);
250  for (auto& row : first) {
251  idx = row.template getId<Key>();
252 
253  if (std::apply(
254  [](auto&... x) {
255  return ((x == soa::RowViewSentinel{static_cast<uint64_t>(x.mMaxRow)}) && ...);
256  },
257  iterators)) {
258  break;
259  }
260 
261  auto result = std::apply(
262  [&](auto&... x) {
263  std::array<bool, sizeof...(T)> results{setValue(x)...};
264  return (results[framework::has_type_at_v<std::decay_t<decltype(x)>>(rest_it_t{})] && ...);
265  },
266  iterators);
267 
268  if (result) {
269  cursor(0, row.globalIndex(), values[framework::has_type_at_v<T>(tables_t{})]...);
270  }
271  }
272  builder.setLabel(label);
273  return builder.finalize();
274  }
275 };
278 struct IndexSparse {
279  template <typename... Cs, typename Key, typename T1, typename... T>
280  static auto indexBuilder(const char* label, framework::pack<Cs...>, Key const&, std::tuple<T1, T...> tables)
281  {
282  static_assert(sizeof...(Cs) == sizeof...(T) + 1, "Number of columns does not coincide with number of supplied tables");
283  using tables_t = framework::pack<T...>;
284  using first_t = T1;
285  auto tail = tuple_tail(tables);
286  TableBuilder builder;
287  auto cursor = framework::FFL(builder.cursor<o2::soa::Table<Cs...>>());
288 
289  std::array<int32_t, sizeof...(T)> values;
290 
291  iterator_tuple_t<std::decay_t<T>...> iterators = std::apply(
292  [](auto&&... x) {
293  return std::make_tuple(x.begin()...);
294  },
295  tail);
296 
297  using rest_it_t = decltype(pack_from_tuple(iterators));
298 
299  auto setValue = [&](auto& x, int idx) -> bool {
300  using type = std::decay_t<decltype(x)>;
301  constexpr auto position = framework::has_type_at_v<type>(rest_it_t{});
302 
303  if constexpr (std::is_same_v<framework::pack_element_t<position, framework::pack<std::decay_t<T>...>>, Key>) {
304  values[position] = idx;
305  return true;
306  } else {
307  lowerBound<Key>(idx, x);
308  if (x == soa::RowViewSentinel{static_cast<uint64_t>(x.mMaxRow)}) {
309  values[position] = -1;
310  return false;
311  } else if (x.template getId<Key>() != idx) {
312  values[position] = -1;
313  return false;
314  } else {
315  values[position] = x.globalIndex();
316  ++x;
317  return true;
318  }
319  }
320  };
321 
322  auto first = std::get<first_t>(tables);
323  for (auto& row : first) {
324  auto idx = -1;
325  if constexpr (std::is_same_v<first_t, Key>) {
326  idx = row.globalIndex();
327  } else {
328  idx = row.template getId<Key>();
329  }
330  std::apply(
331  [&](auto&... x) {
332  (setValue(x, idx), ...);
333  },
334  iterators);
335 
336  cursor(0, row.globalIndex(), values[framework::has_type_at_v<T>(tables_t{})]...);
337  }
338  builder.setLabel(label);
339  return builder.finalize();
340  }
341 };

499  throw std::runtime_error("Partition filter does not match declared table type");

501

503  return new o2::soa::Filtered<T>{{table}, tree};
504  } else {
505  return new o2::soa::Filtered<T>{{table.asArrowTable()}, tree};
506  }
