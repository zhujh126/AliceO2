57 void SimParam::reInit()
58 {
59  //
60  // Reinitializes the parameter class after a change of the gas mixture
61  //
62 
63  if (CommonParam::instance()->isXenon()) {
64  // The range and the binwidth for the sampled TRF
65  mTRFbin = 200;
66  // Start 0.2 mus before the signal
67  mTRFlo = -0.4;
68  // End the maximum drift time after the signal
69  mTRFhi = 3.58;
70  // Standard gas gain
71  mGasGain = 4000.0;
72  } else if (CommonParam::instance()->isArgon()) {
73  // The range and the binwidth for the sampled TRF
74  mTRFbin = 50;
75  // Start 0.2 mus before the signal
76  mTRFlo = 0.02;
77  // End the maximum drift time after the signal
78  mTRFhi = 1.98;
79  // Higher gas gain
80  mGasGain = 8000.0;
81  } else {
82  LOG(FATAL) << "Not a valid gas mixture!\n";
83  }
84  mInvTRFwid = ((float)mTRFbin) / (mTRFhi - mTRFlo); // Inverse of the bin width of the integrated TRF
85 
86  // Create the sampled TRF
87  sampleTRF();
88 }
