51  void init(o2::framework::InitContext& ic) final
52  {
53  mLaneId = ic.services().get<const o2::framework::DeviceSpec>().rank;
54  mDBapi.init(ic.options().get<std::string>("ccdb-uri")); // or http://localhost:8080 for a local installation
55  mWriteToDB = mDBapi.isHostReachable() ? true : false;
56  mUpdateGroupingPar = mLaneId == 0 ? !(ic.options().get<bool>("update-not-grouping-parameter")) : false;
57 
58  // write struct containing grouping parameters to access grouped IDCs to CCDB
59  if (mWriteToDB && mUpdateGroupingPar) {
60  // validity for grouping parameters is from first TF to some really large TF (until it is updated) TODO do somewhere else?!
61  mDBapi.storeAsTFileAny<o2::tpc::ParameterIDCGroupCCDB>(&mIDCFactorization.getGroupingParameter(), "TPC/Calib/IDC/GROUPINGPAR", mMetadata, getFirstTF(), std::numeric_limits<uint32_t>::max());
62  mUpdateGroupingPar = false; // write grouping parameters only once
63  }
64  }

66  void run(o2::framework::ProcessingContext& pc) final
67  {
68  // set the min range of TFs for first TF
69  if (mProcessedTFs == 0) {
70  mTFRange[0] = getCurrentTF(pc);
71  }
72 
73  for (int i = 0; i < mCRUs.size(); ++i) {
74  const DataRef ref = pc.inputs().getByPos(i);
75  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
76  const int cru = tpcCRUHeader->subSpecification - mLaneId * CRU::MaxCRU;
77  mIDCFactorization.setIDCs(pc.inputs().get<std::vector<float>>(ref), cru, mProcessedTFs); // aggregate IDCs
78  }
79  ++mProcessedTFs;
80 
81  if (!(mProcessedTFs % ((mIDCFactorization.getNTimeframes() + 5) / 5))) {
82  LOGP(info, "aggregated TFs: {}", mProcessedTFs);
83  }
84 
85  if (mProcessedTFs == mIDCFactorization.getNTimeframes()) {
86  mTFRange[1] = getCurrentTF(pc); // set the TF for last aggregated TF
87  mProcessedTFs = 0; // reset processed TFs for next aggregation interval
88  mIDCFactorization.factorizeIDCs(); // calculate DeltaIDC, 0D-IDC, 1D-IDC
89 
90  if (mDebug) {
91  LOGP(info, "dumping aggregated and factorized IDCs and FT to file");
92  mIDCFactorization.dumpToFile(fmt::format("IDCFactorized_{:02}.root", getCurrentTF(pc)).data());
93  }
94 
95  // storing to CCDB
96  sendOutput(pc.outputs());
97  }
98  }

139  void sendOutputDebug(DataAllocator& output)
140  {
141  output.snapshot(Output{gDataOriginTPC, TPCFactorizeIDCSpec::getDataDescriptionIDC0()}, mIDCFactorization.getIDCZero());
142  output.snapshot(Output{gDataOriginTPC, TPCFactorizeIDCSpec::getDataDescriptionIDC1()}, mIDCFactorization.getIDCOne());
143  for (unsigned int iChunk = 0; iChunk < mIDCFactorization.getNChunks(); ++iChunk) {
144  output.snapshot(Output{gDataOriginTPC, TPCFactorizeIDCSpec::getDataDescriptionIDCDelta(), o2::header::DataHeader::SubSpecificationType{iChunk}, Lifetime::Timeframe}, mIDCFactorization.getIDCDeltaUncompressed(iChunk));

148  void sendOutput(DataAllocator& output)
149  {
150  if (mSendOutDebug) {
151  sendOutputDebug(output);
152  }
153 
154  if (mWriteToDB) {
155  const long timeStampStart = getFirstTF();
156  const long timeStampEnd = getLastTF();
157  mDBapi.storeAsTFileAny<o2::tpc::IDCZero>(&mIDCFactorization.getIDCZero(), "TPC/Calib/IDC/IDC0", mMetadata, timeStampStart, timeStampEnd);
158  mDBapi.storeAsTFileAny<o2::tpc::IDCOne>(&mIDCFactorization.getIDCOne(), "TPC/Calib/IDC/IDC1", mMetadata, timeStampStart, timeStampEnd);
159 
160  for (unsigned int iChunk = 0; iChunk < mIDCFactorization.getNChunks(); ++iChunk) {
161  switch (mCompressionDeltaIDC) {
162  case IDCDeltaCompression::MEDIUM:
163  default: {
164  auto idcDeltaMediumCompressed = mIDCFactorization.getIDCDeltaMediumCompressed(iChunk);
165  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<short>>(&idcDeltaMediumCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
166  break;
167  }
168  case IDCDeltaCompression::HIGH: {
169  auto idcDeltaHighCompressed = mIDCFactorization.getIDCDeltaHighCompressed(iChunk);
170  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<char>>(&idcDeltaHighCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
171  break;
172  }
173  case IDCDeltaCompression::NO:
174  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<float>>(&mIDCFactorization.getIDCDeltaUncompressed(iChunk), "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
175  break;
176  }
177  }
178  }
179  // reseting aggregated IDCs. This is done for safety, but if all data is received in the next aggregation interval it isnt necessary... remove it?
180  mIDCFactorization.reset();
181  }
