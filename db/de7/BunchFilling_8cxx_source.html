64 {

66  // train with 25ns spacing should have bcSpacing = 1

74 void BunchFilling::setBCTrains(int nTrains, int trainSpacingInBC, int nBC, int bcSpacing, int firstBC)
75 {
76  // add nTrains trains of interacting BCs with bcSpacing within the train and trainSpacingInBC empty slots
77  // between the trains
78  for (int it = 0; it < nTrains; it++) {
79  setBCTrain(nBC, bcSpacing, firstBC);
80  firstBC += nBC * bcSpacing + trainSpacingInBC;
81  }
