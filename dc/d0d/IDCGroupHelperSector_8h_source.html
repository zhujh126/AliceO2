88  void initIDCGroupHelperSector()
89  {
90  for (unsigned int reg = 0; reg < Mapper::NREGIONS; ++reg) {
91  const IDCGroupHelperRegion groupTmp(mGroupingPar.GroupPads[reg], mGroupingPar.GroupRows[reg], mGroupingPar.GroupLastRowsThreshold[reg], mGroupingPar.GroupLastPadsThreshold[reg], reg);
92  mNIDCsPerCRU[reg] = groupTmp.getNIDCsPerIntegrationInterval();
93  mRows[reg] = groupTmp.getNRows();
94  mPadsPerRow[reg] = groupTmp.getPadsPerRow();
95  mOffsRow[reg] = groupTmp.getRowOffset();
96  if (reg > 0) {
97  const unsigned int lastInd = reg - 1;
98  mRegionOffs[reg] = mRegionOffs[lastInd] + mNIDCsPerCRU[lastInd];
99  }
100  }
101  mNIDCsPerSector = static_cast<unsigned int>(std::accumulate(mNIDCsPerCRU.begin(), mNIDCsPerCRU.end(), decltype(mNIDCsPerCRU)::value_type(0)));
102  }
