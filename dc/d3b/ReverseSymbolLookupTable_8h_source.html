34 class ReverseSymbolLookupTable
35 {
36  public:
37  using symbol_t = SymbolStatistics::symbol_t;
38  using count_t = SymbolStatistics::count_t;
39  using iterator_t = const symbol_t*;

41  //TODO(milettri): fix once ROOT cling respects the standard http://wg21.link/p1286r2
42  ReverseSymbolLookupTable() noexcept {}; //NOLINT
43 
44  explicit ReverseSymbolLookupTable(const SymbolStatistics& symbolStats)
45  {
46  LOG(trace) << "start building reverse symbol lookup table";
47 
48  mLut.resize(pow2(symbolStats.getSymbolTablePrecision()));
49  // go over all symbols
50  for (size_t index = 0; index < symbolStats.size(); ++index) {
51  const symbol_t symbol = symbolStats.getMinSymbol() + index;
52  const auto [symFrequency, symCumulated] = symbolStats.at(index);
53  for (count_t cumulative = symCumulated; cumulative < symCumulated + symFrequency; cumulative++) {
54  mLut[cumulative] = symbol;
55  }
56  }
57 // advanced diagnostics for debug builds
58 #if !defined(NDEBUG)
59  LOG(debug2) << "reverseSymbolLookupTableProperties: {"
60  << "elements: " << mLut.size() << ", "
61  << "sizeB: " << mLut.size() * sizeof(typename std::decay_t<decltype(mLut)>::value_type) << "}";
62 #endif
63 
64  if (symbolStats.size() == 1) {
65  LOG(warning) << "SymbolStatistics of empty message passed to " << __func__;
66  }
67 
68  LOG(trace) << "done building reverse symbol lookup table";
69  };
70 
71  inline size_t size() const noexcept { return mLut.size(); };
72  inline symbol_t operator[](count_t cumul) const noexcept
73  {
74  assert(cumul < size());
75  return mLut[cumul];
76  };
77  inline iterator_t begin() const noexcept { return mLut.data(); };
78  inline iterator_t end() const noexcept { return mLut.data() + size(); };
79 
80  private:
81  std::vector<symbol_t> mLut{};
82 };
