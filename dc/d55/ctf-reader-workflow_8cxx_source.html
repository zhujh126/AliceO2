43 // we need to add workflow options before including Framework/runDataProcessing
44 void customize(std::vector<o2::framework::ConfigParamSpec>& workflowOptions)
45 {
46  // option allowing to set parameters
47  std::vector<o2::framework::ConfigParamSpec> options;
48  options.push_back(ConfigParamSpec{"onlyDet", VariantType::String, std::string{DetID::NONE}, {"comma-separated list of detectors to accept. Overrides skipDet"}});
49  options.push_back(ConfigParamSpec{"skipDet", VariantType::String, std::string{DetID::NONE}, {"comma-separate list of detectors to skip"}});
50  options.push_back(ConfigParamSpec{"ctf-input", VariantType::String, "none", {"comma-separated list CTF input files"}});
51  options.push_back(ConfigParamSpec{"loop", VariantType::Int, 1, {"loop N times (infinite for N<=0)"}});
52  options.push_back(ConfigParamSpec{"delay", VariantType::Float, 0.f, {"delay in seconds between consecutive TFs sending"}});
53  options.push_back(ConfigParamSpec{"configKeyValues", VariantType::String, "", {"Semicolon separated key=value strings"}});
54 
55  std::swap(workflowOptions, options);
56 }

61 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
62 {
63  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
64  DetID::mask_t dets;
65  dets.set(); // by default read all
66  WorkflowSpec specs;
67 
68  auto mskOnly = DetID::getMask(configcontext.options().get<std::string>("onlyDet"));
69  auto mskSkip = DetID::getMask(configcontext.options().get<std::string>("skipDet"));
70  if (mskOnly.any()) {
71  dets &= mskOnly;
72  } else {
73  dets ^= mskSkip;
74  }
75 
76  std::string inpNames = configcontext.options().get<std::string>("ctf-input");
77  if (inpNames.empty() || inpNames == "none") {
78  if (!configcontext.helpOnCommandLine()) {
79  throw std::runtime_error("--ctf-input <file,...> is not provided");
80  }
81  inpNames = "";
82  }
83 
84  int loop = configcontext.options().get<int>("loop");
85  if (loop < 1) {
86  loop = 0x7fffffff;
87  }
88  int delayMUS = int32_t(1e6 * configcontext.options().get<float>("delay")); // delay in microseconds
89  if (delayMUS < 0) {
90  delayMUS = 0;
91  }
92 
93  specs.push_back(o2::ctf::getCTFReaderSpec(dets, inpNames, loop, delayMUS));
94 
95  // add decodors for all allowed detectors.
96  if (dets[DetID::ITS]) {
97  specs.push_back(o2::itsmft::getEntropyDecoderSpec(DetID::getDataOrigin(DetID::ITS)));
98  }
99  if (dets[DetID::MFT]) {
100  specs.push_back(o2::itsmft::getEntropyDecoderSpec(DetID::getDataOrigin(DetID::MFT)));
101  }
102  if (dets[DetID::TPC]) {
103  specs.push_back(o2::tpc::getEntropyDecoderSpec());
104  }
105  if (dets[DetID::TRD]) {
106  specs.push_back(o2::trd::getEntropyDecoderSpec());
107  }
108  if (dets[DetID::TOF]) {
109  specs.push_back(o2::tof::getEntropyDecoderSpec());
110  }
111  if (dets[DetID::FT0]) {
112  specs.push_back(o2::ft0::getEntropyDecoderSpec());
113  }
114  if (dets[DetID::FV0]) {
115  specs.push_back(o2::fv0::getEntropyDecoderSpec());
116  }
117  if (dets[DetID::FDD]) {
118  specs.push_back(o2::fdd::getEntropyDecoderSpec());
119  }
120  if (dets[DetID::MID]) {
121  specs.push_back(o2::mid::getEntropyDecoderSpec());
122  }
123  if (dets[DetID::MCH]) {
124  specs.push_back(o2::mch::getEntropyDecoderSpec());
125  }
126  if (dets[DetID::EMC]) {
127  specs.push_back(o2::emcal::getEntropyDecoderSpec());
128  }
129  if (dets[DetID::PHS]) {
130  specs.push_back(o2::phos::getEntropyDecoderSpec());
131  }
132  if (dets[DetID::CPV]) {
133  specs.push_back(o2::cpv::getEntropyDecoderSpec());
134  }
135  if (dets[DetID::ZDC]) {
136  specs.push_back(o2::zdc::getEntropyDecoderSpec());
137  }
138  if (dets[DetID::HMP]) {
139  specs.push_back(o2::hmpid::getEntropyDecoderSpec());
140  }
141 
142  return std::move(specs);
143 }
