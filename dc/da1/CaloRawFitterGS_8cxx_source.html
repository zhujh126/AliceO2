29 void CaloRawFitterGS::init()
30 {
31  //prepare fitting arrays, once per lifetime
32  double k = o2::phos::PHOSSimParams::Instance().mSampleDecayTime;
33  ma0[0] = 1.;
34  mb0[0] = 1.;
35  mb1[0] = 0.;
36  mb2[0] = 0.;
37  mb3[0] = 0.;
38  mb4[0] = 0.;
39  for (int i = 1; i < NMAXSAMPLES; i++) {
40  double xi = k * i;
41  ma0[i] = exp(-xi);
42  mb0[i] = mb0[i - 1] + ma0[i];
43  mb1[i] = 4 * mb1[i - 1] + ma0[i] * xi;
44  mb2[i] = 6 * mb2[i - 1] + ma0[i] * xi * xi;
45  mb3[i] = 4 * mb3[i - 1] + ma0[i] * xi * xi * xi;
46  mb4[i] = mb4[i - 1] + ma0[i] * xi * xi * xi * xi;
47  }
48 }

50 CaloRawFitterGS::FitStatus CaloRawFitterGS::evaluate(gsl::span<short unsigned int> signal)
51 {
52 
53  //Pedestal analysis mode
54  if (mPedestalRun) {
55  int nPed = signal.size();
56  mAmp = 0.;
57  mTime = 0.;
58  for (auto a : signal) {
59  mAmp += a;
60  mTime += a * a;
61  }
62  if (nPed > 0) {
63  mAmp /= nPed;
64  mTime = mTime / nPed - mAmp * mAmp;
65  if (mTime > 0.) {
66  mTime = sqrt(mTime);
67  }
68  }
69  mOverflow = false;
70  mStatus = kOK;
71  return kOK;
72  }
73 
74  mStatus = kNotEvaluated;
75  // Extract amplitude and time
76  mStatus = evalFit(signal);
77  return mStatus;
78 }

80 CaloRawFitterGS::FitStatus CaloRawFitterGS::evalFit(gsl::span<short unsigned int> signal)
81 {
82  // Calculate signal parameters (energy, time, quality) from array of samples
83  // Energy is a maximum sample minus pedestal 9
84  // Time is the first time bin
85  // Signal overflows is there are at least 3 samples of the same amplitude above 900
86 
87  int sigLength = signal.size();
88  if (sigLength == 0) {
89  return kEmptyBunch;
90  }
91  mAmp = 0.;
92  mTime = 0.;
93  mChi2 = 0.;
94  mOverflow = false;
95  FitStatus status = kNotEvaluated;
96 
97  //if pedestal should be subtracted first evaluate it
98  float pedMean = 0;
99  int nPed = 0;
100  if (mPedSubtract) {
101  //remember inverse time order
102  for (auto it = signal.rbegin(); (nPed < mPreSamples) && it != signal.rend(); ++it) {
103  nPed++;
104  pedMean += *it;
105  }
106  if (nPed > 0) {
107  pedMean /= nPed;
108  }
109  }
110 
111  float maxSample = 0.; //maximal sample value
112  int nMax = 0; //number of consequitive maximal samples
113  bool spike = false; //did we observe spike?
114  int ap = -1, app = -1; //remember previous values to evaluate spikes
115  double At = 0., Bt = 0., Ct = 0., y2 = 0.;
116 
117  int nSamples = signal.size();
118  if (mPedSubtract) {
119  nSamples = -mPreSamples;
120  }
121  int firstS = nSamples - 1;
122  for (int i = 0; i < nSamples; i++) {
123  float a = signal[firstS - i] - pedMean; //remember inverse order of samples
124  float xi = i * mDecTime;
125  if (a > maxSample) {
126  mMaxSample = a;
127  nMax = 1;
128  }
129  if (a == maxSample) {
130  nMax++;
131  }
132  //check if there was a spike in previous step?
133  if (app >= 0 && ap >= 0) {
134  spike = (2 * ap - (a + app) > 2 * mSpikeThreshold);
135  }
136  if (spike) {
137  status = kSpike;
138  //Try to recover: subtract last point contribution and replace by average of "app" and "a"
139  float atmp = 0.5 * (app + a);
140  float xiprev = xi - mDecTime;
141  float ss = (atmp - ap) * ma0[i - 1];
142  At += ss; //spike can not appear at 0-th bin
143  Bt += ss * xiprev;
144  Ct += ss * xiprev * xiprev;
145  y2 += (atmp * atmp - ap * ap);
146  } else {
147  app = ap;
148  ap = a;
149  }
150  //Check if in saturation
151  if (maxSample > 900 && nMax > 3) {
152  mOverflow = true;
153  }
154 
155  //to calculate time
156  float st = a * ma0[i];
157  At += st;
158  Bt += st * xi;
159  Ct += st * xi * xi;
160  y2 += a * a;
161  //to calculate amplitude
162  } //Scanned full sample
163 
164  //calculate time, amp and chi2
165  double polB = At - Bt;
166  double polC = Ct - 2. * Bt;
167  if (At == 0.) {
168  if (polB == 0.) {
169  mTime = 999.;
170  status = kFitFailed;
171  } else {
172  mTime = -polC / (2. * polB);
173  }
174  } else {
175  double d = polB * polB - At * polC;
176  if (d >= 0) {
177  mTime = (-polB - sqrt(d)) / (At * mDecTime);
178  } else {
179  mTime = 999.;
180  status = kFitFailed;
181  }
182  }
183  if (status == kFitFailed && !mOverflow) { //in case of overflow try to recover
184  mAmp = 0.;
185  mTime = 999.;
186  mChi2 = 999.;
187  return status;
188  }
189 
190  if (!mOverflow) { //normal sample, calculate amp and chi2 and return
191  double tt = mTime * mDecTime;
192  double tt2 = tt * tt;
193  double expT = exp(tt);
194  double nom = (At * tt2 - 2. * Bt * tt + Ct) * expT; // 1./(k*k) cancel with denom
195  int i = nSamples - 1;
196  double denom = (mb4[i] - tt * (mb3[i] - tt * (mb2[i] - tt * (mb1[i] - tt * mb0[i])))) *
197  expT * expT / (mDecTime * mDecTime); // 1/(k*k) cancel with nom
198  if (denom != 0) {
199  mAmp = nom / denom;
200  mChi2 = (y2 - (2. * (At * tt2 - 2. * tt * Bt + Ct) - mAmp * denom * expT) * mAmp * expT / (mDecTime * mDecTime)) / (i + 1);
201  } else {
202  mAmp = 0.;
203  status = kFitFailed;
204  }
205  return status;
206  } else { // overflow: try iterative procedure but for lowGain only
207  if (!mLowGain) {
208  mAmp = 0.;
209  mTime = 999.;
210  mChi2 = 999.;
211  return kOverflow;
212  }
213 
214  //Try to recalculate parameters replacing overflow/spike values by those expected from the sample shape
215  short nIter = 0;
216  double timeOld = mTime;
217  double ampOld = mAmp;
218  if (status == kFitFailed) { //could not calculate time, amp: set best guess
219  timeOld = 0;
220  ampOld = maxSample;
221  }
222 
223  //Iterative process, not more than NITERATIONS
224  short nMaxIter = o2::phos::PHOSSimParams::Instance().mNIterations;
225  for (short nIter = 0; nIter < nMaxIter; nIter++) {
226  ap = -1;
227  app = -1; //remember previous values to evaluate spikes
228 
229  double expT = exp(mDecTime * timeOld);
230  for (int i = 0; i < nSamples; i++) {
231  float a = signal[firstS - i] - pedMean; //remember inverse order of samples
232  float xi = i * mDecTime;
233  if (a == maxSample) { //overflow, replace with calculated
234  a = ampOld * ma0[i] * (timeOld - i) * (timeOld - i) * expT;
235  }
236  //check if there was a spike in prev step?
237  if (app >= 0 && ap >= 0) {
238  if (2 * ap - (a + app) > 2 * mSpikeThreshold) {
239  //Try to recover: subtract last point contribution and replace by average of "app" and "a"
240  float atmp = ampOld * ma0[i] * (timeOld - i + 1) * (timeOld - i + 1) * expT; //0.5*(app+a) ;
241  float xiprev = xi - mDecTime;
242  float s = (atmp - ap) * ma0[i - 1]; //spike can not appear at 0-th bin
243  At += s;
244  Bt += s * xiprev;
245  Ct += s * xiprev * xiprev;
246  y2 += (atmp * atmp - ap * ap);
247  }
248  } else {
249  app = ap;
250  ap = a;
251  }
252 
253  //to calculate time
254  float ss = a * ma0[i];
255  At += ss;
256  Bt += ss * xi;
257  Ct += ss * xi * xi;
258  y2 += a * a;
259  }
260  //evaluate new time and amp
261 
262  double polB = At - Bt;
263  double polC = Ct - 2. * Bt;
264  if (At == 0.) {
265  if (polB == 0.) {
266  mTime = 999.;
267  status = kFitFailed;
268  } else {
269  mTime = -polC / (2. * polB);
270  }
271  } else {
272  double d = polB * polB - At * polC;
273  if (d >= 0) {
274  mTime = (-polB - sqrt(d)) / (At * mDecTime);
275  } else {
276  mTime = 999.;
277  status = kFitFailed;
278  }
279  }
280  if (status == kFitFailed) { //Can not improve, give up
281  mAmp = 0;
282  mTime = 999.;
283  mChi2 = 999.;
284  mOverflow = false;
285  return status;
286  }
287  double tt = mTime * mDecTime;
288  expT = exp(tt);
289  double nom = (At * tt * tt - 2. * Bt * tt + Ct) * expT; // 1./(k*k) cancel with denom
290  int i = nSamples - 1;
291  double denom = ((mb4[i] - tt * (mb3[i] - tt * (mb2[i] - tt * (mb1[i] - tt * mb0[i]))))) *
292  expT * expT / (mDecTime * mDecTime); // 1/(k*k) cancel with nom
293  if (denom != 0) {
294  mAmp = nom / denom;
295  mChi2 = (y2 - (2. * (At * tt * tt - 2. * tt * Bt + Ct) - mAmp * denom * expT) * mAmp * expT / (mDecTime * mDecTime)) / (i + 1);
296  } else {
297  mAmp = 0;
298  mTime = 999.;
299  mChi2 = 999.;
300  mOverflow = false;
301  return kFitFailed;
302  }
303 
304  //Check modification and quit if ready
305  if (std::abs(mTime - timeOld) < mTimeAccuracy && std::abs(mAmp - ampOld) < ampOld * mAmpAccuracy) {
306  break;
307  }
308 
309  timeOld = mTime;
310  ampOld = mAmp;
311  }
312  }
313 
314  return status;
315 }
