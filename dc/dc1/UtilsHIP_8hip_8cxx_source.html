28 int roundUp(const int numToRound, const int multiple)
29 {
30  if (multiple == 0) {
31  return numToRound;
32  }
33  int remainder{numToRound % multiple};
34  if (remainder == 0) {
35  return numToRound;
36  }
37  return numToRound + multiple - remainder;
38 }

40 int findNearestDivisor(const int numToRound, const int divisor)
41 {
42  if (numToRound > divisor) {
43  return divisor;
44  }
45  int result = numToRound;
46  while (divisor % result != 0) {
47  ++result;
48  }
49 
50  return result;
51 }

83 dim3 utils::host_hip::getBlockSize(const int colsNum, const int rowsNum, const int maxThreadsPerBlock)
84 {
85  const DeviceProperties& deviceProperties = ContextHIP::getInstance().getDeviceProperties();
86  int xThreads = std::max(std::min(colsNum, static_cast<int>(deviceProperties.maxThreadsDim.x)), 1);
87  int yThreads = std::max(std::min(rowsNum, static_cast<int>(deviceProperties.maxThreadsDim.y)), 1);
88  const int totalThreads = roundUp(std::min(xThreads * yThreads, maxThreadsPerBlock),
89  static_cast<int>(deviceProperties.warpSize));
90  if (xThreads > yThreads) {
91  xThreads = findNearestDivisor(xThreads, totalThreads);
92  yThreads = totalThreads / xThreads;
93 
94  } else {
95  yThreads = findNearestDivisor(yThreads, totalThreads);
96  xThreads = totalThreads / yThreads;
97  }
98 
99  return dim3{static_cast<unsigned int>(xThreads), static_cast<unsigned int>(yThreads)};
100 }
