55 void DataReaderVSD::open()
56 {
57  TString ESDFileName = "events_0.root";
58  Warning("GotoEvent", "OPEN");
59  mMaxEv = -1;
60  mCurEv = -1;
61  mFile = TFile::Open(ESDFileName);
62  if (!mFile) {
63  Error("VSD_Reader", "Can not open file '%s' ... terminating.",
64  ESDFileName.Data());
65  gSystem->Exit(1);
66  }
67 
68  assert(mEvDirKeys.size() == 0);
69 
70  TPMERegexp name_re("Event\\d+");
71  TObjLink* lnk = mFile->GetListOfKeys()->FirstLink();
72  while (lnk) {
73  if (name_re.Match(lnk->GetObject()->GetName())) {
74  mEvDirKeys.push_back((TKey*)lnk->GetObject());
75  }
76  lnk = lnk->Next();
77  }
78 
79  mMaxEv = mEvDirKeys.size();
80  if (mMaxEv == 0) {
81  Error("VSD_Reader", "No events to show ... terminating.");
82  gSystem->Exit(1);
83  }
84 }
