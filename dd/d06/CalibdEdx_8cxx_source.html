37 void CalibdEdx::finalizeSlot(Slot& slot)
38 {
39  LOG(INFO) << "Finalizing slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd();
40 
41  const dEdxHistos* container = slot.getContainer();
42  const auto statsASide = container->getHists()[0].getStatisticsData();
43  const auto statsCSide = container->getHists()[1].getStatisticsData();
44 
45  slot.print();
46  LOG(INFO) << "A side, truncated mean statistics: Mean = " << statsASide.mCOG << ", StdDev = " << statsCSide.mStdDev << ", Entries = " << statsASide.mSum;
47  LOG(INFO) << "C side, truncated mean statistics: Mean = " << statsCSide.mCOG << ", StdDev = " << statsCSide.mStdDev << ", Entries = " << statsCSide.mSum;
48 
49  CalibMIP mips{statsASide.mCOG, statsCSide.mCOG};
50 
51  const auto className = o2::utils::MemFileHelper::getClassName(mips);
52  const auto fileName = o2::ccdb::CcdbApi::generateFileName(className);
53  const std::map<std::string, std::string> metaData;
54 
55  // TODO: the timestamp is now given with the TF index, but it will have
56  // to become an absolute time.
57  TFType timeFrame = slot.getTFStart();
58  mInfoVector.emplace_back("TPC/Calib/MIPS", className, fileName, metaData, timeFrame, 99999999999999);
59  mMIPVector.push_back(mips);
60 
61  if (mDebugOutputStreamer) {
62  LOG(INFO) << "Dumping time slot data to file";
63 
64  *mDebugOutputStreamer << "mipPosition"
65  << "timeFrame=" << timeFrame // Initial time frame of time slot
66  << "calibMIP=" << mips // Computed MIP positions
67  << "dEdxHistos=" << slot.getContainer() // dE/dx histograms
68  << "\n";
69  }
70 }
