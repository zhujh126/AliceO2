45  TPCAverageGroupIDCDevice(const int lane, const std::vector<uint32_t>& crus, const unsigned int rangeIDC, const float sigma, const bool debug = false, const bool loadFromFile = false)
46  : mLane{lane}, mCRUs{crus}, mRangeIDC{rangeIDC}, mDebug{debug}, mLoadFromFile{loadFromFile}, mOneDIDCs(rangeIDC)
47  {
48  auto& paramIDCGroup = ParameterIDCGroup::Instance();
49  for (const auto& cru : mCRUs) {
50  const CRU cruTmp(cru);
51  const unsigned int reg = cruTmp.region();
52  mIDCs.emplace(cru, IDCAverageGroup(paramIDCGroup.GroupPads[reg], paramIDCGroup.GroupRows[reg], paramIDCGroup.GroupLastRowsThreshold[reg], paramIDCGroup.GroupLastPadsThreshold[reg], reg, cruTmp.sector(), sigma));
53  mBuffer1DIDCs.emplace(cru, std::deque<std::vector<float>>(std::ceil(static_cast<float>(mRangeIDC) / mMinIDCsPerTF), std::vector<float>(mMinIDCsPerTF, 0)));
54  }
55  }

57  void init(o2::framework::InitContext& ic) final
58  {
59  if (mLoadFromFile) {
60  const char* fName = "IDCGroup.root";
61  TFile fInp(fName, "READ");
62  for (TObject* keyAsObj : *fInp.GetListOfKeys()) {
63  const auto key = dynamic_cast<TKey*>(keyAsObj);
64  const char* name = key->GetName();
65  LOGP(info, "Key name: {} Type: {}", name, key->GetClassName());
66 
67  if (std::strcmp(o2::tpc::IDCAverageGroup::Class()->GetName(), key->GetClassName()) != 0) {
68  LOGP(info, "skipping object. wrong class.");
69  continue;
70  }
71  IDCAverageGroup* idcavg = (IDCAverageGroup*)fInp.Get(name);
72  unsigned int cru = idcavg->getSector() * Mapper::NREGIONS + idcavg->getRegion();
73  // check cru
74  if (std::find(mCRUs.begin(), mCRUs.end(), cru) != mCRUs.end()) {
75  mIDCs[cru].setFromFile(fName, name);
76  mIDCs[cru].processIDCs();
77  }
78  delete idcavg;
79  }
80  }
81  }

83  void run(o2::framework::ProcessingContext& pc) final
84  {
85  LOGP(info, "averaging and grouping IDCs for TF {} for CRUs {} to {} using {} threads", getCurrentTF(pc), mCRUs.front(), mCRUs.back(), mIDCs.begin()->second.getNThreads());
86 
87  if (!mLoadFromFile) {
88  for (int i = 0; i < mCRUs.size(); ++i) {
89  const DataRef ref = pc.inputs().getByPos(i);
90  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
91  const int cru = tpcCRUHeader->subSpecification >> 7;
92  mIDCs[cru].setIDCs(pc.inputs().get<std::vector<float>>(ref));
93  mIDCs[cru].processIDCs();
94 
95  // send the output for one CRU for one TF
96  sendOutput(pc.outputs(), cru);
97  }
98  } else {
99  for (int i = 0; i < mCRUs.size(); ++i) {
100  sendOutput(pc.outputs(), mCRUs[i]);
101  }
102  }
103 
104  if (mDebug) {
105  TFile fOut(fmt::format("IDCGroup_{}_tf_{}.root", mLane, getCurrentTF(pc)).data(), "RECREATE");
106  for (int i = 0; i < mCRUs.size(); ++i) {
107  const DataRef ref = pc.inputs().getByPos(i);
108  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
109  const int cru = tpcCRUHeader->subSpecification >> 7;
110  fOut.WriteObject(&mIDCs[cru], fmt::format("CRU_{}", cru).data());
111  }
112  }
113  }

150  void sendOutput(DataAllocator& output, const uint32_t cru)
151  {
152  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
153 
154  // calculate 1D-IDCs = sum over 2D-IDCs as a function of the integration interval
155  std::vector<float> vec1DIDCs = mIDCs[cru].getIDCGroup().get1DIDCs();
156  output.snapshot(Output{gDataOriginTPC, getDataDescription1DIDC(), subSpec, Lifetime::Timeframe}, vec1DIDCs);
157 
158  // TODO use this and fix #include <boost/container/pmr/polymorphic_allocator.hpp> in ROOT CINT
159  // output.adoptContainer(Output{gDataOriginTPC, getDataDescriptionIDCGroup(), subSpec, Lifetime::Timeframe}, std::move(mIDCs[cru]).getIDCGroupData());
160  output.snapshot(Output{gDataOriginTPC, getDataDescriptionIDCGroup(), subSpec, Lifetime::Timeframe}, mIDCs[cru].getIDCGroup().getData());
161 
162  mBuffer1DIDCs[cru].emplace_back(std::move(vec1DIDCs));
163  mBuffer1DIDCs[cru].pop_front(); // removing oldest 1D-IDCs
164 
165  fill1DIDCs(cru);
166  output.snapshot(Output{gDataOriginTPC, getDataDescription1DIDCEPN(), subSpec, Lifetime::Timeframe}, mOneDIDCs);
167  }
168 
169  void fill1DIDCs(const uint32_t cru)
170  {
171  // fill 1D-IDC vector with mRangeIDC buffered values
172  unsigned int i = mRangeIDC;
173  for (auto it = mBuffer1DIDCs[cru].crbegin(); it != mBuffer1DIDCs[cru].crend(); ++it) {
174  for (auto idc = it->crbegin(); idc != it->crend(); ++idc) {
175  mOneDIDCs[--i] = *idc;
176  if (i == 0) {
177  return;
178  }
179  }
180  }
181  }

184 DataProcessorSpec getTPCAverageGroupIDCSpec(const int ilane, const std::vector<uint32_t>& crus, const unsigned int rangeIDC, const float sigma, const bool debug, const bool loadFromFile)
185 {
186  std::vector<OutputSpec> outputSpecs;
187  std::vector<InputSpec> inputSpecs;
188  outputSpecs.reserve(crus.size());
189  inputSpecs.reserve(crus.size());
190 
191  for (const auto& cru : crus) {
192  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
193  if (!loadFromFile) {
194  inputSpecs.emplace_back(InputSpec{"idcs", gDataOriginTPC, TPCIntegrateIDCDevice::getDataDescription(TPCIntegrateIDCDevice::IDCFormat::Sim), subSpec, Lifetime::Timeframe});
195  }
196  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCAverageGroupIDCDevice::getDataDescriptionIDCGroup(), subSpec});
197  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCAverageGroupIDCDevice::getDataDescription1DIDC(), subSpec});
198  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCAverageGroupIDCDevice::getDataDescription1DIDCEPN(), subSpec});
199  }
200 
201  const auto id = fmt::format("tpc-averagegroup-idc-{:02}", ilane);
202  return DataProcessorSpec{
203  id.data(),
204  inputSpecs,
205  outputSpecs,
206  AlgorithmSpec{adaptFromTask<TPCAverageGroupIDCDevice>(ilane, crus, rangeIDC, sigma, debug, loadFromFile)},
207  }; // end DataProcessorSpec
