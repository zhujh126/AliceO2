21 std::shared_ptr<arrow::Table> ArrowHelpers::joinTables(std::vector<std::shared_ptr<arrow::Table>>&& tables)
22 {
23  if (tables.size() == 1) {
24  return tables[0];
25  }
26  for (auto i = 0u; i < tables.size() - 1; ++i) {
27  if (tables[i]->num_rows() != tables[i + 1]->num_rows()) {
28  throw o2::framework::runtime_error_f("Tables %s and %s have different sizes (%d vs %d) and cannot be joined!",
29  tables[i]->schema()->metadata()->Get("label").ValueOrDie().c_str(),
30  tables[i + 1]->schema()->metadata()->Get("label").ValueOrDie().c_str(),
31  tables[i]->num_rows(),
32  tables[i + 1]->num_rows());
33  }
34  }
35  std::vector<std::shared_ptr<arrow::Field>> fields;
36  std::vector<std::shared_ptr<arrow::ChunkedArray>> columns;
37 
38  for (auto& t : tables) {
39  auto tf = t->fields();
40  std::copy(tf.begin(), tf.end(), std::back_inserter(fields));
41  }
42 
43  auto schema = std::make_shared<arrow::Schema>(fields);
44 
45  if (tables[0]->num_rows() != 0) {
46  for (auto& t : tables) {
47  auto tc = t->columns();
48  std::copy(tc.begin(), tc.end(), std::back_inserter(columns));
49  }
50  }
51  return arrow::Table::Make(schema, columns);
52 }

54 std::shared_ptr<arrow::Table> ArrowHelpers::concatTables(std::vector<std::shared_ptr<arrow::Table>>&& tables)
55 {
56  if (tables.size() == 1) {
57  return tables[0];
58  }
59  std::vector<std::shared_ptr<arrow::ChunkedArray>> columns;
60  assert(tables.size() > 1);
61  std::vector<std::shared_ptr<arrow::Field>> resultFields = tables[0]->schema()->fields();
62  auto compareFields = [](std::shared_ptr<arrow::Field> const& f1, std::shared_ptr<arrow::Field> const& f2) {
63  // Let's do this with stable sorting.
64  return (!f1->Equals(f2)) && (f1->name() < f2->name());
65  };
66  for (size_t i = 1; i < tables.size(); ++i) {
67  auto& fields = tables[i]->schema()->fields();
68  std::vector<std::shared_ptr<arrow::Field>> intersection;
69 
70  std::set_intersection(resultFields.begin(), resultFields.end(),
71  fields.begin(), fields.end(),
72  std::back_inserter(intersection), compareFields);
73  resultFields.swap(intersection);
74  }
75 
76  for (auto& field : resultFields) {
77  arrow::ArrayVector chunks;
78  for (auto& table : tables) {
79  auto ci = table->schema()->GetFieldIndex(field->name());
80  if (ci == -1) {
81  throw std::runtime_error("Unable to find field " + field->name());
82  }
83  auto column = table->column(ci);
84  auto otherChunks = column->chunks();
85  chunks.insert(chunks.end(), otherChunks.begin(), otherChunks.end());
86  }
87  columns.push_back(std::make_shared<arrow::ChunkedArray>(chunks));
88  }
89 
90  auto result = arrow::Table::Make(std::make_shared<arrow::Schema>(resultFields), columns);
91  return result;
92 }

103 arrow::Status getSliceFor(int value, char const* key, std::shared_ptr<arrow::Table> const& input, std::shared_ptr<arrow::Table>& output, uint64_t& offset)
104 {
105  arrow::Datum value_counts;
106 #if ARROW_VERSION_MAJOR > 4
107  auto options = arrow::compute::ScalarAggregateOptions::Defaults();
108 #else
109  auto options = arrow::compute::CountOptions::Defaults();
110 #endif
111  ARROW_ASSIGN_OR_RAISE(value_counts,
112  arrow::compute::CallFunction("value_counts", {input->GetColumnByName(key)},
113  &options));
114  auto pair = static_cast<arrow::StructArray>(value_counts.array());
115  auto values = static_cast<arrow::NumericArray<arrow::Int32Type>>(pair.field(0)->data());
116  auto counts = static_cast<arrow::NumericArray<arrow::Int64Type>>(pair.field(1)->data());
117 
118  int slice;
119  for (slice = 0; slice < values.length(); ++slice) {
120  if (values.Value(slice) == value) {
121  offset = slice;
122  output = input->Slice(slice, counts.Value(slice));
123  return arrow::Status::OK();
124  }
125  }
126  output = input->Slice(0, 0);
127  return arrow::Status::OK();
128 }
