37 std::vector<painter::PadCoordinates> painter::getPadCoordinatesSector()
38 {
39  std::vector<painter::PadCoordinates> padCoords;
40 
41  const auto& regInf = Mapper::instance().getMapPadRegionInfo();
42 
43  for (const auto& padReg : regInf) {
44  const auto npr = padReg.getNumberOfPadRows();
45  const auto ro = padReg.getRowOffset();
46  const auto xm = padReg.getXhelper();
47  const auto ph = padReg.getPadHeight();
48  const auto pw = padReg.getPadWidth();
49  const auto yro = padReg.getRadiusFirstRow();
50  const auto ks = ph / pw * std::tan(1.74532925199432948e-01);
51 
52  for (int irow = 0; irow < npr; ++irow) {
53  const auto npads = std::floor(ks * (irow + ro) + xm);
54  for (int ipad = -npads; ipad < npads; ++ipad) {
55  const auto xPadBottomRight = yro + ph * irow;
56  const auto xPadTopRight = yro + ph * (irow + 1);
57  const auto ri = xPadBottomRight;
58  const auto yPadBottomRight = pw * ipad * xPadBottomRight / (ri + ph / 2);
59  const auto yPadTopRight = pw * ipad * xPadTopRight / (ri + ph / 2);
60  const auto yPadBottomLeft = pw * (ipad + 1) * xPadBottomRight / (ri + ph / 2);
61  const auto yPadTopLeft = pw * (ipad + 1) * xPadTopRight / (ri + ph / 2);
62  auto& padCoord = padCoords.emplace_back();
63  padCoord.xVals = {xPadBottomRight, xPadTopRight, xPadTopRight, xPadBottomRight};
64  padCoord.yVals = {yPadBottomRight, yPadTopRight, yPadTopLeft, yPadBottomLeft};
65  }
66  }
67  }
68 
69  return padCoords;
70 }

79 template <class T>
80 TCanvas* painter::draw(const CalDet<T>& calDet, int nbins1D, float xMin1D, float xMax1D, TCanvas* outputCanvas)
81 {
82  using DetType = CalDet<T>;
83  using CalType = CalArray<T>;
84 
85  static const Mapper& mapper = Mapper::instance();
86 
87  // ===| name and title |======================================================
88  std::string title = calDet.getName();
89  std::string name = calDet.getName();
90  std::replace(name.begin(), name.end(), ' ', '_');
91  std::replace(title.begin(), title.end(), '_', ' ');
92 
93  // ===| define histograms |===================================================
94  // TODO: auto scaling of ranges based on mean and variance?
95  // for the moment use roots auto scaling
96 
97  // set buffer size such that autoscaling uses the full range. This is about 2MB per histogram!
98  const int bufferSize = TH1::GetDefaultBufferSize();
99  TH1::SetDefaultBufferSize(Sector::MAXSECTOR * mapper.getPadsInSector());
100 
101  auto hAside1D = new TH1F(fmt::format("h_Aside_1D_{}", name).data(), fmt::format("{} (A-Side)", title).data(),
102  nbins1D, xMin1D, xMax1D); //TODO: modify ranges
103 
104  auto hCside1D = new TH1F(fmt::format("h_Cside_1D_{}", name).data(), fmt::format("{} (C-Side)", title).data(),
105  nbins1D, xMin1D, xMax1D); //TODO: modify ranges
106 
107  auto hAside2D = new TH2F(fmt::format("h_Aside_2D_{}", name).data(), fmt::format("{} (A-Side);x (cm);y (cm)", title).data(),
108  300, -300, 300, 300, -300, 300);
109 
110  auto hCside2D = new TH2F(fmt::format("h_Cside_2D_{}", name).data(), fmt::format("{} (C-Side);x (cm);y (cm)", title).data(),
111  300, -300, 300, 300, -300, 300);
112 
113  for (ROC roc; !roc.looped(); ++roc) {
114 
115  auto hist2D = hAside2D;
116  auto hist1D = hAside1D;
117  if (roc.side() == Side::C) {
118  hist2D = hCside2D;
119  hist1D = hCside1D;
120  }
121 
122  const int nrows = mapper.getNumberOfRowsROC(roc);
123  for (int irow = 0; irow < nrows; ++irow) {
124  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
125  for (int ipad = 0; ipad < npads; ++ipad) {
126  const auto val = calDet.getValue(roc, irow, ipad);
127  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
128  const int bin = hist2D->FindBin(pos.X(), pos.Y());
129  if (!hist2D->GetBinContent(bin)) {
130  hist2D->SetBinContent(bin, val);
131  }
132  hist1D->Fill(val);
133  }
134  }
135  }
136 
137  if (xMax1D > xMin1D) {
138  hAside2D->SetMinimum(xMin1D);
139  hAside2D->SetMaximum(xMax1D);
140  hCside2D->SetMinimum(xMin1D);
141  hCside2D->SetMaximum(xMax1D);
142  }
143 
144  // ===| Draw histograms |=====================================================
145  auto c = outputCanvas;
146  if (!c) {
147  c = new TCanvas(fmt::format("c_{}", name).data(), title.data(), 1000, 1000);
148  }
149  c->Clear();
150  c->Divide(2, 2);
151 
152  c->cd(1);
153  hAside2D->Draw("colz");
154  hAside2D->SetStats(0);
155  drawSectorsXY(Side::A);
156 
157  c->cd(2);
158  hCside2D->Draw("colz");
159  hCside2D->SetStats(0);
160  drawSectorsXY(Side::C);
161 
162  c->cd(3);
163  hAside1D->Draw();
164 
165  c->cd(4);
166  hCside1D->Draw();
167 
168  // reset the buffer size
169  TH1::SetDefaultBufferSize(bufferSize);
170 
171  // associate histograms to canvas
172  hAside1D->SetBit(TObject::kCanDelete);
173  hCside1D->SetBit(TObject::kCanDelete);
174  hAside2D->SetBit(TObject::kCanDelete);
175  hCside2D->SetBit(TObject::kCanDelete);
176 
177  return c;
178 }

198 template <class T>
199 void painter::fillHistogram2D(TH2& h2D, const CalDet<T>& calDet, Side side)
200 {
201  static const Mapper& mapper = Mapper::instance();
202 
203  for (ROC roc; !roc.looped(); ++roc) {
204  if (roc.side() != side) {
205  continue;
206  }
207 
208  const int nrows = mapper.getNumberOfRowsROC(roc);
209  for (int irow = 0; irow < nrows; ++irow) {
210  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
211  for (int ipad = 0; ipad < npads; ++ipad) {
212  const auto val = calDet.getValue(roc, irow, ipad);
213  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
214  const int bin = h2D.FindBin(pos.X(), pos.Y());
215  if (!h2D.GetBinContent(bin)) {
216  h2D.SetBinContent(bin, val);
217  }
218  }
219  }
220  }
221 }

224 template <class T>
225 void painter::fillHistogram2D(TH2& h2D, const CalArray<T>& calArray)
226 {
227  static const Mapper& mapper = Mapper::instance();
228 
229  const size_t position = calArray.getPadSubsetNumber();
230  const PadSubset padSubset = calArray.getPadSubset();
231  const int nrows = mapper.getNumberOfPadRows(padSubset, position);
232 
233  // ===| fill hist |===========================================================
234  for (int irow = 0; irow < nrows; ++irow) {
235  const int padsInRow = mapper.getNumberOfPadsInRow(padSubset, position, irow);
236  for (int ipad = 0; ipad < padsInRow; ++ipad) {
237  const GlobalPadNumber pad = mapper.getPadNumber(padSubset, position, irow, ipad);
238  const auto val = calArray.getValue(pad);
239  const int cpad = ipad - padsInRow / 2;
240  h2D.Fill(irow, cpad, val);
241  }
242  }
243 }

265 template <class T>
266 TH2* painter::getHistogram2D(const CalArray<T>& calArray)
267 {
268  static const Mapper& mapper = Mapper::instance();
269 
270  const size_t position = calArray.getPadSubsetNumber();
271  const PadSubset padSubset = calArray.getPadSubset();
272 
273  // ===| maximum number of rows and pads |=====================================
274  const int nrows = mapper.getNumberOfPadRows(padSubset, position);
275  const int npads = mapper.getNumberOfPadsInRow(padSubset, position, nrows - 1) + 6;
276 
277  // ===| create histogram |====================================================
278  std::string title = calArray.getName();
279  std::string name = calArray.getName();
280  std::replace(title.begin(), title.end(), '_', ' ');
281  std::replace(name.begin(), name.end(), ' ', '_');
282 
283  if (padSubset == PadSubset::ROC) {
284  title += fmt::format(" ({})", getROCTitle(position));
285  }
286 
287  auto hist = new TH2F(fmt::format("h_{}", name).data(),
288  fmt::format("{};pad row;pad", title).data(),
289  nrows, 0., nrows,
290  npads, -npads / 2, npads / 2);
291 
292  fillHistogram2D(*hist, calArray);
293 
294  return hist;
295 }

309 template <class T>
310 std::vector<TCanvas*> painter::makeSummaryCanvases(const CalDet<T>& calDet, int nbins1D, float xMin1D, float xMax1D, bool onlyFilled, std::vector<TCanvas*>* outputCanvases)
311 {
312 
313  std::vector<TCanvas*> vecCanvases;
314 
315  auto nROCs = calDet.getData().size();
316 
317  if (onlyFilled) {
318  nROCs = 0;
319  for (size_t iroc = 0; iroc < calDet.getData().size(); ++iroc) {
320  const auto& roc = calDet.getCalArray(iroc);
321 
322  if (hasData(roc)) {
323  ++nROCs;
324  }
325  }
326 
327  if (!nROCs) {
328  return vecCanvases;
329  }
330  }
331 
332  // ===| set up canvases |===
333  TCanvas* cSides = nullptr;
334  TCanvas* cROCs1D = nullptr;
335  TCanvas* cROCs2D = nullptr;
336  const std::string_view calName = calDet.getName();
337 
338  if (outputCanvases) {
339  if (outputCanvases->size() < 3) {
340  LOGP(error, "At least 3 canvases are needed to fill the output, only {} given", outputCanvases->size());
341  return vecCanvases;
342  }
343 
344  cSides = outputCanvases->at(0);
345  cROCs1D = outputCanvases->at(1);
346  cROCs2D = outputCanvases->at(2);
347  cSides->Clear();
348  cROCs1D->Clear();
349  cROCs2D->Clear();
350  } else {
351 
352  cROCs1D = new TCanvas(fmt::format("c_ROCs_{}_1D", calName).data(), fmt::format("{} values for each ROC", calName).data(), 1400, 1000);
353  cROCs2D = new TCanvas(fmt::format("c_ROCs_{}_2D", calName).data(), fmt::format("{} values for each ROC", calName).data(), 1400, 1000);
354  }
355  cSides = draw(calDet, nbins1D, xMin1D, xMax1D, cSides);
356  cROCs1D->DivideSquare(nROCs);
357  cROCs2D->DivideSquare(nROCs);
358 
359  vecCanvases.emplace_back(cSides);
360  vecCanvases.emplace_back(cROCs1D);
361  vecCanvases.emplace_back(cROCs2D);
362 
363  // ===| produce plots for each ROC |===
364  size_t pad = 1;
365  for (size_t iroc = 0; iroc < calDet.getData().size(); ++iroc) {
366  const auto& roc = calDet.getCalArray(iroc);
367 
368  if (onlyFilled && !hasData(roc)) {
369  continue;
370  }
371 
372  // ===| 1D histogram |===
373  auto h1D = new TH1F(fmt::format("h_{}_{:02d}", calName, iroc).data(), fmt::format("{} distribution ROC {:02d} ({});ADC value", calName, iroc, getROCTitle(iroc)).data(), nbins1D, xMin1D, xMax1D);
374  for (const auto& val : roc.getData()) {
375  h1D->Fill(val);
376  }
377 
378  // ===| 2D histogram |===
379  auto h2D = painter::getHistogram2D(roc);
380  h2D->SetStats(0);
381  if (xMax1D > xMin1D) {
382  h2D->SetMinimum(xMin1D);
383  h2D->SetMaximum(xMax1D);
384  }
385  h2D->SetUniqueID(iroc);
386 
387  cROCs1D->cd(pad);
388  h1D->Draw();
389 
390  cROCs2D->cd(pad);
391  h2D->Draw("colz");
392 
393  ++pad;
394 
395  // associate histograms to canvas
396  h1D->SetBit(TObject::kCanDelete);
397  h2D->SetBit(TObject::kCanDelete);
398  }
399 
400  return vecCanvases;
401 }

456 {

472  }
473  }

477 //______________________________________________________________________________
478 void painter::drawSectorsXY(Side side, int sectorLineColor, int sectorTextColor)
479 {
480  TLine l;
481  l.SetLineColor(sectorLineColor);
482 
483  TLine l2;
484  l2.SetLineColor(sectorLineColor);
485  l2.SetLineStyle(kDotted);
486 
487  TLatex latSide;
488  latSide.SetTextColor(sectorTextColor);
489  latSide.SetTextAlign(22);
490  latSide.SetTextSize(0.08);
491  latSide.DrawLatex(0, 0, (side == Side::C) ? "C" : "A");
492 
493  TLatex lat;
494  lat.SetTextAlign(22);
495  lat.SetTextSize(0.03);
496  lat.SetTextColor(sectorLineColor);
497 
498  constexpr float phiWidth = float(SECPHIWIDTH);
499  const float rFactor = std::cos(phiWidth / 2.);
500  const float rLow = 83.65 / rFactor;
501  const float rIROCup = 133.3 / rFactor;
502  const float rOROClow = 133.5 / rFactor;
503  const float rOROC12 = 169.75 / rFactor;
504  const float rOROC23 = 207.85 / rFactor;
505  const float rOut = 247.7 / rFactor;
506  const float rText = rLow * rFactor * 3. / 4.;
507 
508  for (Int_t isector = 0; isector < 18; ++isector) {
509  const float sinR = std::sin(phiWidth * isector);
510  const float cosR = std::cos(phiWidth * isector);
511 
512  const float sinL = std::sin(phiWidth * ((isector + 1) % 18));
513  const float cosL = std::cos(phiWidth * ((isector + 1) % 18));
514 
515  const float sinText = std::sin(phiWidth * (isector + 0.5));
516  const float cosText = std::cos(phiWidth * (isector + 0.5));
517 
518  const float xR1 = rLow * cosR;
519  const float yR1 = rLow * sinR;
520  const float xR2 = rOut * cosR;
521  const float yR2 = rOut * sinR;
522 
523  const float xL1 = rLow * cosL;
524  const float yL1 = rLow * sinL;
525  const float xL2 = rOut * cosL;
526  const float yL2 = rOut * sinL;
527 
528  const float xOROCmup1 = rOROClow * cosR;
529  const float yOROCmup1 = rOROClow * sinR;
530  const float xOROCmup2 = rOROClow * cosL;
531  const float yOROCmup2 = rOROClow * sinL;
532 
533  const float xIROCmup1 = rIROCup * cosR;
534  const float yIROCmup1 = rIROCup * sinR;
535  const float xIROCmup2 = rIROCup * cosL;
536  const float yIROCmup2 = rIROCup * sinL;
537 
538  const float xO121 = rOROC12 * cosR;
539  const float yO121 = rOROC12 * sinR;
540  const float xO122 = rOROC12 * cosL;
541  const float yO122 = rOROC12 * sinL;
542 
543  const float xO231 = rOROC23 * cosR;
544  const float yO231 = rOROC23 * sinR;
545  const float xO232 = rOROC23 * cosL;
546  const float yO232 = rOROC23 * sinL;
547 
548  const float xText = rText * cosText;
549  const float yText = rText * sinText;
550 
551  // left side line
552  l.DrawLine(xR1, yR1, xR2, yR2);
553 
554  // IROC inner line
555  l.DrawLine(xR1, yR1, xL1, yL1);
556 
557  // IROC end line
558  l.DrawLine(xIROCmup1, yIROCmup1, xIROCmup2, yIROCmup2);
559 
560  // OROC start line
561  l.DrawLine(xOROCmup1, yOROCmup1, xOROCmup2, yOROCmup2);
562 
563  // OROC1 - OROC2 line
564  l.DrawLine(xO121, yO121, xO122, yO122);
565 
566  // OROC2 - OROC3 line
567  l.DrawLine(xO231, yO231, xO232, yO232);
568 
569  // IROC inner line
570  l.DrawLine(xR2, yR2, xL2, yL2);
571 
572  // sector numbers
573  lat.DrawLatex(xText, yText, fmt::format("{}", isector).data());
574  }
575 }

577 void painter::drawSectorLocalPadNumberPoly(short padTextColor, float lineScalePS)
578 {
579  static const Mapper& mapper = Mapper::instance();
580  const auto coords = getPadCoordinatesSector();
581  TLatex lat;
582  lat.SetTextAlign(12);
583  lat.SetTextSize(0.002f);
584  lat.SetTextColor(padTextColor);
585  gStyle->SetLineScalePS(lineScalePS);
