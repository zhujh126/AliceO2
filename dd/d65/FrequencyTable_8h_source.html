110 void FrequencyTable::addSamples(Source_IT begin, Source_IT end)
111 {
112  if (begin != end) {
113  const auto& [minIter, maxIter] = std::minmax_element(begin, end);
114  addSamples(begin, end, *minIter, *maxIter);
115  } else {
116  LOG(warning) << "Passed empty message to " << __func__; // RS this is ok for empty columns
117  return;
118  }

121 template <typename Source_IT, std::enable_if_t<internal::isIntegralIter_v<Source_IT>, bool>>
122 void FrequencyTable::addSamples(Source_IT begin, Source_IT end, symbol_t min, symbol_t max)
123 {
124  LOG(trace) << "start adding samples";
125  internal::RANSTimer t;
126  t.start();
127 
128  if (begin == end) {
129  LOG(warning) << "Passed empty message to " << __func__; // RS this is ok for empty columns
130  return;
131  }
132 
133  resizeFrequencyTable(min, max);
134 
135  // add new symbols
136  std::for_each(begin, end, [this](symbol_t symbol) { ++this->getSymbol(symbol); });
137 
138  mNumSamples += std::distance(begin, end);
139 
140  t.stop();
141  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
142 
143 #if !defined(NDEBUG)
144  LOG(debug2) << *this;
145 #endif
146  LOG(trace) << "done adding samples";
147 }

149 template <typename Freq_IT, std::enable_if_t<internal::isIntegralIter_v<Freq_IT>, bool>>
150 void FrequencyTable::addFrequencies(Freq_IT begin, Freq_IT end, symbol_t min, symbol_t max)
151 {
152  static_assert(std::is_integral<typename std::iterator_traits<Freq_IT>::value_type>::value);
153 
154  LOG(trace) << "start adding frequencies";
155  internal::RANSTimer t;
156  t.start();
157 
158  if (begin == end) {
159  LOG(warning) << "Passed empty FrequencyTable to " << __func__; // RS this is ok for empty columns
160  return;
161  }
162 
163  // ensure correct size of frequency table and grow it if necessary
164  resizeFrequencyTable(min, max);
165 
166  // either 0 or offset from array start.
167  const size_t offset = std::abs(mMin - min);
168 
169  // ftableA[i] += fTableB[i+offset]
170  std::transform(begin, end,
171  mFrequencyTable.begin() + offset,
172  mFrequencyTable.begin() + offset,
173  [this](typename std::iterator_traits<Freq_IT>::value_type first, count_t second) {
174  mNumSamples += first;
175  return first + second; });
176 
177  t.stop();
178  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
179 
180 #if !defined(NDEBUG)
181  LOG(debug2) << *this;
182 #endif
183 
184  LOG(trace) << "done adding frequencies";
185 }
