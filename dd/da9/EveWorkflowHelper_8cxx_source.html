119  auto creator = [maskTrk, &recoCont, &retVal, nTracksTPCITSO2, nTPCTOFMatches](auto& trk, GID gid, float time, float) {
120  if (gid.getSource() == GID::ITSTPCTOF) {
121  if (maskTrk[GID::TPC] && nTracksTPCITSO2) {
122  const auto& match = recoCont.getTOFMatch(gid);
123  const auto& trkITSTPC = recoCont.getTPCITSTracks();

131  }
132  }
133  } else if constexpr (isTPCTrack<decltype(trk)>()) {
134  time = trk.getTime0();
135  } else if constexpr (isTPCITSTrack<decltype(trk)>()) {
136  if (maskTrk[GID::TPC] && retVal->tpcLinkITS.size()) {
137  retVal->tpcLinkITS[trk.getRefTPC().getIndex()] = trk.getRefITS().getIndex();
138  }
139  } else if constexpr (isTPCTOFTrack<decltype(trk)>()) {

174  }
