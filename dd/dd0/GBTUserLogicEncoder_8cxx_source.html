40 void GBTUserLogicEncoder::processTrigger(const InteractionRecord& ir, uint8_t triggerWord)
41 {
43  auto& vec = mBoards[ir];
44  for (uint8_t ireg = 0; ireg < 2; ++ireg) {
45  uint8_t firedLoc = (mMask >> (4 * ireg)) & 0xF;
46  vec.push_back({raw::sSTARTBIT, triggerWord, static_cast<uint8_t>(ireg + mOffset), firedLoc});
47  }
48  for (uint8_t iloc = 0; iloc < 8; ++iloc) {
49  if (mMask & (1 << iloc)) {
50  vec.push_back({raw::sSTARTBIT | raw::sCARDTYPE, triggerWord, static_cast<uint8_t>(iloc + mOffset), 0});
51  if (triggerWord & (raw::sSOX | raw::sEOX)) {
53  for (int ich = 0; ich < 4; ++ich) {
54  vec.back().patternsBP[ich] = 0xFFFF;
55  vec.back().patternsNBP[ich] = 0xFFFF;
56  }
57  }
58  }
59  }
60 }

62 void GBTUserLogicEncoder::addRegionalBoards(uint8_t activeBoards, InteractionRecord ir)
63 {
65  ir += mElectronicsDelay.BCToLocal + mElectronicsDelay.regToLocal;
66  auto& vec = mBoards[ir];
67  for (uint8_t ireg = 0; ireg < 2; ++ireg) {
68  uint8_t firedLoc = (activeBoards >> (4 * ireg)) & 0xF;
69  if (firedLoc > 0) {
70  vec.push_back({raw::sSTARTBIT, 0, static_cast<uint8_t>(ireg + mOffset), firedLoc});
71  }

75 void GBTUserLogicEncoder::process(gsl::span<const ROBoard> data, const InteractionRecord& ir)
76 {
78  auto& vec = mBoards[ir];
79  uint8_t activeBoards = 0;
80  for (auto& loc : data) {
81  for (int ich = 0; ich < 4; ++ich) {
82  if (loc.patternsBP[ich] && loc.patternsNBP[ich]) {
83  activeBoards |= (1 << (raw::getLocId(loc.boardId) % 8));
84  }
85  }
86  vec.emplace_back(loc);
87  }
88  addRegionalBoards(activeBoards, ir);
89 }

91 void GBTUserLogicEncoder::flush(std::vector<char>& buffer, const InteractionRecord& ir)
92 {
94  std::map<InteractionRecord, std::vector<ROBoard>> tmpBoards;
95  for (auto& item : mBoards) {
96  if (item.first <= ir) {
97  for (auto& loc : item.second) {
98  buffer.emplace_back(loc.statusWord);
99  buffer.emplace_back(loc.triggerWord);
100  addShort(buffer, item.first.bc);
101  buffer.emplace_back((raw::getLocId(loc.boardId) << 4) | loc.firedChambers);
102  buffer.emplace_back(mCrateId << 4);
103  if (raw::isLoc(loc.statusWord)) {
104  for (int ich = 4; ich >= 0; --ich) {
105  if (loc.firedChambers & (1 << ich)) {
106  addShort(buffer, loc.patternsBP[ich]);
107  addShort(buffer, loc.patternsNBP[ich]);
108  }
109  }
110  }
111  }
112  } else {
113  tmpBoards[item.first] = item.second;
114  }
115  }
116  mBoards.swap(tmpBoards);
117 }
