99 std::vector<PNT> getTrackPoints(const o2::track::TrackPar& trc, float minR, float maxR, float maxStep)
100 
101 {
102  // prepare space points from the track param
103  std::vector<PNT> pnts;
104  int nSteps = std::max(2, int((maxR - minR) / maxStep));
105  const auto prop = o2::base::Propagator::Instance();
106  float xMin = trc.getX(), xMax = maxR * maxR - trc.getY() * trc.getY();
107  if (xMax > 0) {
108  xMax = std::sqrt(xMax);
109  }
110  LOG(INFO) << "R: " << minR << " " << maxR << " || X: " << xMin << " " << xMax;
111  float dx = (xMax - xMin) / nSteps;
112  auto tp = trc;
113  float dxmin = std::abs(xMin - tp.getX()), dxmax = std::abs(xMax - tp.getX());
114  bool res = false;
115  if (dxmin > dxmax) { //start from closest end
116  std::swap(xMin, xMax);
117  dx = -dx;
118  }
119  if (!prop->propagateTo(tp, xMin, false, 0.99, maxStep, o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
120  return pnts;
121  }
122  auto xyz = tp.getXYZGlo();
123  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
124  for (int is = 0; is < nSteps; is++) {
125  if (!prop->propagateTo(tp, tp.getX() + dx, false, 0.99, 999., o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
126  return pnts;
127  }
128  xyz = tp.getXYZGlo();
129  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
130  }
131  return pnts;
132 }

134 void O2DPLDisplaySpec::run(ProcessingContext& pc)
135 {
136  if (!this->mEveHostNameMatch) {
137  return;
138  }
139 
140  // filtering out any run which occur before reaching next time interval
141  std::chrono::time_point<std::chrono::high_resolution_clock> currentTime = std::chrono::high_resolution_clock::now();
142  std::chrono::duration<double> elapsed = currentTime - this->mTimeStamp;
143  if (elapsed < this->mTimeInteval) {
144  return; // skip this run - it is too often
145  }
146  this->mTimeStamp = currentTime;
147 
148  o2::globaltracking::RecoContainer recoData;
149  recoData.collectData(pc, *mDataRequest);
150  auto cnt = EveWorkflowHelper::compute(recoData, &(mConfig->configCalib), mClMask, mTrkMask, mTrkMask);
151 
152  std::cout << cnt->globalTracks.size() << std::endl;
153  o2::event_visualisation::VisualisationEvent vEvent;
154  int trackCount = cnt->globalTracks.size(); // how many tracks should be stored
155  if (this->mNumberOfTracks != -1 && this->mNumberOfTracks < trackCount) {
156  trackCount = this->mNumberOfTracks; // less than available
157  }
158 
159  for (unsigned int i = 0; i < trackCount; i++) {
160  auto tr = cnt->globalTracks[i];
161  std::cout << tr->getX() << std::endl;
162  float rMax = 385;
163  float rMin = std::sqrt(tr->getX() * tr->getX() + tr->getY() * tr->getY());
164  auto pnts = getTrackPoints(*tr, rMin, rMax, 4.0);
165 
166  o2::event_visualisation::VisualisationTrack* vTrack = vEvent.addTrack({.charge = 0,
167  .energy = 0.0,
168  .ID = 0,
169  .PID = 0,
170  .mass = 0.0,
171  .signedPT = 0.0,
172  .startXYZ = {0, 0, 0},
173  .endXYZ = {0, 0, 0},
174  .pxpypz = {0, 0, 0},
175  .parentID = 0,
176  .phi = 0.0,
177  .theta = 0.0,
178  .helixCurvature = 0.0,
179  .type = 0,
180  .source = CosmicSource});
181  float dz = 0.0;
182  for (size_t ip = 0; ip < pnts.size(); ip++) {
183  vTrack->addPolyPoint(pnts[ip][0], pnts[ip][1], pnts[ip][2] + dz);
184  }
185  }
186 
187  FileProducer producer(this->mJsonPath, this->mNumberOfFiles);
188  vEvent.toFile(producer.newFileName());
189 }

195 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
196 {
197  WorkflowSpec specs;
198 
199  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
200  bool useMC = cfgc.options().get<bool>("enable-mc") && !cfgc.options().get<bool>("disable-mc");
201  std::string jsonFolder = cfgc.options().get<std::string>("jsons-folder");
202  std::string eveHostName = cfgc.options().get<std::string>("eve-hostname");
203  char hostname[HOST_NAME_MAX];
204  gethostname(hostname, HOST_NAME_MAX);
205  bool eveHostNameMatch = eveHostName.empty() || eveHostName == hostname;
206 
207  int eveDDSColIdx = cfgc.options().get<int>("eve-dds-collection-index");
208  char* colIdx = getenv("DDS_COLLECTION_INDEX");
209  eveHostNameMatch &= eveDDSColIdx == -1 || (colIdx && atoi(colIdx) == eveDDSColIdx);
210 
211  std::chrono::milliseconds timeInterval(cfgc.options().get<int>("time-interval"));
212  int numberOfFiles = cfgc.options().get<int>("number-of_files");
213  int numberOfTracks = cfgc.options().get<int>("number-of_tracks");
214 
215  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-tracks"));
216  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-clusters"));
217  if (!srcTrk.any() && !srcCl.any()) {
218  throw std::runtime_error("No input configured");
219  }
220  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
221  dataRequest->requestTracks(srcTrk, useMC);
222  dataRequest->requestClusters(srcCl, useMC);
223 
224  if (cfgc.options().get<bool>("read-from-files")) {
225  InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, useMC);
226  }
227 
228  specs.emplace_back(DataProcessorSpec{
229  "o2-eve-display",
230  dataRequest->inputs,
231  {},
232  AlgorithmSpec{adaptFromTask<O2DPLDisplaySpec>(useMC, srcTrk, srcCl, dataRequest, jsonFolder, timeInterval, numberOfFiles, numberOfTracks, eveHostNameMatch)}});
233 
234  return std::move(specs);
235 }
