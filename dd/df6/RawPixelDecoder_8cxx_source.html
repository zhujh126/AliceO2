74 template <class Mapping>
75 int RawPixelDecoder<Mapping>::decodeNextTrigger()
76 {
77  mTimerDecode.Start(false);
78  mNChipsFiredROF = 0;
79  mNPixelsFiredROF = 0;
80  mInteractionRecord.clear();
81  int nLinksWithData = 0, nru = mRUDecodeVec.size();
82  do {
83 #ifdef WITH_OPENMP
84  omp_set_num_threads(mNThreads);
85 #pragma omp parallel for schedule(dynamic) reduction(+ \
86  : nLinksWithData, mNChipsFiredROF, mNPixelsFiredROF)
87 #endif
88  for (int iru = 0; iru < nru; iru++) {
89  nLinksWithData += decodeNextTrigger(iru);
90  mNChipsFiredROF += mRUDecodeVec[iru].nChipsFired;
91  int npix = 0;
92  for (int ic = mRUDecodeVec[iru].nChipsFired; ic--;) {
93  npix += mRUDecodeVec[iru].chipsData[ic].getData().size();
94  }
95  mNPixelsFiredROF += npix;
96  }
97 
98  if (nLinksWithData) { // fill some statistics
99  mROFCounter++;
100  mNChipsFired += mNChipsFiredROF;
101  mNPixelsFired += mNPixelsFiredROF;
102  mCurRUDecodeID = 0; // getNextChipData will start from here
103  mLastReadChipID = -1;
104  // set IR and trigger from the 1st non empty link
105  for (const auto& link : mGBTLinks) {
106  if (link.status == GBTLink::DataSeen) {
107  mInteractionRecord = link.ir;
108  mInteractionRecordHB = o2::raw::RDHUtils::getHeartBeatIR(*link.lastRDH);
109  mTrigger = link.trigger;
110  break;
111  }
112  }
113  break;
114  }
115 
116  } while (mNLinksDone < mGBTLinks.size());
117 
118  ensureChipOrdering();
119  mTimerDecode.Stop();
120  return nLinksWithData;
121 }

143 template <class Mapping>
144 int RawPixelDecoder<Mapping>::decodeNextTrigger(int iru)
145 {
146  auto& ru = mRUDecodeVec[iru];
147  ru.clear();
148  int ndec = 0; // number of yet non-empty links
149  for (int il = 0; il < RUDecodeData::MaxLinksPerRU; il++) {
150  auto* link = getGBTLink(ru.links[il]);
151  if (link) {
152  auto res = link->collectROFCableData(mMAP);
153  if (res == GBTLink::DataSeen) { // at the moment process only DataSeen
154  ndec++;
155  } else if (res == GBTLink::StoppedOnEndOfData || res == GBTLink::AbortedOnError) { // this link has exhausted its data or it has to be discarded due to the error
156  mNLinksDone++;
157  }
158  }
159  }
160  if (ndec) {
161  ru.decodeROF(mMAP);
162  }
163  return ndec;
164 }

168 template <class Mapping>
169 void RawPixelDecoder<Mapping>::setupLinks(InputRecord& inputs)
170 {
171  mCurRUDecodeID = NORUDECODED;
172  auto nLinks = mGBTLinks.size();
173  auto origin = (mUserDataOrigin == o2::header::gDataOriginInvalid) ? mMAP.getOrigin() : mUserDataOrigin;
174  auto datadesc = (mUserDataDescription == o2::header::gDataDescriptionInvalid) ? o2::header::gDataDescriptionRawData : mUserDataDescription;
175  std::vector<InputSpec> filter{InputSpec{"filter", ConcreteDataTypeMatcher{origin, datadesc}, Lifetime::Timeframe}};
176 
177  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
178  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
179  {
180  std::vector<InputSpec> dummy{InputSpec{"dummy", ConcreteDataMatcher{origin, datadesc, 0xDEADBEEF}}};
181  for (const auto& ref : InputRecordWalker(inputs, dummy)) {
182  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
183  if (dh->payloadSize == 0) {
184  LOGP(WARNING, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF",
185  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize);
186  return;
187  }
188  }
189  }
190 
191  DPLRawParser parser(inputs, filter);
192  uint32_t currSSpec = 0xffffffff; // dummy starting subspec
193  int linksAdded = 0;
194  for (auto it = parser.begin(); it != parser.end(); ++it) {
195  auto const* dh = it.o2DataHeader();
196  auto& lnkref = mSubsSpec2LinkID[dh->subSpecification];
197  const auto& rdh = *reinterpret_cast<const header::RDHAny*>(it.raw()); // RSTODO this is a hack in absence of generic header getter
198 
199  if (lnkref.entry == -1) { // new link needs to be added
200  lnkref.entry = int(mGBTLinks.size());
201  auto& lnk = mGBTLinks.emplace_back(RDHUtils::getCRUID(rdh), RDHUtils::getFEEID(rdh), RDHUtils::getEndPointID(rdh), RDHUtils::getLinkID(rdh), lnkref.entry);
202  getCreateRUDecode(mMAP.FEEId2RUSW(RDHUtils::getFEEID(rdh))); // make sure there is a RU for this link
203  lnk.verbosity = GBTLink::Verbosity(mVerbosity);
204  lnk.format = mFormat;
205  LOG(INFO) << mSelfName << " registered new link " << lnk.describe() << " RUSW=" << int(mMAP.FEEId2RUSW(lnk.feeID));
206  linksAdded++;
207  }
208  auto& link = mGBTLinks[lnkref.entry];
209  if (currSSpec != dh->subSpecification) { // this is the 1st part for this link in this TF, next parts must follow contiguously!!!
210  link.clear(false, true); // clear link data except statistics
211  currSSpec = dh->subSpecification;
212  }
213  link.cacheData(it.raw(), RDHUtils::getMemorySize(rdh));
214  }
215 
216  if (linksAdded) { // new links were added, update link<->RU mapping, usually is done for 1st TF only
217  if (nLinks) {
218  LOG(WARNING) << mSelfName << " New links appeared although the initialization was already done";
219  for (auto& ru : mRUDecodeVec) { // reset RU->link references since they may have been changed
220  memset(&ru.links[0], -1, RUDecodeData::MaxLinksPerRU * sizeof(int));
221  }
222  }
223  // sort RUs in stave increasing order
224  std::sort(mRUDecodeVec.begin(), mRUDecodeVec.end(), [](const RUDecodeData& ruA, const RUDecodeData& ruB) -> bool { return ruA.ruSWID < ruB.ruSWID; });
225  for (auto i = 0; i < mRUDecodeVec.size(); i++) {
226  mRUEntry[mRUDecodeVec[i].ruSWID] = i;
227  }
228  nLinks = mGBTLinks.size();
229  // attach link to corresponding RU: this can be done once all RUs are created, to make sure their pointers don't change
230  for (int il = 0; il < nLinks; il++) {
231  auto& link = mGBTLinks[il];
232  bool newLinkAdded = (link.ruPtr == nullptr);
233  link.ruPtr = getRUDecode(mMAP.FEEId2RUSW(link.feeID)); // link to RU reference, reattach even it was already set before
234  uint16_t lr, ruOnLr, linkInRU;
235  mMAP.expandFEEId(link.feeID, lr, ruOnLr, linkInRU);
236  if (newLinkAdded) {
237  LOG(INFO) << mSelfName << " Attaching " << link.describe() << " to RU#" << int(mMAP.FEEId2RUSW(link.feeID))
238  << " (stave " << ruOnLr << " of layer " << lr << ')';
239  }
240  link.idInRU = linkInRU;
241  link.ruPtr->links[linkInRU] = il; // RU to link reference
242  }
243  }
244 }

264 template <class Mapping>
265 ChipPixelData* RawPixelDecoder<Mapping>::getNextChipData(std::vector<ChipPixelData>& chipDataVec)
266 {
267  // decode new RU if no cached non-empty chips
268  for (; mCurRUDecodeID < mRUDecodeVec.size(); mCurRUDecodeID++) {
269  auto& ru = mRUDecodeVec[mCurRUDecodeID];
270  if (ru.lastChipChecked < ru.nChipsFired) {
271  auto& chipData = ru.chipsData[ru.lastChipChecked++];
272  assert(mLastReadChipID < chipData.getChipID());
273  mLastReadChipID = chipData.getChipID();
274  chipDataVec[mLastReadChipID].swap(chipData);
275  return &chipDataVec[mLastReadChipID];
276  }
277  }
278  // will need to decode new trigger
279  if (!mDecodeNextAuto || !decodeNextTrigger()) { // no more data to decode
280  return nullptr;
281  }
282  return getNextChipData(chipDataVec);
283 }

308 template <>
309 void RawPixelDecoder<ChipMappingMFT>::ensureChipOrdering()
310 {
311  mOrderedChipsPtr.clear();
312  // define looping order, if mCurRUDecodeID < mRUDecodeVec.size(), this means that decodeNextTrigger() was called before
313  if (mCurRUDecodeID < mRUDecodeVec.size()) { // define sort order
314  for (; mCurRUDecodeID < mRUDecodeVec.size(); mCurRUDecodeID++) {
315  auto& ru = mRUDecodeVec[mCurRUDecodeID];
316  while (ru.lastChipChecked < ru.nChipsFired) {
317  mOrderedChipsPtr.push_back(&ru.chipsData[ru.lastChipChecked++]);
318  }
319  }
320  // sort in decreasing order
321  std::sort(mOrderedChipsPtr.begin(), mOrderedChipsPtr.end(), [](const ChipPixelData* a, const ChipPixelData* b) { return a->getChipID() > b->getChipID(); });
322  }
323 }
