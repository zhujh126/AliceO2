106 template <typename H, typename... T>
107 constexpr auto make_originals_from_type()
108 {
109  using decayed = std::decay_t<H>;
110  if constexpr (sizeof...(T) == 0) {
111  if constexpr (is_type_with_originals_v<decayed>) {
112  return typename decayed::originals{};
113  } else if constexpr (is_type_with_originals_v<typename decayed::table_t>) {
114  return typename decayed::table_t::originals{};
115  } else if constexpr (is_type_with_parent_v<decayed>) {
116  return make_originals_from_type<typename decayed::parent_t>();
117  } else {
118  return framework::pack<decayed>{};
119  }
120  } else if constexpr (is_type_with_originals_v<decayed>) {
121  return framework::concatenate_pack(typename decayed::originals{}, make_originals_from_type<T...>());

123  return framework::concatenate_pack(typename decayed::table_t::originals{}, make_originals_from_type<T...>());
124  } else {
125  return framework::concatenate_pack(framework::pack<decayed>{}, make_originals_from_type<T...>());
126  }
127 }

152 template <typename T, typename ChunkingPolicy = Chunked>
153 class ColumnIterator : ChunkingPolicy
154 {
155  static constexpr char SCALE_FACTOR = std::is_same_v<std::decay_t<T>, bool> ? 3 : 0;
156 
157  public:
162  ColumnIterator(arrow::ChunkedArray const* column)
163  : mColumn{column},
164  mCurrentPos{nullptr},
165  mFirstIndex{0},
166  mCurrentChunk{0}
167  {
168  auto array = getCurrentArray();
169  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset();
170  mLast = mCurrent + array->length();
171  }
172 
173  ColumnIterator() = default;
174  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
175  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
176 
177  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
178  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
179 
181  void nextChunk() const
182  {
183  auto previousArray = getCurrentArray();
184  mFirstIndex += previousArray->length();
185 
186  mCurrentChunk++;
187  auto array = getCurrentArray();
188  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
189  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
190  }
191 
192  void prevChunk() const
193  {
194  auto previousArray = getCurrentArray();
195  mFirstIndex -= previousArray->length();
196 
197  mCurrentChunk--;
198  auto array = getCurrentArray();
199  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
200  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
201  }
202 
203  void moveToChunk(int chunk)
204  {
205  if (mCurrentChunk < chunk) {
206  while (mCurrentChunk != chunk) {
207  nextChunk();
208  }
209  } else {
210  while (mCurrentChunk != chunk) {
211  prevChunk();
212  }
213  }
214  }
215 
217  void moveToEnd()
218  {
219  mCurrentChunk = mColumn->num_chunks() - 1;
220  auto array = getCurrentArray();
221  mFirstIndex = mColumn->length() - array->length();
222  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
223  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
224  }
225 
226  decltype(auto) operator*() const
227  {
228  if constexpr (ChunkingPolicy::chunked) {
229  if (O2_BUILTIN_UNLIKELY(((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast))) {
230  nextChunk();
231  }
232  }
233  if constexpr (std::is_same_v<bool, std::decay_t<T>>) {
234  // FIXME: notice that due to the type punning we cannot simply convert the
235  // masked char to a bool, because it's undefined behavior.
236  // FIXME: check if shifting the masked bit to the first position is better than != 0
237  return (*((char*)mCurrent + (*mCurrentPos >> SCALE_FACTOR)) & (1 << (*mCurrentPos & 0x7))) != 0;
238  } else {
239  return *(mCurrent + (*mCurrentPos >> SCALE_FACTOR));
240  }
241  }
242 
243  // Move to the chunk which containts element pos
244  ColumnIterator<T>& moveToPos()
245  {
246  // If we get outside range of the current chunk, go to the next.
247  if constexpr (ChunkingPolicy::chunked) {
248  while (O2_BUILTIN_UNLIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast)) {
249  nextChunk();
250  }
251  }
252  return *this;
253  }
254 
255  // Move to the chunk which containts element pos
256  ColumnIterator<T>& checkNextChunk()
257  {
258  if constexpr (ChunkingPolicy::chunked) {
259  if (O2_BUILTIN_LIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) <= mLast)) {
260  return *this;
261  }
262  nextChunk();
263  }
264  return *this;
265  }
266 
267  mutable T const* mCurrent;
268  int64_t const* mCurrentPos;
269  mutable T const* mLast;
270  arrow::ChunkedArray const* mColumn;
271  mutable int mFirstIndex;
272  mutable int mCurrentChunk;
273 
274  private:
276  auto getCurrentArray() const
277  {
278  std::shared_ptr<arrow::Array> chunkToUse = mColumn->chunk(mCurrentChunk);
279  if constexpr (std::is_same_v<arrow_array_for_t<T>, arrow::FixedSizeListArray>) {
280  chunkToUse = std::dynamic_pointer_cast<arrow::FixedSizeListArray>(chunkToUse)->values();
281  return std::static_pointer_cast<arrow_array_for_t<value_for_t<T>>>(chunkToUse);
282  } else {
283  return std::static_pointer_cast<arrow_array_for_t<T>>(chunkToUse);
284  }
285  }
286 };
