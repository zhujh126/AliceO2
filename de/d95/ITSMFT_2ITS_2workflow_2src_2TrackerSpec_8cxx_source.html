54 void TrackerDPL::init(InitContext& ic)
55 {
56  mTimer.Stop();
57  mTimer.Reset();
58  auto filename = ic.options().get<std::string>("grp-file");
59  const auto grp = parameters::GRPObject::loadFrom(filename);
60  if (grp) {
61  mGRP.reset(grp);
62  base::Propagator::initFieldFromGRP(grp);
63  auto field = static_cast<field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
64 
65  base::GeometryManager::loadGeometry();
66  GeometryTGeo* geom = GeometryTGeo::Instance();
67  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::T2GRot,
68  o2::math_utils::TransformType::T2G));
69 
70  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
71  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
72  if (o2::utils::Str::pathExists(matLUTFile)) {
73  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
74  o2::base::Propagator::Instance()->setMatLUT(lut);
75  LOG(INFO) << "Loaded material LUT from " << matLUTFile;
76  } else {
77  LOG(INFO) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
78  }
79 
80  auto* chainITS = mRecChain->AddChain<o2::gpu::GPUChainITS>();
81  mRecChain->Init();
82  mVertexer = std::make_unique<Vertexer>(chainITS->GetITSVertexerTraits());
83  mTracker = std::make_unique<Tracker>(chainITS->GetITSTrackerTraits());
84 
85  std::vector<TrackingParameters> trackParams;
86  std::vector<MemoryParameters> memParams;
87 
88  mRunVertexer = true;
89  if (mMode == "async") {
90 
91  trackParams.resize(3);
92  memParams.resize(3);
93  trackParams[0].TrackletMaxDeltaPhi = 0.05f;
94  trackParams[1].TrackletMaxDeltaPhi = 0.1f;
95  trackParams[2].MinTrackLength = 4;
96  trackParams[2].TrackletMaxDeltaPhi = 0.3;
97  LOG(INFO) << "Initializing tracker in async. phase reconstruction with " << trackParams.size() << " passes";
98 
99  } else if (mMode == "sync") {
100 
101  trackParams.resize(1);
102  memParams.resize(1);
103  LOG(INFO) << "Initializing tracker in sync. phase reconstruction with " << trackParams.size() << " passes";
104 
105  } else if (mMode == "cosmics") {
106 
107  mRunVertexer = false;
108  trackParams.resize(1);
109  memParams.resize(1);
110  trackParams[0].MinTrackLength = 4;
111  trackParams[0].TrackletMaxDeltaPhi = o2::its::constants::math::Pi * 0.5f;
112  for (int iLayer = 0; iLayer < o2::its::constants::its2::TrackletsPerRoad; iLayer++) {
113  trackParams[0].TrackletMaxDeltaZ[iLayer] = o2::its::constants::its2::LayersZCoordinate()[iLayer + 1];
114  memParams[0].TrackletsMemoryCoefficients[iLayer] = 0.5f;
115  // trackParams[0].TrackletMaxDeltaZ[iLayer] = 10.f;
116  }
117  for (int iLayer = 0; iLayer < o2::its::constants::its2::CellsPerRoad; iLayer++) {
118  trackParams[0].CellMaxDCA[iLayer] = 10000.f; //cm
119  trackParams[0].CellMaxDeltaZ[iLayer] = 10000.f; //cm
120  memParams[0].CellsMemoryCoefficients[iLayer] = 0.001f;
121  }
122  LOG(INFO) << "Initializing tracker in reconstruction for cosmics with " << trackParams.size() << " passes";
123 
124  } else {
125  throw std::runtime_error(fmt::format("Unsupported ITS tracking mode {:s} ", mMode));
126  }
127  mTracker->setParameters(memParams, trackParams);
128 
129  mVertexer->getGlobalConfiguration();
130  mTracker->getGlobalConfiguration();
131  LOG(INFO) << Form("Using %s for material budget approximation", (mTracker->isMatLUT() ? "lookup table" : "TGeometry"));
132 
133  double origD[3] = {0., 0., 0.};
134  mTracker->setBz(field->getBz(origD));
135  } else {
136  throw std::runtime_error(o2::utils::Str::concat_string("Cannot retrieve GRP from the ", filename));
137  }
138 
139  std::string dictPath = ic.options().get<std::string>("its-dictionary-path");
140  std::string dictFile = o2::base::NameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::ITS, dictPath, "bin");
141  if (o2::utils::Str::pathExists(dictFile)) {
142  mDict.readBinaryFile(dictFile);
143  LOG(INFO) << "Tracker running with a provided dictionary: " << dictFile;
144  } else {
145  LOG(INFO) << "Dictionary " << dictFile << " is absent, Tracker expects cluster patterns";
146  }

149 void TrackerDPL::run(ProcessingContext& pc)
150 {
151  mTimer.Start(false);
152  auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
153  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
154 
155  // code further down does assignment to the rofs and the altered object is used for output
156  // we therefore need a copy of the vector rather than an object created directly on the input data,
157  // the output vector however is created directly inside the message memory thus avoiding copy by
158  // snapshot
159  auto rofsinput = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
160  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "ITSTrackROF", 0, Lifetime::Timeframe}, rofsinput.begin(), rofsinput.end());
161 
162  LOG(INFO) << "ITSTracker pulled " << compClusters.size() << " clusters, " << rofs.size() << " RO frames";
163 
164  const dataformats::MCTruthContainer<MCCompLabel>* labels = nullptr;
165  gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
166  if (mIsMC) {
167  labels = pc.inputs().get<const dataformats::MCTruthContainer<MCCompLabel>*>("labels").release();
168  // get the array as read-only span, a snapshot is send forward
169  mc2rofs = pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("MC2ROframes");
170  LOG(INFO) << labels->getIndexedSize() << " MC label objects , in " << mc2rofs.size() << " MC events";
171  }
172 
173  std::vector<o2::its::TrackITSExt> tracks;
174  auto& allClusIdx = pc.outputs().make<std::vector<int>>(Output{"ITS", "TRACKCLSID", 0, Lifetime::Timeframe});
175  std::vector<o2::MCCompLabel> trackLabels;
176  auto& allTracks = pc.outputs().make<std::vector<o2::its::TrackITS>>(Output{"ITS", "TRACKS", 0, Lifetime::Timeframe});
177  std::vector<o2::MCCompLabel> allTrackLabels;
178 
179  auto& vertROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "VERTICESROF", 0, Lifetime::Timeframe});
180  auto& vertices = pc.outputs().make<std::vector<Vertex>>(Output{"ITS", "VERTICES", 0, Lifetime::Timeframe});
181 
182  auto& irFrames = pc.outputs().make<std::vector<o2::dataformats::IRFrame>>(Output{"ITS", "IRFRAMES", 0, Lifetime::Timeframe});
183 
184  std::uint32_t roFrame = 0;
185  ROframe event(0, 7);
186 
187  bool continuous = mGRP->isDetContinuousReadOut("ITS");
188  LOG(INFO) << "ITSTracker RO: continuous=" << continuous;
189 
190  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance(); // RS: this should come from CCDB
191  int nBCPerTF = continuous ? alpParams.roFrameLengthInBC : alpParams.roFrameLengthTrig;
192 
193  const auto& multEstConf = FastMultEstConfig::Instance(); // parameters for mult estimation and cuts
194  FastMultEst multEst; // mult estimator
195 
196  // snippet to convert found tracks to final output tracks with separate cluster indices
197  auto copyTracks = [](auto& tracks, auto& allTracks, auto& allClusIdx) {
198  for (auto& trc : tracks) {
199  trc.setFirstClusterEntry(allClusIdx.size()); // before adding tracks, create final cluster indices
200  int ncl = trc.getNumberOfClusters(), nclf = 0;
201  uint8_t patt = 0;
202  for (int ic = TrackITSExt::MaxClusters; ic--;) { // track internally keeps in->out cluster indices, but we want to store the references as out->in!!!
203  auto clid = trc.getClusterIndex(ic);
204  if (clid >= 0) {
205  allClusIdx.push_back(clid);
206  nclf++;
207  patt |= 0x1 << ic;
208  }
209  }
210  assert(ncl == nclf);
211  trc.setPattern(patt);
212  allTracks.emplace_back(trc);
213  }
214  };
215 
216  gsl::span<const unsigned char>::iterator pattIt = patterns.begin();
217  for (auto& rof : rofs) {
218  int nclUsed = ioutils::loadROFrameData(rof, event, compClusters, pattIt, mDict, labels);
219  // prepare in advance output ROFRecords, even if this ROF to be rejected
220  int first = allTracks.size();
221 
222  if (nclUsed) {
223  LOG(INFO) << "ROframe: " << roFrame << ", clusters loaded : " << nclUsed;
224 
225  // for vertices output
226  auto& vtxROF = vertROFvec.emplace_back(rof); // register entry and number of vertices in the
227  vtxROF.setFirstEntry(vertices.size()); // dedicated ROFRecord
228  vtxROF.setNEntries(0);
229 
230  if (multEstConf.cutMultClusLow > 0 || multEstConf.cutMultClusHigh > 0) { // cut was requested
231  auto mult = multEst.process(rof.getROFData(compClusters));
232  if (mult < multEstConf.cutMultClusLow || mult > multEstConf.cutMultClusHigh) {
233  LOG(INFO) << "Estimated cluster mult. " << mult << " is outside of requested range "
234  << multEstConf.cutMultClusLow << " : " << multEstConf.cutMultClusHigh << " | ROF " << rof.getBCData();
235  rof.setFirstEntry(first);
236  rof.setNEntries(0);
237  continue;
238  }
239  }
240 
241  std::vector<Vertex> vtxVecLoc;
242  if (mRunVertexer) {
243  mVertexer->clustersToVertices(event);
244  vtxVecLoc = mVertexer->exportVertices();
245  }
246 
247  if (mRunVertexer && (multEstConf.cutMultVtxLow > 0 || multEstConf.cutMultVtxHigh > 0)) { // cut was requested
248  std::vector<o2::dataformats::Vertex<o2::dataformats::TimeStamp<int>>> vtxVecSel;
249  vtxVecSel.swap(vtxVecLoc);
250  for (const auto& vtx : vtxVecSel) {
251  if (vtx.getNContributors() < multEstConf.cutMultVtxLow || (multEstConf.cutMultVtxHigh > 0 && vtx.getNContributors() > multEstConf.cutMultVtxHigh)) {
252  LOG(INFO) << "Found vertex mult. " << vtx.getNContributors() << " is outside of requested range "
253  << multEstConf.cutMultVtxLow << " : " << multEstConf.cutMultVtxHigh << " | ROF " << rof.getBCData();
254  continue; // skip vertex of unwanted multiplicity
255  }
256  vtxVecLoc.push_back(vtx);
257  }
258  if (vtxVecLoc.empty()) { // reject ROF
259  rof.setFirstEntry(first);
260  rof.setNEntries(0);
261  continue;
262  }
263  }
264 
265  if (mRunVertexer) {
266  event.addPrimaryVertices(vtxVecLoc);
267  } else {
268  event.addPrimaryVertex(0.f, 0.f, 0.f);
269  }
270  mTracker->setROFrame(roFrame);
271  mTracker->clustersToTracks(event);
272  tracks.swap(mTracker->getTracks());
273  LOG(INFO) << "Found tracks: " << tracks.size();
274  int number = tracks.size();
275  trackLabels.swap(mTracker->getTrackLabels());
276  rof.setFirstEntry(first);
277  rof.setNEntries(number);
278  copyTracks(tracks, allTracks, allClusIdx);
279  std::copy(trackLabels.begin(), trackLabels.end(), std::back_inserter(allTrackLabels));
280  trackLabels.clear();
281  vtxROF.setNEntries(vtxVecLoc.size());
282  for (const auto& vtx : vtxVecLoc) {
283  vertices.push_back(vtx);
284  }
285  if (number) {
286  irFrames.emplace_back(rof.getBCData(), rof.getBCData() + nBCPerTF - 1);
287  }
288  }
289  roFrame++;
290  }
291 
292  LOG(INFO) << "ITSTracker pushed " << allTracks.size() << " tracks";
293  if (mIsMC) {
294  pc.outputs().snapshot(Output{"ITS", "TRACKSMCTR", 0, Lifetime::Timeframe}, allTrackLabels);
295  pc.outputs().snapshot(Output{"ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe}, mc2rofs);
296  }
297  mTimer.Stop();
298 }

306 DataProcessorSpec getTrackerSpec(bool useMC, const std::string& trModeS, o2::gpu::GPUDataTypes::DeviceType dType)
307 {
308  std::vector<InputSpec> inputs;
309  inputs.emplace_back("compClusters", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
310  inputs.emplace_back("patterns", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
311  inputs.emplace_back("ROframes", "ITS", "CLUSTERSROF", 0, Lifetime::Timeframe);
312 
313  std::vector<OutputSpec> outputs;
314  outputs.emplace_back("ITS", "TRACKS", 0, Lifetime::Timeframe);
315  outputs.emplace_back("ITS", "TRACKCLSID", 0, Lifetime::Timeframe);
316  outputs.emplace_back("ITS", "ITSTrackROF", 0, Lifetime::Timeframe);
317  outputs.emplace_back("ITS", "VERTICES", 0, Lifetime::Timeframe);
318  outputs.emplace_back("ITS", "VERTICESROF", 0, Lifetime::Timeframe);
319  outputs.emplace_back("ITS", "IRFRAMES", 0, Lifetime::Timeframe);
320 
321  if (useMC) {
322  inputs.emplace_back("labels", "ITS", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
323  inputs.emplace_back("MC2ROframes", "ITS", "CLUSTERSMC2ROF", 0, Lifetime::Timeframe);
324  outputs.emplace_back("ITS", "TRACKSMCTR", 0, Lifetime::Timeframe);
325  outputs.emplace_back("ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe);
326  outputs.emplace_back("ITS", "VERTICES", 0, Lifetime::Timeframe);
327  }
328 
329  return DataProcessorSpec{
330  "its-tracker",
331  inputs,
332  outputs,
333  AlgorithmSpec{adaptFromTask<TrackerDPL>(useMC, trModeS, dType)},
334  Options{
335  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the grp file"}},
336  {"its-dictionary-path", VariantType::String, "", {"Path of the cluster-topology dictionary file"}},
337  {"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
338 }
