29 SymbolStatistics::SymbolStatistics(symbol_t min,
30  size_t scaleBits,
31  size_t nUsedAlphabetSymbols,
32  histogram_t&& frequencies) : mFrequencyTable{std::move(frequencies)},
33  mMin{min},
34  mSymbolTablePrecission{scaleBits},
35  mNUsedAlphabetSymbols{nUsedAlphabetSymbols}
36 {
37 
38  using namespace internal;
39  LOG(trace) << "start building symbol statistics";
40  RANSTimer t;
41  t.start();
42 
43  // calculate reonormalization size.
44  mSymbolTablePrecission = [&, this]() {
45  const size_t minScale = MIN_SCALE;
46  const size_t maxScale = MAX_SCALE;
47  size_t calculated = mSymbolTablePrecission > 0 ? mSymbolTablePrecission : static_cast<size_t>(3 * numBitsForNSymbols(this->getNUsedAlphabetSymbols()) / 2 + 2);
48  calculated = std::max(minScale, std::min(maxScale, calculated));
49  if (mSymbolTablePrecission > 0 && calculated != mSymbolTablePrecission) {
50  LOG(warning) << fmt::format("Normalization interval for rANS SymbolTable of {} Bits is outside of allowed range of {} - {} Bits. Setting to {} Bits",
51  mSymbolTablePrecission, minScale, maxScale, calculated);
52  }
53  return calculated;
54  }();
55 
56  //add a special symbol for incompressible data;
57  [this]() {
58  mFrequencyTable.push_back(1);
59  ++mNUsedAlphabetSymbols;
60  }();
61 
62  // range check
63  if (numBitsForNSymbols(mFrequencyTable.size()) > numSymbolsWithNBits(MAX_RANGE)) {
64  const std::string errmsg = fmt::format("Alphabet Range of {} Bits of the source message surpasses maximal allowed range of {} Bits.",
65  numBitsForNSymbols(mFrequencyTable.size()), MAX_RANGE);
66  LOG(error) << errmsg;
67  throw std::runtime_error(errmsg);
68  }
69 
70  buildCumulativeFrequencyTable();
71  rescale();
72 
73  assert(mFrequencyTable.size() > 0);
74  assert(mCumulativeFrequencyTable.size() == mFrequencyTable.size());
75 
76  t.stop();
77  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
78 
79 // advanced diagnostics in debug builds
80 #if !defined(NDEBUG)
81  LOG(debug2) << "SymbolStatistics: {"
82  << "entries: " << mFrequencyTable.size() << ", "
83  << "frequencyTableSizeB: " << mFrequencyTable.size() * sizeof(typename std::decay_t<decltype(mFrequencyTable)>::value_type) << ", "
84  << "CumulativeFrequencyTableSizeB: " << mCumulativeFrequencyTable.size() * sizeof(typename std::decay_t<decltype(mCumulativeFrequencyTable)>::value_type) << "}";
85 #endif
86 
87  if (mFrequencyTable.size() == 1) { // we do this check only after the adding the escape symbol
88  LOG(debug) << "Passed empty message to " << __func__; // RS this is ok for empty columns
89  }
90 
91  LOG(trace) << "done building symbol statistics";
92 }

94 void SymbolStatistics::rescale()
95 {
96 
97  // temporarily extend cumulative frequency Table to obtain total number of entries
98  mCumulativeFrequencyTable.push_back(mCumulativeFrequencyTable.back() + mFrequencyTable.back());
99 
100  auto getFrequency = [this](count_t i) { return mCumulativeFrequencyTable[i + 1] - mCumulativeFrequencyTable[i]; };
101 
102  using namespace internal;
103  LOG(trace) << "start rescaling frequency table";
104  RANSTimer t;
105  t.start();
106 
107  if (mFrequencyTable.empty()) {
108  LOG(warning) << "rescaling Frequency Table for empty message";
109  }
110 
111  const auto sortIdx = [&, this]() {
112  std::vector<size_t> indices;
113  indices.reserve(getNUsedAlphabetSymbols());
114 
115  // we will sort only those memorize only those entries which can be used
116  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
117  if (mFrequencyTable[i] != 0) {
118  indices.push_back(i);
119  }
120  }
121  std::sort(indices.begin(), indices.end(), [&](count_t i, count_t j) { return getFrequency(i) < getFrequency(j); });
122 
123  return indices;
124  }();
125 
126  // resample distribution based on cumulative frequencies
127  const count_t newCumulatedFrequency = pow2(mSymbolTablePrecission);
128  assert(newCumulatedFrequency >= this->getNUsedAlphabetSymbols());
129  const count_t cumulatedFrequencies = mCumulativeFrequencyTable.back();
130  size_t needsShift = 0;
131  for (size_t i = 0; i < sortIdx.size(); i++) {
132  if (static_cast<count_t>(getFrequency(sortIdx[i])) * (newCumulatedFrequency - needsShift) / cumulatedFrequencies >= 1) {
133  break;
134  }
135  needsShift++;
136  }
137 
138  size_t shift = 0;
139  auto beforeUpdate = mCumulativeFrequencyTable[0];
140  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
141  if (mFrequencyTable[i] && static_cast<uint64_t>(mCumulativeFrequencyTable[i + 1] - beforeUpdate) * (newCumulatedFrequency - needsShift) / cumulatedFrequencies < 1) {
142  shift++;
143  }
144  beforeUpdate = mCumulativeFrequencyTable[i + 1];
145  mCumulativeFrequencyTable[i + 1] = (static_cast<uint64_t>(newCumulatedFrequency - needsShift) * mCumulativeFrequencyTable[i + 1]) / cumulatedFrequencies + shift;
146  }
147  assert(shift == needsShift);
148 
149  //verify
150 #if !defined(NDEBUG)
151  assert(mCumulativeFrequencyTable.front() == 0 &&
152  mCumulativeFrequencyTable.back() == newCumulatedFrequency);
153  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
154  if (mFrequencyTable[i] == 0) {
155  assert(mCumulativeFrequencyTable[i + 1] == mCumulativeFrequencyTable[i]);
156  } else {
157  assert(mCumulativeFrequencyTable[i + 1] > mCumulativeFrequencyTable[i]);
158  }
159  }
160 #endif
161 
162  // calculate updated frequencies
163  for (size_t i = 0; i < mFrequencyTable.size(); i++) {
164  mFrequencyTable[i] = getFrequency(i);
165  }
166 
167  // remove added entry to cumulative Frequency table:
168  mCumulativeFrequencyTable.pop_back();
169  assert(mFrequencyTable.size() == mCumulativeFrequencyTable.size());
170 
171  t.stop();
172  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
173 
174  LOG(trace) << "done rescaling frequency table";
175 }

177 void SymbolStatistics::buildCumulativeFrequencyTable()
178 {
179  LOG(trace) << "start building cumulative frequency table";
180 
181  mCumulativeFrequencyTable.resize(mFrequencyTable.size());
182  mCumulativeFrequencyTable[0] = 0;
183  std::partial_sum(mFrequencyTable.begin(), --mFrequencyTable.end(),
184  ++mCumulativeFrequencyTable.begin());
185 
186  LOG(trace) << "done building cumulative frequency table";
187 }
