30 enum ParsingErrors { TRDParsingNoError,
31  TRDParsingUnrecognisedVersion,
32  TRDParsingBadDigt,
33  TRDParsingBadTracklet,
34  TRDParsingDigitEndMarkerWrongState, // read a end marker but we were expecting something else due to
35  TRDParsingDigitMCMHeaderSanityCheckFailure, //essentially we did not see an MCM header see RawData.h for requirement
36  TRDParsingDigitROBDecreasing, // sequential headers must have the same or increasing rob number
37  TRDParsingDigitMCMNotIncreasing, // sequential headers must have increasing mcm number
38  TRDParsingDigitADCMaskMismatch, // mask adc count does not match # of 1s in bitpattern
39  TRDParsingDigitADCMaskAdvanceToEnd, // in advancing to adcmask we have reached the end of the buffer
40  TRDParsingDigitMCMHeaderBypassButStateMCMHeader, // we are reading mcmadc data but the state is mcmheader
41  TRDParsingDigitEndMarkerStateButReadingMCMADCData, // read the endmarker while expecting to read the mcmadcdata
42  TRDParsingDigitADCChannel21, // ADCMask is zero but we are still on a digit.
43  TRDParsingDigitADCChannelGT22, // error allocating digit, so digit channel has error value
44  TRDParsingDigitGT10ADCs, // more than 10 adc data words seen
45  TRDParsingDigitSanityCheck, // adc failed sanity check see RawData.cxx for faiulre reasons
46  TRDParsingDigitExcessTimeBins, // ADC has more than 30 timebins (10 adc words)
47  TRDParsingDigitParsingExitInWrongState, // exiting parsing in the wrong state ... got to the end of the buffer in wrong state.
48  TRDParsingDigitStackMisMatch, // mismatch between rdh and hcheader stack calculation/value
49  TRDParsingDigitLayerMisMatch, // mismatch between rdh and hcheader stack calculation/value
50  TRDParsingDigitSectorMisMatch, // mismatch between rdh and hcheader stack calculation/value
51  TRDParsingTrackletCRUPaddingWhileParsingTracklets, // reading a padding word while expecting tracklet data
52  TRDParsingTrackletBit11NotSetInTrackletHCHeader, // bit 11 not set in hc header for tracklets.
53  TRDParsingTrackletHCHeaderSanityCheckFailure, // HCHeader sanity check failure, see RawData.cxx for reasons.
54  TRDParsingTrackletMCMHeaderSanityCheckFailure, // MCMHeader sanity check failure, see RawData.cxx for reasons.
55  TRDParsingTrackletMCMHeaderButParsingMCMData, // state is still MCMHeader but we are parsing MCMData
56  TRDParsingTrackletStateMCMHeaderButParsingMCMData,
57  TRDParsingTrackletTrackletCountGTThatDeclaredInMCMHeader, //mcmheader tracklet count does not match that in we have parsed.
58  TRDParsingTrackletInvalidTrackletCount, // invalid tracklet count in header vs data
59  TRDParsingTrackletPadRowIncreaseError, // subsequent padrow can not be less than previous one.
60  TRDParsingTrackletColIncreaseError, // subsequent col can not be less than previous one
61  TRDParsingTrackletNoTrackletEndMarker, // got to the end of the buffer with out finding a tracklet end marker.
62  TRDParsingTrackletExitingNoTrackletEndMarker // got to the end of the buffer exiting tracklet parsing with no tracklet end marker
63 };
