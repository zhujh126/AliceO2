93 FIT_CALIBRATION_DEVICE_TEMPLATES
94 void FIT_CALIBRATION_DEVICE_TYPE::_sendOutputs(o2::framework::DataAllocator& outputs)
95 {
96 
97  using clbUtils = o2::calibration::Utils;
98  const auto& objectsToSend = mCalibrator->getStoredCalibrationObjects();
99 
100  uint32_t iSendChannel = 0;
101  for (const auto& [ccdbInfo, calibObject] : objectsToSend) {
102  outputs.snapshot(o2::framework::Output{clbUtils::gDataOriginCDBPayload, "FIT_CALIB", iSendChannel}, *calibObject);
103  outputs.snapshot(o2::framework::Output{clbUtils::gDataOriginCDBWrapper, "FIT_CALIB", iSendChannel}, ccdbInfo);
104  ++iSendChannel;
105  }
106  mCalibrator->initOutput();
107 }
