35 void dEdxHistos::fill(const gsl::span<const TrackTPC> tracks)
36 {
37  for (const auto& track : tracks) {
38 
39  // applying cut
40  if (!mApplyCuts || mCuts.goodTrack(track)) {
41  // filling histogram
42  if (track.hasASideClustersOnly()) {
43  mEntries[0]++;
44  mHist[0].fill(track.getdEdx().dEdxTotTPC);
45  } else if (track.hasCSideClustersOnly()) {
46  mEntries[1]++;
47  mHist[1].fill(track.getdEdx().dEdxTotTPC);
48  }
49  }
50  }

53 void dEdxHistos::merge(const dEdxHistos* other)
54 {
55  for (size_t i = 0; i < mHist.size(); i++) {
56  const auto binCount = mHist[i].getNBins();
57  for (size_t bin = 0; bin < binCount; bin++) {
58  float bin_content = other->getHists()[i].getBinContent(bin);
59  mHist[i].fillBin(bin, bin_content);
60  }
61  }
