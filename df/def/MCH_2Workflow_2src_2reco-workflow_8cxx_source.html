47 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
48 {
49  WorkflowSpec specs;
50 
51  auto disableRootOutput = configcontext.options().get<bool>("disable-root-output");
52  auto disableRootInput = configcontext.options().get<bool>("disable-root-input");
53  auto useMC = !configcontext.options().get<bool>("disable-mc");
54 
55  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
56 
57  if (!disableRootInput) {
58  specs.emplace_back(o2::mch::getDigitReaderSpec(useMC, "mch-sim-digit-reader"));
59  }
60  specs.emplace_back(o2::mch::getPreClusterFinderSpec("mch-precluster-finder"));
61  specs.emplace_back(o2::mch::getClusterFinderOriginalSpec("mch-cluster-finder"));
62  specs.emplace_back(o2::mch::getClusterTransformerSpec());
63  specs.emplace_back(o2::mch::getTrackFinderSpec("mch-track-finder"));
64 
65  if (!disableRootOutput) {
66  if (useMC) {
67  specs.emplace_back(o2::mch::getTrackMCLabelFinderSpec("mch-track-mc-label-finder"));
68  }
69  specs.emplace_back(o2::mch::getTrackWriterSpec(useMC, "mch-track-writer", "mchtracks.root"));
70  }
71 
72  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
73  o2::raw::HBFUtilsInitializer hbfIni(configcontext, specs);
74 
75  // write the configuration used for the reco workflow
76  o2::conf::ConfigurableParam::writeINI("o2mchrecoflow_configuration.ini");
77 
78  return specs;
79 }
